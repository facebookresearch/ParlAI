42 def get_no_affordance_actions(in_room, carrying, other_carrying, other_name):
43  all_actions = []
44  # Drop "a" prefix from all
45  in_room = [i[2:] for i in in_room]

47  other_carrying = [i[2:] for i in other_carrying]
48  for obj in other_carrying:
49  all_actions.append('steal {} from {}'.format(obj, other_name))
50  for obj in in_room:
51  all_actions.append('get {}'.format(obj))
52  for obj in carrying:

54  all_actions.append(f.format(obj))
55  all_actions.append('give {} to {}'.format(obj, other_name))
56  for obj2 in in_room:
57  all_actions.append('put {} in {}'.format(obj, obj2))
58  for obj2 in carrying:
59  if obj2 == obj:
60  continue
61  all_actions.append('put {} in {}'.format(obj, obj2))
62  return list(set(all_actions))
