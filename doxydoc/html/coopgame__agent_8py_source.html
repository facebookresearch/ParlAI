122  """Convert text observaton (string) to a ``torch.autograd.Variable`` of
123  tokens using ``DictionaryAgent``.
124  """
125  text_tokens = self.dict.txt2vec(text)

132  """Convert a ``torch.autograd.Variable`` of tokens into a string."""
133  text_tokens = vec
134  if isinstance(text_tokens, Variable):
135  text_tokens = list(text_tokens.data)

138  def observe(self, observation):
139  """Update state, given a previous reply by other agent. In case of
140  questioner, it can be goal description at start of episode.
141  """
142  self.observation = observation
143 
144  # if episode not done, tokenize, embed and update state
145  # at the end of dialog episode, perform backward pass and step
146  if not observation.get('episode_done', False):
147  text_tokens = self.tokenize(observation['text'])
148  token_embeds = self.listen_net(text_tokens)
149  if 'image' in observation:
150  token_embeds = torch.cat((token_embeds, observation['image']), 1)
151  token_embeds = token_embeds.squeeze(1)
152  self.h_state, self.c_state = self.state_net(
153  token_embeds, (self.h_state, self.c_state)
154  )
155  else:
156  if observation.get('reward', None):
157  for action in self.actions:
158  action.reinforce(observation['reward'])
159  autograd_backward(
160  self.actions, [None for _ in self.actions], retain_graph=True
161  )
162  # clamp all gradients between (-5, 5)
163  for module in self.modules:
164  for parameter in module.parameters():
165  parameter.grad.data.clamp_(min=-5, max=5)
166  self.optimizer.step()
167  else:
168  # start of dialog episode
169  self.optimizer.zero_grad()
170  self.reset()

172  def act(self):
173  """Based on current state, utter a reply (string) for next round."""
174  out_distr = self.speak_net(self.h_state)
175  if self.opt['datatype'] == 'train':
176  action = out_distr.multinomial()
177  else:
178  _, action = out_distr.max(1)
179  action = action.unsqueeze(1)
180  self.actions.append(action)
181  action_text = self.detokenize(action.squeeze(1))
182  return {'text': action_text, 'id': self.id}

185  """Reset internal state (and actions, if specified)."""
186  # TODO(kd): share state across other instances during batch training
187  self.h_state = Variable(torch.zeros(1, self.opt['hidden_size']))
188  self.c_state = Variable(torch.zeros(1, self.opt['hidden_size']))
189  if self.use_cuda:
190  self.h_state, self.c_state = self.h_state.cuda(), self.c_state.cuda()
191 
192  if not retain_actions:
193  self.actions = []

262  def predict(self, tasks, num_tokens):
263  """Extra method to be executed at the end of episode to carry out goal
264  and decide reward on the basis of prediction.
265  """
266  guess_tokens = []
267  for _ in range(num_tokens):
268  # explicit task dependence
269  task_embeds = self.listen_net(tasks)
270  prediction = self.predict_net(task_embeds, (self.h_state, self.c_state))
271  guess_tokens.append(prediction)
272  return guess_tokens
