329  # Uses a weak lock file to try to prevent clobbering between threads
330  if not self._should_use_time_logs():
331  return # sandbox doesn't check logs
332  file_path = os.path.join(parent_dir, TIME_LOGS_FILE_NAME)
333  file_lock = os.path.join(parent_dir, TIME_LOGS_FILE_LOCK)
334  with LockFile(file_lock) as _lock_file:
335  assert _lock_file is not None
336  if os.path.exists(file_path):
337  with open(file_path, 'rb+') as time_log_file:
338  existing_times = pickle.load(time_log_file)
339  # Initial loads should only reset if it's been a day,
340  # otherwise only need to check an hour for safety
341  compare_time = 24 * 60 * 60 if init_load else 60 * 60
342  if (
343  time.time() - existing_times['last_reset'] < compare_time
344  and not force
345  ):
346  return # do nothing if it's been less than a day
347  reset_workers = list(existing_times.keys())
348  reset_workers.remove('last_reset')
349  if len(reset_workers) != 0:
350  self.worker_manager.un_time_block_workers(reset_workers)
351 
352  # Reset the time logs
353  os.remove(file_path)
354  # new time logs
355  with open(file_path, 'wb+') as time_log_file:
356  time_logs = {'last_reset': time.time()}
357  pickle.dump(time_logs, time_log_file, pickle.HIGHEST_PROTOCOL)
