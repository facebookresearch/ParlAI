65 def download(url, path, fname, redownload=False):
66  """
67  Download file using `requests`.
68 
69  If ``redownload`` is set to false, then
70  will not download tar file again if it is present (default ``True``).
71  """
72  outfile = os.path.join(path, fname)
73  download = not os.path.isfile(outfile) or redownload
74  print("[ downloading: " + url + " to " + outfile + " ]")
75  retry = 5
76  exp_backoff = [2 ** r for r in reversed(range(retry))]
77 
78  pbar = tqdm.tqdm(unit='B', unit_scale=True, desc='Downloading {}'.format(fname))
79 
80  while download and retry >= 0:
81  resume_file = outfile + '.part'
82  resume = os.path.isfile(resume_file)
83  if resume:
84  resume_pos = os.path.getsize(resume_file)
85  mode = 'ab'
86  else:
87  resume_pos = 0
88  mode = 'wb'
89  response = None
90 
91  with requests.Session() as session:
92  try:
93  header = (
94  {'Range': 'bytes=%d-' % resume_pos, 'Accept-Encoding': 'identity'}
95  if resume
96  else {}
97  )
98  response = session.get(url, stream=True, timeout=5, headers=header)
99 
100  # negative reply could be 'none' or just missing
101  if resume and response.headers.get('Accept-Ranges', 'none') == 'none':
102  resume_pos = 0
103  mode = 'wb'
104 
105  CHUNK_SIZE = 32768
106  total_size = int(response.headers.get('Content-Length', -1))
107  # server returns remaining size if resuming, so adjust total
108  total_size += resume_pos
109  pbar.total = total_size
110  done = resume_pos
111 
112  with open(resume_file, mode) as f:
113  for chunk in response.iter_content(CHUNK_SIZE):
114  if chunk: # filter out keep-alive new chunks
115  f.write(chunk)
116  if total_size > 0:
117  done += len(chunk)
118  if total_size < done:
119  # don't freak out if content-length was too small
120  total_size = done
121  pbar.total = total_size
122  pbar.update(len(chunk))
123  break
124  except requests.exceptions.ConnectionError:
125  retry -= 1
126  pbar.clear()
127  if retry >= 0:
128  print('Connection error, retrying. (%d retries left)' % retry)
129  time.sleep(exp_backoff[retry])
130  else:
131  print('Retried too many times, stopped retrying.')
132  finally:
133  if response:
134  response.close()
135  if retry < 0:
136  raise RuntimeWarning('Connection broken too many times. Stopped retrying.')
137 
138  if download and retry > 0:
139  pbar.update(done - pbar.n)
140  if done < total_size:
141  raise RuntimeWarning(
142  'Received less data than specified in '
143  + 'Content-Length header for '
144  + url
145  + '.'
146  + ' There may be a download problem.'
147  )
148  move(resume_file, outfile)
149 
150  pbar.close()

211 def cat(file1, file2, outfile, deleteFiles=True):
212  """Concatenate two files to an outfile, possibly deleting the originals."""
213  with open(outfile, 'wb') as wfd:
214  for f in [file1, file2]:
215  with open(f, 'rb') as fd:
216  shutil.copyfileobj(fd, wfd, 1024 * 1024 * 10)
217  # 10MB per writing chunk to avoid reading big file into memory.
218  if deleteFiles:
219  os.remove(file1)
220  os.remove(file2)

347 def download_multiprocess(
348  urls, path, num_processes=32, chunk_size=100, dest_filenames=None, error_path=None
349 ):
350  """
351  Download items in parallel (e.g. for an image + dialogue task)
352 
353  Note: "of threading, multiprocess and pytorch.multiprocessing pick two".
354  These three don't all play well together. On OS X, may hang upon successful finish.
355 
356  :param urls: Array of urls to download
357  :param path: directory to save items in
358  :param num_processes: number of processes to use
359  :param chunk_size: chunk size to use
360  :param dest_filenames: optional array of same length as url with filenames.
361  Images will be saved as path + dest_filename
362  :param error_path: where to save error logs
363  :return: array of tuples of (destination filename, http status code, error
364  message if any). Note that upon failure, file may not actually be created.
365  """
366 
367  pbar = tqdm.tqdm(total=len(urls), position=0)
368 
369  # Resume TODO: isfile() may take too long ?? Should I try in a .tmp file
370  if dest_filenames:
371  if len(dest_filenames) != len(urls):
372  raise Exception(
373  'If specified, destination filenames must equal url array in length.'
374  )
375  else:
376 
377  def _naming_fn(url, url_metadata=None):
378  return hashlib.md5(url.encode('utf-8')).hexdigest()
379 
380  dest_filenames = [_naming_fn(url) for url in urls]
381 
382  items = zip(urls, dest_filenames)
383  remaining_items = [
384  it for it in items if not os.path.isfile(os.path.join(path, it[1]))
385  ]
386  print(
387  f'Of {len(urls)} items, {len(urls) - len(remaining_items)} already existed; only going to download {len(remaining_items)} items.'
388  )
389  pbar.update(len(urls) - len(remaining_items))
390 
391  pool_chunks = (
392  (remaining_items[i : i + chunk_size], path, _download_multiprocess_single)
393  for i in range(0, len(remaining_items), chunk_size)
394  )
395  remaining_chunks_count = math.ceil(float(len(remaining_items) / chunk_size))
396  print(
397  f'Going to download {remaining_chunks_count} chunks with {chunk_size} images per chunk using {num_processes} processes.'
398  )
399 
400  pbar.desc = 'Downloading'
401  all_results = []
402  collected_errors = []
403 
404  with Pool(num_processes) as pool:
405  for idx, chunk_result in enumerate(
406  pool.imap_unordered(_download_multiprocess_map_chunk, pool_chunks, 2)
407  ):
408  all_results.extend(chunk_result)
409  for dest_file, http_status_code, error_msg in chunk_result:
410  if http_status_code != 200:
411  # msg field available as third item in the tuple
412  # not using b/c error log file would blow up
413  collected_errors.append(
414  {
415  'dest_file': dest_file,
416  'status_code': http_status_code,
417  'error': error_msg,
418  }
419  )
420  print(
421  f'Bad download - chunk: {idx}, dest_file: {dest_file}, http status code: {http_status_code}, error_msg: {error_msg}'
422  )
423  pbar.update(len(chunk_result))
424  pbar.close()
425 
426  if error_path:
427  now = time.strftime("%Y%m%d-%H%M%S")
428  error_filename = os.path.join(
429  error_path, 'parlai_download_multiprocess_errors_%s.log' % now
430  )
431 
432  with open(os.path.join(error_filename), 'w+') as error_file:
433  error_file.write(json.dumps(collected_errors))
434  print('Summary of errors written to %s' % error_filename)
435 
436  print(
437  'Of %s items attempted downloading, %s had errors.'
438  % (len(remaining_items), len(collected_errors))
439  )
440 
441  print('Finished downloading chunks.')
442  return all_results

449  Apparently function must be pickable (which apparently means must be

451  imap_unordered. Has to do with how it's passed to the subprocess.

457  """

464 def _download_multiprocess_single(url, path, dest_fname):
465  """
466  Helper function to download an individual item.
467 
468  Unlike download() above, does not deal with downloading chunks of a big
469  file, does not support retries (and does not fail if retries are exhausted).
470 
471  :param url: URL to download from
472  :param path: directory to save in
473  :param dest_fname: destination file name of image
474  :return tuple (dest_fname, http status)
475  """
476 
477  status = None
478  error_msg = None
479  try:
480  # 'User-Agent' header may need to be specified
481  headers = {}
482 
483  # Use smaller timeout to skip errors, but can result in failed downloads
484  response = requests.get(
485  url, stream=False, timeout=10, allow_redirects=True, headers=headers
486  )
487  except Exception as e:
488  # Likely a timeout during fetching but had an error in requests.get()
489  status = 500
490  error_msg = '[Exception during download during fetching] ' + str(e)
491  return dest_fname, status, error_msg
492 
493  if response.ok:
494  try:
495  with open(os.path.join(path, dest_fname), 'wb+') as out_file:
496  # Some sites respond with gzip transport encoding
497  response.raw.decode_content = True
498  out_file.write(response.content)
499  status = 200
500  except Exception as e:
501  # Likely a timeout during download or decoding
502  status = 500
503  error_msg = '[Exception during decoding or writing] ' + str(e)
504  else:
505  # We get here if there is an HTML error page (i.e. a page saying "404
506  # not found" or anything else)
507  status = response.status_code
508  error_msg = '[Response not OK] Response: %s' % response
509 
510  return dest_fname, status, error_msg
