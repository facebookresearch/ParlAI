244  def get(self, episode_idx, entry_idx):
245  conversation = self.convos[episode_idx]
246  conv_len = len(conversation['utterances'])
247  utterance = conversation['utterances'][entry_idx]['text']
248

261  tmp += [annot["name"]]
262  action['label_types'] += [tmp]
