40 def pad(tensor, length, dim=0, pad=0):
41  """Pad tensor to a specific length.
42

44  :param length: new length
45  :param dim: (default 0) dimension to pad
46 
47  :returns: padded tensor if the tensor is shorter than length
48  """

164  def _decode_forced(self, ys, encoder_states):
165  """Decode with teacher forcing."""
166  bsz = ys.size(0)
167  seqlen = ys.size(1)
168 
169  hidden = encoder_states[1]
170  attn_params = (encoder_states[0], encoder_states[2])
171 
172  # input to model is START + each target except the last
173  y_in = ys.narrow(1, 0, seqlen - 1)
174  xs = torch.cat([self._starts(bsz), y_in], 1)
175 
176  scores = []
177  if self.attn_type == 'none':
178  # do the whole thing in one go
179  output, hidden = self.decoder(xs, hidden, attn_params)
180  score = self.output(output)
181  scores.append(score)
182  else:
183  # need to feed in one token at a time so we can do attention
184  # TODO: do we need to do this? actually shouldn't need to since we
185  # don't do input feeding
186  for i in range(seqlen):
187  xi = xs.select(1, i).unsqueeze(1)
188  output, hidden = self.decoder(xi, hidden, attn_params)
189  score = self.output(output)
190  scores.append(score)
191

195  def _decode(self, encoder_states, maxlen):
196  """Decode maxlen tokens."""
197  hidden = encoder_states[1]
198  attn_params = (encoder_states[0], encoder_states[2])
199  bsz = encoder_states[0].size(0)
200 
201  xs = self._starts(bsz) # input start token
202 
203  scores = []
204  for _ in range(maxlen):
205  # generate at most longest_label tokens
206  output, hidden = self.decoder(xs, hidden, attn_params)
207  score = self.output(output)
208  scores.append(score)
209  xs = score.max(2)[1] # next input is current predicted output

214  def _align_inds(self, encoder_states, cand_inds):
215  """Select the encoder states relevant to valid candidates."""
216  enc_out, hidden, attn_mask = encoder_states
217 
218  # LSTM or GRU/RNN hidden state?
219  if isinstance(hidden, torch.Tensor):
220  hid, cell = hidden, None
221  else:
222  hid, cell = hidden
223 
224  if len(cand_inds) != hid.size(1):
225  # if the number of candidates is mismatched from the number of
226  # hidden states, we throw out the hidden states we won't rank with
227  cand_indices = hid.new(cand_inds)
228  hid = hid.index_select(1, cand_indices)
229  if cell is None:
230  hidden = hid
231  else:
232  cell = cell.index_select(1, cand_indices)
233  hidden = (hid, cell)
234 
235  if self.attn_type != 'none':
236  enc_out = enc_out.index_select(0, cand_indices)
237  attn_mask = attn_mask.index_select(0, cand_indices)
238 
239  return enc_out, hidden, attn_mask

241  def _extract_cur(self, encoder_states, index, num_cands):
242  """Extract encoder states at current index and expand them."""
243  enc_out, hidden, attn_mask = encoder_states
244  if isinstance(hidden, torch.Tensor):
245  cur_hid = hidden.select(1, index).unsqueeze(1).expand(-1, num_cands, -1)
246  else:
247  cur_hid = (
248  hidden[0]
249  .select(1, index)
250  .unsqueeze(1)
251  .expand(-1, num_cands, -1)

253  hidden[1]
254  .select(1, index)
255  .unsqueeze(1)
256  .expand(-1, num_cands, -1)
257  .contiguous(),

259

266  def _rank(self, cands, cand_inds, encoder_states):
267  """Rank each cand by the average log-probability of the sequence."""
268  if cands is None:
269  return None
270  encoder_states = self._align_inds(encoder_states, cand_inds)
271 
272  cand_scores = []
273  for batch_idx in range(len(cands)):
274  # we do one set of candidates at a time
275  curr_cs = cands[batch_idx]
276  num_cands = curr_cs.size(0)
277 
278  # select just the one hidden state
279  cur_enc_states = self._extract_cur(encoder_states, batch_idx, num_cands)
280 
281  score = self._decode_forced(curr_cs, cur_enc_states)
282  true_score = F.log_softmax(score, dim=2).gather(2, curr_cs.unsqueeze(2))
283  nonzero = curr_cs.ne(0).float()
284  scores = (true_score.squeeze(2) * nonzero).sum(1)
285  seqlens = nonzero.sum(1)
286  scores /= seqlens
287  cand_scores.append(scores)
288 
289  max_len = max(len(c) for c in cand_scores)
290  cand_scores = torch.cat(
291  [pad(c, max_len, pad=self.NULL_IDX).unsqueeze(0) for c in cand_scores], 0
292  )
293  return cand_scores

295  def forward(
296  self, xs, ys=None, cands=None, prev_enc=None, maxlen=None, seq_len=None
297  ):
298  """Get output predictions from the model.
299 
300  :param xs: (bsz x seqlen) LongTensor input to the encoder
301  :param ys: expected output from the decoder. used for teacher
302  forcing to calculate loss.
303  :param cands: set of candidates to rank
304  :param prev_enc: if you know you'll pass in the same xs multiple
305  times, you can pass in the encoder output from the
306  last forward pass to skip recalcuating the same
307  encoder output.
308  :param maxlen: max number of tokens to decode. if not set, will
309  use the length of the longest label this model
310  has seen. ignored when ys is not None.
311  :param seq_len this is the sequence length of the input (xs), i.e.
312  xs.size(1). we use this to recover the proper
313  output sizes in the case when we distribute over
314  multiple gpus
315 
316  :returns: scores, candidate scores, and encoder states
317  scores contains the model's predicted token scores.
318  (bsz x seqlen x num_features)
319  candidate scores are the score the model assigned to each candidate
320  (bsz x num_cands)
321  encoder states are the (output, hidden, attn_mask) states from the
322  encoder. feed this back in to skip encoding on the next call.
323  """
324  if ys is not None:
325  # keep track of longest label we've ever seen
326  # we'll never produce longer ones than that during prediction
327  self.longest_label = max(self.longest_label, ys.size(1))
328 
329  encoder_states = self._encode(xs, prev_enc)
330 
331  # rank candidates if they are available
332  cand_scores = None
333  if cands is not None:
334  cand_inds = [i for i in range(cands.size(0))]
335  cand_scores = self._rank(cands, cand_inds, encoder_states)
336 
337  if ys is not None:
338  # use teacher forcing
339  scores = self._decode_forced(ys, encoder_states)
340  else:
341  scores = self._decode(encoder_states, maxlen or self.longest_label)
342 
343  if seq_len is not None:
344  # when using multiple gpus, we need to make sure output of
345  # encoder is correct size for gathering; we recover this with
346  # the parameter seq_len
347  if encoder_states[0].size(1) < seq_len:
348  out_pad_tensor = torch.zeros(
349  encoder_states[0].size(0),
350  seq_len - encoder_states[0].size(1),
351  encoder_states[0].size(2),
352  ).cuda()
353  new_out = torch.cat([encoder_states[0], out_pad_tensor], 1)
354  encoder_states = (new_out, encoder_states[1], encoder_states[2])
355 
356  return scores, cand_scores, encoder_states

437  def forward(self, xs):
438  """Encode sequence.
439 
440  :param xs: (bsz x seqlen) LongTensor of input token indices
441 
442  :returns: encoder outputs, hidden state, attention mask
443  encoder outputs are the output state at each step of the encoding.
444  the hidden state is the final hidden state of the encoder.
445  the attention mask is a mask of which input values are nonzero.
446  """
447  bsz = len(xs)
448 
449  # embed input tokens
450  xs = self.input_dropout(xs)
451  xes = self.dropout(self.lt(xs))
452  attn_mask = xs.ne(0)
453  try:
454  x_lens = torch.sum(attn_mask.int(), dim=1)
455  xes = pack_padded_sequence(xes, x_lens, batch_first=True)
456  packed = True
457  except ValueError:
458  # packing failed, don't pack then
459  packed = False
460 
461  encoder_output, hidden = self.rnn(xes)
462  if packed:
463  encoder_output, _ = pad_packed_sequence(encoder_output, batch_first=True)
464  if self.dirs > 1:
465  # project to decoder dimension by taking sum of forward and back
466  if isinstance(self.rnn, nn.LSTM):
467  hidden = (
468  hidden[0].view(-1, self.dirs, bsz, self.hsz).sum(1),
469  hidden[1].view(-1, self.dirs, bsz, self.hsz).sum(1),
470  )
471  else:
472  hidden = hidden.view(-1, self.dirs, bsz, self.hsz).sum(1)
473 
474  return encoder_output, hidden, attn_mask

527  def forward(self, xs, hidden=None, attn_params=None):
528  """Decode from input tokens.
529 
530  :param xs: (bsz x seqlen) LongTensor of input token indices
531  :param hidden: hidden state to feed into decoder. default (None)
532  initializes tensors using the RNN's defaults.
533  :param attn_params: (optional) tuple containing attention parameters,

538  output state of the encoder. for an RNN, this is
539  (bsz, seq_len, num_directions * hiddensize).
540  hidden state will be same dimensions as input
541  hidden state. for an RNN, this is a tensor of sizes
542  (bsz, numlayers * num_directions, hiddensize).
543  """
544  # sequence indices => sequence embeddings
545  xes = self.dropout(self.lt(xs))
546

548  # modify input vectors with attention
549  xes, _attw = self.attention(xes, hidden, attn_params)
550

553

555  # modify output vectors with attention
556  output, _attw = self.attention(output, new_hidden, attn_params)

561 class OutputLayer(nn.Module):
562  """Takes in final states and returns distribution over candidates."""
563 
564  def __init__(
565  self,
566  num_features,
567  embeddingsize,
568  hiddensize,
569  dropout=0,
570  numsoftmax=1,
571  shared_weight=None,
572  padding_idx=-1,
573  ):
574  """Initialize output layer.
575 
576  :param num_features: number of candidates to rank
577  :param hiddensize: (last) dimension of the input vectors
578  :param embeddingsize: (last) dimension of the candidate vectors
579  :param numsoftmax: (default 1) number of softmaxes to calculate.
580  see arxiv.org/abs/1711.03953 for more info.
581  increasing this slows down computation but can
582  add more expressivity to the embeddings.
583  :param shared_weight: (num_features x esz) vector of weights to use as
584  the final linear layer's weight matrix. default
585  None starts with a new linear layer.
586  :param padding_idx: model should output a large negative number for
587  score at this index. if set to -1 (default),
588  this is disabled. if >= 0, subtracts one from
589  num_features and always outputs -1e20 at this
590  index. only used when shared_weight is not None.
591  setting this param helps protect gradient from
592  entering shared embedding matrices.
593  """
594  super().__init__()
595  self.dropout = nn.Dropout(p=dropout)
596 
597  self.padding_idx = padding_idx if shared_weight is not None else -1
598 
599  # embedding to scores
600  if shared_weight is None:
601  # just a regular linear layer
602  self.e2s = nn.Linear(embeddingsize, num_features, bias=True)
603  else:
604  # use shared weights and a bias layer instead
605  if padding_idx == 0:
606  num_features -= 1 # don't include padding
607  shared_weight = shared_weight.narrow(0, 1, num_features)
608  elif padding_idx > 0:
609  raise RuntimeError('nonzero pad_idx not yet implemented')
610  self.weight = Parameter(shared_weight)
611  self.bias = Parameter(torch.Tensor(num_features))
612  self.reset_parameters()
613  self.e2s = lambda x: F.linear(x, self.weight, self.bias)
614 
615  self.numsoftmax = numsoftmax
616  if numsoftmax > 1:
617  self.esz = embeddingsize
618  self.softmax = nn.Softmax(dim=1)
619  self.prior = nn.Linear(hiddensize, numsoftmax, bias=False)
620  self.latent = nn.Linear(hiddensize, numsoftmax * embeddingsize)
621  self.activation = nn.Tanh()
622  else:
623  # rnn output to embedding
624  if hiddensize != embeddingsize:
625  # learn projection to correct dimensions
626  self.o2e = nn.Linear(hiddensize, embeddingsize, bias=True)
627  else:
628  # no need for any transformation here
629  self.o2e = lambda x: x
630 
631  def reset_parameters(self):
632  """Reset bias param."""
633  if hasattr(self, 'bias'):
634  stdv = 1.0 / math.sqrt(self.bias.size(0))
635  self.bias.data.uniform_(-stdv, stdv)
636 
637  def forward(self, input):
638  """Compute scores from inputs.
639 
640  :param input: (bsz x seq_len x num_directions * hiddensize) tensor of
641  states, e.g. the output states of an RNN
642 
643  :returns: (bsz x seqlen x num_cands) scores for each candidate
644  """
645  # next compute scores over dictionary
646  if self.numsoftmax > 1:
647  bsz = input.size(0)
648  seqlen = input.size(1) if input.dim() > 1 else 1
649 
650  # first compute different softmax scores based on input vec
651  # hsz => numsoftmax * esz
652  latent = self.latent(input)
653  active = self.dropout(self.activation(latent))
654  # esz => num_features
655  logit = self.e2s(active.view(-1, self.esz))
656 
657  # calculate priors: distribution over which softmax scores to use
658  # hsz => numsoftmax
659  prior_logit = self.prior(input).view(-1, self.numsoftmax)
660  # softmax over numsoftmax's
661  prior = self.softmax(prior_logit)
662 
663  # now combine priors with logits
664  prob = self.softmax(logit).view(bsz * seqlen, self.numsoftmax, -1)
665  probs = (prob * prior.unsqueeze(2)).sum(1).view(bsz, seqlen, -1)
666  scores = probs.log()
667  else:
668  # hsz => esz, good time for dropout
669  e = self.dropout(self.o2e(input))
670  # esz => num_features
671  scores = self.e2s(e)
672 
673  if self.padding_idx == 0:
674  pad_score = scores.new(scores.size(0), scores.size(1), 1).fill_(-NEAR_INF)
675  scores = torch.cat([pad_score, scores], dim=-1)
676 
677  return scores

686  def __init__(
687  self,
688  attn_type,
689  hiddensize,
690  embeddingsize,
691  bidirectional=False,
692  attn_length=-1,
693  attn_time='pre',
694  ):
695  """Initialize attention layer."""
696  super().__init__()
697  self.attention = attn_type
698 
699  if self.attention != 'none':
700  hsz = hiddensize
701  hszXdirs = hsz * (2 if bidirectional else 1)
702  if attn_time == 'pre':
703  # attention happens on the input embeddings
704  input_dim = embeddingsize
705  elif attn_time == 'post':
706  # attention happens on the output of the rnn
707  input_dim = hsz
708  else:
709  raise RuntimeError('unsupported attention time')
710 
711  # linear layer for combining applied attention weights with input
712  self.attn_combine = nn.Linear(hszXdirs + input_dim, input_dim, bias=False)
713 
714  if self.attention == 'local':
715  # local attention over fixed set of output states
716  if attn_length < 0:
717  raise RuntimeError('Set attention length to > 0.')
718  self.max_length = attn_length
719  # combines input and previous hidden output layer
720  self.attn = nn.Linear(hsz + input_dim, attn_length, bias=False)
721  # combines attention weights with encoder outputs
722  elif self.attention == 'concat':
723  self.attn = nn.Linear(hsz + hszXdirs, hsz, bias=False)
724  self.attn_v = nn.Linear(hsz, 1, bias=False)
725  elif self.attention == 'general':
726  # equivalent to dot if attn is identity
727  self.attn = nn.Linear(hsz, hszXdirs, bias=False)
728 
729  def forward(self, xes, hidden, attn_params):
730  """Compute attention over attn_params given input and hidden states.
731 
732  :param xes: input state. will be combined with applied
733  attention.
734  :param hidden: hidden state from model. will be used to select
735  states to attend to in from the attn_params.
736  :param attn_params: tuple of encoder output states and a mask showing
737  which input indices are nonzero.
738 
739  :returns: output, attn_weights
740  output is a new state of same size as input state `xes`.
741  attn_weights are the weights given to each state in the
742  encoder outputs.
743  """
744  if self.attention == 'none':
745  # do nothing, no attention
746  return xes, None
747 
748  if type(hidden) == tuple:
749  # for lstms use the "hidden" state not the cell state
750  hidden = hidden[0]
751  last_hidden = hidden[-1] # select hidden state from last RNN layer
752 
753  enc_out, attn_mask = attn_params
754  bsz, seqlen, hszXnumdir = enc_out.size()
755  numlayersXnumdir = last_hidden.size(1)
756 
757  if self.attention == 'local':
758  # local attention weights aren't based on encoder states
759  h_merged = torch.cat((xes.squeeze(1), last_hidden), 1)
760  attn_weights = F.softmax(self.attn(h_merged), dim=1)
761 
762  # adjust state sizes to the fixed window size
763  if seqlen > self.max_length:
764  offset = seqlen - self.max_length
765  enc_out = enc_out.narrow(1, offset, self.max_length)
766  seqlen = self.max_length
767  if attn_weights.size(1) > seqlen:
768  attn_weights = attn_weights.narrow(1, 0, seqlen)
769  else:
770  hid = last_hidden.unsqueeze(1)
771  if self.attention == 'concat':
772  # concat hidden state and encoder outputs
773  hid = hid.expand(bsz, seqlen, numlayersXnumdir)
774  h_merged = torch.cat((enc_out, hid), 2)
775  # then do linear combination of them with activation
776  active = F.tanh(self.attn(h_merged))
777  attn_w_premask = self.attn_v(active).squeeze(2)
778  elif self.attention == 'dot':
779  # dot product between hidden and encoder outputs
780  if numlayersXnumdir != hszXnumdir:
781  # enc_out has two directions, so double hid
782  hid = torch.cat([hid, hid], 2)
783  enc_t = enc_out.transpose(1, 2)
784  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
785  elif self.attention == 'general':
786  # before doing dot product, transform hidden state with linear
787  # same as dot if linear is identity
788  hid = self.attn(hid)
789  enc_t = enc_out.transpose(1, 2)
790  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
791 
792  # calculate activation scores, apply mask if needed
793  if attn_mask is not None:
794  # remove activation from NULL symbols
795  attn_w_premask.masked_fill_((1 - attn_mask), -NEAR_INF)
796  attn_weights = F.softmax(attn_w_premask, dim=1)
797 
798  # apply the attention weights to the encoder states
799  attn_applied = torch.bmm(attn_weights.unsqueeze(1), enc_out)
800  # concatenate the input and encoder states
801  merged = torch.cat((xes.squeeze(1), attn_applied.squeeze(1)), 1)
802  # combine them with a linear layer and tanh activation
803  output = torch.tanh(self.attn_combine(merged).unsqueeze(1))
804 
805  return output, attn_weights
