320  def parallel_eval_mode(self):
321  """Parallel function that shuts one agent down and asks the other
322  to do the evaluation if their are two agents. If there is only
323  one agent, it performs the evaluation.

327  def eval_or_shutdown(agent):
328  if self.model_agent is None and agent == self.other_agent:
329  control_msg = {'episode_done': False}
330  control_msg['id'] = 'SYSTEM'
331  control_msg['text'] = OTHER_AGENT_FINISHED_MSG
332  self.other_agent.observe(validate(control_msg))
333  # mark eval agent done
334  self.eval_agent.mturk_manager.mark_workers_done([self.eval_agent])
335  # shutdown other agent
336  self.other_agent.shutdown()
337  else:
338  control_msg = {'episode_done': False}
339  control_msg['id'] = 'SYSTEM'
340  # General mark for this convo
341  control_msg['text'] = GMARK_MSG
342  control_msg['general_mark_score'] = True
343  self.eval_agent.observe(validate(control_msg))
344  act = self.eval_agent.act(timeout=self.max_resp_time)
345  timeout = self.check_timeout(act)
346  if timeout:
347  return
348  while act['text'] not in self.ratings:
349  control_msg['text'] = NAN_MSG
350  self.eval_agent.observe(validate(control_msg))
351  act = self.eval_agent.act(timeout=self.max_resp_time)

353  self.gmark_score = int(act['text'])
