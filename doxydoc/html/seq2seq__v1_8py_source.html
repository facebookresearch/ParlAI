24 class Seq2seqAgent(TorchAgent):
25  """Agent which takes an input sequence and produces an output sequence.
26 
27  This model supports encoding the input and decoding the output via one of
28  several flavors of RNN. It then uses a linear layer (whose weights can
29  be shared with the embedding layer) to convert RNN output states into
30  output tokens. This model supports greedy decoding, selecting the
31  highest probability token at each time step, as well as beam
32  search.
33 
34  For more information, see the following papers:
35  - Neural Machine Translation by Jointly Learning to Align and Translate
36  `(Bahdanau et al. 2014) <arxiv.org/abs/1409.0473>`_
37  - Sequence to Sequence Learning with Neural Networks
38  `(Sutskever et al. 2014) <arxiv.org/abs/1409.3215>`_
39  - Effective Approaches to Attention-based Neural Machine Translation
40  `(Luong et al. 2015) <arxiv.org/abs/1508.04025>`_
41  """

427  def train_step(self, batch):
428  """Train on a single batch of examples."""
429  batchsize = batch.text_vec.size(0)
430  if self.multigpu and batchsize % 2 != 0:
431  # throw out one training example
432  batch = self.truncate_input(batch)
433  # helps with memory usage
434  self._init_cuda_buffer(
435  self.model, self.criterion, batchsize, self.truncate or 180
436  )
437  self.model.train()
438  self.zero_grad()
439 
440  try:
441  seq_len = None if not self.multigpu else batch.text_vec.size(1)
442  out = self.model(batch.text_vec, batch.label_vec, seq_len=seq_len)
443 
444  # generated response
445  scores = out[0]
446  _, preds = scores.max(2)
447 
448  score_view = scores.view(-1, scores.size(-1))
449  loss = self.criterion(score_view, batch.label_vec.view(-1))
450  # save loss to metrics
451  notnull = batch.label_vec.ne(self.NULL_IDX)
452  target_tokens = notnull.long().sum().item()
453  correct = ((batch.label_vec == preds) * notnull).sum().item()
454  self.metrics['correct_tokens'] += correct
455  self.metrics['loss'] += loss.item()
456  self.metrics['num_tokens'] += target_tokens
457  loss /= target_tokens # average loss per token
458  loss.backward()
459  self.update_params()
460  except RuntimeError as e:
461  # catch out of memory exceptions during fwd/bck (skip batch)
462  if 'out of memory' in str(e):
463  print(
464  '| WARNING: ran out of memory, skipping batch. '
465  'if this happens frequently, decrease batchsize or '
466  'truncate the inputs to the model.'
467  )
468  self.metrics['total_skipped_batches'] += 1
469  else:
470  raise e

500  @staticmethod
501  def beam_search(
502  model,
503  batch,
504  beam_size,
505  start=1,
506  end=2,
507  pad=0,
508  min_length=3,
509  min_n_best=5,
510  max_ts=40,
511  block_ngram=0,
512  ):
513  """ Beam search given the model and Batch
514  This function uses model with the following reqs:
515  - model.encoder takes input returns tuple (enc_out, enc_hidden, attn_mask)
516  - model.decoder takes decoder params and returns decoder outputs after attn
517  - model.output takes decoder outputs and returns distr over dictionary
518 
519  Function arguments:
520  model : nn.Module, here defined in modules.py
521  batch : Batch structure with input and labels
522  beam_size : Size of each beam during the search
523  start : start of sequence token
524  end : end of sequence token
525  pad : padding token
526  min_length : minimum length of the decoded sequence
527  min_n_best : minimum number of completed hypothesis generated from each beam
528  max_ts: the maximum length of the decoded sequence
529 
530  Return:
531  beam_preds_scores : list of tuples (prediction, score) for each sample in Batch
532  n_best_preds_scores : list of n_best list of tuples (prediction, score) for
533  each sample from Batch
534  beams : list of Beam instances defined in Beam class, can be used for any
535  following postprocessing, e.g. dot logging.
536  """
537  encoder_states = model.encoder(batch.text_vec)
538  enc_out = encoder_states[0]
539  enc_hidden = encoder_states[1]
540  attn_mask = encoder_states[2]
541  current_device = encoder_states[0][0].device
542 
543  batch_size = len(batch.text_lengths)
544  beams = [
545  Beam(
546  beam_size,
547  min_length=min_length,
548  padding_token=pad,
549  bos_token=start,
550  eos_token=end,
551  min_n_best=min_n_best,
552  cuda=current_device,
553  block_ngram=block_ngram,
554  )
555  for i in range(batch_size)
556  ]
557  decoder_input = (
558  torch.Tensor([start])
559  .detach()
560  .expand(batch_size, 1)
561  .long()
562  .to(current_device)
563  )
564  # repeat encoder_outputs, hiddens, attn_mask
565  decoder_input = decoder_input.repeat(1, beam_size).view(
566  beam_size * batch_size, -1
567  )
568  enc_out = (
569  enc_out.unsqueeze(1)
570  .repeat(1, beam_size, 1, 1)
571  .view(batch_size * beam_size, -1, enc_out.size(-1))
572  )
573  attn_mask = (
574  encoder_states[2]
575  .repeat(1, beam_size)
576  .view(attn_mask.size(0) * beam_size, -1)
577  )
578  repeated_hiddens = []
579  if isinstance(enc_hidden, tuple): # LSTM
580  for i in range(len(enc_hidden)):
581  repeated_hiddens.append(
582  enc_hidden[i].unsqueeze(2).repeat(1, 1, beam_size, 1)
583  )
584  num_layers = enc_hidden[0].size(0)
585  hidden_size = enc_hidden[0].size(-1)
586  enc_hidden = tuple(
587  [
588  repeated_hiddens[i].view(
589  num_layers, batch_size * beam_size, hidden_size
590  )
591  for i in range(len(repeated_hiddens))
592  ]
593  )
594  else: # GRU
595  num_layers = enc_hidden.size(0)
596  hidden_size = enc_hidden.size(-1)
597  enc_hidden = (
598  enc_hidden.unsqueeze(2)
599  .repeat(1, 1, beam_size, 1)
600  .view(num_layers, batch_size * beam_size, hidden_size)
601  )
602 
603  hidden = enc_hidden
604  for ts in range(max_ts):
605  if all((b.done() for b in beams)):
606  break
607  output, hidden = model.decoder(decoder_input, hidden, (enc_out, attn_mask))
608  score = model.output(output)
609  # score contains softmax scores for batch_size * beam_size samples
610  score = score.view(batch_size, beam_size, -1)
611  score = F.log_softmax(score, dim=-1)
612  for i, b in enumerate(beams):
613  if not b.done():
614  b.advance(score[i])
615  decoder_input = torch.cat(
616  [b.get_output_from_current_step() for b in beams]
617  ).unsqueeze(-1)
618  permute_hidden_idx = torch.cat(
619  [
620  beam_size * i + b.get_backtrack_from_current_step()
621  for i, b in enumerate(beams)
622  ]
623  )
624  # permute decoder hiddens with respect to chosen hypothesis now
625  if isinstance(hidden, tuple): # LSTM
626  for i in range(len(hidden)):
627  hidden[i].data.copy_(
628  hidden[i].data.index_select(dim=1, index=permute_hidden_idx)
629  )
630  else: # GRU
631  hidden.data.copy_(
632  hidden.data.index_select(dim=1, index=permute_hidden_idx)
633  )
634  for b in beams:
635  b.check_finished()
636 
637  beam_preds_scores = [list(b.get_top_hyp()) for b in beams]
638  for pair in beam_preds_scores:
639  pair[0] = Beam.get_pretty_hypothesis(pair[0])
640 
641  n_best_beams = [b.get_rescored_finished(n_best=min_n_best) for b in beams]
642  n_best_beam_preds_scores = []
643  for i, beamhyp in enumerate(n_best_beams):
644  this_beam = []
645  for hyp in beamhyp:
646  pred = beams[i].get_pretty_hypothesis(

743  seq_len = None if not self.multigpu else batch.text_vec.size(1)

751  # save loss to metrics
752  notnull = orig_batch.label_vec.ne(self.NULL_IDX)
753  target_tokens = notnull.long().sum().item()
754  correct = ((orig_batch.label_vec == f_preds) * notnull).sum().item()
755  self.metrics['correct_tokens'] += correct
756  self.metrics['loss'] += loss.item()
757  self.metrics['num_tokens'] += target_tokens
758

762  # now select the text of the cands based on their scores
763  cand_choices = self._pick_cands(
764  cand_preds, cand_params[1], orig_batch.candidates

835  """Initialize evaluator."""
836  if opt.get('multigpu'):
837  print(

839  + 'Evaluator Agent. Setting this option to False.'

846  def next_word_probability(self, partial_out):
847  """Return probability distribution over next words.
848 
849  This probability is based on both nn input and partial true output.
850  This is used to calculate the per-word perplexity.
851 
852  Arguments:
853  observation -- input observation dict
854  partial_out -- list of previous "true" words
855 
856  Returns a dict, where each key is a word and each value is a
857  probability score for that word.
858  Unset keys will use a probability of 1e-7.
859 
860  e.g.
861  {'text': 'Run test program.'}, ['hello'] => {'world': 1.0}
862  """
863  obs = self.observation
864  xs = obs['text_vec'].unsqueeze(0)
865  ys = self._vectorize_text(
866  ' '.join(partial_out), False, True, self.truncate
867  ).unsqueeze(0)
868  if (
869  self.prev_enc is not None
870  and self.last_xs is not None
871  and (
872  xs.shape[1] != self.last_xs.shape[1]
873  or (xs == self.last_xs).sum().item() != xs.shape[1]
874  )
875  ):
876  # reset prev_enc, this is a new input
877  self.prev_enc = None
878  self.last_xs = xs
879 
880  self.model.eval()
881  out = self.model(
882  xs,
883  ys=(ys if len(partial_out) > 0 else None),
884  prev_enc=self.prev_enc,
885  maxlen=1,
886  )
887  scores, self.prev_enc = out[0], out[2]
888  # scores is bsz x seqlen x num_words, so select probs of current index
889  probs = F.softmax(scores.select(1, -1), dim=1).squeeze()
890  dist = mydefaultdict(lambda: 1e-7) # default probability for any token
891  for i in range(len(probs)):
892  dist[self.dict[i]] = probs[i].item()
893  return dist
