14 def readFiles(dpath, rfnames):
15  en_fname, de_fname = rfnames
16  url_base = 'https://nlp.stanford.edu/projects/nmt/data/wmt14.en-de/'
17  en_url = url_base + en_fname
18  de_url = url_base + de_fname
19  build_data.download(en_url, dpath, en_fname)
20  build_data.download(de_url, dpath, de_fname)
21  with open(os.path.join(dpath, en_fname), 'r') as f:
22  # We replace '##AT##-##AT##' as a workaround in order to use the
23  # nltk tokenizer specified by DictionaryAgent
24  en = [l[:-1].replace('##AT##-##AT##', '__AT__') for l in f]
25 
26  with open(os.path.join(dpath, de_fname), 'r') as f:
27  de = [l[:-1].replace('##AT##-##AT##', '__AT__') for l in f]
28 
29  return list(zip(de, en))
