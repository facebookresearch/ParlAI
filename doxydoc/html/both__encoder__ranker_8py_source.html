82  def train_step(self, batch):
83  self.biencoder.train_step(batch)
84  # the crossencoder requires batches that are smaller.
85  # split the batch into smaller pieces
86  outc = []
87  step = self.crossencoder_batchsize
88  for start in range(0, len(batch.text_vec), step):
89  mbatch = Batch(
90  text_vec=batch.text_vec[start : start + step],
91  label_vec=batch.label_vec[start : start + step],
92  candidate_vecs=batch.candidate_vecs[start : start + step],
93  candidates=batch.candidates[start : start + step],
94  )
95  outc.append(self.crossencoder.train_step(mbatch))
96  return Output(text=[text for out in outc for text in out.text])

98  def eval_step(self, batch):
99  """ We pass the batch first in the biencoder, then filter with crossencoder
100  """
101  output_biencoder = self.biencoder.eval_step(batch)
102  if output_biencoder is None:

104  new_candidate_vecs = [

107  ]
108  new_candidates = [

111  if cands is not None
112  ]
113  copy_batch = Batch(
114  text_vec=batch.text_vec,
115  candidate_vecs=new_candidate_vecs,
116  candidates=new_candidates,
