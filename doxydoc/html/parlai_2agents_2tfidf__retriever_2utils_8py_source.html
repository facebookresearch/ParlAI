261 def filter_ngram(gram, mode='any'):
262  """Decide whether to keep or discard an n-gram.
263 
264  Args:
265  gram: list of tokens (length N)
266  mode: Option to throw out ngram if
267  'any': any single token passes filter_word
268  'all': all tokens pass filter_word
269  'ends': book-ended by filterable tokens
270  """
271  filtered = [filter_word(w) for w in gram]
272  if mode == 'any':
273  return any(filtered)
274  elif mode == 'all':
275  return all(filtered)
276  elif mode == 'ends':
277  return filtered[0] or filtered[-1]
278  else:
279  raise ValueError('Invalid mode: %s' % mode)
