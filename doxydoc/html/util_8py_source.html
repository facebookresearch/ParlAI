21  def __init__(self, text, assume_persontokens=True, dictionary=None):
22  """
23  Separates text (the dialogue context) into:
24  self.persona_lines: list of strings; the persona lines with "your persona:"
25  removed.
26  self.partner_utts: list of strings; the partner's utterances.
27  self.own_utts: list of strings; the bot's own utterances.
28  All of the above are lowercase and tokenized.
29 
30  Inputs:
31  text: string. Contains several lines separated by \n. The first few lines are
32  persona lines beginning "your persona: ", then the next lines are dialogue.
33  assume_persontokens: If True, assert that the dialogue lines start with
34  __p1__ and __p2__ respectively, and then remove them.
35  dictionary: parlai DictionaryAgent.
36  """
37  p1_token, p2_token = TorchAgent.P1_TOKEN, TorchAgent.P2_TOKEN
38 
39  # Split text into lines
40  lines = [t.strip() for t in text.split('\n')]
41 
42  # Identify the persona lines and remove "your persona"
43  persona_lines = [t for t in lines if "your persona:" in t]
44  persona_lines = [remove_prefix(pl, "your persona:") for pl in persona_lines]
45  persona_lines = [fix_personaline_period(pl) for pl in persona_lines]
46 
47  # Identify the dialogue lines. It's assumed that p1 goes first.
48  utts = lines[len(persona_lines) :]
49  p1_utts = [utts[i] for i in range(0, len(utts), 2)]
50  p2_utts = [utts[i] for i in range(1, len(utts), 2)]
51 
52  # Check for and remove the __p1__ and __p2__ prefixes
53  if assume_persontokens:
54  p1_utts = [remove_prefix(utt, p1_token) for utt in p1_utts]
55  p2_utts = [remove_prefix(utt, p2_token) for utt in p2_utts]
56 
57  # Tokenize and lowercase
58  if dictionary is not None:
59  persona_lines = [" ".join(dictionary.tokenize(pl)) for pl in persona_lines]
60  p1_utts = [" ".join(dictionary.tokenize(utt)) for utt in p1_utts]
61  p2_utts = [" ".join(dictionary.tokenize(utt)) for utt in p2_utts]
62 
63  # Strip trailing whitespace and discard any empty lines
64  self.persona_lines = [l.strip() for l in persona_lines if l.strip() != ""]
65  self.partner_utts = [l.strip() for l in p1_utts if l.strip() != ""]
66  self.own_utts = [l.strip() for l in p2_utts if l.strip() != ""]

90 def fix_personaline_period(line):
91  """
92  Sometimes the tokenized persona lines have a period at the end but no space before
93  the period. This function fixes it, e.g. changes
94  'my favorite color is blue.' to 'my favorite color is blue .'

141  # Go through the candidates, measuring their extrep_2gram level
142  # Optionally print out the original ordering
143  candidates = [] # list of (orig_idx, pred, text, score, extrep_2gram) tuples
144  if verbose:
145  print("\nORIGINAL ORDER:")
146  for idx, (pred, score) in enumerate(n_best_beam_preds):
147  text = dictionary.vec2txt(pred.tolist())
148  text = text.replace('__start__ ', '').replace(' __end__', '')
149  if verbose:
150  print("%i %.4f %s" % (idx, score, text))
151  extrep_2gram = eval_attr(text, history, 'extrep_2gram')
152  candidates.append((idx, pred, text, score, extrep_2gram))
153 
154  # Sort the candidates by ascending repetition. Tiebreak using original ranking.
155  candidates = sorted(candidates, key=lambda x: (x[4], x[0]))
156 
157  # Optionally print out the new ordering
158  if verbose:
159  print("\nSORTED BY EXTREP_2GRAM:")
160  for (idx, _, text, _, extrep_2gram) in candidates:
161  print("%i %.4f %s" % (idx, extrep_2gram, text))
162  print("")
163 
164  # Identify the top-ranked (w.r.t. new ordering) candidate that contains '?'
165  for (_, pred, text, score, _) in candidates:
166  if "?" not in text:
167  continue
168  return (pred, score)
169 
170  # If there was no candidate containing '?', return top-ranked (w.r.t extrep_2gram)
171  (_, pred, score, _, _) = candidates[0]
