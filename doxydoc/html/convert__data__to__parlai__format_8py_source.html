24 def dump_data(opt):
25  # create repeat label agent and assign it to the specified task
26  agent = RepeatLabelAgent(opt)
27  world = create_task(opt, agent)
28  ignorefields = opt.get('ignore_fields', '')
29  if opt['outfile'] is None:
30  outfile = tempfile.mkstemp(
31  prefix='{}_{}_'.format(opt['task'], opt['datatype']), suffix='.txt'
32  )[1]
33  else:
34  outfile = opt['outfile']
35 
36  if opt['num_examples'] == -1:
37  num_examples = world.num_examples()
38  else:
39  num_examples = opt['num_examples']
40  log_timer = TimeLogger()
41 
42  print('[ starting to convert.. ]')
43  print('[ saving output to {} ]'.format(outfile))
44  fw = open(outfile, 'w')
45  for _ in range(num_examples):
46  world.parley()
47  acts = world.get_acts()
48  value = acts[0].get('labels', acts[0].pop('eval_labels', None))
49  acts[0].force_set('labels', value)
50  txt = msg_to_str(acts[0], ignore_fields=ignorefields)
51  fw.write(txt + '\n')
52  if acts[0].get('episode_done', False):
53  fw.write('\n')
54 
55  if log_timer.time() > opt['log_every_n_secs']:
56  text, _log = log_timer.log(world.total_parleys, world.num_examples())
57  print(text)
58 
59  if world.epoch_done():
60  print('EPOCH DONE')
61  break
62  fw.close()
