206 def load_eval_worlds(agent, opt, datatype):
207  """
208  Create a new eval world for the agent and the given opt.
209 
210  Overrides the datatype options for doing this. Handles some magic
211  overrides of other special options for the training script.
212 
213  :param Agent agent:
214  The model being trained.
215 
216  :param Opt opt:
217  The global CLI opts.
218 
219  :param string datatype:
220  The new datatype.
221  """
222  if 'stream' in opt['datatype']:
223  datatype += ':stream'
224  opt = opt.copy()
225  opt['datatype'] = datatype
226  if opt.get('pytorch_teacher_task'):
227  # never use pytorch teachers for evaluation
228  # but don't forget what we were normally using
229  opt['task'] = opt['pytorch_teacher_task']
230  del opt['pytorch_teacher_task']
231  if opt.get('evaltask'):
232  # if a different eval task is specified, use it.
233  opt['task'] = opt['evaltask']
234  if opt.get('eval_batchsize'):
235  # override eval time batchsize
236  opt['batchsize'] = opt['eval_batchsize']
237 
238  tasks = opt['task'].split(',')
239  worlds = []
240  # possibly load agent
241  if opt.get('validation_share_agent', False):
242  valid_agent = create_agent_from_shared(agent.share())
243  else:
244  valid_agent = agent
245  # create worlds
246  for task in tasks:
247  task_opt = opt.copy() # copy opt since we edit the task
248  task_opt['task'] = task
249  valid_world = create_task(task_opt, valid_agent)
250  worlds.append(valid_world)
251 
252  return worlds

255 def _run_single_eval(opt, valid_world, max_exs):
256  # run evaluation on a single world
257  valid_world.reset()
258 
259  cnt = 0
260  max_cnt = max_exs if max_exs > 0 else float('inf')
261  while not valid_world.epoch_done() and cnt < max_cnt:
262  valid_world.parley()
263  if cnt == 0 and opt['display_examples']:
264  print(valid_world.display() + '\n~~')
265  print(valid_world.report())
266  cnt += valid_world.opt['batchsize']
267

269  valid_world.reset() # make sure world doesn't remember valid data
270

274 def run_eval(valid_worlds, opt, datatype, max_exs=-1, write_log=False):
275  """
276  Eval on validation/test data.
277 
278  :param valid_world:
279  list of the pre-created validation worlds.
280  :param opt:
281  the options that specific the task, eval_task, etc
282  :param datatype:
283  the datatype to use, such as "valid" or "test"
284  :param bool write_log:
285  specifies to write metrics to file if the model_file is set
286  :param int max_exs:
287  limits the number of examples if max_exs > 0
288  """
289  if valid_worlds is None:
290  # This isn't the primary worker, so we can just skip evaluation

307  print(metrics)

311  # Write out metrics

473  def validate(self):
474  """
475  Perform a validation run, checking whether we should stop training.
476 
477  :return: boolean indicating whether training should stop
478  :rtype: bool
479  """
480  opt = self.opt
481 
482  if self.valid_worlds is None:
483  # we need to load the world now
484  self.valid_worlds = _maybe_load_eval_worlds(self.agent, opt, 'valid')
485 
486  # run evaluation on valid set
487  valid_report = sync_object(
488  run_eval(self.valid_worlds, opt, 'valid', opt['validation_max_exs'])
489  )
490  v = valid_report.copy()
491  v['train_time'] = self.train_time.time()
492  self.valid_reports.append(v)
493  # logging
494  if opt['tensorboard_log'] and is_primary_worker():
495  self.tb_logger.log_metrics('valid', self.parleys, valid_report)
496  # saving
497  if (
498  opt.get('model_file')
499  and opt.get('save_after_valid')
500  and is_primary_worker()
501  ):
502  print("[ saving model checkpoint: " + opt['model_file'] + ".checkpoint ]")
503  self.save_model('.checkpoint')
504 
505  # send valid metrics to agent if the agent wants them
506  if hasattr(self.agent, 'receive_metrics'):
507  self.agent.receive_metrics(valid_report)
508 
509  # check which metric to look at
510  if 'tasks' in valid_report and '/' in opt['validation_metric']:
511  # if you are multitasking and want your validation metric to be
512  # a metric specific to a subtask, specify your validation metric
513  # as -vmt subtask/metric
514  subtask = opt['validation_metric'].split('/')[0]
515  validation_metric = opt['validation_metric'].split('/')[1]
516  new_valid = valid_report['tasks'][subtask][validation_metric]
517  else:
518  new_valid = valid_report[opt['validation_metric']]
519 
520  # check if this is the best validation so far
521  if (
522  self.best_valid is None
523  or self.valid_optim * new_valid > self.valid_optim * self.best_valid
524  ):
525  print(
526  '[ new best {}: {}{} ]'.format(
527  opt['validation_metric'],
528  new_valid,
529  ' (previous best was {})'.format(self.best_valid)
530  if self.best_valid is not None
531  else '',
532  )
533  )
534  self.best_valid = new_valid
535  self.impatience = 0
536  if opt.get('model_file') and is_primary_worker():
537  print("[ saving best valid model: " + opt['model_file'] + " ]")
538  self.save_model()
539  print(
540  "[ saving best valid metric: " + opt['model_file'] + ".best_valid ]"
541  )
542  _save_best_valid(opt['model_file'], self.best_valid)
543  self.saved = True
544  if (
545  opt['validation_metric'] == 'accuracy'
546  and self.best_valid >= opt['validation_cutoff']
547  ):
548  print('[ task solved! stopping. ]')
549  return True
550  else:
551  self.impatience += 1
552  print(
553  '[ did not beat best {}: {} impatience: {} ]'.format(
554  opt['validation_metric'], round(self.best_valid, 4), self.impatience
555  )
556  )
557  self.validate_time.reset()
558 
559  # check if we are out of patience
560  if (
561  opt['validation_patience'] > 0
562  and self.impatience >= opt['validation_patience']
563  ):
564  print('[ ran out of patience! stopping training. ]')
565  return True
566  return False

568  def _average_dicts(self, all_versions):
569  # instead of a list-of-dicts with like keys, make a dict-of-lists with
570  # keys to reduce
571  to_reduce = {}
572  for d in all_versions:
573  for k, v in d.items():
574  to_reduce.setdefault(k, []).append(v)
575  # now perform the reduction
576  finalized = {}
577  for k, values in to_reduce.items():
578  if k == 'exs' or k == 'total_skipped_batches':
579  # sum across workers
580  finalized[k] = np.sum(values)
581  elif isinstance(values[0], dict):
582  # do the same procedure recursively
583  finalized[k] = self._average_dicts(values)
584  elif isinstance(values[0], str):
585  finalized[k] = values[0]
586  else:
587  # all other cases, take the mean across the workers
588  finalized[k] = np.mean(values)
589  if all(isinstance(v, int) for v in values):
590  finalized[k] = int(finalized[k])
591  return finalized

633  def _compute_eta(self, epochs_completed, time_elapsed):
634  """
635  Compute the estimated seconds remaining in training.
636 
637  :param float epochs_completed: number of epochs already completed.
638  :param float time_elapsed: total time spent already, in seconds.
639  :return: ETA in seconds, or None if not computable
640  """
641  # start off with no estimate
642  eta = None
643 
644  # Determine time_left and num_epochs
645  max_epochs = self.opt.get('num_epochs', 0)
646  if max_epochs > 0 and epochs_completed > 0:
647  epoch_progress = epochs_completed / max_epochs
648  eta = (1 - epoch_progress) * time_elapsed / epoch_progress
649 
650  max_training_time = self.opt.get('max_training_time', -1)
651  if max_training_time > 0:
652  time_left = max_training_time - time_elapsed
653  if eta is None or time_left < eta:
654  eta = time_left
655 
656  return eta
