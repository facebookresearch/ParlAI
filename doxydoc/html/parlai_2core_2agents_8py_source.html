7 """
8 Common Abstract classes for many agents.
9 
10 This module provides a set of basic agents:
11 
12  ``Agent(object)``
13  base class for all other agents, implements the ``observe()`` method
14  which receives an observation/action dict and the ``act()`` method which
15  returns a dict in response.
16 
17  ``Teacher(Agent)``
18  also implements the ``report()`` method for returning metrics. All ParlAI
19  tasks implement the ``Teacher`` class.
20 
21  ``MultiTaskTeacher(Teacher)``
22  creates a set of teachers based on a task string passed to the ``Teacher``,
23  creating multiple teachers within it and alternating between them.
24 
25 All agents are initialized with the following parameters:
26 
27  ``opt`` -- contains any options needed to set up the agent. This generally contains
28  all command-line arguments recognized from ``core.params``, as well as other
29  options that might be set through the framework to enable certain modes.
30 
31  ``shared`` (optional) -- if not ``None``, contains any shared data used to construct
32  this particular instantiation of the agent. This data might have been
33  initialized by another agent, so that different agents can share the same
34  data (possibly in different Processes).
35 
36 This module also provides a utility method:
37 
38  ``create_task_agents(str)``: instantiate task-specific agents (e.g. a teacher)
39  from a given task string (e.g. 'babi:task1k:1' or 'squad'). Used by
40  ``MultiTaskTeacher``.
41 
42 """

134  @classmethod
135  def upgrade_opt(cls, opt_from_disk):
136  """
137  Upgrade legacy options when loading an opt file from disk.
138 
139  This is primarily made available to provide a safe space to handle
140  backwards-compatible behavior. For example, perhaps we introduce a
141  new option today, which wasn't previously available. We can have the
142  argument have a new default, but fall back to the "legacy" compatibility
143  behavior if the option doesn't exist.
144 
145  ``upgrade_opt`` provides an opportunity for such checks for backwards
146  compatibility. It is called shortly after loading the opt file from
147  disk, and is called before the Agent is initialized.
148 
149  Other possible examples include:
150 
151  1. Renaming an option,
152  2. Deprecating an old option,
153  3. Splitting coupled behavior, etc.
154 
155  Implementations of ``upgrade_opt`` should conform to high standards,
156  due to the risk of these methods becoming complicated and difficult to
157  reason about. We recommend the following behaviors:
158 
159  1. ``upgrade_opt`` should only be used to provide backwards
160  compatibility. Other behavior should find a different location.
161  2. Children should always call the parent's ``upgrade_opt`` first.
162  3. ``upgrade_opt`` should always warn when an option was overwritten.
163  4. Include comments annotating the date and purpose of each upgrade.
164  5. Add an integration test which ensures your old work behaves
165  appropriately.
166 
167  :param Opt opt_from_disk:
168  The opt file, as loaded from the ``.opt`` file on disk.
169  :return:
170  The modified options
171  :rtype:
172  Opt
173  """
174  # 2019-07-11: currently a no-op.
175  return opt_from_disk

246 class MultiTaskTeacher(Teacher):
247  """
248  MultiTaskTeacher which teaches multiple tasks.
249 
250  Creates a teacher that is actually a set of teachers each based on a task
251  string -- each of these teachers will get called in turn,
252  either randomly or in order. They are all in the same world (they are the
253  same agent switching tasks).
254 
255  The task string format is described for the ``create_task_agents()``
256  function above.
257  """
258 
259  def __init__(self, opt, shared=None):
260  self.tasks = []
261  self.opt = opt
262 
263  self.id = opt['task']
264  if shared and 'tasks' in shared:
265  self.tasks = [create_agent_from_shared(t) for t in shared['tasks']]
266  else:
267  tasks = opt['task'].split(',')
268  for k in tasks:
269  k = k.strip()
270  if k:
271  opt_singletask = copy.deepcopy(opt)
272  opt_singletask['task'] = k
273  self.tasks.extend(create_task_agent_from_taskname(opt_singletask))
274  self.task_idx = -1
275  self.new_task = True
276  self.random = opt.get('datatype') == 'train'
277  # Make multi-task task probabilities.
278  self.cum_task_weights = [1] * len(self.tasks)
279  self.task_choices = range(len(self.tasks))
280  weights = self.opt.get('multitask_weights', [1])
281  sum = 0
282  for i in self.task_choices:
283  if len(weights) > i:
284  weight = weights[i]
285  else:
286  weight = 1
287  self.cum_task_weights[i] = weight + sum
288  sum += weight
289 
290  def num_examples(self):
291  """Return the number of examples."""
292  if not hasattr(self, 'num_exs'):
293  # num_examples is sum of all examples in all tasks
294  tasks_num_exs = [t.num_examples() for t in self.tasks]
295  if any(num is None for num in tasks_num_exs):
296  self.num_exs = None
297  else:
298  self.num_exs = sum(tasks_num_exs)
299  return self.num_exs
300 
301  def num_episodes(self):
302  """Return the number of episodes."""
303  if not hasattr(self, 'num_eps'):
304  # num_episodes is sum of all num_episodes in all tasks
305  tasks_num_eps = [t.num_episodes() for t in self.tasks]
306  if any(num is None for num in tasks_num_eps):
307  self.num_eps = None
308  else:
309  self.num_eps = sum(tasks_num_eps)
310  return self.num_eps
311 
312  def observe(self, observation):
313  """Make an observation."""
314  return self.tasks[self.task_idx].observe(observation)
315 
316  def act(self):
317  """Act on the previous observation."""
318  if self.new_task:
319  self.new_task = False
320  if self.random:
321  # select random teacher
322  self.task_idx = random.choices(
323  self.task_choices, cum_weights=self.cum_task_weights
324  )[0]
325  else:
326  # do at most one full loop looking for unfinished task
327  for _ in range(len(self.tasks)):
328  self.task_idx = (self.task_idx + 1) % len(self.tasks)
329  if not self.tasks[self.task_idx].epoch_done():
330  # if this task has examples ready, break
331  break
332  if self.tasks[self.task_idx].epoch_done():
333  # all tasks are done, so return empty action table
334  return {'episode_done': True}
335  t = self.tasks[self.task_idx].act()
336  if t['episode_done']:
337  self.new_task = True
338  return t
339 
340  def epoch_done(self):
341  """Return whether all subtasks are completed."""
342  for t in self.tasks:
343  if not t.epoch_done():
344  return False
345  return True
346 
347  # return transformed metrics showing total examples and accuracy if avail.
348  def report(self):
349  """Report aggregated metrics across all subtasks."""
350  return aggregate_metrics(self.tasks)
351 
352  def reset(self):
353  """Reset all subtasks."""
354  for t in self.tasks:
355  t.reset()
356 
357  def reset_metrics(self):
358  """Reset metrics for each subtask."""
359  for t in self.tasks:
360  t.reset_metrics()
361 
362  def save(self):
363  """Save each subtask."""
364  for t in self.tasks:
365  t.save()
366 
367  def share(self):
368  """Shares this teacher by sharing each subtask."""
369  shared = {}
370  shared['class'] = type(self)
371  shared['opt'] = self.opt
372  shared['tasks'] = [t.share() for t in self.tasks]
373  return shared
374 
375  def shutdown(self):
376  """Shutdown each agent."""
377  for t in self.tasks:
378  t.shutdown()

401 def compare_init_model_opts(opt, curr_opt):
402  """Print loud warning when `init_model` opts differ from previous configuration."""
403  if opt.get('init_model') is None:
404  return
405  opt['init_model'] = modelzoo_path(opt['datapath'], opt['init_model'])
406  optfile = opt['init_model'] + '.opt'
407  if not os.path.isfile(optfile):
408  return
409  init_model_opt = load_opt_file(optfile)
410 
411  extra_opts = {}
412  different_opts = {}
413  exempt_opts = [
414  'model_file',
415  'dict_file',
416  'override',
417  'starttime',
418  'init_model',
419  'batchindex',
420  ]
421 
422  # search through init model opts
423  for k, v in init_model_opt.items():
424  if (
425  k not in exempt_opts
426  and k in init_model_opt
427  and init_model_opt[k] != curr_opt.get(k)
428  ):
429  if isinstance(v, list):
430  if init_model_opt[k] != list(curr_opt[k]):
431  different_opts[k] = ','.join([str(x) for x in v])
432  else:
433  different_opts[k] = v
434 
435  # search through opts to load
436  for k, v in curr_opt.items():
437  if k not in exempt_opts and k not in init_model_opt:
438  if isinstance(v, list):
439  extra_opts[k] = ','.join([str(x) for x in v])
440  else:
441  extra_opts[k] = v
442 
443  # print warnings
444  extra_strs = ['{}: {}'.format(k, v) for k, v in extra_opts.items()]
445  if extra_strs:
446  print('\n' + '*' * 75)
447  print(
448  '[ WARNING ] : your model is being loaded with opts that do not '
449  'exist in the model you are initializing the weights with: '
450  '{}'.format(','.join(extra_strs))
451  )
452 
453  different_strs = [
454  '--{} {}'.format(k, v).replace('_', '-') for k, v in different_opts.items()
455  ]
456  if different_strs:
457  print('\n' + '*' * 75)
458  print(
459  '[ WARNING ] : your model is being loaded with opts that differ '
460  'from the model you are initializing the weights with. Add the '
461  'following args to your run command to change this: \n'
462  '\n{}'.format(' '.join(different_strs))
463  )
464  print('*' * 75)

467 def load_agent_module(opt):
468  """
469  Load agent options and module from file if opt file exists.
470 
471  Checks to see if file exists opt['model_file'] + ".opt"; if so, load up the
472  options from the file and use that to create an agent, loading the model
473  type from that file and overriding any options specified in that file when
474  instantiating the agent.
475 
476  If that file does not exist, return None.
477  """
478  model_file = opt['model_file']
479  optfile = model_file + '.opt'
480  if os.path.isfile(optfile):
481  new_opt = load_opt_file(optfile)
482  # TODO we need a better way to say these options are never copied...
483  if 'datapath' in new_opt:
484  # never use the datapath from an opt dump
485  del new_opt['datapath']
486  if 'batchindex' in new_opt:
487  # This saved variable can cause trouble if we switch to BS=1 at test time
488  del new_opt['batchindex']
489  # only override opts specified in 'override' dict
490  if opt.get('override'):
491  for k, v in opt['override'].items():
492  if str(v) != str(new_opt.get(k, None)):
493  print(
494  "[ warning: overriding opt['{}'] to {} ("
495  "previously: {} )]".format(k, v, new_opt.get(k, None))
496  )
497  new_opt[k] = v
498 
499  model_class = get_agent_module(new_opt['model'])
500 
501  # check for model version
502  if hasattr(model_class, 'model_version'):
503  curr_version = new_opt.get('model_version', 0)
504  if curr_version != model_class.model_version():
505  model = new_opt['model']
506  m = (
507  'It looks like you are trying to load an older version of'
508  ' the selected model. Change your model argument to use '
509  'the old version from parlai/agents/legacy_agents: for '
510  'example: `-m legacy:{m}:{v}` or '
511  '`--model parlai.agents.legacy_agents.{m}.{m}_v{v}:{c}`'
512  )
513  if '.' not in model:
514  # give specific error message if it's easy
515  raise RuntimeError(
516  m.format(m=model, v=curr_version, c=model_class.__name__)
517  )
518  else:
519  # otherwise generic one
520  raise RuntimeError(
521  m.format(m='modelname', v=curr_version, c='ModelAgent')
522  )
523 
524  if hasattr(model_class, 'upgrade_opt'):
525  new_opt = model_class.upgrade_opt(new_opt)
526 
527  # add model arguments to new_opt if they aren't in new_opt already
528  for k, v in opt.items():
529  if k not in new_opt:
530  new_opt[k] = v
531  new_opt['model_file'] = model_file
532  if not new_opt.get('dict_file'):
533  new_opt['dict_file'] = model_file + '.dict'
534  elif new_opt.get('dict_file') and not os.path.isfile(new_opt['dict_file']):
535  old_dict_file = new_opt['dict_file']
536  new_opt['dict_file'] = model_file + '.dict'
537  if not os.path.isfile(new_opt['dict_file']):
538  warn_once(
539  'WARNING: Neither the specified dict file ({}) nor the '
540  '`model_file`.dict file ({}) exists, check to make sure either '
541  'is correct. This may manifest as a shape mismatch later '
542  'on.'.format(old_dict_file, new_opt['dict_file'])
543  )
544 
545  # if we want to load weights from --init-model, compare opts with
546  # loaded ones
547  compare_init_model_opts(opt, new_opt)
548  return model_class(new_opt)
549  else:
550  return None

553 def get_agent_module(dir_name):
554  """
555  Return the module for an agent specified by ``--model``.
556 
557  Can be formatted in several different ways:
558 
559  * full: `-m parlai.agents.seq2seq.seq2seq:Seq2seqAgent`
560  * shorthand: -m seq2seq, which will check both paths
561  ``parlai.agents.seq2seq.seq2seq:Seq2seqAgent`` and
562  ``parlai.agents.seq2seq.agents:Seq2seqAgent``
563  * half-shorthand: ``-m seq2seq/variant``, which will check the path
564  `parlai.agents.seq2seq.variant:VariantAgent`
565  * legacy models: ``-m legacy:seq2seq:0``, which will look for the deprecated
566  model at ``parlai.agents.legacy_agents.seq2seq.seq2seq_v0:Seq2seqAgent``
567 
568  The base path to search when using shorthand formats can be changed from
569  "parlai" to "parlai_internal" by prepending "internal:" to the path, e.g.
570  "internal:seq2seq".
571 
572  To use legacy agent versions, you can prepend "legacy:" to model arguments,
573  e.g. "legacy:seq2seq:0" will translate to ``legacy_agents/seq2seq/seq2seq_v0``.
574 
575  To use agents in projects, you can prepend "projects:" and the name of the
576  project folder to model arguments, e.g. "projects:personachat:kvmemnn"
577  will translate to ``projects/personachat/kvmemnn``.
578 
579  :param dir_name: path to model class in one of the above formats.
580  """
581  repo = 'parlai'
582  if dir_name.startswith('internal:'):
583  # To switch to local repo, useful for non-public projects
584  # (make a directory called 'parlai_internal' with your private agents)
585  # this will follow the same paths but look in parlai_internal instead
586  repo = 'parlai_internal'
587  dir_name = dir_name[9:]
588 
589  if dir_name.startswith('legacy:'):
590  # e.g. -m legacy:seq2seq:0
591  # will check legacy_agents.seq2seq.seq2seq_v0:Seq2seqAgent
592  s = dir_name.split(':')
593  if len(s) != 3:
594  raise RuntimeError(
595  'legacy paths should follow pattern '
596  'legacy:model:version; you used {}'
597  ''.format(dir_name)
598  )
599  model_name = s[1] # seq2seq
600  module_name = 'parlai.agents.legacy_agents.{m}.{m}_v{v}'.format(
601  m=model_name, v=s[2]
602  )
603  class_name = name_to_agent_class(model_name)
604  elif dir_name.startswith('projects:'):
605  # e.g. -m projects:personachat:kvmemnn
606  s = dir_name.split(':')
607  if len(s) != 3:
608  raise RuntimeError(
609  'projects paths should follow pattern '
610  'projects:folder:model; you used {}'
611  ''.format(dir_name)
612  )
613  folder_name = s[1]
614  model_name = s[2]
615  module_name = 'projects.{p}.{m}.{m}'.format(m=model_name, p=folder_name)
616  class_name = name_to_agent_class(model_name)
617  elif ':' in dir_name:
618  # e.g. -m "parlai.agents.seq2seq.seq2seq:Seq2seqAgent"
619  s = dir_name.split(':')
620  module_name = s[0]
621  class_name = s[1]
622  elif '/' in dir_name:
623  # e.g. -m my_agent/special_variant
624  # will check parlai.agents.my_agent.special_variant:SpecialVariantAgent
625  sp = dir_name.split('/')
626  module_name = "%s.agents.%s.%s" % (repo, sp[0], sp[1])
627  class_name = name_to_agent_class(sp[1])
628  else:
629  # e.g. -m seq2seq
630  # will check parlai.agents.seq2seq.agents for Seq2seqAgent first
631  # then check parlai.agents.seq2seq.seq2seq for Seq2seqAgent second
632  class_name = name_to_agent_class(dir_name)
633  try:
634  module_name = "%s.agents.%s.agents" % (repo, dir_name)
635  importlib.import_module(module_name) # check if it's there
636  except ImportError:
637  module_name = "%s.agents.%s.%s" % (repo, dir_name, dir_name)
638  my_module = importlib.import_module(module_name)
639  model_class = getattr(my_module, class_name)
640  return model_class

643 def create_agent(opt, requireModelExists=False):
644  """
645  Create an agent from the options ``model``, ``model_params`` and ``model_file``.
646 
647  The input is either of the form
648  ``parlai.agents.ir_baseline.agents:IrBaselineAgent`` (i.e. the path
649  followed by the class name) or else just ``ir_baseline`` which
650  assumes the path above, and a class name suffixed with 'Agent'.
651 
652  If ``model-file`` is available in the options this function can also
653  attempt to load the model from that location instead. This avoids having to
654  specify all the other options necessary to set up the model including its
655  name as they are all loaded from the options file if it exists (the file
656  opt['model_file'] + '.opt' must exist and contain a pickled or json dict
657  containing the model's options).
658  """
659  if opt.get('datapath', None) is None:
660  # add datapath, it is missing
661  from parlai.core.params import ParlaiParser, get_model_name
662 
663  parser = ParlaiParser(add_parlai_args=False)
664  parser.add_parlai_data_path()
665  # add model args if they are missing
666  model = get_model_name(opt)
667  if model is not None:
668  parser.add_model_subargs(model)
669  opt_parser = parser.parse_args("", print_args=False)
670  for k, v in opt_parser.items():
671  if k not in opt:
672  opt[k] = v
673 
674  if opt.get('model_file'):

734 def get_task_module(taskname):
735  """
736  Get the module of the task agent specified by `--task`.
737 
738  Can be formatted in several different ways:
739 
740  * full: ``-t parlai.tasks.babi.agents:DefaultTeacher``
741  * shorthand: ``-t babi``, which will check
742  ``parlai.tasks.babi.agents:DefaultTeacher``
743  * shorthand specific: ``-t babi:task10k``, which will check
744  ``parlai.tasks.babi.agents:Task10kTeacher``
745 
746  The base path to search when using shorthand formats can be changed from
747  "parlai" to "parlai_internal" by prepending "internal:" to the path, e.g.
748  "internal:babi".
749 
750  Options can be sent to the teacher by adding an additional colon,
751  for example ``-t babi:task10k:1`` directs the babi Task10kTeacher to use
752  task number 1.
753 
754  :param taskname: path to task class in one of the above formats.
755  """
756  sp = taskname.strip()
757  repo = 'parlai'
758  if sp.startswith('internal:'):
759  # To switch to local repo, useful for non-public projects
760  # (make a directory called 'parlai_internal' with your private agents)
761  repo = 'parlai_internal'
762  sp = sp[9:]
763  sp = sp.split(':')
764  if '.' in sp[0]:
765  module_name = sp[0]
766  elif sp[0] == 'pytorch_teacher':
767  module_name = 'parlai.core.pytorch_data_teacher'
768  else:
769  task = sp[0].lower()
770  module_name = "%s.tasks.%s.agents" % (repo, task)
771  if len(sp) > 1 and '=' not in sp[1]:
772  sp[1] = sp[1][0].upper() + sp[1][1:]
773  teacher = sp[1]
774  if '.' not in sp[0] and 'Teacher' not in teacher:
775  # Reformat from underscore to CamelCase and append "Teacher" to
776  # class name by default if a complete path is not given.
777  words = teacher.split('_')
778  teacher_name = ''
779  for w in words:
780  teacher_name += w[0].upper() + w[1:]
781  teacher = teacher_name + "Teacher"
782  else:
783  teacher = "DefaultTeacher"
784  my_module = importlib.import_module(module_name)
785  teacher_class = getattr(my_module, teacher)
786  return teacher_class
