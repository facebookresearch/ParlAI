225 class MultipassTeacher(CandidateTeacher):
226  """
227  Multiturn teacher, where each episode goes:
228 
229  call response
230  1 1
231  2 1 2
232  3 1 2 3
233  4 1 2 3 4
234  """
235 
236  def num_examples(self):
237  return super().num_examples() * self.example_size
238 
239  def setup_data(self, fold):
240  raw = super().setup_data(fold)
241  for (t, a, _, cs), _ in raw:
242  split_t = t.split(' ')
243  ans = a[0]
244  for i, bit in enumerate(split_t):
245  label = ans[: 2 * i + 1]
246  cands = [c[: 2 * i + 1] for c in cs]
247  yield (bit, [label], 0, cands), i == 0

250 class MultiturnCandidateTeacher(CandidateTeacher):
251  """Splits inputs/targets by spaces into multiple turns.
252

254  """

256  def setup_data(self, fold):

259  split_t = t.split(' ')

298 class BadExampleTeacher(CandidateTeacher):
299  """
300  Teacher which produces a variety of examples that upset verify_data.py.
301 
302  Useful for checking how models respond when the following assumptions are
303  violated:
304 
305  0. text is empty string
306  1. missing text
307  2. label is empty string
308  3. missing label
309  4. label candidates is empty
310  5. label candidates contains an empty string
311  6. label isn't in the candidates
312  7. missing label candidates
313 
314  Note: this test may come to outlive its purpose in the future. When failing
315  this test, one should consider who is really at fault: the test, or the code.
316  """
317 
318  NUM_CASES = 8
319 
320  def __init__(self, opt, shared=None):
321  super().__init__(opt, shared)
322  # gross hack: override data.get to force things the way we want; otherwise
323  # we can't actually force some of these scenarios.
324  self.data.get = self._wrapperfn(self.data.get)
325 
326  def _wrapperfn(self, oldget):
327  def newget(*args):
328  item, eod = oldget(*args)
329  item = copy.deepcopy(item)
330  newget.case = (newget.case + 1) % self.NUM_CASES
331  case = newget.case
332  if case == 0:
333  # empty string input
334  item['text'] = ''
335  elif case == 1:
336  # not text input
337  del item['text']
338  elif case == 2:
339  # empty string label
340  item['labels'] = ['']
341  elif case == 3:
342  # no label
343  del item['labels']
344  elif case == 4:
345  # no label candidates
346  item['label_candidates'] = []
347  elif case == 5:
348  # extra empty string in labels
349  item['label_candidates'] = list(item['label_candidates']) + ['']
350  elif case == 6:
351  # label candidates doesn't have the label
352  item['label_candidates'] = list(item['label_candidates'])
353  item['label_candidates'].remove(item['labels'][0])
354  elif case == 7:
355  # no label candidates field
356  del item['label_candidates']
357  return item, eod
358 
359  newget.case = random.randint(0, self.NUM_CASES)
360  return newget
