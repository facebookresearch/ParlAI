102  elif fold == "test":
103  self.corpus = self.test

111  offset = (i + j) % len(self.corpus)

220 class MultipassTeacher(CandidateTeacher):
221  """
222  Multiturn teacher, where each episode goes:
223 
224  call response
225  1 1
226  2 1 2
227  3 1 2 3
228  4 1 2 3 4
229  """
230 
231  def num_examples(self):
232  return super().num_examples() * self.example_size
233 
234  def setup_data(self, fold):
235  raw = super().setup_data(fold)
236  for (t, a, _, cs), _ in raw:
237  split_t = t.split(' ')
238  ans = a[0]
239  for i, bit in enumerate(split_t):
240  label = ans[: 2 * i + 1]
241  cands = [c[: 2 * i + 1] for c in cs]
242  yield (bit, [label], 0, cands), i == 0

245 class MultiturnCandidateTeacher(CandidateTeacher):
246  """Splits inputs/targets by spaces into multiple turns.
247 
248  Good for testing models that use the dialog history.
249  """
250 
251  def setup_data(self, fold):
252  raw = super().setup_data(fold)
253  for (t, a, r, cs), _e in raw:
254  split_t = t.split(' ')
255  split_a = a[0].split(' ')
256  split_cs = [c.split(' ') for c in cs]
257  for i in range(len(split_t)):
258  yield (
259  (
260  split_t[i],
261  [' '.join(split_a[: i + 1])],
262  r,
263  [' '.join(c[: i + 1]) for c in split_cs],
264  ),
265  i == 0,
266  )

293 class BadExampleTeacher(CandidateTeacher):
294  """
295  Teacher which produces a variety of examples that upset verify_data.py.
296 
297  Useful for checking how models respond when the following assumptions are
298  violated:
299 
300  0. text is empty string
301  1. missing text
302  2. label is empty string
303  3. missing label
304  4. label candidates is empty
305  5. label candidates contains an empty string
306  6. label isn't in the candidates
307  7. missing label candidates
308 
309  Note: this test may come to outlive its purpose in the future. When failing
310  this test, one should consider who is really at fault: the test, or the code.
311  """
312 
313  NUM_CASES = 8
314 
315  def __init__(self, opt, shared=None):
316  super().__init__(opt, shared)
317  # gross hack: override data.get to force things the way we want; otherwise
318  # we can't actually force some of these scenarios.
319  self.data.get = self._wrapperfn(self.data.get)
320 
321  def _wrapperfn(self, oldget):
322  def newget(*args):
323  item, eod = oldget(*args)
324  item = copy.deepcopy(item)
325  newget.case = (newget.case + 1) % self.NUM_CASES
326  case = newget.case
327  if case == 0:
328  # empty string input
329  item['text'] = ''
330  elif case == 1:
331  # not text input
332  del item['text']
333  elif case == 2:
334  # empty string label
335  item['labels'] = ['']
336  elif case == 3:
337  # no label
338  del item['labels']
339  elif case == 4:
340  # no label candidates
341  item['label_candidates'] = []
342  elif case == 5:
343  # extra empty string in labels
344  item['label_candidates'] = list(item['label_candidates']) + ['']
345  elif case == 6:
346  # label candidates doesn't have the label
347  item['label_candidates'] = list(item['label_candidates'])
348  item['label_candidates'].remove(item['labels'][0])
349  elif case == 7:
350  # no label candidates field
351  del item['label_candidates']
352  return item, eod
353 
354  newget.case = random.randint(0, self.NUM_CASES)
355  return newget
