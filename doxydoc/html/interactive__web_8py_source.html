173  def do_POST(self):
174  """Handle POST request, especially replying to a chat message."""
175  if self.path == '/interact':
176  content_length = int(self.headers['Content-Length'])
177  body = self.rfile.read(content_length)
178  model_response = self._interactive_running(
179  SHARED.get('opt'), body.decode('utf-8')
180  )
181 
182  self.send_response(200)
183  self.send_header('Content-type', 'application/json')
184  self.end_headers()
185  json_str = json.dumps(model_response)
186  self.wfile.write(bytes(json_str, 'utf-8'))
187  elif self.path == '/reset':
188  self.send_response(200)
189  self.send_header('Content-type', 'application/json')
190  self.end_headers()
191  SHARED['agent'].reset()
192  self.wfile.write(bytes("{}", 'utf-8'))
193  else:
194  return self._respond({'status': 500})
