113 def clean_text(words):
114  if len(words) > 0 and words[-1] == ':':
115  return None
116  sentence = ' '.join(words).strip(' -').lower()
117 
118  sentence = CLEAN_BRACKETS_REGEX.sub('', sentence)
119  if len([ch for ch in BRACKETS_CHARACTERS if ch in sentence]) > 0:
120  return None
121 
122  sentence = sentence.replace('\\\'', '\'')
123  if sentence.count('"') % 2 == 1:
124  # There are unmatched double-quotes.
125  # Usually, it means a quote got splitted into separate utterances,
126  # so it's bad example of a dialog
127  return None
128 
129  sentence = normalize_apostrophe(sentence)
130 
131  for (regex, replacement) in CLEANUP_REGEX_RULES:
132  sentence = regex.sub(replacement, sentence)
133  for (pattern, replacement) in CLEANUP_REPLACE_RULES:
134  sentence = sentence.replace(pattern, replacement)
135

137 
138  if (
