59  def setup_data(self, path):
60  def rebuild(entries):
61  if len(entries) == 0:
62  return []
63  # flip the first example
64  flipped = [(SILENCE_TOKEN, [entries[0][0]], 0)]
65  # flip the rest
66  flipped += [
67  (entries[i][1][0], [entries[i + 1][0]], 0)
68  for i in range(len(entries) - 1)
69  ]
70  return flipped
71 
72  # this shows conversations in both directions
73  # we skip examples for which no label is present
74  alternate = []
75  for entry, new in super().setup_data(path):
76  if new:
77  for i, e in enumerate(rebuild(alternate)):
78  if e[1]:
79  yield e, i == 0
80  alternate.clear()
81  else:
82  alternate.append(entry)
83  if entry[1]:
84  yield entry, new
85 
86  # flip the last episode
87  if alternate:
88  for i, e in enumerate(rebuild(alternate)):
89  if e[1]:
