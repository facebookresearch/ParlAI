65  def decode_forced(self, encoder_states, ys):
66  """
67  Decode with a fixed, true sequence, computing loss.
68 
69  Useful for training, or ranking fixed candidates.
70 
71  :param ys:
72  the prediction targets. Contains both the start and end tokens.
73 
74  :type ys:
75  LongTensor[bsz, time]
76 
77  :param encoder_states:
78  Output of the encoder. Model specific types.
79 
80  :type encoder_states:
81  model specific
82 
83  :return:
84  pair (logits, choices) containing the logits and MLE predictions
85 
86  :rtype:
87  (FloatTensor[bsz, ys, vocab], LongTensor[bsz, ys])
88  """
89  bsz = ys.size(0)
90  seqlen = ys.size(1)
91  inputs = ys.narrow(1, 0, seqlen - 1)
92  inputs = torch.cat([self.START.detach().expand(bsz, 1), inputs], 1)
93  latent, _ = self.decoder(inputs, encoder_states)
94  logits = self.output(latent)
95  _, preds = logits.max(dim=2)
96  return logits, preds

98  @abstractmethod
99  def reorder_encoder_states(self, encoder_states, indices):
100  """
101  Reorder encoder states according to a new set of indices.
102 
103  This is an abstract method, and *must* be implemented by the user.
104 
105  Its purpose is to provide beam search with a model-agnostic interface for
106  beam search. For example, this method is used to sort hypotheses,
107  expand beams, etc.
108 
109  For example, assume that encoder_states is an bsz x 1 tensor of values
110 
111  .. code-block:: python
112 
113  indices = [0, 2, 2]
114  encoder_states = [[0.1]
115  [0.2]
116  [0.3]]
117 
118  then the output will be
119 
120  .. code-block:: python
121 
122  output = [[0.1]
123  [0.3]
124  [0.3]]
125 
126  :param encoder_states:
127  output from encoder. type is model specific.
128 
129  :type encoder_states:
130  model specific
131 
132  :param indices:
133  the indices to select over. The user must support non-tensor
134  inputs.
135 
136  :type indices: list[int]
137 
138  :return:
139  The re-ordered encoder states. It should be of the same type as
140  encoder states, and it must be a valid input to the decoder.
141 
142  :rtype:
143  model specific
144  """
145  pass

147  @abstractmethod
148  def reorder_decoder_incremental_state(self, incremental_state, inds):
149  """
150  Reorder incremental state for the decoder.
151 
152  Used to expand selected beams in beam search. Unlike reorder_encoder_states,
153  implementing this method is optional. However, without incremental decoding,
154  decoding a single beam becomes O(n^2) instead of O(n), which can make
155  beam search impractically slow.
156 
157  In order to fall back to non-incremental decoding, just return None from this
158  method.
159 
160  :param incremental_state:
161  second output of model.decoder
162  :type incremental_state:
163  model specific
164  :param inds:
165  indices to select and reorder over.
166  :type inds:
167  LongTensor[n]
168 
169  :return:
170  The re-ordered decoder incremental states. It should be the same
171  type as incremental_state, and usable as an input to the decoder.
172  This method should return None if the model does not support
173  incremental decoding.
174 
175  :rtype:
176  model specific
177  """
178  pass

180  def forward(self, *xs, ys=None, prev_enc=None, maxlen=None, bsz=None):
181  """
182  Get output predictions from the model.
183 
184  :param xs:
185  input to the encoder
186  :type xs:
187  LongTensor[bsz, seqlen]
188  :param ys:
189  Expected output from the decoder. Used
190  for teacher forcing to calculate loss.
191  :type ys:
192  LongTensor[bsz, outlen]
193  :param prev_enc:
194  if you know you'll pass in the same xs multiple times, you can pass
195  in the encoder output from the last forward pass to skip
196  recalcuating the same encoder output.
197  :param maxlen:
198  max number of tokens to decode. if not set, will use the length of
199  the longest label this model has seen. ignored when ys is not None.
200  :param bsz:
201  if ys is not provided, then you must specify the bsz for greedy
202  decoding.
203 
204  :return:
205  (scores, candidate_scores, encoder_states) tuple
206 
207  - scores contains the model's predicted token scores.
208  (FloatTensor[bsz, seqlen, num_features])
209  - candidate_scores are the score the model assigned to each candidate.
210  (FloatTensor[bsz, num_cands])
211  - encoder_states are the output of model.encoder. Model specific types.
212  Feed this back in to skip encoding on the next call.
213  """
214  assert ys is not None, "Greedy decoding in TGModel.forward no longer supported."
215  # TODO: get rid of longest_label
216  # keep track of longest label we've ever seen
217  # we'll never produce longer ones than that during prediction
218  self.longest_label = max(self.longest_label, ys.size(1))
219 
220  # use cached encoding if available
221  encoder_states = prev_enc if prev_enc is not None else self.encoder(*xs)
222 
223  # use teacher forcing
224  scores, preds = self.decode_forced(encoder_states, ys)
225 
226  return scores, preds, encoder_states

458  def report(self):
459  """
460  Report loss and perplexity from model's perspective.
461 
462  Note that this includes predicting __END__ and __UNK__ tokens and may
463  differ from a truly independent measurement.
464  """
465  base = super().report()
466  m = {}
467  num_tok = self.metrics['num_tokens']
468  if num_tok > 0:
469  m['loss'] = self.metrics['loss']
470  if self.metrics['correct_tokens'] > 0:
471  m['token_acc'] = self.metrics['correct_tokens'] / num_tok
472  m['nll_loss'] = self.metrics['nll_loss'] / num_tok
473  try:
474  m['ppl'] = math.exp(m['nll_loss'])
475  except OverflowError:
476  m['ppl'] = float('inf')
477  if self.metrics['total_skipped_batches'] > 0:
478  m['total_skipped_batches'] = self.metrics['total_skipped_batches']
479  for k, v in m.items():
480  # clean up: rounds to sigfigs and converts tensors to floats
481  base[k] = round_sigfigs(v, 4)
482  return base

504  def compute_loss(self, batch, return_output=False):
505  """
506  Compute and return the loss for the given batch.
507 
508  Easily overridable for customized loss functions.
509 
510  If return_output is True, the full output from the call to self.model()
511  is also returned, via a (loss, model_output) pair.
512  """
513  if batch.label_vec is None:
514  raise ValueError('Cannot compute loss without a label.')
515  model_output = self.model(*self._model_input(batch), ys=batch.label_vec)
516  scores, preds, *_ = model_output
517  score_view = scores.view(-1, scores.size(-1))
518  loss = self.criterion(score_view, batch.label_vec.view(-1))
519  # save loss to metrics
520  notnull = batch.label_vec.ne(self.NULL_IDX)
521  target_tokens = notnull.long().sum().item()
522  correct = ((batch.label_vec == preds) * notnull).sum().item()
523  self.metrics['correct_tokens'] += correct
524  self.metrics['nll_loss'] += loss.item()
525  self.metrics['num_tokens'] += target_tokens
526  loss /= target_tokens # average loss per token
527  if return_output:
528  return (loss, model_output)
529  else:
530  return loss

532  def train_step(self, batch):
533  """Train on a single batch of examples."""
534  batchsize = batch.text_vec.size(0)
535  # helps with memory usage
536  self._init_cuda_buffer(batchsize, self.truncate or 256)
537  self.model.train()
538  self.zero_grad()
539 
540  try:
541  loss = self.compute_loss(batch)
542  self.metrics['loss'] += loss.item()
543  self.backward(loss)
544  self.update_params()
545  except RuntimeError as e:
546  # catch out of memory exceptions during fwd/bck (skip batch)
547  if 'out of memory' in str(e):
548  print(
549  '| WARNING: ran out of memory, skipping batch. '
550  'if this happens frequently, decrease batchsize or '
551  'truncate the inputs to the model.'
552  )
553  self.metrics['total_skipped_batches'] += 1
554  # gradients are synced on backward, now this model is going to be
555  # out of sync! catch up with the other workers
556  self._init_cuda_buffer(8, 8, True)
557  else:
558  raise e

560  def eval_step(self, batch):
561  """Evaluate a single batch of examples."""
562  if batch.text_vec is None:
563  return
564  bsz = batch.text_vec.size(0)
565  self.model.eval()
566  cand_scores = None
567 
568  if batch.label_vec is not None:
569  # calculate loss on targets with teacher forcing
570  loss = self.compute_loss(batch) # noqa: F841 we need the side effects
571  self.metrics['loss'] += loss.item()
572 
573  preds = None
574  if self.skip_generation:
575  warn_once(
576  "--skip-generation does not produce accurate metrics beyond ppl",
577  RuntimeWarning,
578  )
579  else:
580  maxlen = self.label_truncate or 256
581  beam_preds_scores, _ = self._generate(batch, self.beam_size, maxlen)
582  preds, scores = zip(*beam_preds_scores)
583 
584  cand_choices = None
585  # TODO: abstract out the scoring here
586  if self.rank_candidates:
587  # compute roughly ppl to rank candidates
588  cand_choices = []
589  encoder_states = self.model.encoder(*self._model_input(batch))
590  for i in range(bsz):
591  num_cands = len(batch.candidate_vecs[i])
592  enc = self.model.reorder_encoder_states(encoder_states, [i] * num_cands)
593  cands, _ = padded_tensor(
594  batch.candidate_vecs[i], self.NULL_IDX, self.use_cuda
595  )
596  scores, _ = self.model.decode_forced(enc, cands)
597  cand_losses = F.cross_entropy(
598  scores.view(num_cands * cands.size(1), -1),
599  cands.view(-1),
600  reduction='none',
601  ).view(num_cands, cands.size(1))
602  # now cand_losses is cands x seqlen size, but we still need to
603  # check padding and such
604  mask = (cands != self.NULL_IDX).float()
605  cand_scores = (cand_losses * mask).sum(dim=1) / (mask.sum(dim=1) + 1e-9)
606  _, ordering = cand_scores.sort()
607  cand_choices.append([batch.candidates[i][o] for o in ordering])
608 
609  text = [self._v2t(p) for p in preds] if preds is not None else None
610  return Output(text, cand_choices)

660  def _generate(self, batch, beam_size, max_ts):
661  """
662  Generate an output with beam search.
663 
664  Depending on the options, this may perform greedy/topk/nucleus generation.
665 
666  :param Batch batch:
667  Batch structure with input and labels
668  :param int beam_size:
669  Size of each beam during the search
670  :param int max_ts:
671  the maximum length of the decoded sequence
672 
673  :return:
674  tuple (beam_pred_scores, n_best_pred_scores, beams)
675 
676  - beam_preds_scores: list of (prediction, score) pairs for each sample in
677  Batch
678  - n_best_preds_scores: list of n_best list of tuples (prediction, score)
679  for each sample from Batch
680  - beams :list of Beam instances defined in Beam class, can be used for any
681  following postprocessing, e.g. dot logging.
682  """
683  model = self.model
684  if isinstance(model, torch.nn.parallel.DistributedDataParallel):
685  model = self.model.module
686  encoder_states = model.encoder(*self._model_input(batch))
687  dev = batch.text_vec.device
688 
689  bsz = len(batch.text_lengths)
690  beams = [self._treesearch_factory(dev) for _ in range(bsz)]
691 
692  # repeat encoder outputs and decoder inputs
693  decoder_input = (
694  torch.LongTensor([self.START_IDX]).expand(bsz * beam_size, 1).to(dev)
695  )
696 
697  inds = torch.arange(bsz).to(dev).unsqueeze(1).repeat(1, beam_size).view(-1)
698  encoder_states = model.reorder_encoder_states(encoder_states, inds)
699  incr_state = None
700 
701  for _ts in range(max_ts):
702  if all((b.is_done() for b in beams)):
703  # exit early if possible
704  break
705 
706  score, incr_state = model.decoder(decoder_input, encoder_states, incr_state)
707  # only need the final hidden state to make the word prediction
708  score = score[:, -1:, :]
709  score = model.output(score)
710  # score contains softmax scores for bsz * beam_size samples
711  score = score.view(bsz, beam_size, -1)
712  score = F.log_softmax(score, dim=-1)
713  for i, b in enumerate(beams):
714  if not b.is_done():
715  b.advance(score[i])
716  incr_state_inds = torch.cat(
717  [
718  beam_size * i + b.get_backtrack_from_current_step()
719  for i, b in enumerate(beams)
720  ]
721  )
722  incr_state = model.reorder_decoder_incremental_state(
723  incr_state, incr_state_inds
724  )
725  decoder_input = torch.index_select(decoder_input, 0, incr_state_inds)
726  selection = torch.cat(
727  [b.get_output_from_current_step() for b in beams]
728  ).unsqueeze(-1)
729  decoder_input = torch.cat([decoder_input, selection], dim=-1)
730 
731  # get the top prediction for each
732  beam_preds_scores = [b.get_top_hyp() for b in beams]
733 
734  return beam_preds_scores, beams

750 class TreeSearch(object):
751  """
752  Abstract Tree Search class.
753 
754  It keeps information about beam_size concurrent, developing hypotheses.
755  Concrete implementations make choices about which token to explore next at
756  each point in the tree. Different choices result in different generation
757  algorithms.
758  """
759 
760  def __init__(
761  self,
762  beam_size,
763  padding_token=0,
764  bos_token=1,
765  eos_token=2,
766  min_length=3,
767  min_n_best=3,
768  device='cpu',
769  ):
770  """
771  Instantiate Beam object.
772 
773  :param beam_size:
774  number of hypothesis in the beam
775  :param padding_token:
776  padding token ID
777  :param bos_token:
778  beginning of sentence token ID
779  :param eos_token:
780  end of sentence token ID
781  :param min_length:
782  minimum length of the predicted sequence
783  :param min_n_best:
784  Search will not be finished until a minimum number of utterances are
785  completed.
786  :param device:
787  What device to use for computations
788  """
789  self.beam_size = beam_size
790  self.min_length = min_length
791  self.eos = eos_token
792  self.bos = bos_token
793  self.pad = padding_token
794  self.device = device
795  # recent score for each hypo in the beam
796  self.scores = None
797  # self.scores values per each time step
798  self.all_scores = [torch.Tensor([0.0] * beam_size).to(self.device)]
799  # backtracking id to hypothesis at previous time step
800  self.bookkeep = []
801  # output tokens at each time step
802  self.outputs = [
803  torch.Tensor(self.beam_size).long().fill_(self.bos).to(self.device)
804  ]
805  # keeps tuples (score, time_step, hyp_id)
806  self.finished = []
807  self.eos_top = False
808  self.eos_top_ts = None
809  self.n_best_counter = 0
810  self.min_n_best = min_n_best
811  self.partial_hyps = [[self.bos] for i in range(beam_size)]
812 
813  def get_output_from_current_step(self):
814  """Get the outputput at the current step."""
815  return self.outputs[-1]
816 
817  def get_backtrack_from_current_step(self):
818  """Get the backtrack at the current step."""
819  return self.bookkeep[-1]
820 
821  @abstractmethod
822  def select_paths(self, logprobs, prior_scores):
823  """
824  Select the next vocabulary item in these beams.
825 
826  :param logprobs:
827  a (beamsize x vocab) tensor of log probabilities. If this is the first
828  turn in the dialogue, it will be a (1 x vocab) tensor.
829  :param prior_scores:
830  a (beamsize) tensor of weights with the cumulative running
831  log-probability of each beam. If the first turn, it will be a (1) tensor.
832 
833  :return:
834  a (hypothesis_ids, token_id, scores) tuple, where:
835 
836  - hypothesis_ids is a LongTensor of hypotheses we're extending. May have
837  repeats, but should always be (beamsize) long.
838  - token_ids is a (beamsize) LongTensor of next-token choices for
839  each of the hypotheses.
840  - scores is a (beamsize) Tensor with the updated cumulative log-probs
841  of each beam.
842  """
843  pass
844 
845  def advance(self, logprobs):
846  """Advance the beam one step."""
847  current_length = len(self.all_scores) - 1
848  if current_length < self.min_length:
849  # penalize all eos probs to make it decode longer
850  for hyp_id in range(logprobs.size(0)):
851  logprobs[hyp_id][self.eos] = neginf(logprobs.dtype)
852 
853  if self.scores is None:
854  self.scores = torch.zeros(1).type_as(logprobs).to(logprobs.device)
855 
856  hyp_ids, tok_ids, self.scores = self.select_paths(logprobs, self.scores)
857  self.all_scores.append(self.scores)
858 
859  self.outputs.append(tok_ids)
860  self.bookkeep.append(hyp_ids)
861  self.partial_hyps = [
862  self.partial_hyps[hyp_ids[i]] + [tok_ids[i].item()]
863  for i in range(self.beam_size)
864  ]
865 
866  # check new hypos for eos label, if we have some, add to finished
867  for hypid in range(self.beam_size):
868  if self.outputs[-1][hypid] == self.eos:
869  # this is finished hypo, adding to finished
870  eostail = _HypothesisTail(
871  timestep=len(self.outputs) - 1,
872  hypid=hypid,
873  score=self.scores[hypid],
874  tokenid=self.eos,
875  )
876  self.finished.append(eostail)
877  self.n_best_counter += 1
878 
879  if self.outputs[-1][0] == self.eos:
880  self.eos_top = True
881  if self.eos_top_ts is None:
882  self.eos_top_ts = len(self.outputs) - 1
883 
884  def is_done(self):
885  """Return whether beam search is complete."""
886  return self.eos_top and self.n_best_counter >= self.min_n_best
887 
888  def get_top_hyp(self):
889  """
890  Get single best hypothesis.
891 
892  :return: hypothesis sequence and the final score
893  """
894  return self._get_rescored_finished(n_best=1)[0]
895 
896  def _get_hyp_from_finished(self, hypothesis_tail):
897  """
898  Extract hypothesis ending with EOS at timestep with hyp_id.
899 
900  :param timestep:
901  timestep with range up to len(self.outputs) - 1
902 
903  :param hyp_id:
904  id with range up to beam_size - 1
905 
906  :return:
907  hypothesis sequence
908  """
909  hyp_idx = []
910  endback = hypothesis_tail.hypid
911  for i in range(hypothesis_tail.timestep, -1, -1):
912  hyp_idx.append(
913  _HypothesisTail(
914  timestep=i,
915  hypid=endback,
916  score=self.all_scores[i][endback],
917  tokenid=self.outputs[i][endback],
918  )
919  )
920  endback = self.bookkeep[i - 1][endback]
921 
922  return hyp_idx
923 
924  def _get_pretty_hypothesis(self, list_of_hypotails):
925  """Return hypothesis as a tensor of token ids."""
926  return torch.stack([ht.tokenid for ht in reversed(list_of_hypotails)])
927 
928  def _get_rescored_finished(self, n_best=None):
929  """
930  Return finished hypotheses according to adjusted scores.
931 
932  Score adjustment is done according to the Google NMT paper, which
933  penalizes long utterances.
934 
935  :param n_best:
936  number of finalized hypotheses to return
937 
938  :return:
939  list of (tokens, score) pairs, in sorted order, where:
940  - tokens is a tensor of token ids
941  - score is the adjusted log probability of the entire utterance
942  """
943  # if we never actually finished, force one
944  if not self.finished:
945  self.finished.append(
946  _HypothesisTail(
947  timestep=len(self.outputs) - 1,
948  hypid=0,
949  score=self.all_scores[-1][0],
950  tokenid=self.eos,
951  )
952  )
953 
954  rescored_finished = []
955  for finished_item in self.finished:
956  current_length = finished_item.timestep + 1
957  # these weights are from Google NMT paper
958  length_penalty = math.pow((1 + current_length) / 6, 0.65)
959  rescored_finished.append(
960  _HypothesisTail(
961  timestep=finished_item.timestep,
962  hypid=finished_item.hypid,
963  score=finished_item.score / length_penalty,
964  tokenid=finished_item.tokenid,
965  )
966  )
967 
968  # Note: beam size is almost always pretty small, so sorting is cheap enough
969  srted = sorted(rescored_finished, key=attrgetter('score'), reverse=True)
970 
971  if n_best is not None:
972  srted = srted[:n_best]
973 
974  return [
975  (self._get_pretty_hypothesis(self._get_hyp_from_finished(hyp)), hyp.score)
976  for hyp in srted
977  ]

981  """
982  Greedy search.
983 
984  Picks the highest probability utterance at each step. Only works with
985  --beam-size 1.

1000 class BeamSearch(TreeSearch):
1001  """Beam search."""
1002 
1003  def select_paths(self, logprobs, prior_scores):
1004  """Select the next vocabulary item in these beams."""
1005  # if numel is 1, then this is the first time step, only one hyp is expanded
1006  if prior_scores.numel() == 1:
1007  logprobs = logprobs[0:1]
1008 
1009  # beam search actually looks over all hypotheses together so we flatten
1010  beam_scores = logprobs + prior_scores.unsqueeze(1).expand_as(logprobs)
1011  flat_beam_scores = beam_scores.view(-1)
1012  best_scores, best_idxs = torch.topk(flat_beam_scores, self.beam_size, dim=-1)
1013  voc_size = logprobs.size(-1)
1014 
1015  # get the backtracking hypothesis id as a multiple of full voc_sizes
1016  hyp_ids = best_idxs / voc_size
1017  # get the actual word id from residual of the same division
1018  tok_ids = best_idxs % voc_size
1019 
1020  return (hyp_ids, tok_ids, best_scores)

1023 class TopKSampling(TreeSearch):
1024  """
1025  Top-K sampling (Fan et al., 2018).
1026 
1027  Samples from a truncated distribution where only the most probable K words
1028  are considered at each time.
1029 
1030  Typical values of k are 2, 10, 50.
1031 
1032  See https://arxiv.org/abs/1805.04833 for details.
1033  """
1034 
1035  def __init__(self, k, *args, **kwargs):
1036  super().__init__(*args, **kwargs)
1037  self.k = k
1038 
1039  def select_paths(self, logprobs, prior_scores):
1040  values, indices = logprobs.topk(self.k, dim=-1)
1041  probs = torch.softmax(values, dim=-1)
1042  choices = torch.multinomial(probs, 1)[:, 0]
1043  hyp_ids = torch.arange(logprobs.size(0)).to(logprobs.device)
1044  tok_ids = indices[hyp_ids, choices]
1045  scores = values[hyp_ids, choices]
1046  best_scores = prior_scores.expand_as(scores) + scores
1047  return (hyp_ids, tok_ids, best_scores)

1050 class NucleusSampling(TreeSearch):
1051  """
1052  Nucelus, aka top-p sampling (Holtzman et al., 2019).
1053 
1054  Samples from a truncated distribution which covers a fixed CDF proportion
1055  of the original distribution.
1056 
1057  Typical values of p are 0.3 and 0.9.
1058 
1059  See https://arxiv.org/abs/1904.09751 for details.
1060  """
1061 
1062  def __init__(self, p, *args, **kwargs):
1063  super().__init__(*args, **kwargs)
1064  self.p = p
1065 
1066  def select_paths(self, logprobs, prior_scores):
1067  # Unlike the other treesearch methods, we have to switch to linspace
1068  # for the probabilities in order to compute the CDF.
1069  probs = torch.softmax(logprobs, dim=-1)
1070  sprobs, sinds = probs.sort(dim=-1, descending=True)
1071  # The subtraction here is so that we always include the first word to
1072  # go over p. For example, if the most probable token has a prob of 0.5, and
1073  # p = 0.3, then we need still need to include that first token.
1074  mask = (sprobs.cumsum(dim=-1) - sprobs[:, :1]) >= self.p
1075  sprobs[mask] = 0
1076  sprobs.div_(sprobs.sum(dim=-1).unsqueeze(1))
1077  choices = torch.multinomial(sprobs, 1)[:, 0]
1078  hyp_ids = torch.arange(logprobs.size(0)).to(logprobs.device)
1079  tok_ids = sinds[hyp_ids, choices]
1080  # Convert back to logspace.
1081  scores = sprobs[hyp_ids, choices].log()
1082  best_scores = prior_scores.expand_as(scores) + scores
1083  return (hyp_ids, tok_ids, best_scores)
