129  """Compute hashed ngram counts of text."""
130  row, col, data = [], [], []
131  # Tokenize
132  tokens = tokenize(utils.normalize(text))

135  ngrams = tokens.ngrams(n=ngram, uncased=True, filter_fn=utils.filter_ngram)

241 def get_tfidf_matrix_t(cnts):
242  """Convert the word count matrix into tfidf one (torch version).
243 
244  tfidf = log(tf + 1) * log((N - Nt + 0.5) / (Nt + 0.5))
245  * tf = term frequency in document
246  * N = number of documents
247  * Nt = number of occurences of term in all documents
248  """
249  Nt = get_doc_freqs_t(cnts)
250  idft = ((cnts.size(1) - Nt + 0.5) / (Nt + 0.5)).log()
251  idft[idft < 0] = 0
252  tft = sparse_log1p(cnts)
253  inds, vals = tft._indices(), tft._values()
254  for i, ind in enumerate(inds[0]):
255  vals[i] *= idft[ind]
256  tfidft = tft
257  return tfidft

260 def get_tfidf_matrix(cnts):
261  """Convert the word count matrix into tfidf one.
262 
263  tfidf = log(tf + 1) * log((N - Nt + 0.5) / (Nt + 0.5))
264  * tf = term frequency in document
265  * N = number of documents
266  * Nt = number of occurences of term in all documents
267  """
268  Nt = get_doc_freqs(cnts)
269  idfs = np.log((cnts.shape[1] - Nt + 0.5) / (Nt + 0.5))
270  idfs[idfs < 0] = 0
271  idfs = sp.diags(idfs, 0)
272  tfs = cnts.log1p()
273  tfidfs = idfs.dot(tfs)
274 
275  return tfidfs
