193  def _get_data(self):

195  # all of these colors are bolded
196  RESET = '\033[0m'
197  RED = '\033[1;91m'
198  YELLOW = '\033[1;93m'
199  GREEN = '\033[1;92m'
200  BLUE = '\033[1;96m'
201  CYAN = '\033[1;94m'
202  MAGENTA = '\033[1;95m'
203 
204  # only use colors if we're outputting to a terminal
205  USE_COLORS = _sys.stdout.isatty()
206  if not USE_COLORS:
207  RESET = RED = YELLOW = GREEN = BLUE = CYAN = MAGENTA = ''
208 
209  # generate the rainbow stars
210  rainbow = [RED, YELLOW, GREEN, CYAN, BLUE, MAGENTA]
211  size = 78 // len(rainbow)
212  stars = ''.join([color + '*' * size for color in rainbow])
213  stars += RESET
214 
215  if not os.path.exists(self.data_path):
216  os.makedirs(self.data_path)
217  if not os.path.isfile(os.path.join(self.data_path, 'train.csv')):
218  raise RuntimeError(

220  '<https://www.kaggle.com/c/jigsaw-toxic-comment-classification-challenge/data>. '
221  '\nIt cannot be automatically downloaded, as one must agree to '
222  'the competition rules outlined on the website before '

233  def build(self, opt):
234  self._get_data()
235  version = 'v1.0'
236  read_path = self.data_path
237  if not build_data.built(self.data_path, version):
238  print('[building data from : ' + read_path + ']')
239  build_data.make_dir(self.data_path)
240  # Read in data
241  train = pd.read_csv(os.path.join(read_path, 'train.csv'))
242  test = pd.read_csv(os.path.join(read_path, 'test.csv'))
243  test_labels = pd.read_csv(os.path.join(read_path, 'test_labels.csv'))
244 
245  # Labels for test data; value of -1 indicates it was not used for scoring
246  test_labels = test_labels[
247  (test_labels.toxic != -1)
248  & (test_labels.severe_toxic != -1)
249  & (test_labels.obscene != -1)
250  & (test_labels.threat != -1)
251  & (test_labels.insult != -1)
252  & (test_labels.identity_hate != -1)
253  ]
254 
255  # Merge test data with labels
256  test = pd.merge(test_labels, test, on='id')
257 
258  # Split 10% of train data to be valid
259  test['data_type'] = 'test'
260  train['data_type'] = 'train'
261  valid_set = train.sample(frac=0.1, random_state=42)
262  valid_set['data_type'] = 'valid'
263  train.update(valid_set)
264 
265  # Combine test and train into one data frame
266  total_data = pd.concat([test, train], ignore_index=True, sort=False)
267  # Rename comment_text to text for the act dict
268  total_data.rename(columns={'comment_text': 'text'}, inplace=True)
269 
270  # Use the different categories to get binary classification
271  total_data['sensitive'] = (
272  total_data['severe_toxic']
273  + total_data['toxic']
274  + total_data['obscene']
275  + total_data['threat']
276  + total_data['insult']
277  + total_data['identity_hate']
278  )
279 
280  total_data.loc[total_data['sensitive'] < 1, 'is_sensitive'] = 0
281  total_data.loc[total_data['sensitive'] >= 1, 'is_sensitive'] = 1
282 
283  # Drop unecessary column
284  total_data = total_data.drop(columns=['id'])
285 
286  self.data_to_json(total_data, 'wiki-toxic-comments-default.json')
287 
288  # Partition 80/10/10 according to arXiv:1811.12900 [cs.CL]
289  # 90 train 10 test in paper
290  # split based on command line flag
291  original_train = total_data[
292  (total_data['data_type'] == 'valid')
293  | (total_data['data_type'] == 'train')
294  ].copy()
295  l_td = len(original_train)
296 
297  original_train.iloc[: int(0.8 * l_td)]['data_type'] = 'train'
298  original_train.iloc[int(0.8 * l_td) : int(0.9 * l_td)]['data_type'] = 'test'
299  original_train.iloc[int(0.9 * l_td) :]['data_type'] = 'valid'
300 
301  self.data_to_json(original_train, 'wiki-toxic-comments-partition.json')
302 
303  # Merge the 3 files to get a list of dicts as follows:
304  # [
305  # {
306  # 'toxic': 0 or 1 ,
307  # 'severe_toxic': 0 or 1,
308  # 'obscene': 0 or 1,

328  def _balance_data(self, data_list):
329  # NOTE: this assumes that there are more OK labels than NOT OK labels
330  ok = [x for x in data_list if x['is_sensitive'] == 0]
331  notok = [x for x in data_list if x['is_sensitive'] == 1]
332 
333  new_not_ok = []
334  while len(new_not_ok) < len(ok):
335  new_not_ok.append(self.fixed_random.choice(notok))
336 
337  new_data = ok + new_not_ok
338  self.fixed_random.shuffle(new_data)
339  return new_data
