190  def _check_sequence(self, toks, idx, node):
191  """
192  Check if words from the sequence are in the trie.
193 
194  This checks phrases made from
195  toks[i], toks[i:i+2] ... toks[i:i + self.max_len]
196  """
197  right = min(idx + self.max_len, len(toks))
198  for i in range(idx, right):
199  if toks[i] in node:
200  node = node[toks[i]]
201  if self.END in node:
202  return ' '.join(toks[j] for j in range(idx, i + 1))
203  else:
