401  def act(self, timeout=None, blocking=True):
402  """Sends a message to other agents in the world. If blocking, this
403  will wait for the message to come in so it can be sent. Otherwise
404  it will return None.
405  """
406  if not blocking:
407  # if this is the first act since last sent message start timing
408  if self.message_request_time is None:
409  self.request_message()
410  self.message_request_time = time.time()
411 
412  # If checking timeouts
413  if timeout:
414  # If time is exceeded, timeout
415  if time.time() - self.message_request_time > timeout:
416  return self.prepare_timeout()
417 
418  # Get a new message, if it's not None reset the timeout
419  msg = self.get_new_act_message()
420  if msg is not None and self.message_request_time is not None:
421  self.message_request_time = None
422  return msg
423  else:
424  self.request_message()
425  self.message_request_time = time.time()
426 
427  # Timeout in seconds, after which the HIT is expired automatically
428  if timeout:
429  start_time = time.time()
430 
431  # Wait for agent's new message
432  while True:
433  msg = self.get_new_act_message()
434  self.message_request_time = None
435  if msg is not None:
436  return msg
437 
438  # Check if the Turker waited too long to respond
439  if timeout:
440  current_time = time.time()
441  if (current_time - start_time) > timeout:
442  self.message_request_time = None
443  return self.prepare_timeout()
444  time.sleep(shared_utils.THREAD_SHORT_SLEEP)

569  def wait_completion_timeout(self, iterations):
570  """Suspends the thread waiting for hit completion for some number of
571  iterations on the THREAD_MTURK_POLLING_SLEEP time"""
572 
573  # Determine number of sleep iterations for the amount of time
574  # we want to wait before syncing with MTurk. Start with 10 seconds
575  # of waiting
576  iters = (
577  shared_utils.THREAD_MTURK_POLLING_SLEEP / shared_utils.THREAD_MEDIUM_SLEEP
578  )
579  i = 0
580  # Wait for the desired number of MTURK_POLLING_SLEEP iterations
581  while not self.hit_is_complete and i < iters * iterations:
582  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
583  i += 1
584  return

586  def wait_for_hit_completion(self, timeout=None):
587  """Waits for a hit to be marked as complete"""
588  # Timeout in seconds, after which the HIT will be expired automatically
589  if timeout:
590  if timeout < 0:
591  # Negative timeout is for testing, wait for packet to send
592  time.sleep(1)
593  self.mturk_manager.free_workers([self])
594  return True
595  start_time = time.time()
596  wait_periods = 1
597  self.wait_completion_timeout(wait_periods)
598  sync_attempts = 0
599  while (
600  not self.hit_is_complete
601  and self.mturk_manager.get_agent_work_status(self.assignment_id)
602  != self.ASSIGNMENT_DONE
603  ):
604  if sync_attempts < 8:
605  # Scaling on how frequently to poll, doubles time waited on
606  # every failure
607  wait_periods *= 2
608  sync_attempts += 1
609  else:
610  # Okay we've waited for 45 mins and the HIT still isn't up
611  self.disconnected = True
612  # Check if the Turker already returned/disconnected
613  if self.hit_is_returned or self.disconnected:
614  self.mturk_manager.free_workers([self])
615  return False
616  if timeout:
617  current_time = time.time()
618  if (current_time - start_time) > timeout:
619  shared_utils.print_and_log(
620  logging.INFO,
621  "Timeout waiting for ({})_({}) to complete {}.".format(
622  self.worker_id, self.assignment_id, self.conversation_id
623  ),
624  )
625  self.set_hit_is_abandoned()
626  self.mturk_manager.free_workers([self])
627  return False
628  shared_utils.print_and_log(
629  logging.DEBUG,
630  'Waiting for ({})_({}) to complete {}...'.format(
631  self.worker_id, self.assignment_id, self.conversation_id
632  ),
633  )
634  self.wait_completion_timeout(wait_periods)
635 
636  shared_utils.print_and_log(
637  logging.INFO,
638  'Conversation ID: {}, Agent ID: {} - HIT is done.'.format(
639  self.conversation_id, self.id
640  ),
641  )
642  self.mturk_manager.free_workers([self])
643  return True
