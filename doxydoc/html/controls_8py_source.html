30 class NIDFFeats(object):
31  """
32  An object to hold a vector containing the NIDF values for all words in the
33  vocabulary. The vector is contstructed when first needed.
34  """
35 
36  def __init__(self):
37  self.NIDF_FEATS = None # will be vector length vocab_size containing NIDF vals
38 
39  def make_feat_vec(self, dict):
40  """
41  Construct the NIDF feature vector for the given dict.
42  """
43  print("Constructing NIDF feature vector...")
44  self.NIDF_FEATS = torch.zeros((len(dict)))
45  num_oovs = 0
46  for idx in range(len(dict)):
47  word = dict[idx]
48  if word in word2nidf:
49  # Leave emoji (these appear in Twitter dataset) as NIDF=0
50  # (so we don't encourage emoji when we set WD weight high for NIDF)
51  if word[0] == '@' and word[-1] == '@':
52  continue
53  nidf = word2nidf[word] # between 0 and 1
54  self.NIDF_FEATS[idx] = nidf
55  else:
56  # print("WARNING: word %s has no NIDF; marking it as NIDF=0" % word)
57  num_oovs += 1 # If we don't have NIDF for this word, set as 0
58  print(
59  'Done constructing NIDF feature vector; of %i words in dict there '
60  'were %i words with unknown NIDF; they were marked as NIDF=0.'
61  % (len(dict), num_oovs)
62  )
63 
64  def get_feat_vec(self, dict):
65  """
66  Return the NIDF feature vector. If necessary, construct it first.
67  """
68  if self.NIDF_FEATS is None:
69  self.make_feat_vec(dict)
70  return self.NIDF_FEATS

121 def intrep_frac(lst):
122  """Returns the fraction of items in the list that are repeated"""
123  if len(lst) == 0:
124  return 0
125  num_rep = 0
126  for idx in range(len(lst)):
127  if lst[idx] in lst[:idx]:
128  num_rep += 1
129  return num_rep / len(lst)

132 def extrep_frac(lst1, lst2):
133  """Returns the fraction of items in lst1 that are in lst2"""
134  if len(lst1) == 0:
135  return 0
136  num_rep = len([x for x in lst1 if x in lst2])
137  return num_rep / len(lst1)

154 def matching_ngram_completions(comparison_seq, hypothesis, n):
155  """
156  Return the list of words that if appended to hypothesis, would create a n-gram that
157  already exists in comparison_seq. For efficiency, this function represents words
158  as integers not strings.
159 
160  Inputs:
161  comparison_seq: list of integers
162  hypothesis: list of integers or None
163  n: integer
164 
165  Output:
166  bad_words: list of integers
167  """
168  if hypothesis is None or len(hypothesis) < n - 1 or len(comparison_seq) < n:
169  return []
170  hypothesis = [int(i) for i in hypothesis] # cast to list of ints
171  comparison_seq = [int(i) for i in comparison_seq] # cast to list of ints
172  n_minus_1_gram = hypothesis[-(n - 1) :] # list of ints length n-1
173  bad_words = [
174  comparison_seq[i]
175  for i in range(n - 1, len(comparison_seq))
176  if comparison_seq[i - (n - 1) : i] == n_minus_1_gram
177  ] # list of ints
178  return bad_words

183 # These functions compute the decoding features for weighted decoding.
184 #
185 # Given a conversational history and a hypothesis (i.e. a partially generated response),
186 # these functions update the weighted decoding feature vector (of length vocab_size) by

188 #
189 # All these functions have the following inputs and outputs:
190 #
191 # Inputs:
192 # dict: parlai DictionaryAgent
193 # hypothesis: a list of integers. This is the partially generated response,
194 # represented via word indices.
195 # history: a ConvAI2History. This represents the conversation history.
196 # wt: a float. This is the weight for the weighted decoding feature.
197 # feat: a vector length vocab_size. This will ultimately contain the sum of all the
198 # weighted decoding features.
199 # Output:
200 # feat: a vector length vocab_size. This is the feature vector, now with the new
201 # weighted decoding feature added (multiplied by wt).

205 def intrep_word_used_before(dict, hypothesis, history, wt, feat, remove_stopwords):
206  """
207  Weighted decoding feature function. See explanation above.
208  This feature is 1 for words that have already appeared within the hypothesis,
209  0 otherwise.
210 
211  Additional inputs:
212  remove_stopwords: bool. If True, stopwords are not included when identifying words
213  that have already appeared.
214  """
215  if hypothesis is not None:
216  if remove_stopwords:
217  hypothesis = [idx for idx in hypothesis if dict[idx] not in STOPWORDS]
218  if len(hypothesis) > 0:
219  feat[hypothesis] += wt
220  return feat

224  """
225  Weighted decoding feature function. See explanation above.

244  This feature is 1 for words that have already been used earlier in the conversation;

322 def lastutt_sim_arora_word(dict, hypothesis, history, wt, feat):
323  """
324  Weighted decoding feature function. See explanation above.
325  Given a word w, this feature is equal to
326  cos_sim(word_emb(w), sent_emb(l))
327  the cosine similarity between the GloVe vector for word w, and the Arora-style
328  sentence embedding for the partner's last utterance l.
329  """
330  partner_utts = history.partner_utts
331  if len(partner_utts) == 0: # if bot goes first then do nothing
332  return feat
333  last_utt = partner_utts[-1] # string
334  if last_utt.strip().lower() == "__silence__": # if bot goes first then do nothing
335  return feat
336 
337  # Get last_utt_emb, which is a tensor shape (glove_dim)
338  last_utt_emb = sent_embedder.embed_sent(dict.tokenize(last_utt))
339  if last_utt_emb is None:
340  return feat
341 
342  # Get cosine similarities, which is a tensor shape (vocab_size)
343  sims = sent_embedder.get_word_sims(last_utt, last_utt_emb, dict)
344 
345  feat += wt * sims
346  return feat

442 def get_wd_features(dict, hypothesis, history, wd_features, wd_weights):
443  """
444  Given a conversational history and a hypothesis (i.e. partially generated response),
445  compute the Weighted Decoding features for all words in the vocabulary.
446 
447  Inputs:
448  dict: parlai DictionaryAgent
449  hypothesis: list of ints or None
450  history: a ConvAI2History. This represents the conversation history.
451  wd_features: list of strings; the names of the WD features we want to use
452  wd_weights: list of floats; the weights corresponding to the WD features.
453  Returns:
454  wd_feat_vec: tensor shape (vocab_size), containing weighted sum of the feature
455  functions, for each candidate continuation word
456  """
457  wd_feat_vec = torch.zeros((len(dict)))
458  for f, w in zip(wd_features, wd_weights):
459  wd_feat_vec = WDFEATURE2UPDATEFN[f]((dict, hypothesis, history, w, wd_feat_vec))
460  return wd_feat_vec

555 def avg_nidf(utt, history):
556  """
557  Sentence-level attribute function. See explanation above.
558  Returns the mean NIDF of the words in utt.
559  """
560  words = utt.split()
561  problem_words = [w for w in words if w not in word2nidf]
562  ok_words = [w for w in words if w in word2nidf]
563  if len(ok_words) == 0:
564  print(
565  "WARNING: For all the words in the utterance '%s', we do not have the "
566  "NIDF score. Marking as avg_nidf=1." % utt
567  )
568  return 1 # rarest possible sentence
569  nidfs = [word2nidf[w] for w in ok_words]
570  avg_nidf = sum(nidfs) / len(nidfs)
571  if len(problem_words) > 0:
572  print(
573  "WARNING: When calculating avg_nidf for the utterance '%s', we don't "
574  "know NIDF for the following words: %s" % (utt, str(problem_words))
575  )
576  assert avg_nidf >= 0 and avg_nidf <= 1
577  return avg_nidf

588 def lastutt_sim_arora_sent(utt, history):
589  """
590  Sentence-level attribute function. See explanation above.
591 
592  Returns
593  cos_sim(sent_emb(last_utt), sent_emb(utt))
594  the cosine similarity of the Arora-style sentence embeddings for the current
595  response (utt) and the partner's last utterance (last_utt, which is in history).
596 
597  - If there is no last_utt (i.e. utt is the first utterance of the conversation),
598  returns None.
599  - If one or both of utt and last_utt are all-OOV; thus we can't compute sentence
600  embeddings, return the string 'oov'.
601  """
602  partner_utts = history.partner_utts
603  if len(partner_utts) == 0:
604  # print('WARNING: returning lastuttsim = None because bot goes first')
605  return None
606  last_utt = partner_utts[-1] # string

732 def get_qn_bucket_probs():
733  """
734  Assuming we have 11 CT question buckets (0 to 10), compute P(bucket|question=1) and
735  P(bucket|question=0); this is needed so we can probabilistically assign incoming
736  training examples to buckets.
737 
738  Returns:
739  prob_bucket_given_qn: list of floats length 11; P(bucket|question=1)
740  prob_bucket_given_notqn: list of floats length 11; P(bucket|question=0)
741  """
742  prob_qn = 41101 / 131438 # P(question=1), computed across ConvAI2 dataset. ~31%
743 
744  # Compute P(bucket), i.e. the total sizes of the buckets.
745  # This is done by assuming that buckets 1 to 10 are equal in size, but bucket 0
746  # is larger because we have more non-questions than questions. Therefore:
747  # P(question=1) = P(bucket=1, question=1) + ... + P(bucket=10, question=1)
748  # = prob_bucket_n * 0.1 + ... + prob_bucket_n * 1
749  # = 5.5 * prob_bucket_n
750  # Thus we can derive the value for prob_bucket_n and prob_bucket_0:
751  prob_bucket_n = prob_qn / 5.5 # P(bucket=n) for n=1,...,10
752  prob_bucket_0 = 1 - 10 * prob_bucket_n # P(bucket=0)
753  prob_bucket = [prob_bucket_0] + [prob_bucket_n] * 10 # list length 11, P(bucket)
754 
755  # Compute P(bucket|qn=1) and P(bucket|qn=0) using Bayes Rule:
756  # P(bucket|qn=1) = P(bucket) * P(qn=1|bucket) / P(qn=1)
757  # P(bucket|qn=0) = P(bucket) * P(qn=0|bucket) / P(qn=0)
758  prob_bucket_given_qn = [pb * (i / 10) / prob_qn for i, pb in enumerate(prob_bucket)]
759  prob_bucket_given_notqn = [
760  pb * ((10 - i) / 10) / (1 - prob_qn) for i, pb in enumerate(prob_bucket)
761  ]
762 
763  return prob_bucket_given_qn, prob_bucket_given_notqn
764 
765 
766 PROB_BUCKET_GIVEN_QN, PROB_BUCKET_GIVEN_NOTQN = get_qn_bucket_probs()

769 def bucket_question(ex, ctrl, num_buckets):
770  """
771  Given an example (where the target response may or may not be a question) and its
772  history, probabilistically determine what question-asking CT bucket to use.
773 
774  Inputs:
775  ex: message dictionary containing a bool field 'question'
776  ctrl: string. The name of the CT control. Should be 'question'.
777  num_buckets: int. The number of question-asking CT buckets. Assumed to be 11.
778  Returns:
779  out: int. bucket number.
780  """
781  assert num_buckets == 11
782  is_qn = int(ex['question'])
783  assert is_qn in [0, 1]
784  is_qn = bool(is_qn)
785  if is_qn: # Sample from P(bucket|qn=1)
786  out = np.random.choice(range(num_buckets), 1, p=PROB_BUCKET_GIVEN_QN)
787  else: # Sample from P(bucket|qn=0)
788  out = np.random.choice(range(num_buckets), 1, p=PROB_BUCKET_GIVEN_NOTQN)
789  out = int(out[0])
790  return out

793 def sort_into_bucket(val, bucket_lbs):
794  """
795  Returns the highest bucket such that val >= lower bound for that bucket.
796 
797  Inputs:
798  val: float. The value to be sorted into a bucket.
799  bucket_lbs: list of floats, sorted ascending.
800 
801  Returns:
802  bucket_id: int in range(num_buckets); the bucket that val belongs to.
803  """
804  num_buckets = len(bucket_lbs)
805  for bucket_id in range(num_buckets - 1, -1, -1): # iterate descending
806  lb = bucket_lbs[bucket_id]
807  if val >= lb:
808  return bucket_id
809  raise ValueError('val %f is not >= any of the lower bounds: %s' % (val, bucket_lbs))
