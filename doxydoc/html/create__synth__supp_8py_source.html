75 def create_supp(opt):
76  """Evaluates a model.
77 
78  :param opt: tells the evaluation function how to run
79  :param bool print_parser: if provided, prints the options that are set within the
80  model after loading the model
81  :return: the final result of calling report()
82  """
83  # Create model and assign it to the specified task
84  agent = create_agent(opt, requireModelExists=True)
85  world = create_task(opt, agent)
86 
87  # Extract supp examples from misses on deploy set
88  num_seen = 0
89  num_misses = 0
90  num_supp = 0
91  num_supp_correct = 0
92  examples = []
93  while not world.epoch_done():
94  world.parley()
95  # Examples are considered one at a time
96  num_seen += 1
97  if num_seen % 1000 == 0:
98  print(f"{num_seen}/{world.num_examples()}")
99  report = world.report()
100  if report['accuracy'] < 1.0:
101  # Example is a miss (i.e., model got it wrong)
102  num_misses += 1
103  if random.random() < opt['conversion_rate']:
104  # Example will be converted (e.g., bot recognized mistake and asked)
105  num_supp += 1
106  texts = world.acts[0]['text'].split('\n')
107  context = texts[-1]
108  memories = texts[:-1]
109  candidates = world.acts[0]['label_candidates']
110  # Reward of 1 indicates positive, -1 indicates negative (for training)
111  # For now, we only train with positives, and the reward field is unused
112  reward = 1
113 
114  if random.random() < opt['conversion_acc']:
115  # Example will be converted correctly (e.g., good user response)
116  num_supp_correct += 1
117  response = world.acts[0]['eval_labels'][0]
118  else:
119  # Example will be converted incorrectly (e.g., bad user response)
120  response = random.choice(
121  world.acts[0]['label_candidates'][: NUM_INLINE_CANDS - 1]
122  )
123 
124  example = Parley(context, response, reward, candidates, memories)
125  examples.append(example)
126  world.reset_metrics()
127 
128  print("EPOCH DONE")
129  print(f"Model file: {opt['model_file']}")
130  print(f"Deploy file: {opt['task']}")
131  print(f"Supp file: {opt['outfile']}")
132  print(f"Deploy size (# examples seen): {num_seen}")
133  print(f"Supp size (# examples converted): {num_supp}")
134 
135  acc = 1 - (num_misses / num_seen)
136  print(f"Accuracy (% of deploy): {acc * 100:.1f}% ({num_misses} misses)")
137  print(
138  f"Conversion rate (% of misses): {num_supp/num_misses * 100:.2f}% "
139  f"({num_supp}/{num_misses})"
140  )
141  print(
142  f"Conversion acc (% of converted): {num_supp_correct/num_supp * 100:.2f}% "
143  f"({num_supp_correct}/{num_supp})"
144  )
145 
146  with open(opt['outfile'], 'w') as outfile:
147  for ex in examples:
148  outfile.write(json.dumps(ex.to_dict()) + '\n')
149 
150 
151 if __name__ == '__main__':
152  random.seed(42)
153  parser = setup_args()
