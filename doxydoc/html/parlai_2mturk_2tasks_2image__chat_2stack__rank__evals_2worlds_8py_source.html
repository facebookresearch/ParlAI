147  def parley(self):
148  """CHOOSER is given an image and 2 responses, and is asked
149  for more engaging response.
150  """
151  # Initial Message Value
152  control_msg = {'episode_done': False}
153  control_msg['id'] = 'SYSTEM'
154 
155  """We only have to worry about 1 agent"""
156  agent = self.agent
157 
158  """First, we give CHOOSER the image
159  """
160  while self.turn_idx < self.num_images:
161  print(self.world_tag + ' is at turn {}...'.format(self.turn_idx))
162  # Send image to turker
163  config = config_first if self.d_rnd == 'first_response' else config_second
164  control_msg['description'] = config['task_description']
165  self.example_num, example = agent.example_generator.pop_example()
166  img = load_image(
167  os.path.join(self.image_path, '{}.jpg'.format(example['image_hash']))
168  )
169  buffered = BytesIO()
170  img.save(buffered, format="JPEG")
171  encoded = str(base64.b64encode(buffered.getvalue()).decode('ascii'))
172  control_msg['image'] = encoded
173  """
174  Setup responses for ranking
175  """
176  responses = []
177  if self.dummy_eval:
178  # getting the first and second responses from eval
179  responses = [
180  ('comment', example['dialog'][1][1]),
181  ('comment', example['dialog'][2][1]),
182  ]
183  personality = example['dialog'][2][0]
184  else:
185  responses = [(ck, example[ck]) for ck in [self.ck1, self.ck2]]
186  personality = example.get('personality', '')
187 
188  random.shuffle(responses)
189  control_msg['responses'] = [c[1] for c in responses]
190  d_hist = ''
191  if self.d_rnd == 'first_response':
192  d_hist = [example['dialog'][1][1]]
193  elif self.d_rnd == 'second_response':
194  d_hist = [example['dialog'][1][1], example['dialog'][2][1]]
195  control_msg['d_hist'] = d_hist
196  control_msg['d_rnd'] = self.d_rnd
197  if self.show_personality:
198  control_msg['personality'] = (
199  '<b><span style="color:blue">'
200  '{}\n</span></b>'.format(personality.strip())
201  )
202 
203  best_pick = None
204  control_msg['text'] = PICK_BEST_MSG.format(self.turn_idx + 1)
205  control_msg['new_eval'] = True
206  agent.observe(validate(control_msg))
207  time.sleep(1)
208  act = agent.act(timeout=self.max_resp_time)
209  # First timeout check
210  self.check_timeout(act)
211  if self.chat_done:
212  break
213  try:
214  best_idx = int(act['chosen'])
215  reason = act['text']
216  best_pick = responses[best_idx]
217  except Exception:
218  # Agent disconnected
219  break
220 
221  example['rank_choices'] = responses
222  example['dialog_round_evaluated'] = self.d_rnd
223  example['best_pick'] = best_pick
224  example['reason'] = reason
