80 def get_word_stats(text, agent_dict, bins=[0, 100, 1000, 100000]):
81  """
82  Function which takes text sequence and dict, returns word freq and length statistics
83 
84  :param sequence: text sequence
85  :param agent_dict: can be external dict or dict from the model
86  :param bins: list with range boundaries
87  :return: freqs dictionary, num words, avg word length, avg char length
88  """
89  pred_list = agent_dict.tokenize(text)
90  pred_freq = [agent_dict.freq[word] for word in pred_list]
91  freqs = {i: 0 for i in bins}
92  for f in pred_freq:
93  for b in bins:
94  if f <= b:
95  freqs[b] += 1
96  break
97 
98  wlength = len(pred_list)
99  clength = len(text) # including spaces
100  return freqs, len(pred_freq), wlength, clength

103 def update_sent_attr_stats(sent_attrs, history, prediction):
104  """
105  Update the sent_attrs dict with the attributes of a prediction with given history.
106 
107  Inputs:
108  sent_attrs: dictionary mapping each attr (a string) to a list of floats
109  (the scores).
110  history: a ConvAI2History
111  prediction: string. the response text for which we measure sent attributes
112  """
113  for attr in sent_attrs.keys():
114  attr_score = eval_attr(prediction, history, attr)
115  sent_attrs[attr].append(attr_score)
116  return sent_attrs
