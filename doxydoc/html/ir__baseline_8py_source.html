33 class MaxPriorityQueue(Sequence):
34  """Fixed-size priority queue keeping the max_size largest items."""
35 
36  def __init__(self, max_size):
37  """Initialize priority queue.
38 
39  :param max_size: maximum capacity of priority queue.
40  """
41  self.capacity = max_size
42  self.lst = []
43 
44  def add(self, item, priority=None):
45  """Add element to the queue, with a separate priority if desired.
46 
47  Element will not be added if the queue is at capacity and the element
48  has lower priority than the lowest currently in the queue.
49 
50  :param item: item to add to queue.
51  :param priority: priority to use for item. if None (default), will use
52  the item itself to calculate its own priority.
53  """
54  if priority is None:
55  priority = item
56  if len(self.lst) < self.capacity:
57  heapq.heappush(self.lst, (priority, item))
58  elif priority > self.lst[0][0]:
59  heapq.heapreplace(self.lst, (priority, item))
60 
61  def __getitem__(self, key):
62  """Get item at specified index.
63 
64  :param key: integer index into priority queue, 0 <= index < max_size.
65 
66  :returns: item stored at the specified index.
67  """
68  return sorted(self.lst)[key][1]
69 
70  def __len__(self):
71  """Return length of priority queue."""
72  return len(self.lst)
73 
74  def __str__(self):
75  """Return str representation of the priority queue in list form."""
76  return str([v for _, v in sorted(self.lst)])
77 
78  def __repr__(self):
79  """Return repr representation of the priority queue in list form."""
80  return repr([v for _, v in sorted(self.lst)])

150 def score_match(query_rep, text, length_penalty, dictionary=None, debug=False):
151  """Calculate the score match between the query representation the text.
152 
153  :param query_rep: base query representation to match text again.
154  :param text: string to comapre against query_rep for matching tokens
155  :param length_penalty: scores are divided by the norm taken to this power
156  :dictionary: optional dictionary to use to tokenize text
157  :debug: flag to enable printing every match
158 
159  :returns: float score of match
160  """
161  if text == "":
162  return 0
163  if not dictionary:
164  words = text.lower().split(' ')
165  else:
166  words = [w for w in dictionary.tokenize(text.lower())]
167  score = 0
168  rw = query_rep['words']
169  used = {}
170  for w in words:
171  if w in rw and w not in used:
172  score += rw[w]
173  if debug:
174  print("match: " + w)
175  used[w] = True
176  norm = math.sqrt(len(used))
177  norm = math.pow(norm * query_rep['norm'], length_penalty)
178  if norm > 1:
179  score /= norm
180  return score

183 def rank_candidates(query_rep, cands, length_penalty, dictionary=None):
184  """Rank candidates given representation of query.
185 
186  :param query_rep: base query representation to match text again.
187  :param cands: strings to compare against query_rep for matching tokens
188  :param length_penalty: scores are divided by the norm taken to this power
189  :dictionary: optional dictionary to use to tokenize text
190 
191  :returns: ordered list of candidate strings in score-ranked order
192  """
193  if True:
194  mpq = MaxPriorityQueue(100)
195  for c in cands:
196  score = score_match(query_rep, c, length_penalty, dictionary)
197  mpq.add(c, score)
198  return list(reversed(mpq))
199  else:
200  cands = list(cands)
201  score = [0] * len(cands)
202  for i, c in enumerate(cands):
203  score[i] = -score_match(query_rep, c, length_penalty, dictionary)
204  r = [i[0] for i in sorted(enumerate(score), key=lambda x: x[1])]
205  res = []
206  for i in range(min(100, len(score))):
207  res.append(cands[r[i]])
208  return res

315  def build_query_representation(self, query):
316  """Build representation of query, e.g. words or n-grams.
317 
318  :param query: string to represent.
319 
320  :returns: dictionary containing 'words' dictionary (token => frequency)
321  and 'norm' float (square root of the number of tokens)
322  """
323  rep = {}
324  rep['words'] = {}
325  words = [w for w in self.dictionary.tokenize(query.lower())]
326  rw = rep['words']
327  used = {}
328  for w in words:
329  if len(self.dictionary.freqs()) > 0:
330  rw[w] = 1.0 / (1.0 + math.log(1.0 + self.dictionary.freqs()[w]))
331  else:
332  if w not in stopwords:

334  used[w] = True
