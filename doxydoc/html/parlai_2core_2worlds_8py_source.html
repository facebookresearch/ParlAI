6 """
7 Worlds are the basic environments which define how agents interact with one another.
8 
9  ``World(object)`` provides a generic parent class, including ``__enter__``
10  and ``__exit__`` statements which allow you to guarantee that the shutdown
11  method is called.
12 
13  ``DialogPartnerWorld(World)`` provides a two-agent turn-based dialog setting.
14 
15  ``MultiAgentDialogWorld(World)`` provides a multi-agent setting.
16 
17  ``MultiWorld(World)`` creates a set of environments (worlds) for the same agent
18  to multitask over, a different environment will be chosen per episode.
19 
20  ``HogwildWorld(World)`` is a container that creates another world within itself for
21  every thread, in order to have separate simulated environments for each one.
22  Each world gets its own agents initialized using the ``share()`` parameters
23  from the original agents.
24 
25  ``BatchWorld(World)`` is a container for doing minibatch training over a world by
26  collecting batches of N copies of the environment (each with different state).
27 
28 
29 All worlds are initialized with the following parameters:
30 
31  ``opt`` -- contains any options needed to set up the agent. This generally contains
32  all command-line arguments recognized from core.params, as well as other
33  options that might be set through the framework to enable certain modes.
34  ``agents`` -- the set of agents that should be attached to the world,
35  e.g. for DialogPartnerWorld this could be the teacher (that defines the
36  task/dataset) and the learner agent. This is ignored in the case of
37  sharing, and the shared parameter is used instead to initalize agents.
38  ``shared`` (optional) -- if not None, contains any shared data used to construct
39  this particular instantiation of the world. This data might have been
40  initialized by another world, so that different agents can share the same
41  data (possibly in different Processes).
42 """

337  """
338  Basic world where each agent gets a turn in a round-robin fashion.
339 
340  Each agent receives as input the actions of all other agents since its last `act()`.
341  """

353  def parley(self):
354  """
355  Perform a turn for every agent.
356 
357  For each agent, get an observation of the last action each of the
358  other agents took. Then take an action yourself.
359  """

361  for index, agent in enumerate(self.agents):
362  acts[index] = agent.act()
363  for other_agent in self.agents:
364  if other_agent != agent:
365  other_agent.observe(validate(acts[index]))
366  self.update_counters()

454  def parley(self):
455  """For each agent: act, execute and observe actions in world."""
456  acts = self.acts
457  for index, agent in enumerate(self.agents):
458  # The agent acts.
459  acts[index] = agent.act()
460  # We execute this action in the world.
461  self.execute(agent, acts[index])
462  # All agents (might) observe the results.
463  for other_agent in self.agents:
464  obs = self.observe(other_agent, acts[index])
465  if obs is not None:
466  other_agent.observe(obs)
467  self.update_counters()

470 class MultiWorld(World):
471  """
472  Container for multiple worlds.
473 
474  Container for a set of worlds where each world gets a turn
475  in a round-robin fashion. The same user_agents are placed in each,
476  though each world may contain additional agents according to the task
477  that world represents.
478  """
479 
480  def __init__(self, opt, agents=None, shared=None, default_world=None):
481  super().__init__(opt)
482  self.worlds = []
483  for index, k in enumerate(opt['task'].split(',')):
484  k = k.strip()
485  if k:
486  opt_singletask = copy.deepcopy(opt)
487  opt_singletask['task'] = k
488  if shared:
489  # Create worlds based on shared data.
490  s = shared['worlds'][index]
491  self.worlds.append(s['world_class'](s['opt'], None, s))
492  else:
493  # Agents are already specified.
494  self.worlds.append(
495  create_task_world(
496  opt_singletask, agents, default_world=default_world
497  )
498  )
499  self.world_idx = -1
500  self.new_world = True
501  self.parleys = -1
502  self.random = opt.get('datatype', None) == 'train'
503  # Make multi-task task probabilities.
504  self.cum_task_weights = [1] * len(self.worlds)
505  self.task_choices = range(len(self.worlds))
506  weights = self.opt.get('multitask_weights', [1])
507  sum = 0
508  for i in self.task_choices:
509  if len(weights) > i:
510  weight = weights[i]
511  else:
512  weight = 1
513  self.cum_task_weights[i] = weight + sum
514  sum += weight

563  def parley_init(self):
564  """
565  Update the current subworld.
566 
567  If we are in the middle of an episode, keep the same world and finish this
568  episode. If we have finished this episode, pick a new world (either in a
569  random or round-robin fashion).
570  """
571  self.parleys = self.parleys + 1
572  if self.world_idx >= 0 and self.worlds[self.world_idx].episode_done():
573  self.new_world = True
574  if self.new_world:
575  self.new_world = False
576  self.parleys = 0
577  if self.random:
578  # select random world
579  self.world_idx = random.choices(
580  self.task_choices, cum_weights=self.cum_task_weights
581  )[0]
582  else:
583  # do at most one full loop looking for unfinished world
584  for _ in range(len(self.worlds)):
585  self.world_idx = (self.world_idx + 1) % len(self.worlds)
586  if not self.worlds[self.world_idx].epoch_done():
587  # if this world has examples ready, break
588  break

630 def _override_opts_in_shared(table, overrides):
631  """
632  Override all shared dicts.
633 
634  Looks recursively for ``opt`` dictionaries within shared dict and overrides
635  any key-value pairs with pairs from the overrides dict.
636  """
637  if 'opt' in table:
638  # change values if an 'opt' dict is available
639  for k, v in overrides.items():
640  table['opt'][k] = v
641  for k, v in table.items():
642  # look for sub-dictionaries which also might contain an 'opt' dict
643  if type(v) == dict and k != 'opt' and 'opt' in v:
644  _override_opts_in_shared(v, overrides)
645  elif type(v) == list:
646  for item in v:
647  if type(item) == dict and 'opt' in item:
648  # if this is a list of agent shared dicts, we want to iterate
649  _override_opts_in_shared(item, overrides)
650  else:
651  # if this is e.g. list of candidate strings, stop right away
652  break
653  return table

656 class BatchWorld(World):
657  """
658  BatchWorld contains many copies of the same world.
659 
660  Create a separate world for each item in the batch, sharing
661  the parameters for each.
662 
663  The underlying world(s) it is batching can be either
664  ``DialogPartnerWorld``, ``MultiAgentWorld``, ``ExecutableWorld`` or
665  ``MultiWorld``.
666  """
667 
668  def __init__(self, opt, world):
669  super().__init__(opt)
670  self.opt = opt
671  self.random = opt.get('datatype', None) == 'train'
672  self.world = world
673  self.worlds = []
674  for i in range(opt['batchsize']):
675  # make sure that any opt dicts in shared have batchindex set to i
676  # this lets all shared agents know which batchindex they have,
677  # which is needed for ordered data (esp valid/test sets)
678  shared = world.share()
679  shared['batchindex'] = i
680  for agent_shared in shared.get('agents', ''):
681  agent_shared['batchindex'] = i
682  # TODO: deprecate override_opts
683  _override_opts_in_shared(shared, {'batchindex': i})
684  self.worlds.append(shared['world_class'](opt, None, shared))
685  self.batch_observations = [None] * len(self.world.get_agents())
686  self.first_batch = None
687  self.acts = [None] * len(self.world.get_agents())
688 
689  def batch_observe(self, index, batch_actions, index_acting):
690  """Observe corresponding actions in all subworlds."""
691  batch_observations = []
692  for i, w in enumerate(self.worlds):
693  agents = w.get_agents()
694  observation = None
695  if batch_actions[i] is None:
696  # shouldn't send None, should send empty observations
697  batch_actions[i] = [{}] * len(self.worlds)
698  if hasattr(w, 'observe'):
699  # The world has its own observe function, which the action
700  # first goes through (agents receive messages via the world,
701  # not from each other).
702  observation = w.observe(agents[index], validate(batch_actions[i]))
703  else:
704  if index == index_acting:
705  return None # don't observe yourself talking
706  observation = validate(batch_actions[i])
707  observation = agents[index].observe(observation)
708  if observation is None:
709  raise ValueError('Agents should return what they observed.')
710  batch_observations.append(observation)
711  return batch_observations
712 
713  def batch_act(self, agent_idx, batch_observation):
714  """Act in all subworlds."""
715  # Given batch observation, do update for agents[index].
716  # Call update on agent
717  a = self.world.get_agents()[agent_idx]
718  if hasattr(a, 'batch_act') and not (
719  hasattr(a, 'use_batch_act') and not a.use_batch_act
720  ):
721  batch_actions = a.batch_act(batch_observation)
722  # Store the actions locally in each world.

845 class HogwildProcess(Process):
846  """
847  Process child used for ``HogwildWorld``.
848 
849  Each ``HogwildProcess`` contain its own unique ``World``.
850  """
851 
852  def __init__(self, tid, opt, shared, sync):
853  self.numthreads = opt['numthreads']
854  opt = copy.deepcopy(opt)
855  opt['numthreads'] = 1 # don't let threads create more threads!
856  self.opt = opt
857  self.shared = shared
858  self.shared['threadindex'] = tid

861  a['threadindex'] = tid

865  def run(self):
866  """
867  Run a parley loop.
868 
869  Runs normal parley loop for as many examples as this thread can get
870  ahold of via the semaphore ``queued_sem``.
871  """
872  world = self.shared['world_class'](self.opt, None, self.shared)
873  if self.opt.get('batchsize', 1) > 1:
874  world = BatchWorld(self.opt, world)
875  self.sync['threads_sem'].release()
876  with world:
877  while True:
878  if self.sync['term_flag'].value:
879  break # time to close
880  self.sync['queued_sem'].acquire()
881  self.sync['threads_sem'].release()
882 
883  # check if you need to reset before moving on
884  if self.sync['epoch_done_ctr'].value < 0:
885  with self.sync['epoch_done_ctr'].get_lock():
886  # increment the number of finished threads
887  self.sync['epoch_done_ctr'].value += 1
888  if self.sync['epoch_done_ctr'].value == 0:
889  # make sure reset sem is clean
890  for _ in range(self.numthreads):
891  self.sync['reset_sem'].acquire(block=False)
892  world.reset() # keep lock for this!
893 
894  while self.sync['epoch_done_ctr'].value < 0:
895  # only move forward once other threads have finished reset
896  time.sleep(0.1)
897 
898  # process an example or wait for reset
899  if not world.epoch_done() or self.opt.get('datatype').startswith(
900  'train', False
901  ):
902  # do one example if any available
903  world.parley()
904  with self.sync['total_parleys'].get_lock():
905  self.sync['total_parleys'].value += 1
906  else:
907  # during valid/test, we stop parleying once at end of epoch
908  with self.sync['epoch_done_ctr'].get_lock():
909  # increment the number of finished threads
910  self.sync['epoch_done_ctr'].value += 1
911  # send control back to main thread
912  self.sync['threads_sem'].release()
913  # we didn't process anything
914  self.sync['queued_sem'].release()
915  # wait for reset signal
916  self.sync['reset_sem'].acquire()

919 class HogwildWorld(World):
920  """
921  Creates a separate world for each thread (process).
922 
923  Maintains a few shared objects to keep track of state:
924 
925  - A Semaphore which represents queued examples to be processed. Every call
926  of parley increments this counter; every time a Process claims an
927  example, it decrements this counter.
928 
929  - A Condition variable which notifies when there are no more queued
930  examples.
931 
932  - A boolean Value which represents whether the inner worlds should shutdown.
933 
934  - An integer Value which contains the number of unprocessed examples queued
935  (acquiring the semaphore only claims them--this counter is decremented
936  once the processing is complete).
937  """
938 
939  def __init__(self, opt, world):
940  super().__init__(opt)
941  self.inner_world = world
942  self.numthreads = opt['numthreads']
943 
944  self.sync = { # syncronization primitives
945  # semaphores for counting queued examples
946  'queued_sem': Semaphore(0), # counts num exs to be processed
947  'threads_sem': Semaphore(0), # counts threads
948  'reset_sem': Semaphore(0), # allows threads to reset
949  # flags for communicating with threads
950  'reset_flag': Value('b', False), # threads should reset
951  'term_flag': Value('b', False), # threads should terminate
952  # counters
953  'epoch_done_ctr': Value('i', 0), # number of done threads
954  'total_parleys': Value('l', 0), # number of parleys in threads
955  }
956 
957  self.threads = []
958  for i in range(self.numthreads):
959  self.threads.append(HogwildProcess(i, opt, world.share(), self.sync))
960  time.sleep(0.05) # delay can help prevent deadlock in thread launches
961  for t in self.threads:
962  t.start()
963 
964  for _ in self.threads:
965  # wait for threads to launch
966  # this makes sure that no threads get examples before all are set up
967  # otherwise they might reset one another after processing some exs
968  self.sync['threads_sem'].acquire()
969 
970  print(f'[ {self.numthreads} threads initialized ]')
