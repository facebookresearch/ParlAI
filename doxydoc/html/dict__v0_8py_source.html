50 def find_ngrams(token_dict, text, n):
51  """Break text into ngrams that appear in ``token_dict``.
52 
53  :param token_dict: ``dict`` to check for ngrams
54  :param text: ``str`` to look for ngrams in
55  :param n: ``int`` max size of ngrams
56  """
57  # base case
58  if n <= 1:
59  return text
60  # tokens committed to output
61  saved_tokens = []
62  # tokens remaining to be searched in sentence
63  search_tokens = text[:]
64  # tokens stored until next ngram found
65  next_search = []
66  while len(search_tokens) >= n:
67  ngram = ' '.join(search_tokens[:n])
68  if ngram in token_dict:
69  # first, search previous unmatched words for smaller ngrams
70  sub_n = min(len(next_search), n - 1)
71  saved_tokens.extend(find_ngrams(token_dict, next_search, sub_n))
72  next_search.clear()
73  # then add this ngram
74  saved_tokens.append(ngram)
75  # then pop this ngram from the remaining words to search
76  search_tokens = search_tokens[n:]
77  else:
78  next_search.append(search_tokens.pop(0))
79  remainder = next_search + search_tokens
80  sub_n = min(len(remainder), n - 1)
81  saved_tokens.extend(find_ngrams(token_dict, remainder, sub_n))
82  return saved_tokens

559  def sort(self, trim=True):
560  """Sorts the dictionary, so that the elements with the lowest index have
561  the highest counts. This reindexes the dictionary according to the
562  sorted frequencies, breaking ties alphabetically by token.
563 
564  :param bool trim: If True, truncate the dictionary based on minfreq and
565  maxtokens.
566  """
567  # sort first by count, then alphabetically
568  if trim:
569  self.remove_tail(self.minfreq)
570  sorted_pairs = sorted(self.freq.items(), key=lambda x: (-x[1], x[0]))
571  new_tok2ind = {}
572  new_ind2tok = {}
573  for i, (tok, _) in enumerate(sorted_pairs):
574  new_tok2ind[tok] = i
575  new_ind2tok[i] = tok
576  self.tok2ind = new_tok2ind
577  self.ind2tok = new_ind2tok
578  if trim:
579  self.resize_to_max(self.maxtokens)
580  assert len(self.freq) == len(self.ind2tok) == len(self.tok2ind)
581  return sorted_pairs

593  def txt2vec(self, text, vec_type=list):
594  """Converts a string to a vector (list of ints).
595 
596  First runs a sentence tokenizer, then a word tokenizer.
597 
598  ``vec_type`` is the type of the returned vector if the input is a string.
599  """
600  if vec_type == list or vec_type == tuple or vec_type == set:
601  res = vec_type((self[token] for token in self.tokenize(str(text))))
602  elif vec_type == np.ndarray:
603  res = np.fromiter((self[token] for token in self.tokenize(text)), np.int)
604  else:
605  raise RuntimeError('Type {} not supported by dict'.format(vec_type))
606  return res

608  def vec2txt(self, vector, delimiter=' '):
609  """Converts a vector (iterable of ints) into a string, with each token
610  separated by the delimiter (default ``' '``).
611  """
612  text = delimiter.join(self[int(idx)] for idx in vector)
613  # if we used a BPE tokenizer we need to rejoin the encodings
614  if self.tokenizer == 'bpe' and not self.opt.get('bpe_debug', False):
615  text = text.replace('@@ ', '')
616  # It's also possible that we get a BPE encoding on the end of the word
617  if text.endswith('@@'):
618  text = text[:-2]
619  text = text.replace('__newln__', '\n')
620  return text

662  For technical details, please refer to https://arxiv.org/abs/1508.07909.
663  This class just wraps around the official subword-nmt repository.
664

667  in a second pass, calling tokenize() again to get processed output.
668  """

672 
673  If `codecs_filename` already exists, loads the pretrained codecs.
674  If it does not, codecs will be saved there after a call to `finalize()`.
675

677  """
678  if not BPE_INSTALLED:
679  raise RuntimeError(
680  "Please run \"pip install 'git+https://github.com/rsennrich"
681  "/subword-nmt.git#egg=subword-nmt'\""
682  )
683 
684  self.splitter = re.compile(r'\w+|[^\w\s]', re.UNICODE)
685

688  self._load_from_codecs()
689 
690  def _load_from_codecs(self):
691  with open(self.codecs, 'r') as codecs_file:
692  self.bpe = apply_bpe.BPE(codecs_file)
693 
694  def tokenize(self, text):
695  """Tokenize the text with bpe if codecs are already finalized.
696 
697  Otherwise, returns the regularly split tokens that will train the bpe.
698 
699  :param text: str. Raw text to tokenize.
700  :return: a list of tokens. Will use BPE once finalized.
701  """
702  text = text.replace('\n', ' __newln__ ')
703  tokens = self.splitter.findall(text)
704 
705  if hasattr(self, 'bpe'):
706  return self.bpe.segment_tokens(tokens)
707  else:
708  return tokens
709 
710  def finalize(self, frequencies, num_symbols=30000, minfreq=2):
711  """Build the codecs.
712 
713  :param: dictionary of (token: frequency) pairs
714  :param num_symbols: Number of BPE symbols. Recommend 30000-40000.
715  If <= 0, default 30000 will be used.
716  :param minfreq: Minimum frequency of a token before forced BPE
717  decomposition. If <= 0 will use subword-nmt default of 2.
718  """
719  if hasattr(self, 'bpe'):
720  # we already finalized the codecs
721  return False
722 
723  print('Dictionary: saving bpe codecs to {}'.format(self.codecs))
724 
725  dictionary = ("{} {}".format(k, v) for k, v in frequencies.items())
726 
727  if num_symbols <= 0:
728  num_symbols = 30000
729  if minfreq <= 0:
730  minfreq = 2
731  with open(self.codecs, 'w') as outstream:
732  learn_bpe.learn_bpe(
733  dictionary,
734  outstream,
735  num_symbols=num_symbols,
736  min_frequency=minfreq,
737  is_dict=True,
738  )
739
