361  def choose_best_caption(
362  self, image_features, personalities, candidates, candidates_encoded=None, k=1
363  ):
364  """
365  Choose the best caption for each example.
366 
367  :param image_features:
368  list of tensors of image features
369  :param personalities:
370  list of personalities
371  :param candidates:
372  list of candidates, one set per example
373  :param candidates_encoded:
374  optional; if specified, a fixed set of encoded candidates that is
375  used for each example
376  :param k:
377  number of ranked candidates to return. if < 1, we return the ranks
378  of all candidates in the set.
379 
380  :return:
381  a set of ranked candidates for each example
382  """
383  self.eval()
384  context_encoded, _ = self.forward(image_features, personalities, None)
385  context_encoded = context_encoded.detach()
386  one_cand_set = True
387  if candidates_encoded is None:
388  one_cand_set = False
389  candidates_encoded = [
390  self.forward(None, None, c)[1].detach() for c in candidates
391  ]
392  chosen = []
393  for img_index in range(len(context_encoded)):
394  context_encoding = context_encoded[img_index : img_index + 1, :]
395  scores = torch.mm(
396  candidates_encoded[img_index]
397  if not one_cand_set
398  else candidates_encoded,
399  context_encoding.transpose(0, 1),
400  )
401  if k >= 1:
402  _, index_top = torch.topk(scores, k, dim=0)
403  else:
404  _, index_top = torch.topk(scores, scores.size(0), dim=0)
405  chosen.append(
406  [
407  candidates[img_index][idx] if not one_cand_set else candidates[idx]
408  for idx in index_top.unsqueeze(1)
409  ]
410  )
411 
412  return chosen

414  def eval_batch_of_100(self, context_encoded, captions_encoded):
415  """
416  Evaluate a batch of 100 examples.
417 
418  The captions of the other examples are used as negatives.
419 
420  :param context_encoded:
421  the encoded context
422  :param captions_encoded:
423  the encoded captions
424 
425  :return:
426  the total loss, the total number of correct examples, and the
427  total number of examples evaluated.
428  """
429  total_loss = 0
430  total_ok = 0
431  num_examples = 0
432  for i in range(0, len(context_encoded), 100):
433  if i + 100 > len(context_encoded):
434  break
435  num_examples += 100
436  loss, num_correct = self.evaluate_one_batch(
437  context_encoded[i : i + 100, :], captions_encoded[i : i + 100, :]
438  )
439  total_loss += loss.data.cpu().item()
440  total_ok += num_correct.data.cpu().item()
441  return total_loss, total_ok, num_examples

443  def evaluate_one_batch(self, context_encoded, captions_encoded, during_train=False):
444  """
445  Compute loss - and number of correct examples - for one batch.
446 
447  :param context_encoded:
448  the encoded context
449  :param captions_encoded:
450  the encoded captions
451  :param during_train:
452  true if training, else False
453 
454  :return:
455  the batch loss and the number of correct examples
456  """
457  if not during_train:
458  self.zero_grad()
459  self.eval()
460  dot_products = context_encoded.mm(captions_encoded.t())
461  log_prob = torch.nn.functional.log_softmax(dot_products, dim=1)
462  targets = torch.arange(0, len(context_encoded), dtype=torch.long)
463  if self.use_cuda:
464  targets = targets.cuda()
465  loss = torch.nn.functional.nll_loss(log_prob, targets)
466  num_correct = (log_prob.max(dim=1)[1] == targets).float().sum()
467  return loss, num_correct

508  def captions_to_tensor(self, captions):
509  """
510  Tokenize a list of sentences into a 2D float tensor.
511 
512  :param captions:
513  list of sentences to tokenize
514 
515  :return:
516  a (batchsize X truncate_length) tensor representation of the captions,
517  and a similarly sized mask tensor
518  """
519  max_length = self.truncate_length
520  indexes = []
521  for c in captions:
522  vec = self.dictionary.txt2vec(c)
523  if len(vec) > max_length:
524  vec = vec[:max_length]
525  indexes.append(self.dictionary.txt2vec(c))
526  longest = max([len(v) for v in indexes])
527  res = torch.LongTensor(len(captions), longest).fill_(
528  self.dictionary.tok2ind[self.dictionary.null_token]
529  )
530  mask = torch.FloatTensor(len(captions), longest).fill_(0)
531  for i, inds in enumerate(indexes):
532  res[i, 0 : len(inds)] = torch.LongTensor(inds)
533  mask[i, 0 : len(inds)] = torch.FloatTensor([1] * len(inds))
534  if self.use_cuda:
535  res = res.cuda()
536  mask = mask.cuda()
537  return res, mask
