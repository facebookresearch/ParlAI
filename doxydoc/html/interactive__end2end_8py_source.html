110  def get_chosen_topic_passages(self, chosen_topic):
111  retrieved_txt_format = []
112  if chosen_topic in self.wiki_map:
113  retrieved_txt = self.wiki_map[chosen_topic]
114  retrieved_txts = retrieved_txt.split('\n')
115 
116  if len(retrieved_txts) > 1:
117  combined = ' '.join(retrieved_txts[2:])
118  sentences = self.sent_tok.tokenize(combined)
119  total = 0
120  for sent in sentences:
121  if total >= 10:
122  break
123  if len(sent) > 0:
124  retrieved_txt_format.append(' '.join([chosen_topic, sent]))
125  total += 1
126 
127  if len(retrieved_txt_format) > 0:
128  passages = '\n'.join(retrieved_txt_format)
129  else:
130  passages = ''
131 
132  return passages

134  def get_passages(self, act):
135  """Format passages retrieved by taking the first paragraph of the
136  top `num_retrieved` passages.
137  """
138  retrieved_txt = act.get('text', '')
139  cands = act.get('text_candidates', [])
140  if len(cands) > 0:
141  retrieved_txts = cands[: self.opt['num_retrieved']]
142  else:
143  retrieved_txts = [retrieved_txt]
144 
145  retrieved_txt_format = []
146  for ret_txt in retrieved_txts:
147  paragraphs = ret_txt.split('\n')
148  if len(paragraphs) > 2:
149  sentences = self.sent_tok.tokenize(paragraphs[2])
150  for sent in sentences:
151  delim = ' ' + TOKEN_KNOWLEDGE + ' '
152  retrieved_txt_format.append(delim.join([paragraphs[0], sent]))
153 
154  if len(retrieved_txt_format) > 0:
155  passages = '\n'.join(retrieved_txt_format)
156  else:
157  passages = ''
158 
159  return passages

161  def retriever_act(self, history):
162  """Combines and formats texts retrieved by the TFIDF retriever for the
163  chosen topic, the last thing the wizard said, and the last thing the
164  apprentice said.
165  """
166  # retrieve on chosen topic
167  chosen_topic_txts = None
168  if self.ret_history.get('chosen_topic'):
169  chosen_topic_txts = self.get_chosen_topic_passages(
170  self.ret_history['chosen_topic']
171  )
172 
173  # retrieve on apprentice
174  apprentice_txts = None
175  if self.ret_history.get('apprentice'):
176  apprentice_act = {
177  'text': self.ret_history['apprentice'],
178  'episode_done': True,
179  }
180  self.retriever.observe(apprentice_act)
181  apprentice_txts = self.get_passages(self.retriever.act())
182 
183  # retrieve on wizard
184  wizard_txts = None
185  if self.ret_history.get('wizard'):
186  wizard_act = {'text': self.ret_history['wizard'], 'episode_done': True}
187  self.retriever.observe(wizard_act)
188  wizard_txts = self.get_passages(self.retriever.act())
189 
190  # combine everything
191  combined_txt = ''
192  if chosen_topic_txts:
193  combined_txt += chosen_topic_txts
194  if wizard_txts:
195  combined_txt += '\n' + wizard_txts
196  if apprentice_txts:
197  combined_txt += '\n' + apprentice_txts
198 
199  return combined_txt

210  def maintain_retrieved_texts(self, history, observation):
211  """Maintain texts retrieved by the retriever to mimic the set-up
212  from the data collection for the task.
213  """
214  if 'chosen_topic' not in history:

223  history['apprentice'] = ''
224  if 'next_wizard' in history:
225  del history['next_wizard']
226  history['episode_done'] = False
227 
228  # save chosen topic
229  if 'chosen_topic' in observation:
230  history['chosen_topic'] = observation['chosen_topic']
231  if 'text' in observation:
232  history['apprentice'] = observation['text']
233  if 'next_wizard' in history:
234  history['wizard'] = history['next_wizard']
235  # save last thing wizard said (for next time)
236  if 'labels' in observation:
237  history['next_wizard'] = observation['labels'][0]
238  elif 'eval_labels' in observation:
239  history['next_wizard'] = observation['eval_labels'][0]
240 
241  history['episode_done'] = observation['episode_done']
