37 def setup_title_to_passage(opt):
38  print('[ Setting up Title to Passage Dict ]')
39  saved_dp = os.path.join(os.getcwd() + '/data/', 'title_to_passage.pkl')
40  if os.path.exists(saved_dp):
41  print('[ Loading from saved location, {} ]'.format(saved_dp))
42  with open(saved_dp, 'rb') as f:
43  title_to_passage = pickle.load(f)
44  return title_to_passage
45  topics_path = '{}/personas_with_wiki_links.txt'.format(os.getcwd())
46  topics = []
47  with open(topics_path) as f:
48  text = f.read()
49  personas = text.split('\n\n')
50  for persona in personas:
51  persona = persona.split('\n')
52  for i in range(1, len(persona)):
53  p_i = persona[i]
54  if 'https' in p_i:
55  topic = unquote(p_i[p_i.rfind('/') + 1 :]).replace('_', ' ')
56  topics.append(topic)
57  ordered_opt = opt.copy()
58  ordered_opt['datatype'] = 'train:ordered:stream'
59  ordered_opt['batchsize'] = 1
60  ordered_opt['numthreads'] = 1
61  ordered_opt['task'] = 'wikipedia:full:key-value'
62  teacher = create_task_agent_from_taskname(ordered_opt)[0]
63  title_to_passage = {}
64  i = 0
65  length = teacher.num_episodes()
66  pbar = tqdm.tqdm(total=length)
67  while not teacher.epoch_done():
68  pbar.update(1)
69  i += 1
70  action = teacher.act()
71  title = action['text']
72  if title in topics:
73  text = action['labels'][0]
74  title_to_passage[title] = text
75  pbar.close()
76  print('[ Finished Building Title to Passage dict; saving now]')
77  with open(saved_dp, 'wb') as f:
78  pickle.dump(title_to_passage, f)
