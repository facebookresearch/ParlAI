228  def _init_embeddings(self, log=True):
229  """Copy embeddings from the pretrained embeddings to the lookuptable.
230 
231  :param weight: weights of lookup table (nn.Embedding/nn.EmbeddingBag)
232  :param emb_type: pretrained embedding type
233  """
234  weight = self.model.lt.weight
235  emb_type = self.opt.get('embedding_type', 'random')
236  if emb_type == 'random':
237  return
238  embs, name = TorchAgent._get_embtype(self, emb_type)
239  cnt = 0
240  for w, i in self.dict.tok2ind.items():
241  if w in embs.stoi:
242  vec = TorchAgent._project_vec(
243  self, embs.vectors[embs.stoi[w]], weight.size(1)
244  )
245  weight.data[i] = vec
246  cnt += 1

249  'Initialized embeddings for {} tokens ({}%) from {}.'

332  def get_negs(self, xs, ys):
333  negs = []
334  cache_sz = len(self.ys_cache) - 1
335  if cache_sz < 1:
336  return negs
337  k = self.opt['neg_samples']
338  for _i in range(1, k * 3):
339  index = random.randint(0, cache_sz)
340  neg = self.ys_cache[index]
341  if not self.same(ys, neg):
342  negs.append(neg)
343  if len(negs) >= k:
344  break
345  if self.opt['parrot_neg'] > 0:
346  utt = self.history['last_utterance']
347  if len(utt) > 2:
348  query = torch.LongTensor(utt).unsqueeze(0)
349  negs.append(query)
350  return negs

382  def input_dropout(self, xs, ys, negs):
383  def dropout(x, rate):
384  xd = []
385  for i in x[0]:
386  if random.uniform(0, 1) > rate:
387  xd.append(i)
388  if len(xd) == 0:
389  # pick one random thing to put in xd
390  xd.append(x[0][random.randint(0, x.size(1) - 1)])
391  return torch.LongTensor(xd).unsqueeze(0)
392 
393  rate = self.opt.get('input_dropout')
394  xs2 = dropout(xs, rate)
395  ys2 = dropout(ys, rate)
396  negs2 = []
397  for n in negs:
398  negs2.append(dropout(n, rate))
399  return xs2, ys2, negs2

401  def predict(self, xs, ys=None, cands=None, cands_txt=None, obs=None):
402  """Produce a prediction from our model.
403 
404  Update the model using the targets if available, otherwise rank
405  candidates as well if they are available and param is set.
406  """
407  is_training = ys is not None
408  if is_training:
409  negs = self.get_negs(xs, ys)
410  if is_training and len(negs) > 0:
411  self.model.train()
412  self.optimizer.zero_grad()
413  if self.opt.get('input_dropout', 0) > 0:
414  xs, ys, negs = self.input_dropout(xs, ys, negs)
415  xe, ye = self.model(xs, ys, negs)
416  if self.debugMode:
417  # print example
418  print("inp: " + self.v2t(xs.squeeze()))
419  print("pos: " + self.v2t(ys.squeeze()))
420  for c in negs:
421  print("neg: " + self.v2t(c.squeeze()))
422  print("---")
423  y = -torch.ones(xe.size(0))
424  y[0] = 1
425  loss = self.criterion(xe, ye, y)
426  loss.backward()
427  self.optimizer.step()
428  pred = nn.CosineSimilarity().forward(xe, ye)
429  metrics = self.compute_metrics(loss.item(), pred.data.squeeze())
430  return [{'metrics': metrics}]
431  else:
432  self.model.eval()
433  if cands is None or cands[0] is None:
434  # cannot predict without candidates.
435  if self.fixedCands:
436  cands = [self.fixedCands]
437  cands_txt = [self.fixedCands_txt]
438  else:
439  return [{'text': 'I dunno.'}]
440  # test set prediction uses fixed candidates
441  if self.fixedX is None:
442  xe, ye = self.model(xs, ys, self.fixedCands)
443  self.fixedX = ye
444  else:
445  # fixed candidate embed vectors are cached, dont't recompute
446  blah = torch.LongTensor([1])
447  xe, ye = self.model(xs, ys, [blah])
448  ye = self.fixedX
449  else:
450  # test set prediction uses candidates
451  xe, ye = self.model(xs, ys, cands[0])
452  pred = nn.CosineSimilarity().forward(xe, ye)
453  # This is somewhat costly which we could avoid if we do not evalute ranking.
454  # i.e. by only doing: val,ind = pred.max(0)
455  val, ind = pred.sort(descending=True)
456  # predict the highest scoring candidate, and return it.
457  ypred = cands_txt[0][ind.data[0]]
458  tc = []
459  for i in range(min(100, ind.size(0))):
460  tc.append(cands_txt[0][ind.data[i]])
461  ret = [{'text': ypred, 'text_candidates': tc}]
462  return ret
463  return [{'id': self.getID()}]

465  def vectorize(self, observations):
466  """Convert a list of observations into input & target tensors."""
467 
468  def valid(obs):
469  # check if this is an example our model should actually process
470  return 'text2vec' in obs and len(obs['text2vec']) > 0
471 
472  try:
473  # valid examples and their indices
474  valid_inds, exs = zip(
475  *[(i, ex) for i, ex in enumerate(observations) if valid(ex)]
476  )
477  except ValueError:
478  # zero examples to process in this batch, so zip failed to unpack
479  return None, None, None, None
480 
481  # `x` text is already tokenized and truncated
482  # sort by length so we can use pack_padded
483  parsed_x = [ex['text2vec'] for ex in exs]
484  x_lens = [len(x) for x in parsed_x]
485  ind_sorted = sorted(range(len(x_lens)), key=lambda k: -x_lens[k])
486 
487  exs = [exs[k] for k in ind_sorted]
488  valid_inds = [valid_inds[k] for k in ind_sorted]
489  parsed_x = [parsed_x[k] for k in ind_sorted]
490 
491  labels_avail = any(['labels' in ex for ex in exs])
492 
493  max_x_len = max([len(x) for x in parsed_x])
494  for x in parsed_x:
495  x += [self.NULL_IDX] * (max_x_len - len(x))
496  xs = torch.LongTensor(parsed_x)
497 
498  # set up the target tensors
499  ys = None
500  labels = None
501  if labels_avail:
502  # randomly select one of the labels to update on, if multiple
503  labels = [random.choice(ex.get('labels', [''])) for ex in exs]
504  # parse each label and append END
505  parsed_y = [deque(maxlen=self.truncate) for _ in labels]
506  for dq, y in zip(parsed_y, labels):
507  dq.extendleft(reversed(self.parse(y)))
508  max_y_len = max(len(y) for y in parsed_y)
509  for y in parsed_y:
510  y += [self.NULL_IDX] * (max_y_len - len(y))
511  ys = torch.LongTensor(parsed_y)
512 
513  cands = []
514  cands_txt = []
515  if ys is None:
516  # only build candidates in eval mode.
517  for o in observations:
518  if o.get('label_candidates', False):
519  cs = []
520  ct = []
521  for c in o['label_candidates']:
522  cs.append(torch.LongTensor(self.parse(c)).unsqueeze(0))
523  ct.append(c)
524  cands.append(cs)
525  cands_txt.append(ct)
526  else:
527  cands.append(None)
528  cands_txt.append(None)
529  return xs, ys, cands, cands_txt

540  def batch_act(self, observations):
541  batchsize = len(observations)
542  # initialize a table of replies with this agent's id
543  # convert the observations into batches of inputs and targets
544  # valid_inds tells us the indices of all valid examples
545  # e.g. for input [{}, {'text': 'hello'}, {}, {}], valid_inds is [1]
546  # since the other three elements had no 'text' field
547  xs, ys, cands, cands_txt = self.vectorize(observations)
548  batch_reply = self.predict(xs, ys, cands, cands_txt, observations)
549  while len(batch_reply) < batchsize:
550  batch_reply.append({'id': self.getID()})
551  self.add_to_ys_cache(ys)
552  return batch_reply
