77 @contextlib.contextmanager
78 def override_print(suppress=False, prefix=None):
79  """
80  Context manager to override the print to suppress or modify output.
81  Recommended usage is to call this with suppress=True for all non-primary workers,
82  or call with a prefix of rank on all workers.
83  >>> with override_print(prefix="rank{}".format(rank)):
84  ... my_computation()
85  :param bool suppress:
86  if true, all future print statements are noops.
87  :param str prefix:
88  if not None, this string is prefixed to all future print statements.
89  """
90  builtin_print = builtins.print
91 
92  def new_print(*args, **kwargs):
93  if suppress:
94  # do nothing
95  return
96  elif prefix:
97  return builtin_print(prefix, *args, **kwargs)
98  else:
99  # default to normal print
100  return builtin_print(*args, **kwargs)
101 
102  # override the print for now
103  builtins.print = new_print
104  yield
105  # bring it back at the end of the context
106  builtins.print = builtin_print

109 def all_gather_list(data, max_size=16384):
110  """
111  Gather arbitrary data from all nodes into a list.
112 
113  Similar to `~torch.distributed.all_gather` but for arbitrary Python
114  data. Note that *data* must be picklable.
115 
116  :param data:
117  data from the local worker to be gathered on other workers
118  :param int max_size:
119  maximum size of the data to be gathered across workers
120 
121  :returns:
122  a list containing [data1, data2, ...] of all workers
123  """
124  if not is_distributed():
125  # fall back to just keeping things basic if we're not distributed
126  return [data]
127 
128  # stolen shamelessly from fairseq
129  # https://github.com/pytorch/fairseq/blob/c37250ab1c845919af721cd3f5c4cec2993aefe1/fairseq/distributed_utils.py#L116-L170
130  rank = dist.get_rank()
131  world_size = dist.get_world_size()
132 
133  buffer_size = max_size * world_size
134  if (
135  not hasattr(all_gather_list, '_buffer')
136  or all_gather_list._buffer.numel() < buffer_size
137  ):
138  all_gather_list._buffer = torch.cuda.ByteTensor(buffer_size)
139 
140  buffer = all_gather_list._buffer
141  buffer.zero_()
142 
143  enc = pickle.dumps(data)
144  enc_size = len(enc)
145  if enc_size + 2 > max_size:
146  raise ValueError('encoded data exceeds max_size: {}'.format(enc_size + 2))
147  assert max_size < 255 * 256
148 
149  buffer_rank = buffer[rank * max_size : (rank + 1) * max_size]
150  buffer_rank[0] = enc_size // 255 # this encoding works for max_size < 65k
151  buffer_rank[1] = enc_size % 255
152  buffer_rank[2 : enc_size + 2] = torch.ByteTensor(list(enc))
153 
154  dist.all_reduce(buffer)
155 
156  result = []
157  for i in range(world_size):
158  out_buffer = buffer[i * max_size : (i + 1) * max_size]
159  size = (255 * out_buffer[0].item()) + out_buffer[1].item()
160  if size > 0:
161  try:
162  result.append(pickle.loads(bytes(out_buffer[2 : size + 2].tolist())))
163  except pickle.UnpicklingError:
164  raise RuntimeError(
165  'There was an unpickling error in all_gather_list. This likely '
166  'means your workers got out of syncronization (e.g. one is '
167  'expecting to sync and another is not.)'
168  )
169 
170  return result

173 def sync_object(data, max_size=16384):
174  """
175  Sync an object among all workers.
176 
177  All workers will return the same value for `data` when returning from this
178  method, always using the primary worker's version. Useful for ensuring control
179  flow decisions are made the same.
180 
181  :param object data:
182  The object to synchronize. Must be pickleable.
183  :param int max_size:
184  The maximum size of this object in bytes. Large values than 255^2 are not
185  supported.
186 
187  :return: the synchronized data
188  """
189  if not is_distributed():
190  return data
191 
192  # prepare the buffer
193  if not hasattr(sync_object, '_buffer') or sync_object._buffer.numel() < max_size:
194  # cuda is safe because distributed mode is only okay with CUDA
195  sync_object._buffer = torch.cuda.ByteTensor(max_size)
196 
197  buffer = sync_object._buffer
198 
199  if is_primary_worker():
200  enc = pickle.dumps(data)
201  enc_size = len(enc)
202  if (enc_size + 2 > max_size) or (enc_size > 255 * 255):
203  # can't store the size in the first 2 bytes
204  raise ValueError('encoded data exceeds max_size')
205 
206  buffer[0] = enc_size // 255
207  buffer[1] = enc_size % 255
208  buffer[2 : enc_size + 2] = torch.ByteTensor(list(enc))
209 
210  dist.broadcast(buffer, 0)
211 
212  if not is_primary_worker():
213  # deserialize the data
214  enc_size = buffer[0].item() * 255 + buffer[1].item()
215  try:
216  data = pickle.loads(bytes(buffer[2 : enc_size + 2].tolist()))
217  except pickle.UnpicklingError:
218  raise RuntimeError(
219  'There was an unpickling error in sync_object. This likely '
220  'means your workers got out of syncronization (e.g. one is '
221  'expecting to sync and another is not.)'
222  )
223 
224  return data

227 def check_synced_parameters(model):
228  """
229  Check that all parameters across all workers are the same.
230 
231  Always returns True, or raises an AssertionError if they are not
232  synchronized.
233 
234  :param torch.nn.Module model: A pytorch model.
235  :return: True
236  """
237  if not is_distributed():
238  # if things aren't distributed, of course things are in sync
239  return True
240 
241  # compute the local norm:
242  norm2 = sum((p.data ** 2).sum().float() for p in model.parameters()).item()
243  all_versions = all_gather_list(norm2)
244  if not all(n == norm2 for n in all_versions):
245  raise AssertionError(
246  "Some models parameters were out of sync. Got the following norms: {}".format(
247  " ".join(str(x) for x in all_versions)
248  )
249  )
250 
251  return True
