82  def setup_data_with_feedback(self, path):
83  """Reads data in the fbdialog format.
84  This method is very similar to FbDialogTeacher.setup_data(..).
85  The difference is that in this method the feedback is appended to the query
86  from the current example; in the default setup the feedback is appended to
87  the x from the next example.
88 
89  The data would look something like this:
90 
91  Mary moved to the bedroom.
92  Mary travelled to the garden.
93  Where is John?
94  No, that's wrong.
95  [labels: garden]
96 
97  To append feedback to the current example, modify the task name like this:
98  python examples/display_data.py -t dbll_babi:task:2_p0.5:f
99  Default setup:
100  python examples/display_data.py -t dbll_babi:task:2_p0.5
101 
102  """
103  print("[loading fbdialog data:" + path + "]")
104  with open(path) as read:
105  start = True
106  x = ''
107 
108  y = None
109 
110  reward = 0
111  dialog_index = 0
112  read_feedback = False
113  for line in read:
114  line = line.strip().replace('\\n', '\n')
115  if len(line) == 0:
116  continue
117 
118  # first, get conversation index -- '1' means start of episode
119  space_idx = line.find(' ')
120  conv_id = line[:space_idx]
121 
122  # split line into constituent parts, if available:
123  # x<tab>y<tab>reward<tab>label_candidates
124  # where y, reward, and label_candidates are optional
125  split = line[space_idx + 1 :].split('\t')
126 
127  # remove empty items and strip each one
128  for i in range(len(split)):
129  word = split[i].strip()
130  if len(word) == 0:
131  split[i] = ''
132  else:
133  split[i] = word
134  # Empty reward string same as None
135  if len(split) > 2 and split[2] == '':
136  split[2] = None
137 
138  # now check if we're at a new episode
139  if conv_id == '1':
140  dialog_index += 1
141  x = x.strip()
142  if x:
143  yield [x, None, reward], start
144  start = True
145  reward = 0
146  # start a new episode
147  if self.cloze:
148  x = 'Fill in the blank in the last sentence.\n{x}'.format(
149  x=split[0]
150  )
151  else:
152  x = split[0]
153  else:
154  if x:
155  # otherwise add current x to what we have so far
156  x = '{x}\n{next_x}'.format(x=x, next_x=split[0])
157  else:
158  x = split[0]
159  if len(split) > 2 and split[2]:
160  reward += float(split[2])
161

164  # split labels

166

172  else:
