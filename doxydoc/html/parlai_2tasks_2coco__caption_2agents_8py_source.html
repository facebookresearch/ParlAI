24 """
25  Agents for MSCOCO Image Captioning Task
26 
27  There are two versions of the task - one comprising MSCOCO 2014 splits
28  (from the 2015 task competition), and one comprising MSCOCO 2017 splits
29 
30  For the 2014 splits, we use the train, val, and test split of Karpathy et.
31  al, "Deep visual-semantic alignments for generating image descriptions"
32  (splits from here: https://cs.stanford.edu/people/karpathy/deepimagesent/).
33  This split has ~82k train images, 5k validation images, and 5k test images.
34  The val and test images are taken from the original validation set of ~40k.
35 
36  For 2017, we use the splits from the official MSCOCO Image Captioning 2017
37  task.
38 
39 """
40 # There is no real dialog in this task, so for the purposes of display_data, we
41 # include a generic question that applies to all images.
42 QUESTION = "Describe the above picture in a sentence."

63  train_caps = [x['caption'] for x in train]

362  def get(self, episode_idx, entry_idx=0):
363  action = {'episode_done': True}
364 
365  if self.use_intro:
366  action['text'] = QUESTION
367 
368  if self.version == '2014':
369  ep = self.annotation[episode_idx]
370  action['labels'] = [s['raw'] for s in ep['sentences']]
371  action['image_id'] = ep['cocoid']
372  action['split'] = ep['split']
373  if not self.datatype.startswith('train'):
374  if self.num_cands > 0:
375  labels = action['labels']
376  cands_to_sample = [c for c in self.cands if c not in labels]
377  cands = (
378  random.Random(episode_idx).sample(
379  cands_to_sample, self.num_cands
380  )
381  ) + labels
382  random.shuffle(cands)
383  action['label_candidates'] = cands
384  else:
385  action['label_candidates'] = self.cands
386  else:
387  if not self.datatype.startswith('test'):
388  # test set annotations are not available for this dataset
389  anno = self.annotation[episode_idx]
390  action['labels'] = [anno['caption']]
391  action['image_id'] = anno['image_id']
392  if not self.datatype.startswith('train'):
393  if self.num_cands == -1:
394  candidates = self.cands
395  else:
396  # Can only randomly select from validation set
397  candidates = random.Random(episode_idx).choices(
398  self.cands, k=self.num_cands
399  )
400  if anno['caption'] not in candidates:
401  candidates.pop(0)
402  else:
403  candidates.remove(anno['caption'])
404 
405  candidate_labels = [anno['caption']]
406  candidate_labels += candidates
407  action['label_candidates'] = candidate_labels
408  else:
409  if self.num_cands == -1:
410  candidates = self.cands
411  else:
412  # Can only randomly select from validation set
413  candidates = random.Random(episode_idx).choices(
414  self.cands, k=self.num_cands
415  )
416  action['label_candidates'] = candidates
417  action['image_id'] = self.test_info['images'][episode_idx]['id']
418 
419  return action
