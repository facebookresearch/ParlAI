305  def get_rep(self, encodings, batchsize=None):
306  """
307  Get the multimodal representation of the encodings.
308 
309  :param encodings:
310  list of encodings
311  :param batchsize:
312  size of batch
313 
314  :return:
315  final multimodal representations
316  """
317  if not self.multimodal:
318  rep = self.sum_encodings(encodings)
319  else:
320  if self.multimodal_combo == "sum":
321  encodings = self.sum_encodings(encodings).unsqueeze(1)
322  elif self.multimodal_combo == "concat":

328  if rep is None:

332  def choose_best_response(
333  self,
334  image_features,
335  personalities,
336  dialogue_histories,
337  candidates,
338  candidates_encoded=None,
339  k=1,
340  batchsize=None,
341  ):
342  """
343  Choose the best response for each example.
344 
345  :param image_features:
346  list of tensors of image features
347  :param personalities:
348  list of personalities
349  :param dialogue_histories:
350  list of dialogue histories, one per example
351  :param candidates:
352  list of candidates, one set per example
353  :param candidates_encoded:
354  optional; if specified, a fixed set of encoded candidates that is
355  used for each example
356  :param k:
357  number of ranked candidates to return. if < 1, we return the ranks
358  of all candidates in the set.
359 
360  :return:
361  a set of ranked candidates for each example
362  """
363  self.eval()
364  _, _, encoded = self.forward(
365  image_features, personalities, dialogue_histories, None, batchsize=batchsize
366  )
367  encoded = encoded.detach()
368  one_cand_set = True
369  if candidates_encoded is None:
370  one_cand_set = False
371  candidates_encoded = [
372  self.forward_text_encoder(c).detach() for c in candidates
373  ]
374  chosen = [
375  self.choose_topk(
376  idx if not one_cand_set else 0,
377  encoded,
378  candidates,
379  candidates_encoded,
380  one_cand_set,
381  k,
382  )
383  for idx in range(len(encoded))
384  ]
385  return chosen

387  def choose_topk(
388  self, idx, encoded, candidates, candidates_encoded, one_cand_set, k
389  ):
390  """
391  Choose top k best responses for a single example.
392 
393  :param idx:
394  idx of example in encoded
395  :param encoded:
396  full matrix of encoded representations (for the whole batch)
397  :param candidates:
398  list of candidates
399  :param candidates_encoded:
400  encoding of the candidates
401  :param one_cand_set:
402  true if there is one set of candidates for each example
403  :param k:
404  how many ranked responses to return
405 
406  :return:
407  ranked list of k responses
408  """
409  encoding = encoded[idx : idx + 1, :]
410  scores = torch.mm(
411  candidates_encoded[idx] if not one_cand_set else candidates_encoded,
412  encoding.transpose(0, 1),
413  )
414  if k >= 1:
415  _, index_top = torch.topk(scores, k, dim=0)
416  else:
417  _, index_top = torch.topk(scores, scores.size(0), dim=0)
418  return [
419  candidates[idx][idx2] if not one_cand_set else candidates[idx2]
420  for idx2 in index_top.unsqueeze(1)
421  ]

423  def get_loss(self, total_encoded, labels_encoded):
424  """
425  Compute loss over batch.
426 
427  :param total_encoded:
428  encoding of the examples
429  :param labels_encoded:
430  encoding of the labels
431 
432  :return:
433  total batch loss, and number of correct examples
434  """
435  loss = None
436  num_correct = None
437  if labels_encoded is not None:
438  dot_products = total_encoded.mm(
439  labels_encoded.t()
440  ) # batch_size * batch_size
441  log_prob = torch.nn.functional.log_softmax(dot_products, dim=1)
442  targets = torch.arange(0, len(total_encoded), dtype=torch.long)
443  if self.use_cuda:
444  targets = targets.cuda()
445  loss = torch.nn.functional.nll_loss(log_prob, targets)
446  num_correct = (log_prob.max(dim=1)[1] == targets).float().sum()
447  return loss, num_correct

494  def __init__(
495  self,
496  n_heads,
497  n_layers,
498  hidden_dim,
499  ffn_size,
500  reduction=True,
501  attention_dropout=0.0,
502  relu_dropout=0.0,
503  learn_positional_embeddings=False,
504  ):
505  super().__init__()
506  self.ffn_size = ffn_size
507  self.n_layers = n_layers
508  self.n_heads = n_heads
509  self.out_dim = hidden_dim
510  self.dim = hidden_dim
511  self.reduction = reduction
512  assert hidden_dim % n_heads == 0, "MM-Combiner dim must be multiple of n_heads"
513  n_positions = 1024
514  self.position_embeddings = nn.Embedding(n_positions, hidden_dim)
515  if not learn_positional_embeddings:
516  create_position_codes(
517  n_positions, hidden_dim, out=self.position_embeddings.weight
518  )
519  else:
520  nn.init.normal_(self.position_embeddings.weight, 0, hidden_dim ** -0.5)
521 
522  self.layers = nn.ModuleList()
523  for _ in range(self.n_layers):
524  self.layers.append(
525  TransformerEncoderLayer(
526  n_heads, hidden_dim, ffn_size, attention_dropout, relu_dropout
527  )
528  )

530  def forward(self, tensor, mask):
531  """
532  Forward pass.
533 
534  :param tensor:
535  a [bsz, seq_len, hidden_dim] FloatTensor
536  :param mask:
537  a [bsz, seq_len] ByteTensor filled with 1 when inside the sequence and 0 outside.
538 
539  :return:
540  output: a [bsz, hidden_dim] FloatTensor of encodings
541  mask: the same as before
542  """
543  seq_len = tensor.size(1)
544  positions = tensor.new(seq_len).long()
545  positions = torch.arange(seq_len, out=positions).unsqueeze(0)
546  tensor = tensor + self.position_embeddings(positions).expand_as(tensor)
547 
548  tensor *= mask.unsqueeze(-1).float()
549  for i in range(self.n_layers):
550  tensor = self.layers[i](tensor, mask)
551 
552  if self.reduction:
553  divisor = mask.float().sum(dim=1).unsqueeze(-1).clamp(min=1e-20)
554  output = tensor.sum(dim=1) / divisor
555  return output
556  else:
557  output = tensor
558  return output, mask
