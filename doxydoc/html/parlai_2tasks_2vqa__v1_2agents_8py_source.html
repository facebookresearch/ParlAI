178  def tokenize_mcb(self, s):
179  t_str = s.lower()
180  for i in [
181  r'\?',
182  r'\!',
183  r'\'',
184  r'\"',
185  r'\$',
186  r'\:',
187  r'\@',
188  r'\(',
189  r'\)',
190  r'\,',
191  r'\.',
192  r'\;',
193  ]:
194  t_str = re.sub(i, '', t_str)
195  for i in [r'\-', r'\/']:
196  t_str = re.sub(i, ' ', t_str)
197  q_list = re.sub(r'\?', '', t_str.lower()).split(' ')
198  q_list = list(filter(lambda x: len(x) > 0, q_list))
199  return q_list

214  def act(self):
215  """Add any words passed in the 'text' field of the observation to this
216  dictionary.
217  """
218  mc_label = self.observation.get('mc_label', self.observation.get('labels', []))

224  def encode_question(self, examples, training):
225  minwcount = self.opt.get('minwcount', 0)
226  maxlength = self.opt.get('maxlength', 16)
227  for ex in examples:
228  words = self.tokenize_mcb(ex['text'])
229  if training:
230  words_unk = [
231  w if self.freq.get(w, 0) > minwcount else self.unk_token
232  for w in words
233  ]
234  else:
235  words_unk = [w if w in self.tok2ind else self.unk_token for w in words]
236  ex['question_wids'] = [self.tok2ind[self.null_token]] * maxlength
237  for k, w in enumerate(words_unk):
238  if k < maxlength:
239  ex['question_wids'][k] = self.tok2ind[w]
240  return examples

242  def encode_answer(self, examples):
243  for ex in examples:
244  if self.opt.get('samplingans', True):
245  labels = ex.get('labels', ex.get('eval_labels'))
246  ans_count = Counter(labels).most_common()
247  valid_ans = []
248  valid_count = []
249  for ans in ans_count:
250  if ans[0] in self.ans2ind:
251  valid_ans.append(self.ans2ind[ans[0]])
252  valid_count.append(ans[1])
253  if not valid_ans:
254  ex['answer_aid'] = 0
255  else:
256  probs = valid_count / np.sum(valid_count)
257  ex['answer_aid'] = int(np.random.choice(valid_ans, p=probs))
258  else:
259  ex['answer_aid'] = self.ans2ind[ex['mc_label'][0]]
260  return examples

299  def save(self, filename=None, append=False, sort=True):
300  """Save dictionary to file.
301  Format is 'token<TAB>count' for every token in the dictionary, sorted
302  by count with the most frequent words first.
303 
304  If ``append`` (default ``False``) is set to ``True``, appends instead
305  of overwriting.
306 
307  If ``sort`` (default ``True``), then first sort the dictionary before
308  saving.
309  """
310  cw = sorted([(count, w) for w, count in self.ansfreq.items()], reverse=True)
311  final_exs = cw[: self.opt.get('nans', 2000)]
312  final_list = dict([(w, c) for c, w in final_exs])
313  self.ansfreq = defaultdict(int)
314  for ans, ques in self.ans2ques.items():
315  if ans in final_list:
316  for que in ques:
317  self.add_to_ques_dict(que)
318  self.add_to_ans_dict(ans)
319 
320  filename = self.opt['dict_file'] if filename is None else filename
321  print('Dictionary: saving dictionary to {}'.format(filename))
322  # if sort:
323  # self.sort()
324 
325  with open(filename, 'a' if append else 'w') as write:
326  for i in range(len(self.ind2tok)):
327  tok = self.ind2tok[i]
328  cnt = self.freq[tok]
329  write.write('{tok}\t{cnt}\n'.format(tok=escape(tok), cnt=cnt))
330
