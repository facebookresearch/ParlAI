88  def _forward_padded(self, x, x_mask):
89  """Slower (significantly), but more precise,
90  encoding that handles padding."""
91  # Compute sorted sequence lengths
92  lengths = x_mask.data.eq(0).long().sum(1).squeeze()
93  _, idx_sort = torch.sort(lengths, dim=0, descending=True)
94  _, idx_unsort = torch.sort(idx_sort, dim=0)
95 
96  lengths = list(lengths[idx_sort])
97  idx_sort = Variable(idx_sort)
98  idx_unsort = Variable(idx_unsort)
99 
100  # Sort x
101  x = x.index_select(0, idx_sort)
102 
103  # Transpose batch and sequence dims
104  x = x.transpose(0, 1).contiguous()
105 
106  # Pack it up
107  rnn_input = nn.utils.rnn.pack_padded_sequence(x, lengths)
108 
109  # Encode all layers
110  outputs = [rnn_input]
111  for i in range(self.num_layers):
112  rnn_input = outputs[-1]
113 
114  # Apply dropout to input
115  if self.dropout_rate > 0:
116  dropout_input = F.dropout(
117  rnn_input.data, p=self.dropout_rate, training=self.training
118  )
119  rnn_input = nn.utils.rnn.PackedSequence(
120  dropout_input, rnn_input.batch_sizes
121  )

123

126  outputs[i] = nn.utils.rnn.pad_packed_sequence(o)[0]
127

131  else:
132  output = outputs[-1]

136  output = output.index_select(0, idx_unsort)
137

139  if self.dropout_output and self.dropout_rate > 0:
140  output = F.dropout(output, p=self.dropout_rate, training=self.training)
141  return output

144 class SeqAttnMatch(nn.Module):
145  """Given sequences X and Y, match sequence Y to each element in X.
146  * o_i = sum(alpha_j * y_j) for i in X
147  * alpha_j = softmax(y_j * x_i)
148  """
149 
150  def __init__(self, input_size, identity=False):
151  super(SeqAttnMatch, self).__init__()
152  if not identity:
153  self.linear = nn.Linear(input_size, input_size)
154  else:
155  self.linear = None
156 
157  def forward(self, x, y, y_mask):
158  """Input shapes:
159  x = batch * len1 * h
160  y = batch * len2 * h
161  y_mask = batch * len2
162  Output shapes:
163  matched_seq = batch * len1 * h
164  """
165  # Project vectors
166  if self.linear:
167  x_proj = self.linear(x.view(-1, x.size(2))).view(x.size())
168  x_proj = F.relu(x_proj)
169  y_proj = self.linear(y.view(-1, y.size(2))).view(y.size())
170  y_proj = F.relu(y_proj)
171  else:
172  x_proj = x
173  y_proj = y
174 
175  # Compute scores
176  scores = x_proj.bmm(y_proj.transpose(2, 1))
177 
178  # Mask padding
179  y_mask = y_mask.unsqueeze(1).expand(scores.size())
180  scores.data.masked_fill_(y_mask.data, -float('inf'))
181 
182  # Normalize with softmax
183  alpha_flat = F.softmax(scores.view(-1, y.size(1)))
184  alpha = alpha_flat.view(-1, x.size(1), y.size(1))
185 
186  # Take weighted average
187  matched_seq = alpha.bmm(y)
188  return matched_seq

192  """A bilinear attention layer over a sequence X w.r.t y:
193  * o_i = softmax(x_i'Wy) for x_i in X.
194 
195  Optionally don't normalize output weights.
196  """
197

199  super(BilinearSeqAttn, self).__init__()
200  if not identity:

202  else:
203  self.linear = None

205  def forward(self, x, y, x_mask):
206  """
207  x = batch * len * h1
208  y = batch * h2
209  x_mask = batch * len
210  """
211  Wy = self.linear(y) if self.linear is not None else y
212  xWy = x.bmm(Wy.unsqueeze(2)).squeeze(2)
213  xWy.data.masked_fill_(x_mask.data, -float('inf'))
214  self.xWy = xWy
215  if self.training:
216  # In training we output log-softmax for NLL
217  alpha = F.log_softmax(xWy)
218  else:
219  # ...Otherwise 0-1 probabilities
220  # alpha = F.softmax(xWy)
221 
222  # Note: We found better eval performance with unnormalized weights
223  # here
224  alpha = xWy.exp()
225  return alpha
