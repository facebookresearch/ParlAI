73 def calculate_mturk_cost(payment_opt):
74  """MTurk Pricing: https://requester.mturk.com/pricing
75  20% fee on the reward and bonus amount (if any) you pay Workers.
76  HITs with 10 or more assignments will be charged an additional
77  20% fee on the reward you pay Workers.
78 
79  Example payment_opt format for paying reward:
80  {
81  'type': 'reward',
82  'num_total_assignments': 1,
83  'reward': 0.05 # in dollars
84  }
85 
86  Example payment_opt format for paying bonus:
87  {
88  'type': 'bonus',
89  'amount': 1000 # in dollars
90  }
91  """
92  total_cost = 0
93  if payment_opt['type'] == 'reward':
94  mult = 1.2
95  total_cost = payment_opt['num_total_assignments'] * payment_opt['reward'] * mult
96  elif payment_opt['type'] == 'bonus':
97  total_cost = payment_opt['amount'] * 1.2
98  return total_cost

183 def find_qualification(qualification_name, is_sandbox, must_be_owned=True):
184  """Query amazon to find the existing qualification name, return the Id,
185  otherwise return none.
186  If must_be_owned is true, it only returns qualifications owned by the user.
187  Will return False if it finds another's qualification
188  """
189  client = get_mturk_client(is_sandbox)
190 
191  # Search for the qualification owned by the current user
192  response = client.list_qualification_types(
193  Query=qualification_name, MustBeRequestable=True, MustBeOwnedByCaller=True
194  )
195 
196  for qualification in response['QualificationTypes']:
197  if qualification['Name'] == qualification_name:
198  return qualification['QualificationTypeId']
199 
200  # Qualification was not found to exist, check to see if someone else has it
201  response = client.list_qualification_types(
202  Query=qualification_name, MustBeRequestable=True, MustBeOwnedByCaller=False
203  )
204 
205  for qualification in response['QualificationTypes']:
206  if qualification['Name'] == qualification_name:
207  if must_be_owned:
208  print(
209  'Sorry, the qualification name {} is already owned, '
210  'please use a different name for your qualification.'
211  ''.format(qualification_name)
212  )
213  return False
214  return qualification['QualificationTypeId']
215  return None

383  policy_json = '''{{

400  TopicArn=arn, AttributeName='Policy', AttributeValue=policy_json
