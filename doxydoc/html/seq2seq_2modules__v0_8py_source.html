18 def pad(tensor, length, dim=0):
19  if tensor.size(dim) < length:
20  return torch.cat(
21  [
22  tensor,
23  tensor.new(
24  *tensor.size()[:dim],
25  length - tensor.size(dim),
26  *tensor.size()[dim + 1 :],
27  ).zero_(),
28  ],
29  dim=dim,
30  )
31  else:
32  return tensor

100  def unbeamize_hidden(self, hidden, beam_size, batch_size):
101  """
102  Creates a view of the hidden where batch axis is collapsed with beam axis,
103  we need to do this for batched beam search, i.e. we emulate bigger mini-batch
104  :param hidden: hidden state of the decoder
105  :param beam_size: beam size, i.e. num of hypothesis
106  :param batch_size: number of samples in the mini batch
107  :return: view of the hidden
108  """
109  if isinstance(hidden, tuple):
110  num_layers = hidden[0].size(0)
111  hidden_size = hidden[0].size(-1)
112  return (
113  hidden[0].view(num_layers, batch_size * beam_size, hidden_size),
114  hidden[1].view(num_layers, batch_size * beam_size, hidden_size),
115  )
116  else: # GRU
117  num_layers = hidden.size(0)
118  hidden_size = hidden.size(-1)
119  return hidden.view(num_layers, batch_size * beam_size, hidden_size)

125  def forward(
126  self,
127  xs,
128  ys=None,
129  cands=None,
130  valid_cands=None,
131  prev_enc=None,
132  rank_during_training=False,
133  beam_size=1,
134  topk=1,
135  ):
136  """Get output predictions from the model.
137 
138  Arguments:
139  xs -- input to the encoder
140  ys -- expected output from the decoder
141  cands -- set of candidates to rank, if applicable
142  valid_cands -- indices to match candidates with their appropriate xs
143  prev_enc -- if you know you'll pass in the same xs multiple times and
144  the model is in eval mode, you can pass in the encoder output from
145  the last forward pass to skip recalcuating the same encoder output
146  rank_during_training -- (default False) if set, ranks any available
147  cands during training as well
148  """
149  input_xs = xs
150  nbest_beam_preds, nbest_beam_scores = None, None
151  bsz = len(xs)
152  if ys is not None:
153  # keep track of longest label we've ever seen
154  # we'll never produce longer ones than that during prediction
155  self.longest_label = max(self.longest_label, ys.size(1))
156 
157  if prev_enc is not None:
158  enc_out, hidden, attn_mask = prev_enc
159  else:
160  enc_out, hidden = self.encoder(xs)
161  attn_mask = xs.ne(0).float() if self.attn_type != 'none' else None
162  encoder_states = (enc_out, hidden, attn_mask)
163  start = self.START.detach()
164  starts = start.expand(bsz, 1)
165 
166  predictions = []
167  scores = []
168  cand_preds, cand_scores = None, None
169  if self.rank and cands is not None:
170  decode_params = (start, hidden, enc_out, attn_mask)
171  if self.training:
172  if rank_during_training:
173  cand_preds, cand_scores = self.ranker.forward(
174  cands, valid_cands, decode_params=decode_params
175  )
176  else:
177  cand_preds, cand_scores = self.ranker.forward(
178  cands, valid_cands, decode_params=decode_params
179  )
180 
181  if ys is not None:
182  y_in = ys.narrow(1, 0, ys.size(1) - 1)
183  xs = torch.cat([starts, y_in], 1)
184  if self.attn_type == 'none':
185  preds, score, hidden = self.decoder(xs, hidden, enc_out, attn_mask)
186  predictions.append(preds)
187  scores.append(score)
188  else:
189  for i in range(ys.size(1)):
190  xi = xs.select(1, i)
191  preds, score, hidden = self.decoder(xi, hidden, enc_out, attn_mask)
192  predictions.append(preds)
193  scores.append(score)
194  else:
195  # here we do search: supported search types: greedy, beam search
196  if beam_size == 1:
197  done = [False for _ in range(bsz)]
198  total_done = 0
199  xs = starts
200 
201  for _ in range(self.longest_label):
202  # generate at most longest_label tokens
203  preds, score, hidden = self.decoder(
204  xs, hidden, enc_out, attn_mask, topk
205  )
206  scores.append(score)
207  xs = preds
208  predictions.append(preds)
209 
210  # check if we've produced the end token
211  for b in range(bsz):
212  if not done[b]:
213  # only add more tokens for examples that aren't done
214  if preds.data[b][0] == self.END_IDX:
215  # if we produced END, we're done
216  done[b] = True
217  total_done += 1
218  if total_done == bsz:
219  # no need to generate any more
220  break
221 
222  elif beam_size > 1:
223  enc_out, hidden = (
224  encoder_states[0],
225  encoder_states[1],
226  ) # take it from encoder
227  enc_out = enc_out.unsqueeze(1).repeat(1, beam_size, 1, 1)
228  # create batch size num of beams
229  data_device = enc_out.device
230  beams = [
231  Beam(
232  beam_size,
233  3,
234  0,
235  1,
236  2,
237  min_n_best=beam_size / 2,
238  cuda=data_device,
239  )
240  for _ in range(bsz)
241  ]
242  # init the input with start token
243  xs = starts
244  # repeat tensors to support batched beam
245  xs = xs.repeat(1, beam_size)
246  attn_mask = input_xs.ne(0).float()
247  attn_mask = attn_mask.unsqueeze(1).repeat(1, beam_size, 1)
248  repeated_hidden = []
249 
250  if isinstance(hidden, tuple):
251  for i in range(len(hidden)):
252  repeated_hidden.append(
253  hidden[i].unsqueeze(2).repeat(1, 1, beam_size, 1)
254  )
255  hidden = self.unbeamize_hidden(
256  tuple(repeated_hidden), beam_size, bsz
257  )
258  else: # GRU
259  repeated_hidden = hidden.unsqueeze(2).repeat(1, 1, beam_size, 1)
260  hidden = self.unbeamize_hidden(repeated_hidden, beam_size, bsz)
261  enc_out = self.unbeamize_enc_out(enc_out, beam_size, bsz)
262  xs = xs.view(bsz * beam_size, -1)
263  for step in range(self.longest_label):
264  if all((b.done() for b in beams)):
265  break
266  out = self.decoder(xs, hidden, enc_out)
267  scores = out[1]
268  scores = scores.view(bsz, beam_size, -1) # -1 is a vocab size
269  for i, b in enumerate(beams):
270  b.advance(F.log_softmax(scores[i, :], dim=-1))
271  xs = torch.cat(
272  [b.get_output_from_current_step() for b in beams]
273  ).unsqueeze(-1)
274  permute_hidden_idx = torch.cat(
275  [
276  beam_size * i + b.get_backtrack_from_current_step()
277  for i, b in enumerate(beams)
278  ]
279  )
280  new_hidden = out[2]
281  if isinstance(hidden, tuple):
282  for i in range(len(hidden)):
283  hidden[i].data.copy_(
284  new_hidden[i].data.index_select(
285  dim=1, index=permute_hidden_idx
286  )
287  )
288  else: # GRU
289  hidden.data.copy_(
290  new_hidden.data.index_select(
291  dim=1, index=permute_hidden_idx
292  )
293  )
294 
295  for b in beams:
296  b.check_finished()
297  beam_pred = [
298  b.get_pretty_hypothesis(b.get_top_hyp()[0])[1:] for b in beams
299  ]
300  # these beam scores are rescored with length penalty!
301  beam_scores = torch.stack([b.get_top_hyp()[1] for b in beams])
302  pad_length = max([t.size(0) for t in beam_pred])
303  beam_pred = torch.stack(
304  [pad(t, length=pad_length, dim=0) for t in beam_pred], dim=0
305  )
306 
307  # prepare n best list for each beam
308  n_best_beam_tails = [
309  b.get_rescored_finished(n_best=len(b.finished)) for b in beams
310  ]
311  nbest_beam_scores = []
312  nbest_beam_preds = []
313  for i, beamtails in enumerate(n_best_beam_tails):
314  perbeam_preds = []
315  perbeam_scores = []
316  for tail in beamtails:
317  perbeam_preds.append(
318  beams[i].get_pretty_hypothesis(
319  beams[i].get_hyp_from_finished(tail)
320  )
321  )
322  perbeam_scores.append(tail.score)
323  nbest_beam_scores.append(perbeam_scores)
324  nbest_beam_preds.append(perbeam_preds)
325 
326  if self.beam_log_freq > 0.0:
327  num_dump = round(bsz * self.beam_log_freq)
328  for i in range(num_dump):
329  dot_graph = beams[i].get_beam_dot(dictionary=self.dict)
330  dot_graph.write_png(
331  os.path.join(
332  self.beam_dump_path,
333  "{}.png".format(self.beam_dump_filecnt),
334  )
335  )
336  self.beam_dump_filecnt += 1
337 
338  predictions = beam_pred
339  scores = beam_scores
340 
341  if isinstance(predictions, list):
342  predictions = torch.cat(predictions, 1)
343  if isinstance(scores, list):
344  scores = torch.cat(scores, 1)
345 
346  return (
347  predictions,
348  scores,
349  cand_preds,
350  cand_scores,
351  encoder_states,
352  nbest_beam_preds,
353  nbest_beam_scores,
354  )

357 class Encoder(nn.Module):
358  def __init__(
359  self,
360  num_features,
361  padding_idx=0,
362  rnn_class='lstm',
363  emb_size=128,
364  hidden_size=128,
365  num_layers=2,
366  dropout=0.1,
367  bidirectional=False,
368  shared_lt=None,
369  shared_rnn=None,
370  sparse=False,
371  ):
372  super().__init__()
373 
374  self.dropout = nn.Dropout(p=dropout)
375  self.layers = num_layers
376  self.dirs = 2 if bidirectional else 1
377  self.hsz = hidden_size
378 
379  if shared_lt is None:
380  self.lt = nn.Embedding(
381  num_features, emb_size, padding_idx=padding_idx, sparse=sparse
382  )
383  else:
384  self.lt = shared_lt
385 
386  if shared_rnn is None:
387  self.rnn = rnn_class(
388  emb_size,
389  hidden_size,
390  num_layers,
391  dropout=dropout,
392  batch_first=True,
393  bidirectional=bidirectional,
394  )
395  elif bidirectional:
396  raise RuntimeError('Cannot share decoder with bidir encoder.')
397  else:
398  self.rnn = shared_rnn
399 
400  def forward(self, xs):
401  bsz = len(xs)
402 
403  # embed input tokens
404  xes = self.dropout(self.lt(xs))
405  try:
406  x_lens = [x for x in torch.sum((xs > 0).int(), dim=1).data]
407  xes = pack_padded_sequence(xes, x_lens, batch_first=True)
408  packed = True
409  except ValueError:
410  # packing failed, don't pack then
411  packed = False
412 
413  encoder_output, hidden = self.rnn(xes)
414  if packed:
415  encoder_output, _ = pad_packed_sequence(encoder_output, batch_first=True)
416  if self.dirs > 1:
417  # take elementwise max between forward and backward hidden states
418  # NOTE: currently using max, but maybe should use Linear
419  if isinstance(self.rnn, nn.LSTM):
420  hidden = (
421  hidden[0].view(-1, self.dirs, bsz, self.hsz).max(1)[0],
422  hidden[1].view(-1, self.dirs, bsz, self.hsz).max(1)[0],
423  )
424  else:
425  hidden = hidden.view(-1, self.dirs, bsz, self.hsz).max(1)[0]
426 
427  return encoder_output, hidden

430 class Decoder(nn.Module):
431  def __init__(
432  self,
433  num_features,
434  padding_idx=0,
435  rnn_class='lstm',
436  emb_size=128,
437  hidden_size=128,
438  num_layers=2,
439  dropout=0.1,
440  bidir_input=False,
441  share_output=True,
442  attn_type='none',
443  attn_length=-1,
444  attn_time='pre',
445  sparse=False,
446  numsoftmax=1,
447  softmax_layer_bias=False,
448  ):
449  super().__init__()
450 
451  if padding_idx != 0:
452  raise RuntimeError(
453  'This module\'s output layer needs to be fixed '
454  'if you want a padding_idx other than zero.'
455  )
456 
457  self.dropout = nn.Dropout(p=dropout)
458  self.layers = num_layers
459  self.hsz = hidden_size
460  self.esz = emb_size
461 
462  self.lt = nn.Embedding(
463  num_features, emb_size, padding_idx=padding_idx, sparse=sparse
464  )
465  self.rnn = rnn_class(
466  emb_size, hidden_size, num_layers, dropout=dropout, batch_first=True
467  )
468 
469  # rnn output to embedding
470  if hidden_size != emb_size and numsoftmax == 1:
471  # self.o2e = RandomProjection(hidden_size, emb_size)
472  # other option here is to learn these weights
473  self.o2e = nn.Linear(hidden_size, emb_size, bias=False)
474  else:
475  # no need for any transformation here
476  self.o2e = lambda x: x
477  # embedding to scores, use custom linear to possibly share weights
478  shared_weight = self.lt.weight if share_output else None
479  self.e2s = Linear(
480  emb_size, num_features, bias=softmax_layer_bias, shared_weight=shared_weight
481  )
482  self.shared = shared_weight is not None
483 
484  self.attn_type = attn_type
485  self.attn_time = attn_time
486  self.attention = AttentionLayer(
487  attn_type=attn_type,
488  hidden_size=hidden_size,
489  emb_size=emb_size,
490  bidirectional=bidir_input,
491  attn_length=attn_length,
492  attn_time=attn_time,
493  )
494 
495  self.numsoftmax = numsoftmax
496  if numsoftmax > 1:
497  self.softmax = nn.Softmax(dim=1)
498  self.prior = nn.Linear(hidden_size, numsoftmax, bias=False)
499  self.latent = nn.Linear(hidden_size, numsoftmax * emb_size)
500  self.activation = nn.Tanh()
501 
502  def forward(self, xs, hidden, encoder_output, attn_mask=None, topk=1):
503  xes = self.dropout(self.lt(xs))
504  if self.attn_time == 'pre':
505  xes = self.attention(xes, hidden, encoder_output, attn_mask)
506  if xes.dim() == 2:
507  # if only one token inputted, sometimes needs unsquezing
508  xes.unsqueeze_(1)
509  output, new_hidden = self.rnn(xes, hidden)
510  if self.attn_time == 'post':
511  output = self.attention(output, new_hidden, encoder_output, attn_mask)
512 
513  if self.numsoftmax > 1:
514  bsz = xs.size(0)
515  seqlen = xs.size(1) if xs.dim() > 1 else 1
516  latent = self.latent(output)
517  active = self.dropout(self.activation(latent))
518  logit = self.e2s(active.view(-1, self.esz))
519 
520  prior_logit = self.prior(output).view(-1, self.numsoftmax)
521  prior = self.softmax(prior_logit) # softmax over numsoftmax's
522 
523  prob = self.softmax(logit).view(bsz * seqlen, self.numsoftmax, -1)
524  probs = (prob * prior.unsqueeze(2)).sum(1).view(bsz, seqlen, -1)
525  scores = probs.log()
526  else:
527  e = self.dropout(self.o2e(output))
528  scores = self.e2s(e)
529 
530  # select top scoring index, excluding the padding symbol (at idx zero)
531  # we can do topk sampling from renoramlized softmax here, default topk=1 is greedy
532  if topk == 1:
533  _max_score, idx = scores.narrow(2, 1, scores.size(2) - 1).max(2)
534  elif topk > 1:
535  max_score, idx = torch.topk(
536  F.softmax(scores.narrow(2, 1, scores.size(2) - 1), 2),
537  topk,
538  dim=2,
539  sorted=False,
540  )
541  probs = F.softmax(
542  scores.narrow(2, 1, scores.size(2) - 1).gather(2, idx), 2
543  ).squeeze(1)
544  dist = torch.distributions.categorical.Categorical(probs)
545  samples = dist.sample()
546  idx = idx.gather(-1, samples.unsqueeze(1).unsqueeze(-1)).squeeze(-1)
547  preds = idx.add_(1)
548 
549  return preds, scores, new_hidden

552 class Ranker(object):
553  def __init__(self, decoder, padding_idx=0, attn_type='none'):
554  super().__init__()
555  self.decoder = decoder
556  self.NULL_IDX = padding_idx
557  self.attn_type = attn_type
558 
559  def forward(self, cands, cand_inds, decode_params):
560  start, hidden, enc_out, attn_mask = decode_params
561 
562  hid, cell = (hidden, None) if isinstance(hidden, torch.Tensor) else hidden
563  if len(cand_inds) != hid.size(1):
564  cand_indices = start.detach().new(cand_inds)
565  hid = hid.index_select(1, cand_indices)
566  if cell is None:
567  hidden = hid
568  else:
569  cell = cell.index_select(1, cand_indices)
570  hidden = (hid, cell)
571  enc_out = enc_out.index_select(0, cand_indices)
572  attn_mask = attn_mask.index_select(0, cand_indices)
573 
574  cand_scores = []
575 
576  for i in range(len(cands)):
577  curr_cs = cands[i]
578 
579  n_cs = curr_cs.size(0)
580  starts = start.expand(n_cs).unsqueeze(1)
581  scores = 0
582  seqlens = 0
583  # select just the one hidden state
584  if isinstance(hidden, torch.Tensor):
585  nl = hidden.size(0)
586  hsz = hidden.size(-1)
587  cur_hid = hidden.select(1, i).unsqueeze(1).expand(nl, n_cs, hsz)
588  else:
589  nl = hidden[0].size(0)
590  hsz = hidden[0].size(-1)
591  cur_hid = (
592  hidden[0]
593  .select(1, i)
594  .unsqueeze(1)
595  .expand(nl, n_cs, hsz)
596  .contiguous(),
597  hidden[1]
598  .select(1, i)
599  .unsqueeze(1)
600  .expand(nl, n_cs, hsz)
601  .contiguous(),
602  )
603 
604  cur_enc, cur_mask = None, None
605  if attn_mask is not None:
606  cur_mask = attn_mask[i].unsqueeze(0).expand(n_cs, attn_mask.size(-1))
607  cur_enc = enc_out[i].unsqueeze(0).expand(n_cs, enc_out.size(1), hsz)
608  # this is pretty much copied from the training forward above
609  if curr_cs.size(1) > 1:
610  c_in = curr_cs.narrow(1, 0, curr_cs.size(1) - 1)
611  xs = torch.cat([starts, c_in], 1)
612  else:
613  xs, c_in = starts, curr_cs
614  if self.attn_type == 'none':
615  preds, score, cur_hid = self.decoder(xs, cur_hid, cur_enc, cur_mask)
616  true_score = F.log_softmax(score, dim=2).gather(2, curr_cs.unsqueeze(2))
617  nonzero = curr_cs.ne(0).float()
618  scores = (true_score.squeeze(2) * nonzero).sum(1)
619  seqlens = nonzero.sum(1)
620  else:
621  for i in range(curr_cs.size(1)):
622  xi = xs.select(1, i)
623  ci = curr_cs.select(1, i)
624  preds, score, cur_hid = self.decoder(xi, cur_hid, cur_enc, cur_mask)
625  true_score = F.log_softmax(score, dim=2).gather(
626  2, ci.unsqueeze(1).unsqueeze(2)
627  )
628  nonzero = ci.ne(0).float()
629  scores += true_score.squeeze(2).squeeze(1) * nonzero
630  seqlens += nonzero
631 
632  scores /= seqlens # **len_penalty?
633  cand_scores.append(scores)
634

662

672  if not self.shared:

674  self.weight.data.uniform_(-stdv, stdv)
675  if self.bias is not None:

680  if self.shared:
681  # detach weight to prevent gradients from changing weight
682  # (but need to detach every time so weights are up to date)
683  weight = weight.detach()

719 class AttentionLayer(nn.Module):
720  def __init__(
721  self,
722  attn_type,
723  hidden_size,
724  emb_size,
725  bidirectional=False,
726  attn_length=-1,
727  attn_time='pre',
728  ):
729  super().__init__()
730  self.attention = attn_type
731 
732  if self.attention != 'none':
733  hsz = hidden_size
734  hszXdirs = hsz * (2 if bidirectional else 1)
735  if attn_time == 'pre':
736  # attention happens on the input embeddings
737  input_dim = emb_size
738  elif attn_time == 'post':
739  # attention happens on the output of the rnn
740  input_dim = hsz
741  else:
742  raise RuntimeError('unsupported attention time')
743  self.attn_combine = nn.Linear(hszXdirs + input_dim, input_dim, bias=False)
744 
745  if self.attention == 'local':
746  # local attention over fixed set of output states
747  if attn_length < 0:
748  raise RuntimeError('Set attention length to > 0.')
749  self.max_length = attn_length
750  # combines input and previous hidden output layer
751  self.attn = nn.Linear(hsz + input_dim, attn_length, bias=False)
752  # combines attention weights with encoder outputs
753  elif self.attention == 'concat':
754  self.attn = nn.Linear(hsz + hszXdirs, hsz, bias=False)
755  self.attn_v = nn.Linear(hsz, 1, bias=False)
756  elif self.attention == 'general':
757  # equivalent to dot if attn is identity
758  self.attn = nn.Linear(hsz, hszXdirs, bias=False)
759 
760  def forward(self, xes, hidden, enc_out, attn_mask=None):
761  if self.attention == 'none':
762  return xes
763 
764  if type(hidden) == tuple:
765  # for lstms use the "hidden" state not the cell state
766  hidden = hidden[0]
767  last_hidden = hidden[-1] # select hidden state from last RNN layer
768 
769  if self.attention == 'local':
770  if enc_out.size(1) > self.max_length:
771  offset = enc_out.size(1) - self.max_length
772  enc_out = enc_out.narrow(1, offset, self.max_length)
773  h_merged = torch.cat((xes.squeeze(1), last_hidden), 1)
774  attn_weights = F.softmax(self.attn(h_merged), dim=1)
775  if attn_weights.size(1) > enc_out.size(1):
776  attn_weights = attn_weights.narrow(1, 0, enc_out.size(1))
777  else:
778  hid = last_hidden.unsqueeze(1)
779  if self.attention == 'concat':
780  hid = hid.expand(
781  last_hidden.size(0), enc_out.size(1), last_hidden.size(1)
782  )
783  h_merged = torch.cat((enc_out, hid), 2)
784  active = F.tanh(self.attn(h_merged))
785  attn_w_premask = self.attn_v(active).squeeze(2)
786  elif self.attention == 'dot':
787  if hid.size(2) != enc_out.size(2):
788  # enc_out has two directions, so double hid
789  hid = torch.cat([hid, hid], 2)
790  attn_w_premask = torch.bmm(hid, enc_out.transpose(1, 2)).squeeze(1)
791  elif self.attention == 'general':
792  hid = self.attn(hid)
793  attn_w_premask = torch.bmm(hid, enc_out.transpose(1, 2)).squeeze(1)
794  # calculate activation scores
795  if attn_mask is not None:
796  # remove activation from NULL symbols
797  attn_w_premask -= (1 - attn_mask) * 1e20
798  attn_weights = F.softmax(attn_w_premask, dim=1)
799 
800  attn_applied = torch.bmm(attn_weights.unsqueeze(1), enc_out)
801  merged = torch.cat((xes.squeeze(1), attn_applied.squeeze(1)), 1)
802  output = F.tanh(self.attn_combine(merged).unsqueeze(1))
803  return output
