97  """Adds a optional transformer layer and a linear layer on top of BERT."""

99  def __init__(
100  self,
101  bert_model,
102  output_dim,
103  add_transformer_layer=False,
104  layer_pulled=-1,
105  aggregation="first",
106  ):
107  super(BertWrapper, self).__init__()
108  self.layer_pulled = layer_pulled
109  self.aggregation = aggregation
110  self.add_transformer_layer = add_transformer_layer
111  # deduce bert output dim from the size of embeddings
112  bert_output_dim = bert_model.embeddings.word_embeddings.weight.size(1)
113 
114  if add_transformer_layer:
115  config_for_one_layer = BertConfig(
116  0,
117  hidden_size=bert_output_dim,
118  num_attention_heads=int(bert_output_dim / 64),
119  intermediate_size=3072,
120  hidden_act='gelu',
121  )
122  self.additional_transformer_layer = BertLayer(config_for_one_layer)
123  self.additional_linear_layer = torch.nn.Linear(bert_output_dim, output_dim)
124  self.bert_model = bert_model

126  def forward(self, token_ids, segment_ids, attention_mask):
127  """Forward pass."""
128  output_bert, output_pooler = self.bert_model(
129  token_ids, segment_ids, attention_mask
130  )
131  # output_bert is a list of 12 (for bert base) layers.
132  layer_of_interest = output_bert[self.layer_pulled]
133  dtype = next(self.parameters()).dtype
134  if self.add_transformer_layer:
135  # Follow up by yet another transformer layer
136  extended_attention_mask = attention_mask.unsqueeze(1).unsqueeze(2)
137  extended_attention_mask = (~extended_attention_mask).to(dtype) * neginf(
138  dtype
139  )
140  embedding_layer = self.additional_transformer_layer(
141  layer_of_interest, extended_attention_mask
142  )
143  else:
144  embedding_layer = layer_of_interest
145 
146  if self.aggregation == "mean":
147  # consider the average of all the output except CLS.
148  # obviously ignores masked elements
149  outputs_of_interest = embedding_layer[:, 1:, :]
150  mask = attention_mask[:, 1:].type_as(embedding_layer).unsqueeze(2)
151  sumed_embeddings = torch.sum(outputs_of_interest * mask, dim=1)
152  nb_elems = torch.sum(attention_mask[:, 1:].type(dtype), dim=1).unsqueeze(1)
153  embeddings = sumed_embeddings / nb_elems
154  elif self.aggregation == "max":
155  # consider the max of all the output except CLS
156  outputs_of_interest = embedding_layer[:, 1:, :]
157  mask = (~attention_mask[:, 1:]).type(dtype).unsqueeze(2) * neginf(dtype)
158  embeddings, _ = torch.max(outputs_of_interest + mask, dim=1)
159  else:
160  # easiest, we consider the output of "CLS" as the embedding
161  embeddings = embedding_layer[:, 0, :]
162 
163  # We need this in case of dimensionality reduction
164  result = self.additional_linear_layer(embeddings)
165 
166  # Sort of hack to make it work with distributed: this way the pooler layer
167  # is used for grad computation, even though it does not change anything...
168  # in practice, it just adds a very (768*768) x (768*batchsize) matmul
169  result += 0 * torch.sum(output_pooler)
170  return result

173 def surround(idx_vector, start_idx, end_idx):

281  def step(self, closure=None):
282  """
283  Perform a single optimization step.
284 
285  :param closure:
286  A closure that reevaluates the model and returns the loss.
287  """
288  loss = None
289  if closure is not None:
290  loss = closure()
291 
292  for group in self.param_groups:
293  for p in group['params']:
294  if p.grad is None:
295  continue
296  grad = p.grad.data
297  if grad.is_sparse:
298  raise RuntimeError(
299  'Adam does not support sparse gradients, please '
300  'consider SparseAdam instead'
301  )
302 
303  state = self.state[p]
304 
305  # State initialization
306  if len(state) == 0:
307  # Exponential moving average of gradient values
308  state['next_m'] = torch.zeros_like(p.data)
309  # Exponential moving average of squared gradient values
310  state['next_v'] = torch.zeros_like(p.data)
311 
312  next_m, next_v = state['next_m'], state['next_v']
313  beta1, beta2 = group['b1'], group['b2']
314 
315  # Add grad clipping
316  if group['max_grad_norm'] > 0:
317  clip_grad_norm_(p, group['max_grad_norm'])
318 
319  # Decay the first and second moment running average coefficient
320  # In-place operations to update the averages at the same time
321  next_m.mul_(beta1).add_(1 - beta1, grad)
322  next_v.mul_(beta2).addcmul_(1 - beta2, grad, grad)
323  update = next_m / (next_v.sqrt() + group['e'])
324 
325  # Just adding the square of the weights to the loss function is *not*
326  # the correct way of using L2 regularization/weight decay with Adam,
327  # since that will interact with the m and v parameters in strange ways.
328  #
329  # Instead we want to decay the weights in a manner that doesn't interact
330  # with the m/v parameters. This is equivalent to adding the square
331  # of the weights to the loss with plain (non-momentum) SGD.
332  if group['weight_decay'] > 0.0:
333  update += group['weight_decay'] * p.data
334  lr = group['lr']
335 
336  update_with_lr = lr * update
337  p.data.add_(-update_with_lr)
338  return loss
