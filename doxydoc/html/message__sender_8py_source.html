60 def create_text_message(text, quick_replies=None):
61  """Return a list of text messages from the given text. If the message is
62  too long it is split into multiple messages.
63  quick_replies should be a list of options made with create_reply_option.
64  """
65 
66  def _message(text_content, replies):
67  payload = {'text': text_content[:MAX_TEXT_CHARS]}
68  if replies:
69  payload['quick_replies'] = replies
70  return payload
71 
72  tokens = [s[:MAX_TEXT_CHARS] for s in text.split(' ')]
73  splits = []
74  cutoff = 0
75  curr_length = 0
76  if quick_replies:
77  assert (
78  len(quick_replies) <= MAX_QUICK_REPLIES
79  ), 'Number of quick replies {} greater than the max of {}'.format(
80  len(quick_replies), MAX_QUICK_REPLIES
81  )
82  for i in range(len(tokens)):
83  if tokens[i] == '[*SPLIT*]':
84  if ' '.join(tokens[cutoff : i - 1]).strip() != '':
85  splits.append(_message(' '.join(tokens[cutoff:i]), None))
86  cutoff = i + 1
87  curr_length = 0
88  if curr_length + len(tokens[i]) > MAX_TEXT_CHARS:
89  splits.append(_message(' '.join(tokens[cutoff:i]), None))
90  cutoff = i
91  curr_length = 0
92  curr_length += len(tokens[i]) + 1
93  if cutoff < len(tokens):
94  splits.append(_message(' '.join(tokens[cutoff:]), quick_replies))
95  return splits

168  def send_fb_payload(

170  ):
171  """Sends a payload to messenger, processes it if we can"""
172  api_address = 'https://graph.facebook.com/v2.6/me/messages'
173  if payload['type'] == 'list':
174  data = create_compact_list_message(payload['data'])
175  elif payload['type'] in ['image', 'video', 'file', 'audio']:
176  data = create_attachment(payload)
177  else:
178  data = payload['data']
179 
180  message = {
181  "messaging_type": 'RESPONSE',
182  "recipient": {"id": receiver_id},
183  "message": {"attachment": data},
184  }
185  if quick_replies is not None:

187  message['message']['quick_replies'] = quick_replies
188  if persona_id is not None:
189  payload['persona_id'] = persona_id

191  result = response.json()
192  if 'error' in result:
193  if result['error']['code'] == 1200:
194  # temporary error please retry
195  response = requests.post(
