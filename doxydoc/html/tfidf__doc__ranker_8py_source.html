22 class TfidfDocRanker(object):
23  """Loads a pre-weighted inverted index of token/document terms.
24  Scores new queries by taking sparse dot products.
25  """

27  def __init__(self, tfidf_path=None, strict=True):
28  """
29  Args:
30  tfidf_path: path to saved model file
31  strict: fail on empty queries or continue (and return empty result)
32  """
33  # Load from disk
34  logger.info('Loading %s' % tfidf_path)
35  matrix, metadata = utils.load_sparse_csr(tfidf_path)
36  self.doc_mat = matrix
37  self.ngrams = metadata['ngram']
38  self.hash_size = metadata['hash_size']
39  self.tokenizer = tokenizers.get_class(metadata['tokenizer'])()
40  self.doc_freqs = metadata['doc_freqs'].squeeze()
41  self.doc_dict = metadata.get('doc_dict', None)
42  self.num_docs = self.doc_mat.shape[1] - 1
43  self.strict = strict

45  def get_doc_index(self, doc_id):
46  """Convert doc_id --> doc_index"""
47  return self.doc_dict[0][doc_id] if self.doc_dict else doc_id
48 
49  def get_doc_id(self, doc_index):
50  """Convert doc_index --> doc_id"""
51  return self.doc_dict[1][doc_index] if self.doc_dict else doc_index

53  def closest_docs(self, query, k=1, matrix=None):
54  """Closest docs by dot product between query and documents
55  in tfidf weighted word vector space.
56 
57  matrix arg can be provided to be used instead of internal doc matrix.
58  """
59  spvec = self.text2spvec(query)
60  res = spvec * matrix if matrix is not None else spvec * self.doc_mat
61 
62  if len(res.data) <= k:
63  o_sort = np.argsort(-res.data)
64  else:
65  o = np.argpartition(-res.data, k)[0:k]
66  o_sort = o[np.argsort(-res.data[o])]
67 
68  doc_scores = res.data[o_sort]
69  doc_ids = res.indices[o_sort]
70  return doc_ids, doc_scores

72  def batch_closest_docs(self, queries, k=1, num_workers=None):
73  """Process a batch of closest_docs requests multithreaded.
74  Note: we can use plain threads here as scipy is outside of the GIL.
75  """
76  with ThreadPool(num_workers) as threads:
77  closest_docs = partial(self.closest_docs, k=k)
78  results = threads.map(closest_docs, queries)
79  return results

81  def parse(self, query):
82  """Parse the query into tokens (either ngrams or tokens)."""
83  tokens = self.tokenizer.tokenize(query)
84  return tokens.ngrams(n=self.ngrams, uncased=True, filter_fn=utils.filter_ngram)

86  def text2spvec(self, query):
87  """Create a sparse tfidf-weighted word vector from query.
88 
89  tfidf = log(tf + 1) * log((N - Nt + 0.5) / (Nt + 0.5))
90  """
91  # Get hashed ngrams
92  words = self.parse(utils.normalize(query))
93  wids = [utils.hash(w, self.hash_size) for w in words]
94 
95  if len(wids) == 0:
96  if self.strict:
97  raise RuntimeError('No valid word in: %s' % query)
98  else:
99  logger.warning('No valid word in: %s' % query)
100  return sp.csr_matrix((1, self.hash_size))
101 
102  # Count TF
103  wids_unique, wids_counts = np.unique(wids, return_counts=True)
104  tfs = np.log1p(wids_counts)
105 
106  # Count IDF
107  Ns = self.doc_freqs[wids_unique]
108  idfs = np.log((self.num_docs - Ns + 0.5) / (Ns + 0.5))
109  idfs[idfs < 0] = 0
110 
111  # TF-IDF
112  data = np.multiply(tfs, idfs)
113 
114  # One row, sparse csr matrix
115  indptr = np.array([0, len(wids_unique)])
116  spvec = sp.csr_matrix((data, wids_unique, indptr), shape=(1, self.hash_size))
117 
118  return spvec
