54  """Extracts training data for the negative response classifier (NRC) from Mturk logs
55 
56  input: file of logs (in ParlaiDialog format) from Mturk task 1 with turn-by-turn
57  quality ratings 1-5
58  output: file of episodes (self-feeding format) w/ +1/-1 ratings indicating
59  positive/negative example
60  """
61  examples = []
62  positives = opt['positives'].split(',')
63  negatives = opt['negatives'].split(',')
64  assert len(set(positives).intersection(set(negatives))) == 0
65 
66  num_episodes = 0
67  num_parleys = 0
68  for episode in extract_parlai_episodes(opt['infile']):
69  num_episodes += 1
70  history = []
71  for parley in episode:
72  num_parleys += 1
73 
74  # Update history (not including stock control flow responses)
75  if parley.context.startswith(INITIAL_PROMPT):
76  # Conversation prompt, first utterance
77  # Begin history
78  history = [parley.response]
79  elif parley.context.startswith(EXP_REQUEST):
80  # Asked for y_exp, got y_exp
81  # Messed up, so blast history
82  history = []
83  elif parley.context.startswith(NEWTOPIC):
84  # Asked for new topic, got a first utterance
85  # Begin new history
86  history = [parley.response]
87  elif parley.context.startswith(RAT_REQUEST):
88  # Concatenate history and add speaker tokens as necessary
89  # history_size refers to the total number of utterances
90  # (history_size == 0 means predict sentiment from '__null__')
91  # response that's being classified (so if history_size == 0 then
92  # classify based only on the response w/o any extra context).
93  # Note that the response being classified should always be preceded by
94  # __p1__ (the human), not __p2__ (the bot).
95  if opt['history_size'] < 0:
96  utterances = history
97  elif opt['history_size'] == 0:
98  utterances = ['__null__']
99  else:
100  utterances = history[-opt['history_size'] :]
101  context = add_person_tokens(utterances, last_speaker=1)
102 
103  if parley.response in positives:
104  label = 1
105  elif parley.response in negatives:
106  label = -1
107  else:
108  label = 0
109 
110  if label:
111  example = Parley(context, label)
112  examples.append(example)
113 
114  elif CONTINUE in parley.context:
115  # if response was negative, history will get blasted in EXP_REQUEST
116  # if we're here, response was neutral/positive, so continue the history
117  history.append(parley.context[parley.context.rindex(':') + 1 :])
118  history.append(parley.response)
119  else:
120  history.append(parley.context)
121  history.append(parley.response)
