98 def load_cands(path):
99  """Load global fixed set of candidate labels that the teacher provides
100  every example (the true labels for a specific example are also added to
101  this set, so that it's possible to get the right answer).
102  """
103  if path is None:
104  return None
105  cands = []
106  lines_have_ids = False
107  cands_are_replies = False
108  cnt = 0
109  with open(path) as read:
110  for line in read:
111  line = line.strip().replace('\\n', '\n')
112  if len(line) > 0:
113  cnt = cnt + 1
114  # If lines are numbered we strip them of numbers.
115  if cnt == 1 and line[0:2] == '1 ':
116  lines_have_ids = True
117  # If tabs then the label_candidates are all the replies.
118  if '\t' in line and not cands_are_replies:
119  cands_are_replies = True
120  cands = []
121  if lines_have_ids:
122  space_idx = line.find(' ')
123  line = line[space_idx + 1 :]
124  if cands_are_replies:
125  sp = line.split('\t')
126  if len(sp) > 1 and sp[1] != '':
127  cands.append(sp[1])
128  else:
129  cands.append(line)
130  else:
131  cands.append(line)
132  return cands

598  def predict(self, xs, ys=None, cands=None, cands_txt=None, obs=None):

600  Update the model using the targets if available, otherwise rank
601  candidates as well if they are available and param is set.
602  """
603  self.start = time.time()
604  if xs is None:
605  return [{}]
606  is_training = ys is not None
607  if is_training: #

609  if len(negs) > 0:
610  self.model.train()
611  self.zero_grad()
612  if self.opt['loss'] == 'cosine':
613  xe, ye = self.model(xs, obs[0]['mem'], ys, negs)
614  y = Variable(-torch.ones(xe.size(0)))
615  y[0] = 1
616  loss = self.criterion(xe, ye, y)
617  else:
618  x = self.model(xs, obs[0]['mem'], ys, negs)
619  y = Variable(torch.LongTensor([0]))

621  loss.backward()
622  self.update_params()

629  else:

633  )

780  def batchify(self, observations):
781  """Convert a list of observations into input & target tensors."""
782 
783  def valid(obs):
784  # check if this is an example our model should actually process
785  return 'query' in obs and len(obs['query']) > 0
786 
787  try:
788  # valid examples and their indices
789  valid_inds, exs = zip(
790  *[(i, ex) for i, ex in enumerate(observations) if valid(ex)]
791  )
792  except ValueError:
793  # zero examples to process in this batch, so zip failed to unpack
794  return None, None, None, None
795 
796  # `x` text is already tokenized and truncated
797  # sort by length so we can use pack_padded
798  parsed_x = [ex['query'] for ex in exs]
799  x_lens = [len(x) for x in parsed_x]
800  ind_sorted = sorted(range(len(x_lens)), key=lambda k: -x_lens[k])
801 
802  exs = [exs[k] for k in ind_sorted]
803  valid_inds = [valid_inds[k] for k in ind_sorted]
804  parsed_x = [parsed_x[k] for k in ind_sorted]
805 
806  labels_avail = any(['labels' in ex for ex in exs])
807 
808  max_x_len = max([len(x) for x in parsed_x])
809  for x in parsed_x:
810  x += [self.NULL_IDX] * (max_x_len - len(x))
811  xs = torch.LongTensor(parsed_x)
812  xs = Variable(xs)
813 
814  # set up the target tensors
815  ys = None
816  labels = None
817  if labels_avail:
818  # randomly select one of the labels to update on, if multiple
819  labels = [random.choice(ex.get('labels', [''])) for ex in exs]
820  # parse each label and append END
821  parsed_y = [deque(maxlen=self.truncate) for _ in labels]
822  for dq, y in zip(parsed_y, labels):
823  dq.extendleft(reversed(self.parse(y)))

825  for y in parsed_y:
826  y += [self.NULL_IDX] * (max_y_len - len(y))
827  if len(parsed_y[0]) == 0:
828  return None, None, None, None
829  else:

832

839  cs = []
