170 def parse_configuration_file(config_path):
171  """Read the config file for an experiment to get ParlAI settings.
172 
173  :param config_path:
174  path to config
175 
176  :return:
177  parsed configuration dictionary
178  """
179  result = {}
180  result["configs"] = {}
181  with open(config_path) as f:
182  cfg = yaml.load(f.read())
183  # get world path
184  result["world_path"] = cfg.get("world_module")
185  if not result["world_path"]:
186  raise ValueError("Did not specify world module")
187  result["overworld"] = cfg.get("overworld")
188  if not result["overworld"]:
189  raise ValueError("Did not specify overworld")
190  result["page_id"] = cfg.get("page_id")
191  if not result["page_id"]:
192  raise ValueError("Did not specify page_id")
193  result["max_workers"] = cfg.get("max_workers")
194  if not result["max_workers"]:
195  raise ValueError("Did not specify max_workers")
196  result["task_name"] = cfg.get("task_name")
197  if not result["task_name"]:
198  raise ValueError("Did not specify task name")
199  task_world = cfg.get("tasks")
200  if task_world is None or len(task_world) == 0:
201  raise ValueError("task not in config file")
202  # get task file
203  for task_name, configuration in task_world.items():
204  if "task_world" not in configuration:
205  raise ValueError("{} does not specify a task".format(task_name))
206  result["configs"][task_name] = WorldConfig(
207  world_name=task_name,
208  onboarding_name=configuration.get("onboard_world"),
209  task_name=configuration.get("task_world"),
210  max_time_in_pool=configuration.get("timeout") or 300,
211  agents_required=configuration.get("agents_required") or 1,
212  backup_task=configuration.get("backup_task"),
213  )
214  # get world options
215  result["world_opt"] = cfg.get("opt", {})
216 
217  return result
