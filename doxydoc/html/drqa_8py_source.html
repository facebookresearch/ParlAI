210  """Update or predict on a batch of examples.
211  More efficient than act().
212  """
213  batchsize = len(observations)
214  batch_reply = [{'id': self.getID()} for _ in range(batchsize)]

217  examples = [self._build_ex(obs) for obs in observations]
218  valid_inds = [i for i in range(batchsize) if examples[i] is not None]
219  examples = [ex for ex in examples if ex is not None]
220 
221  # If all examples are invalid, return an empty batch.
222  if len(examples) == 0:
223  return batch_reply
224 
225  # Else, use what we have (hopefully everything).
226  batch = batchify(
227  examples,
228  null=self.word_dict[self.word_dict.null_token],
229  cuda=self.opt['cuda'],
230  )
231 
232  # Either train or predict

236  self.model.update(batch)
237  except RuntimeError as e:

252  for i in range(len(predictions)):

255  batch_reply[valid_inds[i]]['candidate_scores'] = [scores[i]]

277  def _build_ex(self, ex):
278  """Find the token span of the answer in the context for this example.
279  If a token span cannot be found, return None. Otherwise, torchify.
280  """
281  # Check if empty input (end of epoch)
282  if 'text' not in ex:
283  return
284 
285  # Split out document + question
286  inputs = {}
287  fields = ex['text'].strip().split('\n')
288 
289  # Data is expected to be text + '\n' + question
290  if len(fields) < 2:
291  raise RuntimeError('Invalid input. Is task a QA task?')
292 
293  paragraphs, question = fields[:-1], fields[-1]
294 
295  if len(fields) > 2 and self.opt.get('subsample_docs', 0) > 0 and 'labels' in ex:
296  paragraphs = self._subsample_doc(
297  paragraphs, ex['labels'], self.opt.get('subsample_docs', 0)
298  )
299 
300  document = ' '.join(paragraphs)
301  inputs['document'], doc_spans = self.word_dict.span_tokenize(document)
302  inputs['question'] = self.word_dict.tokenize(question)
303  inputs['target'] = None
304 
305  # Find targets (if labels provided).
306  # Return if we were unable to find an answer.
307  if 'labels' in ex:
308  if 'answer_starts' in ex:
309  # randomly sort labels and keep the first match
310  labels_with_inds = list(zip(ex['labels'], ex['answer_starts']))
311  random.shuffle(labels_with_inds)
312  for ans, ch_idx in labels_with_inds:
313  # try to find an answer_start matching a tokenized answer
314  start_idx = bisect.bisect_left(
315  list(x[0] for x in doc_spans), ch_idx
316  )
317  end_idx = start_idx + len(self.word_dict.tokenize(ans)) - 1
318  if end_idx < len(doc_spans):
319  inputs['target'] = (start_idx, end_idx)
320  break
321  else:
322  inputs['target'] = self._find_target(inputs['document'], ex['labels'])
323  if inputs['target'] is None:
324  return
325 
326  # Vectorize.
327  inputs = vectorize(self.opt, inputs, self.word_dict, self.feature_dict)
328 
329  # Return inputs with original text + spans (keep for prediction)
330  return inputs + (document, doc_spans)

332  def _find_target(self, document, labels):
333  """Find the start/end token span for all labels in document.
334  Return a random one for training.
335  """
336 
337  def _positions(d, l):
338  for i in range(len(d)):
339  for j in range(i, min(len(d) - 1, i + len(l))):
340  if l == d[i : j + 1]:
341  yield (i, j)
342 
343  targets = []
344  for label in labels:
345  targets.extend(_positions(document, self.word_dict.tokenize(label)))
346  if len(targets) == 0:
347  return
348  return targets[np.random.choice(len(targets))]

350  def _subsample_doc(self, paras, labels, subsample):
351  """Subsample paragraphs from the document (mostly for training speed).
352  """
353  # first find a valid paragraph (with a label)
354  pi = -1
355  for ind, p in enumerate(paras):
356  for l in labels:
357  if p.find(l):
358  pi = ind
359  break
360  if pi == -1:
361  # failed
362  return paras[0:1]
363  new_paras = []
364  if pi > 0:
365  for _i in range(min(subsample, pi - 1)):
366  ind = random.randint(0, pi - 1)
367  new_paras.append(paras[ind])
368  new_paras.append(paras[pi])
369  if pi < len(paras) - 1:
370  for _i in range(min(subsample, len(paras) - 1 - pi)):
371  ind = random.randint(pi + 1, len(paras) - 1)
372  new_paras.append(paras[ind])
373  return new_paras
