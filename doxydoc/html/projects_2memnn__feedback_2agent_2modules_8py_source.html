168 class Embed(nn.Embedding):
169  def __init__(self, *args, position_encoding=False, **kwargs):
170  self.position_encoding = position_encoding
171  super().__init__(*args, **kwargs)
172 
173  def forward(self, lengths, indices):
174  lengths_mat = lengths.data
175  indices = indices.data
176  if lengths.dim() == 1 or lengths.size(1) == 1:
177  lengths_mat = lengths_mat.squeeze().unsqueeze(0)
178 
179  input = torch.LongTensor(
180  lengths_mat.size(0), lengths_mat.size(1), torch.max(lengths_mat)
181  )
182  pad = self.padding_idx if self.padding_idx is not None else 0
183  input.fill_(pad)
184  emb_list = []
185  offset = 0
186  for i, row in enumerate(lengths_mat):
187  for j, length in enumerate(row):
188  if length > 0:
189  input[i, j, :length] = indices[offset : offset + length]
190  offset += length
191  input = Variable(input)
192 
193  for i, row in enumerate(lengths_mat):
194  emb = super().forward(input[i, :, :])
195  if self.position_encoding:
196  emb = emb * Variable(self.position_tensor(row, emb))
197  emb = torch.sum(emb, dim=1).squeeze(1)
198  for j, length in enumerate(row):
199  if length > 0:
200  emb[j] /= length
201  emb_list.append(emb)
202  embs = torch.stack(emb_list)
203 
204  if lengths.dim() == 1:
205  embs = embs.squeeze(0)
206  elif lengths.size(1) == 1:
207  embs = embs.squeeze().unsqueeze(1)
208  return embs

210  @staticmethod
211  @lru_cache(maxsize=32)
212  def position_matrix(J, d):
213  m = torch.Tensor(J, d)
214  for k in range(1, d + 1):
215  for j in range(1, J + 1):
216  m[j - 1, k - 1] = (1 - j / J) - (k / d) * (1 - 2 * j / J)
217  return m

219  @staticmethod
220  def position_tensor(sentence_lengths, embeddings):
221  t = torch.zeros(embeddings.size())
222  embedding_dim = embeddings.size()[-1]
223  for i, length in enumerate(sentence_lengths):
224  if length > 0:
225  t[i, :length, :] = Embed.position_matrix(length, embedding_dim)
226  return t

235  def forward(
236  self,
237  query_embeddings,
238  in_memory_embeddings,
239  out_memory_embeddings,
240  attention_mask=None,
241  ):
242  attention = torch.bmm(
243  in_memory_embeddings, query_embeddings.unsqueeze(2)
244  ).squeeze(2)
245  if attention_mask is not None:
246  # exclude masked elements from the softmax
247  attention = (
248  attention_mask.float() * attention
249  + (1 - attention_mask.float()) * -1e20
250  )
251  probs = softmax(attention).unsqueeze(1)
252  memory_output = torch.bmm(probs, out_memory_embeddings).squeeze(1)
253  query_embeddings = self.linear(query_embeddings)
254  output = memory_output + query_embeddings
255  return output
