167  def extract(self, image, path=None):
168  # check whether initialize CNN network.
169  if not self.netCNN:

171  # extract the image feature

194  def load(self, path):
195  """Load from a given path."""
196  opt = self.opt
197  mode = opt.get('image_mode', 'raw')
198  is_zip = False
199  if mode is None or mode == 'none':
200  # don't need to load images
201  return None
202  elif '.zip' in path:
203  # assume format path/to/file.zip/image_name.jpg
204  is_zip = True
205  sep = path.index('.zip') + 4
206  zipname = path[:sep]
207  file_name = path[sep + 1 :]
208  path = ZipFile(zipname, 'r').open(file_name)
209  if opt['task'] != 'pytorch_teacher':
210  task = opt['task']
211  else:
212  task = opt['image_load_task']
213  prepath = os.path.join(opt['datapath'], task)
214  imagefn = ''.join(zipname.strip('.zip').split('/')[-2:]) + path.name
215  if mode == 'raw':
216  # raw just returns RGB values
217  return Image.open(path).convert('RGB')
218  elif mode == 'ascii':
219  # convert images to ascii ¯\_(ツ)_/¯
220  return self._img_to_ascii(path)
221  else:
222  # otherwise, looks for preprocessed version under 'mode' directory
223  if not is_zip:
224  prepath, imagefn = os.path.split(path)
225  dpath = os.path.join(prepath, mode)
226  if not os.path.exists(dpath):
227  build_data.make_dir(dpath)
228  imagefn = imagefn.split('.')[0]
229  new_path = os.path.join(prepath, mode, imagefn)
230  if not os.path.isfile(new_path):
231  return self.extract(Image.open(path).convert('RGB'), new_path)
232  else:
233  return self.torch.load(new_path)
