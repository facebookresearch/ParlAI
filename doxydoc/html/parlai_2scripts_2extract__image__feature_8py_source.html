65 def get_dataset_class(opt):
66  """ To use a custom Pytorch Dataset, specify it on the command line:
67  ``--dataset parlai.tasks.vqa_v1.agents:VQADataset``
68 
69  Note that if the dataset is named ``DefaultDataset``, then you do
70  not need to specify its name following the colon; e.g., it
71  would just be:
72  ``--dataset parlai.tasks.vqa_v1.agents``
73  """
74  dataset_name = opt.get('pytorch_teacher_dataset')
75  sp = dataset_name.strip().split(':')
76  module_name = sp[0]
77  if len(sp) > 1:
78  dataset = sp[1]
79  else:
80  dataset = 'DefaultDataset'
81  my_module = importlib.import_module(module_name)
82  return getattr(my_module, dataset)

107  while not world.epoch_done():
108  world.parley()
109  pbar.update()
110  pbar.close()
111  elif opt.get('use_hdf5_extraction', False):
112  # TODO Deprecate
113  '''One can specify a Pytorch Dataset for custom image loading'''
114  nw = opt.get('numworkers', 1)
115  im = opt.get('image_mode', 'raw')
116  opt['batchsize'] = 1
117  opt['extract_image'] = True
118  bsz = 1
119  try:
120  import torch

122  except ImportError:
123  raise ImportError('Need to install Pytorch: go to pytorch.org')
124 
125  dataset = get_dataset_class(opt)(opt)
126  pre_image_path, _ = os.path.split(dataset.image_path)
127  image_path = os.path.join(pre_image_path, opt.get('image_mode'))
128  images_built_file = image_path + '.built'
129 
130  if not os.path.exists(image_path) or not os.path.isfile(images_built_file):
131  '''Image features have not been computed yet'''
132  opt['num_load_threads'] = 20
133  agent = RepeatLabelAgent(opt)
134  if opt['task'] == 'pytorch_teacher':
135  if opt.get('pytorch_teacher_task'):
136  opt['task'] = opt['pytorch_teacher_task']

139  world = create_task(opt, agent)
140  exs_seen = 0
141  total_exs = world.num_examples()

143  print('[ Computing and Saving Image Features ]')
144  while exs_seen < total_exs:
145  world.parley()
146  exs_seen += bsz
147  pbar.update(bsz)
148  pbar.close()
149  print('[ Feature Computation Done ]')
150  with open(images_built_file, 'w') as write:
151  write.write(str(datetime.datetime.today()))

153  dataloader = DataLoader(
154  dataset,
155  batch_size=bsz,

157  num_workers=nw,
158  collate_fn=lambda batch: batch[0],
159  )
160 
161  dataset_shape = None
162  image_id_to_index = {}

164  attention = opt.get('attention', False)

176  hdf5_file = h5py.File(hdf5_path, 'w')

178  iterator = tqdm.tqdm(
179  dataloader, unit='batch', unit_scale=True, total=total_exs // bsz
180  )
181  for ex in iterator:

191  if not attention:

193  img = img.sum(3).sum(2).div(nb_regions).view(-1, 2048)
194 
195  if dataset_shape is None:
196  if attention:

198  else:
199  dataset_shape = (num_images, img.size(1))
