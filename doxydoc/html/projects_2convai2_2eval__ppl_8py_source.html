6 """Base script for running official ConvAI2 validation eval for perplexity.
7 This uses a the version of the dataset which does not contain candidates.
8 Leaderboard scores will be run in the same form but on a hidden test set.
9 
10 The official vocabulary for the competition is based on using the
11 "split_tokenize" method on in the ParlAI core dictionary (parlai/core/dict.py)
12 and is built on the training and validation sets of the "convai2" task.
13 This dictionary contains a total of 19304 tokens. The test set contains some
14 tokens which are not in this dictionary--this tokens will not be provided, but
15 we will also *SKIP* calculating perplexity on these tokens. The model should
16 still produce a good guess for the remaining tokens in the sentence, so
17 handling unknown words or expanding the vocabulary with pre-trained or
18 multitasked embeddings are legitimate strategies that may or may not impact the
19 score of the models.
20 
21 Note that this tokenizer will also be used during the perplexity evaluation:
22 the model will be asked to predict one word at a time according to this
23 tokenizer's parsing of the text.
24 
25 This requires agents to implement the following function:
26 
27 def next_word_probability(self, partial_out):
28  Return probability distribution over next words given a partial true output.
29  This is used to calculate the per-word perplexity.
30 
31  Arguments:
32  partial_out -- list of previous "true" words
33 
34  Returns a dict, where each key is a word and each value is a probability
35  score for that word. Unset keys assume a probability of zero.
36 
37  e.g.
38  {'text': 'Run test program.'}, ['hello'] => {'world': 1.0}
39 """

58 class WordFrequencyEntry(Agent):
59  """This is an example entry which tries to use the RepeatLabelAgent.
60  Since no labels are given to the model, it will guess something useless.
61 
62  It builds the official dictionary first, so that it can provide a minimum
63  probablity for each word as well as use the official tokenizer.
64  """

66  def __init__(self, opt, shared=None):
67  super().__init__(opt, shared)
68  if not shared:
69  # build official eval dictionary
70  self.dict = build_dict()
71  else:
72  # only build dict once
73  self.dict = shared['dict']
74  max_freq = self.dict.max_freq()
75  # set probability of each word, skipping the invalid words like __NULL__
76  # (which have frequency more than max_freq)
77  self.freqs = {k: f for k, f in self.dict.freqs().items() if f <= max_freq}
78 
79  def share(self):
80  shared = super().share()
81  # share dict with other threads instead of rebuilding in each
82  shared['dict'] = self.dict
83  return shared

85  def next_word_probability(self, partial_out):
86  """Example implementation of next word probability."""
87  obs = self.observation
88  # initialize probabilities with inverse word frequency
89  freqs = self.freqs.copy()
90 
91  # increase likelihood of predicting input words
92  tokens = self.dict.tokenize(obs.get('text', ''))
93  for t in tokens:
94  if t in freqs:
95  freqs[t] += 10000
96  return freqs
