15 class Seq2seqAgent(TorchGeneratorAgent):
16  """Agent which takes an input sequence and produces an output sequence.
17 
18  This model supports encoding the input and decoding the output via one of
19  several flavors of RNN. It then uses a linear layer (whose weights can
20  be shared with the embedding layer) to convert RNN output states into
21  output tokens. This model supports greedy decoding, selecting the
22  highest probability token at each time step, as well as beam
23  search.
24 
25  For more information, see the following papers:
26  - Neural Machine Translation by Jointly Learning to Align and Translate
27  `(Bahdanau et al. 2014) <arxiv.org/abs/1409.0473>`_
28  - Sequence to Sequence Learning with Neural Networks
29  `(Sutskever et al. 2014) <arxiv.org/abs/1409.3215>`_
30  - Effective Approaches to Attention-based Neural Machine Translation
31  `(Luong et al. 2015) <arxiv.org/abs/1508.04025>`_
32  """

166  opt['hiddensize'],

168  start_idx=self.START_IDX,

171  longest_label=states.get('longest_label', 1),
172  **kwargs,
173  )
174 
175  if opt.get('dict_tokenizer') == 'bpe' and opt['embedding_type'] != 'random':
176  print('skipping preinitialization of embeddings for bpe')
177  elif not states and opt['embedding_type'] != 'random':
178  # `not states`: only set up embeddings if not loading model
179  self._copy_embeddings(model.decoder.lt.weight, opt['embedding_type'])
180  if opt['lookuptable'] in ['unique', 'dec_out']:
181  # also set encoder lt, since it's not shared

184  )

186  if states:

188  model.load_state_dict(states['model'])

191  print('Seq2seq: fixing embedding weights.')

193  model.encoder.lt.weight.requires_grad = False
194  if opt['lookuptable'] in ['dec_out', 'all']:
