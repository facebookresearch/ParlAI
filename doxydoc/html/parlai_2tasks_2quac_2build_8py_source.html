43 def _handle_paragraph(each):
44  output = []
45  story = each['context'].replace('\n', '\\n')
46  for idx, q_a in enumerate(each['qas']):
47  question_txt = ''
48  if idx == 0:

50  else:
51  question_txt = q_a['question']
52  starts, labels = _parse_answers(q_a)
53  output.append(
54  OUTPUT_FORMAT.format(
55  question=question_txt,
56  continuation=MAP_CONTINUATION.get(q_a['followup']),
57  affirmation=MAP_AFFIRMATION.get(q_a['yesno']),
58  start=starts,
59  labels=labels,
60  )
61  )
62  if idx < len(each['qas']) - 1:
63  output.append('\n')
64  output.append('\t\tepisode_done:True\n')
