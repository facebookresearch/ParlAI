373  """Save observation for act.
374  If multiple observations are from the same episode, concatenate them.
375  """
376  # shallow copy observation (deep copy can be expensive)
377  obs = Message(observation.copy()) # TODO: all teachers should return
378  # messages, so this should be eventually unecessary
379  seq_len = self.opt['seq_len']
380  is_training = True
381  if 'labels' not in obs:
382  is_training = False
383 
384  if is_training:
385  if 'text' in obs:
386  if self.use_person_tokens:
387  obs.force_set('text', 'PERSON1 ' + obs['text'])
388  vec = self.parse(obs['text'])
389  vec.append(self.END_IDX)
390  self.next_observe += vec
391  if 'labels' in obs:
392  if self.use_person_tokens:
393  labels = [
394  'PERSON2 ' + label for label in obs['labels'] if label != ''
395  ]
396  obs.force_set('labels', tuple(labels))
397  vec = self.parse(obs['labels'][0])
398  vec.append(self.END_IDX)
399  self.next_observe += vec
400  if len(self.next_observe) < (seq_len + 1):
401  # not enough to return to make a batch
402  # we handle this case in vectorize
403  # labels indicates that we are training
404  self.observation = {'labels': ''}
405  return self.observation
406  else:
407  vecs_to_return = []
408  total = len(self.next_observe) // (seq_len + 1)
409  for _ in range(total):
410  observe = self.next_observe[: (seq_len + 1)]
411  self.next_observe = self.next_observe[(seq_len + 1) :]
412  vecs_to_return.append(observe)
413  dict_to_return = {'text': '', 'labels': '', 'text2vec': vecs_to_return}
414  self.observation = Message(dict_to_return)
415  return dict_to_return
416  else:
417  if 'text' in obs:
418  if self.use_person_tokens:
419  obs.force_set('text', 'PERSON1 ' + obs['text'])
420  if 'eval_labels' in obs:
421  if self.use_person_tokens:
422  eval_labels = [
423  'PERSON2 ' + label
424  for label in obs['eval_labels']
425  if label != ''
426  ]
427  obs.force_set('eval_labels', tuple(eval_labels))
428  self.observation = obs
429  return obs

438  def get_target_loss(self, data, hidden, targets):
439  """Calculates the loss with respect to the targets, token by token,
440  where each output token is conditioned on either the input or the
441  previous target token.
442  """
443  loss = 0.0
444  bsz = data.size(0)
445 
446  # during interactive mode, when no targets exist, we return 0
447  if targets is None:
448  return loss
449 
450  # feed in inputs without end token
451  output, hidden = self.model(data.transpose(0, 1), hidden)
452  self.hidden = self.repackage_hidden(hidden)
453  # feed in end tokens
454  output, hidden = self.model(Variable(self.ends[:bsz].view(1, bsz)), self.hidden)
455  self.hidden = self.repackage_hidden(hidden)
456  output_flat = output.view(-1, len(self.dict))
457  loss += self.criterion(output_flat, targets.select(1, 0).view(-1)).data
458 
459  for i in range(1, targets.size(1)):
460  output, hidden = self.model(
461  targets.select(1, i - 1).view(1, bsz), self.hidden, no_pack=True
462  )
463  self.hidden = self.repackage_hidden(hidden)
464  output_flat = output.view(-1, len(self.dict))
465  loss += self.criterion(output_flat, targets.select(1, i).view(-1)).data
466 
467  return loss

469  def get_predictions(self, data):
470  """Generates predictions word by word until we either reach the end token
471  or some max length (opt['truncate_pred']).
472  """
473  token_list = []
474  bsz = data.size(0)
475  done = [False for _ in range(bsz)]
476  total_done = 0
477  hidden = self.model.init_hidden(bsz)
478 
479  i = 0
480  word_idx = None
481  while total_done < bsz and i <= self.opt['truncate_pred']:
482  if i == 0:
483  # feed in input without end tokens
484  output, hidden = self.model(data.transpose(0, 1), hidden)
485  hidden = self.repackage_hidden(hidden)
486  # feed in end tokens
487  output, hidden = self.model(
488  Variable(self.ends[:bsz].view(1, bsz)), hidden
489  )
490  else:
491  output, hidden = self.model(
492  Variable(word_idx.view(1, bsz)), hidden, no_pack=True
493  )
494  hidden = self.repackage_hidden(hidden)
495  word_weights = output.squeeze().data.exp()
496  if bsz > 1:
497  _, word_idx = torch.max(word_weights, 1)
498  else:
499  if self.sampling_mode:
500  unk_idx = self.dict[self.dict.unk_token]
501  # make word_weights have smaller norm so that calculated
502  # norm does not blow up
503  word_weights = word_weights.div(1e10)
504  # make word_weights have L2 norm 1
505  ww_norm = torch.norm(word_weights, p=2)
506  word_weights = word_weights.div(ww_norm)
507  # square distribution
508  word_weights = torch.mul(word_weights, word_weights)
509  # sample distribution
510  word_idx = torch.multinomial(word_weights, 1)
511  # do not produce UNK token
512  while word_idx == unk_idx:
513  word_idx = torch.multinomial(word_weights, 1)
514  else:
515  _, word_idx = torch.max(word_weights, 0)
516  # mark end indices for items in batch
517  word_idx = word_idx.view(-1)
518  for k in range(word_idx.size(0)):
519  if not done[k]:
520  if int(word_idx[k]) == self.END_IDX:
521  done[k] = True
522  total_done += 1
523  token_list.append(word_idx.view(bsz, 1))
524  i += 1
525 
526  return torch.cat(token_list, 1)

528  def predict(self, data, hidden, targets=None, is_training=True, y_lens=None):
529  """Produce a prediction from our model."""
530  output = None
531  predictions = None
532  if is_training:
533  self.model.train()
534  self.zero_grad()
535  output, hidden = self.model(data, hidden)
536  loss = self.criterion(output.view(-1, len(self.dict)), targets.view(-1))
537  # save loss to metrics
538  target_tokens = targets.ne(self.NULL_IDX).float().sum().item()
539  self.metrics['lmloss'] += loss.double().item()
540  self.metrics['lm_num_tokens'] += target_tokens
541  # average loss per token
542  loss /= target_tokens
543  loss.backward(retain_graph=True)
544  self.update_params()
545  else:
546  self.model.eval()
547  predictions = self.get_predictions(data)
548  bsz = data.size(0)
549  if bsz != self.batchsize:
550  self.hidden = self.model.init_hidden(bsz)
551  if targets is not None:
552  loss = self.get_target_loss(data, self.hidden, targets)
553  self.metrics['loss'] += loss
554  self.metrics['num_tokens'] += sum(y_lens)
555 
556  return output, hidden, predictions

558  def vectorize(self, observations, seq_len, is_training):
559  """Convert a list of observations into input & target tensors."""
560  labels = None
561  valid_inds = None
562  y_lens = None
563  if is_training:
564  for obs in observations:
565  if obs:
566  if 'text2vec' in obs:
567  self.next_batch += obs['text2vec']
568  if len(self.next_batch) <= self.batchsize:
569  return None, None, None, None, None
570  else:
571  data_list = []
572  targets_list = []
573  # total is the number of batches
574  total = len(self.next_batch) // self.batchsize
575  for _ in range(total):
576  batch = self.next_batch[: self.batchsize]
577  self.next_batch = self.next_batch[self.batchsize :]
578 
579  source = torch.LongTensor(batch).t().contiguous()
580  data = Variable(source[:seq_len])
581  targets = Variable(source[1:])
582 
583  if self.use_cuda:
584  data = data.cuda()
585  targets = targets.cuda()
586 
587  data_list.append(data)
588  targets_list.append(targets)
589  else:
590  # here we get valid examples and pad them with zeros
591  xs, ys, labels, valid_inds, _, y_lens = PaddingUtils.pad_text(
592  observations, self.dict, end_idx=self.END_IDX, null_idx=self.NULL_IDX
593  )
594 
595  if self.use_cuda:
596  if xs is not None:
597  xs = Variable(torch.LongTensor(xs)).cuda()
598  if ys is not None:
599  ys = Variable(torch.LongTensor(ys)).cuda()
600  else:
601  if xs is not None:
602  xs = Variable(torch.LongTensor(xs))
603  if ys is not None:
604  ys = Variable(torch.LongTensor(ys))
605  data_list = [xs]
606  targets_list = [ys]
607 
608  return data_list, targets_list, labels, valid_inds, y_lens

610  def batch_act(self, observations):
611  batch_reply = [{'id': self.getID()} for _ in range(len(observations))]
612  if any(['labels' in obs for obs in observations]):
613  # if we are starting a new training epoch, reinitialize hidden
614  if not self.is_training:
615  self.hidden = self.model.init_hidden(self.batchsize)
616  self.is_training = True
617  data_list, targets_list, _c, _v, y_lens = self.vectorize(
618  observations, self.opt['seq_len'], self.is_training
619  )
620  else:
621  # if we just finished training, reinitialize hidden
622  if self.is_training:
623  self.hidden = self.model.init_hidden(self.batchsize)
624  self.is_training = False
625  data_list, targets_list, labels, valid_inds, y_lens = self.vectorize(
626  observations, self.opt['seq_len'], self.is_training
627  )
628 
629  if data_list is None:
630  # not enough data to batch act yet, return empty responses
631  return batch_reply
632 
633  batch_reply = []
634  # during evaluation, len(data_list) is always 1
635  # during training, len(dat_list) >= 0: vectorize returns a list
636  # containing all batches available at the time it is called
637  for i in range(len(data_list)):
638  temp_dicts = [{'id': self.getID()} for _ in range(len(observations))]
639  # ignore case when we do not return any valid indices
640  if data_list[i] is not None:
641  output, hidden, predictions = self.predict(
642  data_list[i], self.hidden, targets_list[i], self.is_training, y_lens
643  )
644  self.hidden = self.repackage_hidden(hidden)
645 
646  if predictions is not None:
647  # map predictions back to the right order
648  PaddingUtils.map_predictions(
649  predictions.cpu(),
650  valid_inds,
651  temp_dicts,
652  observations,
653  self.dict,
654  self.END_IDX,
655  report_freq=self.opt['report_freq'],
656  )
657 
658  batch_reply += temp_dicts
