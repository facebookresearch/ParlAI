466 def test_learning_rate_resuming(self, args):
467  """Test learning rate resumes correctly."""
468  mdl = args['model']
469  with testing_utils.tempdir() as tmpdir:
470  model_file = os.path.join(tmpdir, 'model')
471  stdout1, valid1, test1 = testing_utils.train_model(
472  dict(model_file=model_file, lr_scheduler='invsqrt', **args)
473  )
474  stdout2, valid2, test2 = testing_utils.train_model(
475  dict(model_file=model_file, lr_scheduler='invsqrt', **args)
476  )
477  # make sure the number of updates is being tracked correctly
478  self.assertGreater(
479  valid2['total_train_updates'],
480  valid1['total_train_updates'],
481  '({}) Number of updates is not increasing'.format(mdl),
482  )
483  # make sure the learning rate is decreasing
484  self.assertLess(
485  valid2['lr'],
486  valid1['lr'],
487  '({}) Learning rate is not decreasing'.format(mdl),
488  )
489  # but make sure we're not loading the scheduler if we're fine
490  # tuning
491  stdout3, valid3, test3 = testing_utils.train_model(
492  dict(
493  init_model=os.path.join(tmpdir, 'model'),
494  model_file=os.path.join(tmpdir, 'newmodel'),
495  lr_scheduler='invsqrt',
496  **args,
497  )
498  )
499  self.assertEqual(
500  valid3['total_train_updates'],
501  valid1['total_train_updates'],
502  '({}) Finetuning LR scheduler reset failed '
503  '(total_train_updates).'.format(mdl),
504  )
505  self.assertEqual(
506  valid3['lr'],
507  valid1['lr'],
508  '({}) Finetuning LR scheduler reset failed ' '(lr).'.format(mdl),
509  )
510  # and make sure we're not loading the scheduler if it changes
511  stdout4, valid4, test4 = testing_utils.train_model(
512  dict(
513  init_model=os.path.join(tmpdir, 'model'),
514  model_file=os.path.join(tmpdir, 'newmodel2'),
515  lr_scheduler='reduceonplateau',
516  **args,
517  )
518  )
519  self.assertEqual(
520  valid4['total_train_updates'],
521  valid1['total_train_updates'],
522  '({}) LR scheduler change reset failed (total_train_updates).'
523  '\n{}'.format(mdl, stdout4),
524  )
525  self.assertEqual(
526  valid4['lr'],
527  1e-3,
528  '({}) LR is not correct in final resume.\n{}'.format(mdl, stdout4),
