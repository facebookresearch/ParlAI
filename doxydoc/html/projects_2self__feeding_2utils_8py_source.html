102 def add_person_tokens(responses, first_speaker=None, last_speaker=1):
103  """Converts a list of responses into a single tag-separated string
104  Args:
105  responses: list of responses (strings)
106  first_speaker: either 1 or 2; the owner of the first response
107  last_speaker: either 1 or 2; the owner of the last response
108  NOTE: if first_speaker is provided, it overrides last_speaker
109  Output:
110  text: the concatenated text
111 
112  e.g.,
113  responses = ["How are you?", "I'm doing fine!", "I'm glad to hear it!"]
114  result = add_person_tokens(responses)
115  result: "__p1__ How are you? __p2__ I'm doing fine! __p1__ I'm glad to
116  hear it!"
117  """
118  if first_speaker is None:
119  first_speaker = (last_speaker + len(responses)) % 2 + 1
120  speaker = first_speaker
121  text = ''
122  for response in responses:
123  tag = f"__p{speaker}__"
124  text += ' ' + tag + ' ' + response
125  speaker = 1 if speaker == 2 else 2
126  return text.strip()

156 def episode_to_examples(episode, histsz):
157  """Converts an episode (list of Parleys) into self-feeding compatible examples
158 
159  WARNING: we no longer require a histz when making a self-feeding file. Shortening of
160  the history is typically done in the teacher file or in interactive mode.
161  """
162  examples = []

165  # Update memories and history
166  # memories.extend(parley.memories)
167  history.append(parley.context)
168 
169  # Concatenate history and add speaker tokens as necessary
170  # if history_size == 1, the bot (p2) only sees the immediately
171  # preceding utterance (the prompt from the human, p1).
172  if histsz < 0:
173  utterances = history
174  context = add_person_tokens(utterances, last_speaker=1)
175  elif histsz == 0:
176  context = '__null__'
177  else:
178  utterances = history[-histsz:]
179  context = add_person_tokens(utterances, last_speaker=1)
