250  def _get_batch_train_metrics(self, scores):
251  """
252  Get fast metrics calculations if we train with batch candidates.
253 
254  Specifically, calculate accuracy ('train_accuracy'), average rank,
255  and mean reciprocal rank.
256  """
257  batchsize = scores.size(0)
258  # get accuracy
259  targets = scores.new_empty(batchsize).long()
260  targets = torch.arange(batchsize, out=targets)
261  nb_ok = (scores.max(dim=1)[1] == targets).float().sum().item()
262  self.metrics['train_accuracy'] += nb_ok
263  # calculate mean_rank
264  above_dot_prods = scores - scores.diag().view(-1, 1)
265  ranks = (above_dot_prods > 0).float().sum(dim=1) + 1
266  mrr = 1.0 / (ranks + 0.00001)
267  self.metrics['rank'] += torch.sum(ranks).item()
268  self.metrics['mrr'] += torch.sum(mrr).item()

270  def _get_train_preds(self, scores, label_inds, cands, cand_vecs):
271  """Return predictions from training."""
272  # TODO: speed these calculations up
273  batchsize = scores.size(0)
274  if self.rank_top_k > 0:
275  _, ranks = scores.topk(
276  min(self.rank_top_k, scores.size(1)), 1, largest=True
277  )
278  else:
279  _, ranks = scores.sort(1, descending=True)
280  for b in range(batchsize):
281  rank = (ranks[b] == label_inds[b]).nonzero()
282  rank = rank.item() if len(rank) == 1 else scores.size(1)
283  self.metrics['rank'] += 1 + rank
284  self.metrics['mrr'] += 1.0 / (1 + rank)
285 
286  ranks = ranks.cpu()
287  # Here we get the top prediction for each example, but do not
288  # return the full ranked list for the sake of training speed
289  preds = []
290  for i, ordering in enumerate(ranks):
291  if cand_vecs.dim() == 2: # num cands x max cand length
292  cand_list = cands
293  elif cand_vecs.dim() == 3: # batchsize x num cands x max cand length
294  cand_list = cands[i]
295  if len(ordering) != len(cand_list):
296  # We may have added padded cands to fill out the batch;
297  # Here we break after finding the first non-pad cand in the
298  # ranked list
299  for x in ordering:
300  if x < len(cand_list):
301  preds.append(cand_list[x])
302  break
303  else:
304  preds.append(cand_list[ordering[0]])
305 
306  return Output(preds)

335  def train_step(self, batch):
336  """Train on a single batch of examples."""
337  if batch.text_vec is None and batch.image is None:
338  return
339  batchsize = (
340  batch.text_vec.size(0)
341  if batch.text_vec is not None
342  else batch.image.size(0)
343  )
344  self.model.train()
345  self.zero_grad()
346 
347  cands, cand_vecs, label_inds = self._build_candidates(
348  batch, source=self.candidates, mode='train'
349  )
350  try:
351  scores = self.score_candidates(batch, cand_vecs)
352  loss = self.criterion(scores, label_inds)
353  self.backward(loss)
354  self.update_params()
355  except RuntimeError as e:
356  # catch out of memory exceptions during fwd/bck (skip batch)
357  if 'out of memory' in str(e):
358  print(
359  '| WARNING: ran out of memory, skipping batch. '
360  'if this happens frequently, decrease batchsize or '
361  'truncate the inputs to the model.'
362  )
363  return Output()
364  else:
365  raise e
366 
367  # Update loss
368  self.metrics['loss'] += loss.item()
369  self.metrics['examples'] += batchsize
370 
371  # Get train predictions
372  if self.candidates == 'batch':
373  self._get_batch_train_metrics(scores)
374  return Output()
375  if not self.opt.get('train_predict', False):
376  warn_once(
377  "Some training metrics are omitted for speed. Set the flag "
378  "`--train-predict` to calculate train metrics."
379  )
380  return Output()
381  return self._get_train_preds(scores, label_inds, cands, cand_vecs)

383  def eval_step(self, batch):
384  """Evaluate a single batch of examples."""
385  if batch.text_vec is None and batch.image is None:
386  return
387  batchsize = (
388  batch.text_vec.size(0)
389  if batch.text_vec is not None
390  else batch.image.size(0)
391  )
392  self.model.eval()
393 
394  cands, cand_vecs, label_inds = self._build_candidates(
395  batch, source=self.eval_candidates, mode='eval'
396  )
397 
398  cand_encs = None
399  if self.encode_candidate_vecs and self.eval_candidates in ['fixed', 'vocab']:
400  # if we cached candidate encodings for a fixed list of candidates,
401  # pass those into the score_candidates function
402  if self.eval_candidates == 'fixed':
403  cand_encs = self.fixed_candidate_encs
404  elif self.eval_candidates == 'vocab':
405  cand_encs = self.vocab_candidate_encs
406 
407  scores = self.score_candidates(batch, cand_vecs, cand_encs=cand_encs)
408  if self.rank_top_k > 0:
409  _, ranks = scores.topk(
410  min(self.rank_top_k, scores.size(1)), 1, largest=True
411  )
412  else:
413  _, ranks = scores.sort(1, descending=True)
414 
415  # Update metrics
416  if label_inds is not None:
417  loss = self.criterion(scores, label_inds)
418  self.metrics['loss'] += loss.item()
419  self.metrics['examples'] += batchsize
420  for b in range(batchsize):
421  rank = (ranks[b] == label_inds[b]).nonzero()
422  rank = rank.item() if len(rank) == 1 else scores.size(1)
423  self.metrics['rank'] += 1 + rank
424  self.metrics['mrr'] += 1.0 / (1 + rank)
425 
426  ranks = ranks.cpu()
427  max_preds = self.opt['cap_num_predictions']
428  cand_preds = []
429  for i, ordering in enumerate(ranks):
430  if cand_vecs.dim() == 2:
431  cand_list = cands
432  elif cand_vecs.dim() == 3:
433  cand_list = cands[i]
434  # using a generator instead of a list comprehension allows
435  # to cap the number of elements.
436  cand_preds_generator = (
437  cand_list[rank] for rank in ordering if rank < len(cand_list)
438  )
439  cand_preds.append(list(islice(cand_preds_generator, max_preds)))
440 
441  if (
442  self.opt.get('repeat_blocking_heuristic', True)
443  and self.eval_candidates == 'fixed'
444  ):
445  cand_preds = self.block_repeats(cand_preds)
446 
447  preds = [cand_preds[i][0] for i in range(batchsize)]
448  return Output(preds, cand_preds)

450  def block_repeats(self, cand_preds):
451  """Heuristic to block a model repeating a line from the history."""
452  history_strings = []
453  for h in self.history.history_raw_strings:
454  # Heuristic: Block any given line in the history, splitting by '\n'.
455  history_strings.extend(h.split('\n'))
456 
457  new_preds = []
458  for cp in cand_preds:
459  np = []
460  for c in cp:
461  if c not in history_strings:
462  np.append(c)
463  new_preds.append(np)
464  return new_preds

483  def _build_candidates(self, batch, source, mode):
484  """
485  Build a candidate set for this batch.
486 
487  :param batch:
488  a Batch object (defined in torch_agent.py)
489  :param source:
490  the source from which candidates should be built, one of
491  ['batch', 'batch-all-cands', 'inline', 'fixed']
492  :param mode:
493  'train' or 'eval'
494 
495  :return: tuple of tensors (label_inds, cands, cand_vecs)
496 
497  label_inds: A [bsz] LongTensor of the indices of the labels for each
498  example from its respective candidate set
499  cands: A [num_cands] list of (text) candidates
500  OR a [batchsize] list of such lists if source=='inline'
501  cand_vecs: A padded [num_cands, seqlen] LongTensor of vectorized candidates
502  OR a [batchsize, num_cands, seqlen] LongTensor if source=='inline'
503 
504  Possible sources of candidates:
505 
506  * batch: the set of all labels in this batch
507  Use all labels in the batch as the candidate set (with all but the
508  example's label being treated as negatives).
509  Note: with this setting, the candidate set is identical for all
510  examples in a batch. This option may be undesirable if it is possible
511  for duplicate labels to occur in a batch, since the second instance of
512  the correct label will be treated as a negative.
513  * batch-all-cands: the set of all candidates in this batch
514  Use all candidates in the batch as candidate set.
515  Note 1: This can result in a very large number of candidates.
516  Note 2: In this case we will deduplicate candidates.
517  Note 3: just like with 'batch' the candidate set is identical
518  for all examples in a batch.
519  * inline: batch_size lists, one list per example
520  If each example comes with a list of possible candidates, use those.
521  Note: With this setting, each example will have its own candidate set.
522  * fixed: one global candidate list, provided in a file from the user
523  If self.fixed_candidates is not None, use a set of fixed candidates for
524  all examples.
525  Note: this setting is not recommended for training unless the
526  universe of possible candidates is very small.
527  * vocab: one global candidate list, extracted from the vocabulary with the
528  exception of self.NULL_IDX.
529  """
530  label_vecs = batch.label_vec # [bsz] list of lists of LongTensors
531  label_inds = None
532  batchsize = (
533  batch.text_vec.size(0)
534  if batch.text_vec is not None
535  else batch.image.size(0)
536  )
537 
538  if label_vecs is not None:
539  assert label_vecs.dim() == 2
540 
541  if source == 'batch':
542  warn_once(
543  '[ Executing {} mode with batch labels as set of candidates. ]'
544  ''.format(mode)
545  )
546  if batchsize == 1:
547  warn_once(
548  "[ Warning: using candidate source 'batch' and observed a "
549  "batch of size 1. This may be due to uneven batch sizes at "
550  "the end of an epoch. ]"
551  )
552  if label_vecs is None:
553  raise ValueError(
554  "If using candidate source 'batch', then batch.label_vec cannot be "
555  "None."
556  )
557 
558  cands = batch.labels
559  cand_vecs = label_vecs
560  label_inds = label_vecs.new_tensor(range(batchsize))
561 
562  elif source == 'batch-all-cands':
563  warn_once(
564  '[ Executing {} mode with all candidates provided in the batch ]'
565  ''.format(mode)
566  )
567  if batch.candidate_vecs is None:
568  raise ValueError(
569  "If using candidate source 'batch-all-cands', then batch."
570  "candidate_vecs cannot be None. If your task does not have "
571  "inline candidates, consider using one of "
572  "--{m}={{'batch','fixed','vocab'}}."
573  "".format(m='candidates' if mode == 'train' else 'eval-candidates')
574  )
575  # initialize the list of cands with the labels
576  cands = []
577  all_cands_vecs = []
578  # dictionary used for deduplication
579  cands_to_id = {}
580  for i, cands_for_sample in enumerate(batch.candidates):
581  for j, cand in enumerate(cands_for_sample):
582  if cand not in cands_to_id:
583  cands.append(cand)
584  cands_to_id[cand] = len(cands_to_id)
585  all_cands_vecs.append(batch.candidate_vecs[i][j])
586  cand_vecs, _ = padded_tensor(
587  all_cands_vecs,
588  self.NULL_IDX,
589  use_cuda=self.use_cuda,
590  fp16friendly=self.fp16,
591  )
592  label_inds = label_vecs.new_tensor(
593  [cands_to_id[label] for label in batch.labels]
594  )
595 
596  elif source == 'inline':
597  warn_once(
598  '[ Executing {} mode with provided inline set of candidates ]'
599  ''.format(mode)
600  )
601  if batch.candidate_vecs is None:
602  raise ValueError(
603  "If using candidate source 'inline', then batch.candidate_vecs "
604  "cannot be None. If your task does not have inline candidates, "
605  "consider using one of --{m}={{'batch','fixed','vocab'}}."
606  "".format(m='candidates' if mode == 'train' else 'eval-candidates')
607  )
608 
609  cands = batch.candidates
610  cand_vecs = padded_3d(
611  batch.candidate_vecs,
612  self.NULL_IDX,
613  use_cuda=self.use_cuda,
614  fp16friendly=self.fp16,
615  )
616  if label_vecs is not None:
617  label_inds = label_vecs.new_empty((batchsize))
618  bad_batch = False
619  for i, label_vec in enumerate(label_vecs):
620  label_vec_pad = label_vec.new_zeros(cand_vecs[i].size(1)).fill_(
621  self.NULL_IDX
622  )
623  if cand_vecs[i].size(1) < len(label_vec):
624  label_vec = label_vec[0 : cand_vecs[i].size(1)]
625  label_vec_pad[0 : label_vec.size(0)] = label_vec
626  label_inds[i] = self._find_match(cand_vecs[i], label_vec_pad)
627  if label_inds[i] == -1:
628  bad_batch = True
629  if bad_batch:
630  if self.ignore_bad_candidates and not self.is_training:
631  label_inds = None
632  else:
633  raise RuntimeError(
634  'At least one of your examples has a set of label candidates '
635  'that does not contain the label. To ignore this error '
636  'set `--ignore-bad-candidates True`.'
