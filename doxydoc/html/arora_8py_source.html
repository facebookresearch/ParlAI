27 class SentenceEmbedder(object):
28  """
29  A class to produce Arora-style sentence embeddings
30  sent_emb(s)
31  where s is a sentence. Also gives relatedness scores
32  cos_sim(word_emb(w), sent_emb(s))
33  for words w with GloVe embeddings word_emb(w).
34 
35  See: "A Simple But Tough-To-Beat Baseline For Sentence Embeddings",
36  Arora et al, 2017, https://openreview.net/pdf?id=SyK00v5xx
37  """
38 
39  def __init__(
40  self, word2prob, arora_a, glove_name, glove_dim, first_sv, glove_cache
41  ):
42  """
43  Inputs:
44  word2prob: dict mapping words to their unigram probs
45  arora_a: a float. Is the constant (called "a" in the paper)
46  used to compute Arora sentence embeddings.
47  glove_name: the version of GloVe to use, e.g. '840B'
48  glove_dim: the dimension of the GloVe embeddings to use, e.g. 300
49  first_sv: np array shape (glove_dim). The first singular value,
50  used to compute Arora sentence embeddings. Can be None.
51  glove_cache: The path to where the glove vectors are stored.
52  """
53  self.word2prob = word2prob
54  self.arora_a = arora_a
55  self.glove_name = glove_name
56  self.glove_dim = glove_dim
57  self.glove_cache = glove_cache
58  self.first_sv = first_sv
59  if self.first_sv is not None:
60  self.first_sv = torch.tensor(self.first_sv) # convert to torch tensor
61 
62  self.min_word_prob = min(word2prob.values()) # prob of rarest word
63  self.tt_embs = None # will be torchtext.vocab.GloVe object
64  self.emb_matrix = None # will be np array shape (vocab_size, glove_dim)
65 
66  # Initialize a cache, which holds up to 64 sentences, along with their
67  # corresponding word similarity scores (i.e. cosine sim for every word in the
68  # vocab). This enables us to repeatedly retrieve sims for sentences we have
69  # already processed (useful for batched beam search).
70  self.cache_limit = 64
71  self.cache_sent2sims = {} # maps sent to sims. holds up to cache_limit.
72  self.cache_sentqueue = deque() # list of sents. add to right, remove from left
73

76  Loads torchtext GloVe embs from file and stores in self.tt_embs.
77  """
78  print('Loading torchtext GloVe embs (for Arora sentence embs)...')
79  self.tt_embs = vocab.GloVe(
80  name=self.glove_name, dim=self.glove_dim, cache=self.glove_cache
81  )
82  print('Finished loading torchtext GloVe embs')

84  def get_emb_matrix(self, dictionary):
85  """
86  Construct an embedding matrix containing pretrained GloVe vectors for all words
87  in dictionary, and store in self.emb_matrix. This is needed for
88  response-relatedness weighted decoding.
89 
90  Inputs:
91  dictionary: ParlAI dictionary
92  """
93  print(
94  'Constructing GloVe emb matrix for response-relatedness weighted '
95  'decoding...'
96  )
97  self.emb_matrix = []
98  oov_indices = [] # list of dictionary indices for all OOV words
99  for idx in range(len(dictionary)):
100  word = dictionary[idx]
101  if word in self.tt_embs.stoi:
102  word_emb = self.tt_embs.vectors[self.tt_embs.stoi[word]]
103  else:
104  # If word is OOV, enter a zero vector instead.
105  # This means that the cosine similarity will always be zero.
106  word_emb = torch.zeros(self.glove_dim)
107  oov_indices.append(idx)
108  self.emb_matrix.append(word_emb)
109  self.emb_matrix = np.stack(self.emb_matrix) # (vocab_size, glove_dim)
110  print(
111  'Done constructing GloVe emb matrix; found %i OOVs of %i words'
112  % (len(oov_indices), len(dictionary))
113  )
114 
115  # Get the norm of each of the word vectors. This is needed for cosine sims.
116  # self.emb_matrix_norm is a np array shape (vocab_size)
117  self.emb_matrix_norm = np.linalg.norm(self.emb_matrix, axis=1)
118 
119  # For the OOV words which have zero vectors,
120  # set the norm to 1.0 so we don't have divide-by-zero errors
121  for idx in oov_indices:
122  self.emb_matrix_norm[idx] = 1.0

124  def get_word_sims(self, sent, sent_emb, dictionary):
125  """
126  Given a sentence and its Arora-style sentence embedding, compute the cosine
127  similarities to it, for all words in the dictionary.
128 
129  Inputs:
130  sent: string. Used only for caching lookup purposes.
131  sent_emb: torch Tensor shape (glove_dim).
132  dictionary: ParlAI dictionary
133 
134  Returns:
135  sims: torch Tensor shape (vocab_size), containing the cosine sims.
136  """
137  # If we haven't initialized the GloVe emb matrix yet, do so
138  if self.emb_matrix is None:
139  self.get_emb_matrix(dictionary)
140 
141  # If we have already computed sims for this sentence, return it
142  if sent in self.cache_sent2sims:
143  sims = self.cache_sent2sims[sent]
144  return sims
145 
146  # Compute the cosine similarities. Implementation from here:
147  # https://codereview.stackexchange.com/questions/55717/efficient-numpy-cosine-distance-calculation
148  dotted = self.emb_matrix.dot(sent_emb) # shape (vocab_size)
149  sent_emb_norm = np.linalg.norm(sent_emb) # norm of the sent emb. scalar
150  norms = np.multiply(self.emb_matrix_norm, sent_emb_norm) # shape (vocab_size)
151  sims = np.divide(dotted, norms) # divide dot prods by norms. shape (vocab_size)
152  sims = torch.tensor(sims) # convert to torch Tensor, shape (vocab_size)
153 
154  # Cache sims in self.cache_sent2sims
155  self.cache_sentqueue.append(sent) # append sent to right
156  self.cache_sent2sims[sent] = sims # add (sent, sims) pair to cache
157  if len(self.cache_sentqueue) > self.cache_limit:
158  to_remove = self.cache_sentqueue.popleft() # remove from left
159  del self.cache_sent2sims[to_remove] # remove from cache
160  assert len(self.cache_sent2sims) == len(self.cache_sentqueue)
161  assert len(self.cache_sent2sims) <= self.cache_limit
162 
163  return sims

165  def embed_sent(self, sent, rem_first_sv=True):
166  """
167  Produce a Arora-style sentence embedding for a given sentence.
168 
169  Inputs:
170  sent: tokenized sentence; a list of strings
171  rem_first_sv: If True, remove the first singular value when you compute the
172  sentence embddings. Otherwise, don't remove it.
173  Returns:
174  sent_emb: tensor length glove_dim, or None.
175  If sent_emb is None, that's because all of the words were OOV for GloVe.
176  """
177  # If we haven't loaded the torchtext GloVe embeddings, do so
178  if self.tt_embs is None:
179  self.get_glove_embs()
180 
181  # Lookup glove embeddings for words
182  tokens = [t for t in sent if t in self.tt_embs.stoi] # in-vocab tokens
183  # glove_oov_tokens = [t for t in sent if t not in self.tt_embs.stoi]
184  # if len(glove_oov_tokens)>0:
185  # print("WARNING: tokens OOV for glove: ", glove_oov_tokens)
186  if len(tokens) == 0:
187  print(
188  'WARNING: tried to embed utterance %s but all tokens are OOV for '
189  'GloVe. Returning embedding=None' % sent
190  )
191  return None
192  word_embs = [
193  self.tt_embs.vectors[self.tt_embs.stoi[t]] for t in tokens
194  ] # list of torch Tensors shape (glove_dim)
195 
196  # Get unigram probabilities for the words. If we don't have a word in word2prob,
197  # assume it's as rare as the rarest word in word2prob.
198  unigram_probs = [
199  self.word2prob[t] if t in self.word2prob else self.min_word_prob
200  for t in tokens
201  ] # list of floats
202  # word2prob_oov_tokens = [t for t in tokens if t not in self.word2prob]
203  # if len(word2prob_oov_tokens)>0:
204  # print('WARNING: tokens OOV for word2prob, so assuming they are '
205  # 'maximally rare: ', word2prob_oov_tokens)
206 
207  # Calculate the weighted average of the word embeddings
208  smooth_inverse_freqs = [
209  self.arora_a / (self.arora_a + p) for p in unigram_probs
210  ] # list of floats
211  sent_emb = sum(
212  [word_emb * wt for (word_emb, wt) in zip(word_embs, smooth_inverse_freqs)]
213  ) / len(
214  word_embs
215  ) # torch Tensor shape (glove_dim)
216 
217  # Remove the first singular value from sent_emb
218  if rem_first_sv:
219  sent_emb = remove_first_sv(sent_emb, self.first_sv)
220 
221  return sent_emb

224 def remove_first_sv(emb, first_sv):
225  """
226  Projects out the first singular value (first_sv) from the embedding (emb).
227 
228  Inputs:
229  emb: torch Tensor shape (glove_dim)
230  first_sv: torch Tensor shape (glove_dim)
231 
232  Returns:
233  new emb: torch Tensor shape (glove_dim)
234  """
235  # Calculate dot prod of emb and first_sv using torch.mm:
236  # (1, glove_dim) x (glove_dim, 1) -> (1,1) -> float
237  dot_prod = torch.mm(torch.unsqueeze(emb, 0), torch.unsqueeze(first_sv, 1)).item()
238  return emb - first_sv * dot_prod

241 def get_word_counts(opt, count_inputs):
242  """Goes through the dataset specified in opt, returns word counts and all utterances
243 
244  Inputs:
245  count_inputs: If True, include both input and reply when counting words and
246  utterances. Otherwise, only include reply text.
247 
248  Returns:
249  word_counter: a Counter mapping each word to the total number of times it appears
250  total_count: int. total word count, i.e. the sum of the counts for each word
251  all_utts: list of strings. all the utterances that were used for counting words
252  """
253  # Create repeat label agent and assign it to the specified task
254  agent = RepeatLabelAgent(opt)
255  world = create_task(opt, agent)
256 
257  # Count word frequency for all words in dataset
258  word_counter = Counter()
259  total_count = 0
260  all_utts = []
261  log_timer = TimeLogger()
262  while True:
263  world.parley()
264 
265  # Count words in reply
266  reply = world.acts[0].get('labels', world.acts[0].get('eval_labels'))[0]
267  words = reply.split()
268  word_counter.update(words)
269  total_count += len(words)
270  all_utts.append(reply)
271 
272  # Optionally count words in input text
273  if count_inputs:
274  input = world.acts[0]['text']
275  input = input.split('\n')[-1] # e.g. in ConvAI2, this removes persona
276  words = input.split()
277  word_counter.update(words)
278  total_count += len(words)
279  all_utts.append(input)
280 
281  if log_timer.time() > opt['log_every_n_secs']:
282  text, _log = log_timer.log(world.total_parleys, world.num_examples())
283  print(text)
284 
285  if world.epoch_done():
286  print('EPOCH DONE')
287  break
288 
289  assert total_count == sum(word_counter.values())
290 
291  return word_counter, total_count, all_utts

294 def learn_arora(opt):
295  """
296  Go through ConvAI2 data and collect word counts, thus compute the unigram
297  probability distribution. Use those probs to compute weighted sentence embeddings
298  for all utterances, thus compute first principal component.
299 
300  Save all info to arora.pkl file.
301  """
302  arora_file = os.path.join(opt['datapath'], 'controllable_dialogue', 'arora.pkl')
303 
304  opt['task'] = 'fromfile:parlaiformat'
305  opt['log_every_n_secs'] = 2
306 
307  print('Getting word counts from ConvAI2 train set...')
308  opt['datatype'] = 'train:ordered'
309  opt['fromfile_datapath'] = os.path.join(
310  opt['datapath'], 'controllable_dialogue', 'ConvAI2_parlaiformat', 'train.txt'
311  )
312  # Do include inputs because ConvAI2 train set reverses every convo:
313  word_counter_train, total_count_train, all_utts_train = get_word_counts(
314  opt, count_inputs=False
315  )
316 
317  print('Getting word counts from ConvAI2 val set...')
318  opt['datatype'] = 'valid'
319  opt['fromfile_datapath'] = os.path.join(
320  opt['datapath'], 'controllable_dialogue', 'ConvAI2_parlaiformat', 'valid.txt'
321  )
322  # Don't include inputs because ConvAI2 val set doesn't reverses convos:
323  word_counter_valid, total_count_valid, all_utts_valid = get_word_counts(
324  opt, count_inputs=True
325  )
326 
327  # Merge word counts
328  word_counter = word_counter_train
329  for word, count in word_counter_valid.items():
330  word_counter[word] += count
331  total_count = total_count_train + total_count_valid
332 
333  # Merge all_utts
334  all_utts = all_utts_train + all_utts_valid
335 
336  # Compute unigram prob for every word
337  print("Computing unigram probs for all words...")
338  word2prob = {w: c / total_count for w, c in word_counter.items()}
339 
340  # Settings for sentence embedder
341  arora_a = 0.0001
342  glove_name = '840B'
343  glove_dim = 300
344  glove_cache = modelzoo_path(opt['datapath'], 'models:glove_vectors')
345 
346  # Embed every sentence, without removing first singular value
347  print('Embedding all sentences...')
348  sent_embedder = SentenceEmbedder(
349  word2prob,
350  arora_a,
351  glove_name,
352  glove_dim,
353  first_sv=None,
354  glove_cache=glove_cache,
355  )
356  utt_embs = []
357  log_timer = TimeLogger()
358  for n, utt in enumerate(all_utts):
359  utt_emb = sent_embedder.embed_sent(utt.split(), rem_first_sv=False)
360  utt_embs.append(utt_emb)
361  if log_timer.time() > opt['log_every_n_secs']:
362  text, _log = log_timer.log(n, len(all_utts))
363  print(text)
364 
365  # Use SVD to calculate singular vector
366  # https://docs.scipy.org/doc/numpy-1.13.0/reference/generated/numpy.linalg.svd.html
367  print('Calculating SVD...')
368  utt_embs = np.stack(utt_embs, axis=0) # shape (num_embs, glove_dim)
369  U, s, V = np.linalg.svd(utt_embs, full_matrices=False)
370  first_sv = V[0, :] # first row of V. shape (glove_dim)
371 
372  # Remove singular vector from all embs to get complete Arora-style sent embs
373  print('Removing singular vec from all sentence embeddings...')
374  utt_embs_adj = [
375  remove_first_sv(torch.Tensor(emb), torch.Tensor(first_sv)).numpy()
376  for emb in utt_embs
377  ] # list of np arrays shape (glove_dim)
378 
379  # Make dict mapping ConvAI2 dataset utterances to Arora sent emb
380  # We save this to file for convenience (e.g. if you want to inspect)
381  utt2emb = {utt: emb for (utt, emb) in zip(all_utts, utt_embs_adj)}
382 
383  # Save unigram distribution, first singular value, hyperparameter value for a,
384  # info about GloVe vectors used, and full dict of utt->emb to file
385  print("Saving Arora embedding info to %s..." % arora_file)
386  with open(arora_file, "wb") as f:
387  pickle.dump(
388  {
389  'word2prob': word2prob, # dict: string to float between 0 and 1
390  'first_sv': first_sv, # np array shape (glove_dim)
391  'arora_a': arora_a, # float, 0.0001
392  'glove_name': glove_name, # string, '840B'
393  'glove_dim': glove_dim, # int, 300
394  'utt2emb': utt2emb, # dict: string to np array shape (glove_dim)
395  },
396  f,
397  )
