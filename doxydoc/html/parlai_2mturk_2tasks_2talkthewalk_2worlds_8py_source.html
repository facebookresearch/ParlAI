363  x, y = cell

378  def fill_initial(self, new_g, old_g, size):
379  for i in (0, size - 1):
380  for j in range(size):
381  new_g[i, j] = 0
382  for j in (0, size - 1):
383  for i in range(size):
384  new_g[i, j] = 0
385  for i in range(1, size, 2):
386  for j in range(1, size, 2):
387  new_g[i, j] = old_g[(i - 1) // 2, (j - 1) // 2]
388  for i in range(1, size - 1, 2):
389  for j in range(2, size - 1, 2):
390  new_g[i, j] = (new_g[i, j - 1] + new_g[i, j + 1]) / 2
391  for i in range(2, size - 1, 2):
392  for j in range(1, size - 1, 2):
393  new_g[i, j] = (new_g[i - 1, j] + new_g[i + 1, j]) / 2
394  return new_g

396  def fill_neighbors(self, grid, size):
397  for i in range(size):
398  for j in range(size):
399  if grid[i, j] == -1:
400  neighbors = self.neighbor_coords((i, j), size)
401  neighbor_sum = sum((grid[k, l] for k, l in neighbors))
402  grid[i, j] = neighbor_sum / len(neighbors)
