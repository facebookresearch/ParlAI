224  def test_pyt_batchsort_field(self):
225  """
226  Test that the batchsort actually works for Pytorch Data Teacher
227 
228  That is, for every batch except the last one, check that the length
229  of each example (determined by batchsort_field) is within
230  `batch-length-range`
231  """
232  # First, check that batchsort itself works
233  defaults = unit_test_parser_defaults.copy()
234  defaults['datatype'] = 'train:stream:ordered'
235  defaults['pytorch_teacher_batch_sort'] = True
236  defaults['pytorch_teacher_task'] = 'babi:task1k:1'
237  defaults['batchsize'] = 50
238  max_range = defaults['batch_length_range']
239 
240  def verify_batch_lengths(defaults):
241  with testing_utils.capture_output() as _, testing_utils.tempdir() as tmpdir:
242  # Get processed act from agent
243  parser = train_setup_args()
244  defaults['model_file'] = os.path.join(tmpdir, 'model')
245  defaults['dict_file'] = os.path.join(tmpdir, 'model.dict')
246  parser.set_defaults(**defaults)
247  opt = parser.parse_args()
248  build_dict(opt)
249  agent = create_agent(opt)
250  world_data = create_task(opt, agent)
251  batch_sort_acts = []
252  # first epoch
253  while len(batch_sort_acts) < 900 / 50:
254  world_data.parley()
255  batch_sort_acts.append(world_data.acts[0])
256  teacher = world_data.world.get_agents()[0]
257  teacher.reset_data()
258  # second epoch
259  while len(batch_sort_acts) < 1800 / 50:
260  world_data.parley()
261  batch_sort_acts.append(world_data.acts[0])
262  world_data.shutdown()
263  field = defaults['batch_sort_field']
264  lengths = [
265  [ep_length(b[field]) for b in bb if field in b]
266  for bb in batch_sort_acts[:-2]
267  ] # exclude last batch
268  # verify batch lengths
269  for batch_lens in lengths:
270  self.assertLessEqual(
271  max(batch_lens) - min(batch_lens),
272  max_range,
273  'PytorchDataTeacher batching does not give '
274  'batches with similar sized examples, when '
275  'sorting by `{}` field.'.format(defaults['batch_sort_field']),
276  )
277 
278  defaults['batch_sort_field'] = 'text'
279  verify_batch_lengths(defaults)
280  defaults['batch_sort_field'] = 'text_vec'
281  defaults['pytorch_preprocess'] = True
282  verify_batch_lengths(defaults)
