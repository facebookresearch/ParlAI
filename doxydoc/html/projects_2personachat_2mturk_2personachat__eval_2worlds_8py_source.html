535  def is_exact_match(self, act, ag, tolerance=0):
536  if act['episode_done']:
537  return False
538 
539  control_msg = {'episode_done': False}
540  control_msg['id'] = 'SYSTEM'
541 
542  text = act['text']
543  if text not in ['', ' ', ' ', ' ']:
544  n_word_match = 0

546  per_parse = per.split(' ')
547  regular_words = ['', ' ', 'I', 'I\'m', 'My', 'i']
548  for r_w in regular_words:
549  if r_w in per_parse:
550  per_parse.remove(r_w)
551  per_subseq = [
552  ' '.join(per_parse[i : i + len(per_parse) - tolerance])
553  for i in range(tolerance + 1)
554  ]
555  for pp in per_subseq:
556  if pp in ['', ' ', ' ', ' ']:
557  per_subseq.remove(pp)

559  if n_word_match > 0:
560  control_msg[
