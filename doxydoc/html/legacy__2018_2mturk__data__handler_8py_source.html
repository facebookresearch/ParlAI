144  def _get_connection(self):
145  """Returns a singular database connection to be shared amongst all
146  calls
147  """
148  curr_thread = threading.get_ident()
149  if curr_thread not in self.conn or self.conn[curr_thread] is None:
150  try:
151  conn = sqlite3.connect(self.db_path)
152  conn.row_factory = sqlite3.Row
153  self.conn[curr_thread] = conn
154  except sqlite3.Error as e:
155  shared_utils.print_and_log(
156  logging.ERROR,
157  "Could not get db connection, failing: {}".format(repr(e)),
158  should_print=True,
159  )
160  raise e
161  return self.conn[curr_thread]

558  def log_approve_assignment(self, assignment_id):
559  """Update assignment state to reflect approval, update worker state to
560  increment number of accepted assignments
561  """
562  with self.table_access_condition:
563  conn = self._get_connection()
564  c = conn.cursor()
565 
566  c.execute(
567  'SELECT * FROM assignments WHERE assignment_id = ?;', (assignment_id,)
568  )
569  assignment = c.fetchone()
570  if assignment is None:
571  return
572  status = assignment['status']
573  worker_id = assignment['worker_id']
574  if status == 'Approved':
575  return # assign already approved
576 
577  # handle the approving part
578  c.execute(
579  """UPDATE assignments SET status = ?
580  WHERE assignment_id = ?;""",
581  ('Approved', assignment_id),
582  )
583 
584  # This was a rejection override
585  if status == 'Rejected':
586  c.execute(
587  """UPDATE workers SET approved = approved + 1,
588  rejected = rejected - 1
589  WHERE worker_id = ?;""",
590  (worker_id,),
591  )
592  else:
593  c.execute(
594  """UPDATE workers SET approved = approved + 1
595  WHERE worker_id = ?;""",
596  (worker_id,),
597  )
598 
599  conn.commit()
