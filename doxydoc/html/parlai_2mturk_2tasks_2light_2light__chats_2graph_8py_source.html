221  def evaluate_constraint_set(self, graph, args, constraints):
222  """Check a particular set of arguments against the given constraints
223  returns (True, cons_passed, None) if the constraints all pass
224  returns (False, cons_passed, failure_action) if any constraint fails
225  """
226  cons_passed = 0
227  failure_action = None
228  for constraint in constraints:
229  con_args = [args[i] for i in ([0] + constraint['in_args'])]
230  con_args += constraint['args']
231  con_obj = CONSTRAINTS[constraint['type']]
232  if con_obj.evaluate_constraint(graph, con_args):
233  cons_passed += 1
234  else:
235  spec_fail = constraint.get('spec_fail', 'failed')
236  failure_action = con_obj.get_failure_action(graph, con_args, spec_fail)
237  break
238  if failure_action is None:
239  return True, cons_passed, None
240  else:
241  return False, cons_passed, failure_action
