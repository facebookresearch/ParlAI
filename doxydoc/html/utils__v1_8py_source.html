38 def padded_tensor(items, pad_idx=0, use_cuda=False, left_padded=False, max_len=None):
39  """Create a right-padded matrix from an uneven list of lists.
40 
41  Returns (padded, lengths), where padded is the padded matrix, and lengths
42  is a list containing the lengths of each row.
43 
44  Matrix is right-padded (filled to the right) by default, but can be
45  left padded if the flag is set to True.
46 
47  Matrix can also be placed on cuda automatically.
48 
49  :param list[iter[int]] items: List of items
50  :param bool sort: If True, orders by the length
51  :param int pad_idx: the value to use for padding
52  :param bool use_cuda: if true, places `padded` on GPU
53  :param bool left_padded:
54  :param int max_len: if None, the max length is the maximum item length
55 
56  :returns: (padded, lengths) tuple
57  :rtype: (Tensor[int64], list[int])
58  """
59  # hard fail if we don't have torch
60  if not __TORCH_AVAILABLE:
61  raise ImportError(
62  "Cannot use padded_tensor without torch; go to http://pytorch.org"
63  )
64 
65  # number of items
66  n = len(items)
67  # length of each item
68  lens = [len(item) for item in items]
69  # max in time dimension
70  t = max(lens) if max_len is None else max_len
71 
72  # if input tensors are empty, we should expand to nulls
73  t = max(t, 1)
74 
75  if isinstance(items[0], torch.Tensor):
76  # keep type of input tensors, they may already be cuda ones
77  output = items[0].new(n, t)
78  else:
79  output = torch.LongTensor(n, t)
80  output.fill_(pad_idx)
81 
82  for i, (item, length) in enumerate(zip(items, lens)):
83  if length == 0:
84  # skip empty items
85  continue
86  if not isinstance(item, torch.Tensor):
87  # put non-tensors into a tensor
88  item = torch.LongTensor(item)
89  if left_padded:
90  # place at end
91  output[i, t - length :] = item
92  else:
93  # place at beginning
94  output[i, :length] = item
95 
96  if use_cuda:
97  output = output.cuda()
98  return output, lens

101 def argsort(keys, *lists, descending=False):
102  """Reorder each list in lists by the (descending) sorted order of keys.
103 
104  :param iter keys: Keys to order by.
105  :param list[list] lists: Lists to reordered by keys's order.
106  Correctly handles lists and 1-D tensors.
107  :param bool descending: Use descending order if true.
108 
109  :returns: The reordered items.
110  """
111  ind_sorted = sorted(range(len(keys)), key=lambda k: keys[k])
112  if descending:
113  ind_sorted = list(reversed(ind_sorted))

116  # watch out in case we don't have torch installed

120  output.append([lst[i] for i in ind_sorted])
