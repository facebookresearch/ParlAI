128  def _reduce(self, embs, input):
129  # last dimension is embedding, do operation over dim before that
130  if self.reduction == 'sum':
131  return embs.sum(-2)
132  elif self.reduction == 'mean':
133  # this is more fair than mean(-2) since mean includes null tokens
134  sum = embs.sum(-2)
135  lens = (
136  input.ne(self.padding_idx).sum(-1).unsqueeze(-1).float().clamp_(min=1)
137  )
138  return sum / lens
139  else:
140  raise RuntimeError(
141  'reduction method {} not supported'.format(self.reduction)
142  )

144  def forward(self, input):
145  """
146  Return BOW embedding with PE reweighting if enabled.
147 
148  :param input:
149  (bsz x seqlen) LongTensor
150 
151  :returns:
152  (bsz x esz) FloatTensor
153  """
154  embs = super().forward(input)
155  if self.position_encoding:
156  if embs.dim() == 3:
157  num_mems, seqlen, embdim = embs.size()
158  pe = self.position_matrix(seqlen, embdim, embs.is_cuda)
159  for i in range(num_mems):
160  embs[i] *= pe
161  else:
162  bsz, num_mems, seqlen, embdim = embs.size()
163  pe = self.position_matrix(seqlen, embdim, embs.is_cuda)
164  for i in range(num_mems):
165  embs[:, i] *= pe
166  return self._reduce(embs, input)

168  @staticmethod
169  @lru_cache(maxsize=128)
170  def position_matrix(J, d, use_cuda):
171  """
172  Build matrix of position encoding coeffiencents.
173 
174  See https://papers.nips.cc/paper/5846-end-to-end-memory-networks,
175  section 4.1 Model Details: Sentence Representation.
176 
177  :param J:
178  number of words in the sequence
179 
180  :param d:
181  dimension of the embedding
182 
183  :returns:
184  Position Encoding matrix
185  """
186  m = torch.Tensor(J, d)
187  for k in range(1, d + 1):
188  for j in range(1, J + 1):
189  m[j - 1, k - 1] = (1 - j / J) - (k / d) * (1 - 2 * j / J)
190  if use_cuda:
191  m = m.cuda()
192  return m

195 class Hop(nn.Module):
196  """Memory Network hop outputs attention-weighted sum of memory embeddings.
197 
198  0) rotate the query embeddings
199  1) compute the dot product between the input vector and each memory vector
200  2) compute a softmax over the memory scores
201  3) compute the weighted sum of the memory embeddings using the probabilities
202  4) add the query embedding to the memory output and return the result
203  """
204 
205  def __init__(self, embedding_size, rotate=True):
206  """Initialize linear rotation."""
207  super().__init__()
208  if rotate:
209  self.rotate = nn.Linear(embedding_size, embedding_size, bias=False)
210  else:
211  self.rotate = lambda x: x
212  self.softmax = nn.Softmax(dim=1)
213 
214  def forward(self, query_embs, in_mem_embs, out_mem_embs, pad_mask):
215  """
216  Compute MemNN Hop step.
217 
218  :param query_embs:
219  (bsz x esz) embedding of queries
220 
221  :param in_mem_embs:
222  bsz list of (num_mems x esz) embedding of memories for activation
223 
224  :param out_mem_embs:
225  bsz list of (num_mems x esz) embedding of memories for outputs
226 
227  :param pad_mask
228  (bsz x num_mems) optional mask indicating which tokens correspond to
229  padding
230 
231  :returns:
232  (bsz x esz) output state
233  """
234  # rotate query embeddings
235  attn = torch.bmm(query_embs.unsqueeze(1), in_mem_embs).squeeze(1)
236  if pad_mask is not None:
237  attn[pad_mask] = neginf(attn.dtype)
238  probs = self.softmax(attn)
239  memory_output = torch.bmm(probs.unsqueeze(1), out_mem_embs).squeeze(1)
240  output = memory_output + self.rotate(query_embs)
241  return output
