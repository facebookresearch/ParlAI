187  def _score(self, output, cands):
188  if cands.dim() == 2:
189  return torch.matmul(output, cands.t())
190  elif cands.dim() == 3:
191  return torch.bmm(output.unsqueeze(1), cands.transpose(1, 2)).squeeze(1)
192  else:
193  raise RuntimeError(

241  def score_candidates(self, batch, cand_vecs, cand_encs=None):
242  """Score candidates."""
243  # convoluted check that not all memories are empty
244  if (
245  self.opt['use_memories']
246  and batch.memory_vecs is not None
247  and sum(len(m) for m in batch.memory_vecs)
248  ):
249  mems = padded_3d(
250  batch.memory_vecs, use_cuda=self.use_cuda, pad_idx=self.NULL_IDX
251  )
252  else:
253  mems = None
254 
255  if cand_encs is not None:
256  # we pre-encoded the candidates, do not re-encode here
257  cand_vecs = None
258 
259  context_h, cands_h = self.model(xs=batch.text_vec, mems=mems, cands=cand_vecs)
260 
261  if cand_encs is not None:
262  cands_h = cand_encs
263  scores = self._score(context_h, cands_h)
264 
265  return scores
