315  def _handle_message_read(self, event):
316  # If the message was sent by another user (as in during a conversation)
317  # then we need to propogate the read back to that user.
318  self._log_debug('Messages {} marked as read.'.format(event['read']))
319  reader = event['sender']['id']
320  agent_state = self.get_agent_state(reader)
321  if agent_state is None:
322  return
323  agent = agent_state.get_active_agent()
324  if agent is not None:
325  for partner in agent.message_partners:
326  # We don't know who sent the message that was seen, but we can
327  # send a message observed event to everyone else in the chat
328  self.message_sender.send_read(partner.id)

394  def _on_new_message(self, message):
395  """Put an incoming message onto the correct agent's message queue.
396 
397  :param message:
398  message to put on queue
399  """
400  agent_id = message['sender']['id']
401  if agent_id not in self.messenger_agent_states:
402  self._on_first_message(message)
403  return
404 
405  agent_state = self.get_agent_state(agent_id)
406  if agent_state.get_active_agent() is None:
407  # return agent to overworld
408  if (
409  'text' in message['message']
410  and message['message']['text'].upper() == 'EXIT'
411  ):
412  # remove agent from agent_pool
413  to_remove = []
414  for world_type, _time in agent_state.time_in_pool.items():
415  to_remove.append(world_type)
416  for world_type in to_remove:
417  self.remove_agent_from_pool(
418  agent_state, world_type, mark_removed=False
419  )
420  self.after_agent_removed(agent_state.get_id())
421  agent_state.set_active_agent(agent_state.get_overworld_agent())
422  else:
423  self.observe_message(
424  agent_id,
425  'Please wait while we pair you with another person. '
426  'If you wish to exit, type *EXIT*.',
427  )
428  self.message_sender.typing_on(agent_id)
429  else:
430  # If an agent is in a solo world, we can put a typing indicator
431  # and mark the message as read
432  agent = agent_state.get_active_agent()
433  if len(agent.message_partners) == 0:
434  self.handle_bot_read(agent.id)
435  agent.put_data(message)

597  def check_timeout_in_pool(self, world_type, agent_pool, max_time_in_pool):
598  """Check for timed-out agents in pool.
599 
600  :param world_type:
601  string world type
602  :param agent_pool:
603  list of ``AgentState``s
604  :param max_time_in_pool:
605  int maximum time allowed for agent to be in pool
606  """
607  for agent_state in agent_pool:
608  time_in_pool = agent_state.time_in_pool.get(world_type)
609  if time_in_pool and time.time() - time_in_pool > max_time_in_pool:
610  # remove agent from agent_pool
611  self.remove_agent_from_pool(agent_state, world_type)
612  # put agent back in overworld
613  agent_state.set_active_agent(agent_state.get_overworld_agent())
614 
615  agent_state.stored_data['removed_after_timeout'] = True
616  self.after_agent_removed(agent_state.messenger_id)
617 
618  # reset wait message state
619  agent_state.stored_data['seen_wait_message'] = False
620 
621  elif time_in_pool and time.time() - time_in_pool > 30:
622  # tell agent that a match is taking longer than
623  # expected
624  if (
625  not agent_state.stored_data.get('seen_wait_message')
626  or not agent_state.stored_data['seen_wait_message']
627  ):
628  self.observe_message(
629  agent_state.messenger_id,
630  'Pairing is taking longer than expected. '
631  'If you wish to exit, type *EXIT*.',
632  )
633  self.message_sender.typing_on(agent_state.messenger_id)
634  agent_state.stored_data['seen_wait_message'] = True

644  def _done_callback(fut):
645  """Log and raise exception of task world, if there is one.
646 
647  Additionally, set active agent to overworld agent.
648  """
649  e = fut.exception()
650  if e is not None:
651  shared_utils.print_and_log(
652  logging.ERROR,
653  'World {} had error {}'.format(world_type, repr(e)),
654  should_print=True,
655  )
656  traceback.print_exc(file=sys.stdout)
657  for agent in agents:
658  self.observe_message(
659  agent.id, 'Sorry, this world closed. Returning to overworld.'
660  )
661  else:
662  shared_utils.print_and_log(
663  logging.INFO,
664  'World {} had no error'.format(world_type),
665  should_print=True,
666  )
667  self.active_worlds[task_id] = None
668  for agent in agents:
669  self.after_agent_removed(agent.id)
670  agent_state = self.get_agent_state(agent.id)
671  agent_state.set_active_agent(agent_state.get_overworld_agent())
672 
673  self.running = True
674  while self.running:
