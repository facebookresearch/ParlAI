24  def __init__(
25  self,
26  beam_size,
27  min_length=3,
28  padding_token=0,
29  bos_token=1,
30  eos_token=2,
31  min_n_best=3,
32  cuda='cpu',

34  """Instantiate Beam object.
35 
36  :param beam_size: number of hypothesis in the beam
37  :param min_length: minimum length of the predicted sequence
38  :param padding_token: Set to 0 as usual in ParlAI
39  :param bos_token: Set to 1 as usual in ParlAI
40  :param eos_token: Set to 2 as usual in ParlAI
41  :param min_n_best: Beam will not be done unless this amount of finished
42  hypothesis (with EOS) is done
43  :param cuda: What device to use for computations
44  """
45  self.beam_size = beam_size
46  self.min_length = min_length
47  self.eos = eos_token
48  self.bos = bos_token
49  self.pad = padding_token
50  self.device = cuda
51  # recent score for each hypo in the beam
52  self.scores = torch.Tensor(self.beam_size).float().zero_().to(self.device)
53  # self.scores values per each time step
54  self.all_scores = [torch.Tensor([0.0] * beam_size).to(self.device)]
55  # backtracking id to hypothesis at previous time step
56  self.bookkeep = []
57  # output tokens at each time step
58  self.outputs = [
59  torch.Tensor(self.beam_size).long().fill_(padding_token).to(self.device)
60  ]
61  # keeps tuples (score, time_step, hyp_id)
62  self.finished = []
63  self.HypothesisTail = namedtuple(
64  'HypothesisTail', ['timestep', 'hypid', 'score', 'tokenid']
65  )
66  self.eos_top = False
67  self.eos_top_ts = None
68  self.n_best_counter = 0
69  self.min_n_best = min_n_best
70 
71  def get_output_from_current_step(self):
72  return self.outputs[-1]
73 
74  def get_backtrack_from_current_step(self):
75  return self.bookkeep[-1]

77  def advance(self, softmax_probs):
78  voc_size = softmax_probs.size(-1)
79  if len(self.bookkeep) == 0:
80  # the first step we take only the first hypo into account since all
81  # hypos are the same initially
82  beam_scores = softmax_probs[0]
83  else:
84  # we need to sum up hypo scores and curr softmax scores before topk
85  # [beam_size, voc_size]
86  beam_scores = softmax_probs + self.scores.unsqueeze(1).expand_as(
87  softmax_probs
88  )
89  for i in range(self.outputs[-1].size(0)):
90  # if previous output hypo token had eos
91  # we penalize those word probs to never be chosen
92  if self.outputs[-1][i] == self.eos:
93  # beam_scores[i] is voc_size array for i-th hypo
94  beam_scores[i] = -1e20
95 
96  flatten_beam_scores = beam_scores.view(-1) # [beam_size * voc_size]
97  with torch.no_grad():
98  best_scores, best_idxs = torch.topk(
99  flatten_beam_scores, self.beam_size, dim=-1
100  )
101 
102  self.scores = best_scores
103  self.all_scores.append(self.scores)
104  # get the backtracking hypothesis id as a multiple of full voc_sizes
105  hyp_ids = best_idxs / voc_size
106  # get the actual word id from residual of the same division
107  tok_ids = best_idxs % voc_size
108 
109  self.outputs.append(tok_ids)
110  self.bookkeep.append(hyp_ids)
111 
112  # check new hypos for eos label, if we have some, add to finished
113  for hypid in range(self.beam_size):
114  if self.outputs[-1][hypid] == self.eos:
115  # this is finished hypo, adding to finished
116  eostail = self.HypothesisTail(
117  timestep=len(self.outputs) - 1,
118  hypid=hypid,
119  score=self.scores[hypid],
120  tokenid=self.eos,
121  )
122  self.finished.append(eostail)
123  self.n_best_counter += 1
124 
125  if self.outputs[-1][0] == self.eos:
126  self.eos_top = True
127  if self.eos_top_ts is None:
128  self.eos_top_ts = len(self.outputs) - 1
129 
130  def done(self):
131  return self.eos_top and self.n_best_counter >= self.min_n_best

144  def get_hyp_from_finished(self, hypothesis_tail):
145  """Extract hypothesis ending with EOS at timestep with hyp_id.
146 
147  :param timestep: timestep with range up to len(self.outputs)-1
148  :param hyp_id: id with range up to beam_size-1
149  :return: hypothesis sequence
150  """
151  assert self.outputs[hypothesis_tail.timestep][hypothesis_tail.hypid] == self.eos
152  assert hypothesis_tail.tokenid == self.eos
153  hyp_idx = []
154  endback = hypothesis_tail.hypid
155  for i in range(hypothesis_tail.timestep, -1, -1):
156  hyp_idx.append(

159  hypid=endback,
160  score=self.all_scores[i][endback],
161  tokenid=self.outputs[i][endback],
162  )
163  )
164  endback = self.bookkeep[i - 1][endback]
165 
166  return hyp_idx

177  def get_rescored_finished(self, n_best=None):
178  """
179 
180  :param n_best: how many n best hypothesis to return
181  :return: list with hypothesis
182  """
183  rescored_finished = []
184  for finished_item in self.finished:
185  current_length = finished_item.timestep + 1
186  # these weights are from Google NMT paper
187  length_penalty = math.pow((1 + current_length) / 6, 0.65)
188  rescored_finished.append(
189  self.HypothesisTail(
190  timestep=finished_item.timestep,
191  hypid=finished_item.hypid,
192  score=finished_item.score / length_penalty,
193  tokenid=finished_item.tokenid,
194  )
195  )
196 
197  srted = sorted(rescored_finished, key=attrgetter('score'), reverse=True)
198 
199  if n_best is not None:
200  srted = srted[:n_best]
201 
202  return srted

225  def get_beam_dot(self, dictionary=None, n_best=None):
226  """Creates pydot graph representation of the beam.
227 
228  :param outputs: self.outputs from the beam
229  :param dictionary: tok 2 word dict to save words in the tree nodes
230  :returns: pydot graph
231  """
232  try:
233  import pydot
234  except ImportError:
235  print("Please install pydot package to dump beam visualization")
236 
237  graph = pydot.Dot(graph_type='digraph')
238  outputs = [i.tolist() for i in self.outputs]
239  bookkeep = [i.tolist() for i in self.bookkeep]
240  all_scores = [i.tolist() for i in self.all_scores]
241  if n_best is None:
242  n_best = int(self.beam_size / 2)
243 
244  # get top nbest hyp
245  top_hyp_idx_n_best = []
246  n_best_colors = ['aquamarine', 'chocolate1', 'deepskyblue', 'green2', 'tan']
247  sorted_finished = self.get_rescored_finished(n_best=n_best)
248  for hyptail in sorted_finished:
249  # do not include EOS since it has rescored score not from original
250  # self.all_scores, we color EOS with black
251  top_hyp_idx_n_best.append(self.get_hyp_from_finished(hyptail))
252 
253  # create nodes
254  for tstep, lis in enumerate(outputs):
255  for hypid, token in enumerate(lis):
256  if tstep == 0:
257  hypid = 0 # collapse all __NULL__ nodes
258  node_tail = self.HypothesisTail(
259  timestep=tstep,
260  hypid=hypid,
261  score=all_scores[tstep][hypid],
262  tokenid=token,
263  )
264  color = 'white'
265  rank = None
266  for i, hypseq in enumerate(top_hyp_idx_n_best):
267  if node_tail in hypseq:
268  if n_best <= 5: # color nodes only if <=5
269  color = n_best_colors[i]
270  rank = i
271  break
272  label = (
273  "<{}".format(
274  dictionary.vec2txt([token]) if dictionary is not None else token
275  )
276  + " : "
277  + "{:.{prec}f}>".format(all_scores[tstep][hypid], prec=3)
278  )
279 
280  graph.add_node(
281  pydot.Node(
282  node_tail.__repr__(),
283  label=label,
284  fillcolor=color,
285  style='filled',
286  xlabel='{}'.format(rank) if rank is not None else '',
287  )
288  )
289 
290  # create edges
291  for revtstep, lis in reversed(list(enumerate(bookkeep))):
292  for i, prev_id in enumerate(lis):
293  from_node = graph.get_node(
294  '"{}"'.format(
295  self.HypothesisTail(
296  timestep=revtstep,
297  hypid=prev_id,
298  score=all_scores[revtstep][prev_id],
299  tokenid=outputs[revtstep][prev_id],
300  ).__repr__()
301  )
302  )[0]
303  to_node = graph.get_node(
304  '"{}"'.format(
305  self.HypothesisTail(
306  timestep=revtstep + 1,
307  hypid=i,
308  score=all_scores[revtstep + 1][i],
309  tokenid=outputs[revtstep + 1][i],
310  ).__repr__()
311  )
312  )[0]
313  newedge = pydot.Edge(from_node.get_name(), to_node.get_name())
314  graph.add_edge(newedge)
315 
316  return graph

383 def round_sigfigs(x, sigfigs=4):
384  try:
385  if x == 0:
386  return 0
387  return round(x, -math.floor(math.log10(abs(x)) - sigfigs + 1))
388  except (RuntimeError, TypeError):
389  # handle 1D torch tensors
390  # if anything else breaks here please file an issue on Github
391  if hasattr(x, 'item'):
392  return round_sigfigs(x.item(), sigfigs)
393  else:
394  return round_sigfigs(x[0], sigfigs)
395  except (ValueError, OverflowError) as ex:
396  if x in [float('inf'), float('-inf')] or x != x: # inf or nan
397  return x
398  else:
399  raise ex

407  @classmethod
408  def pad_text(
409  cls,
410  observations,
411  dictionary,
412  end_idx=None,
413  null_idx=0,
414  dq=False,
415  eval_labels=True,
416  truncate=None,
417  ):
418  """We check that examples are valid, pad with zeros, and sort by length
419  so that we can use the pack_padded function. The list valid_inds
420  keeps track of which indices are valid and the order in which we sort
421  the examples.
422  dq -- whether we should use deque or list
423  eval_labels -- whether or not we want to consider eval labels
424  truncate -- truncate input and output lengths
425  """
426 
427  def valid(obs):
428  # check if this is an example our model should actually process
429  return 'text' in obs and len(obs['text']) > 0
430 
431  try:
432  # valid examples and their indices
433  valid_inds, exs = zip(
434  *[(i, ex) for i, ex in enumerate(observations) if valid(ex)]
435  )
436  except ValueError:
437  # zero examples to process in this batch, so zip failed to unpack
438  return None, None, None, None, None, None
439 
440  # `x` text is already tokenized and truncated
441  # sort by length so we can use pack_padded
442  if any(['text2vec' in ex for ex in exs]):
443  parsed_x = [ex['text2vec'] for ex in exs]
444  else:
445  parsed_x = [dictionary.txt2vec(ex['text']) for ex in exs]
446 
447  if len(parsed_x) > 0 and not isinstance(parsed_x[0], deque):
448  if dq:
449  parsed_x = [deque(x, maxlen=truncate) for x in parsed_x]
450  elif truncate is not None and truncate > 0:
451  parsed_x = [x[-truncate:] for x in parsed_x]
452 
453  x_lens = [len(x) for x in parsed_x]
454  ind_sorted = sorted(range(len(x_lens)), key=lambda k: -x_lens[k])
455 
456  exs = [exs[k] for k in ind_sorted]
457  valid_inds = [valid_inds[k] for k in ind_sorted]
458  parsed_x = [parsed_x[k] for k in ind_sorted]
459  end_idxs = [x_lens[k] for k in ind_sorted]
460 
461  eval_labels_avail = any(['eval_labels' in ex for ex in exs])
462  labels_avail = any(['labels' in ex for ex in exs])
463  if eval_labels:
464  some_labels_avail = eval_labels_avail or labels_avail
465  else:
466  some_labels_avail = labels_avail
467 
468  max_x_len = max(x_lens)
469 
470  # pad with zeros
471  if dq:
472  parsed_x = [
473  x
474  if len(x) == max_x_len
475  else x + deque((null_idx,)) * (max_x_len - len(x))
476  for x in parsed_x
477  ]
478  else:
479  parsed_x = [
480  x if len(x) == max_x_len else x + [null_idx] * (max_x_len - len(x))
481  for x in parsed_x
482  ]
483  xs = parsed_x
484 
485  # set up the target tensors
486  ys = None
487  labels = None
488  y_lens = None
489  if some_labels_avail:
490  # randomly select one of the labels to update on (if multiple)
491  if labels_avail:
492  labels = [random.choice(ex.get('labels', [''])) for ex in exs]
493  else:
494  labels = [random.choice(ex.get('eval_labels', [''])) for ex in exs]
495  # parse each label and append END
496  if dq:
497  parsed_y = [deque(maxlen=truncate) for _ in labels]
498  for deq, y in zip(parsed_y, labels):
499  deq.extendleft(reversed(dictionary.txt2vec(y)))
500  else:
501  parsed_y = [dictionary.txt2vec(label) for label in labels]
502  if end_idx is not None:
503  for y in parsed_y:
504  y.append(end_idx)
505 
506  y_lens = [len(y) for y in parsed_y]
507  max_y_len = max(y_lens)
508 
509  if dq:
510  parsed_y = [
511  y
512  if len(y) == max_y_len
513  else y + deque((null_idx,)) * (max_y_len - len(y))
514  for y in parsed_y
515  ]
516  else:
517  parsed_y = [
518  y if len(y) == max_y_len else y + [null_idx] * (max_y_len - len(y))
519  for y in parsed_y
520  ]
521  ys = parsed_y
522 
523  return xs, ys, labels, valid_inds, end_idxs, y_lens

576 class SharedTable(MutableMapping):
577  """Provides a simple shared-memory table of integers, floats, or strings.
578  Use this class as follows:
579 
580  .. code-block:: python
581 
582  tbl = SharedTable({'cnt': 0})
583  with tbl.get_lock():
584  tbl['startTime'] = time.time()
585  for i in range(10):
586  with tbl.get_lock():
587  tbl['cnt'] += 1
588  """
589 
590  types = {int: ctypes.c_int, float: ctypes.c_float, bool: ctypes.c_bool}
591 
592  def __init__(self, init_dict=None):
593  """Create a shared memory version of each element of the initial
594  dictionary. Creates an empty array otherwise, which will extend
595  automatically when keys are added.
596 
597  Each different type (all supported types listed in the ``types`` array
598  above) has its own array. For each key we store an index into the
599  appropriate array as well as the type of value stored for that key.
600  """
601  # idx is dict of {key: (array_idx, value_type)}
602  self.idx = {}
603  # arrays is dict of {value_type: array_of_ctype}
604  self.arrays = {}
605  self.tensors = {}
606 
607  if init_dict:
608  sizes = {typ: 0 for typ in self.types.keys()}
609  for k, v in init_dict.items():
610  if is_tensor(v):
611  # add tensor to tensor dict--don't try to put in rawarray
612  self.tensors[k] = v
613  continue
614  elif type(v) not in sizes:
615  raise TypeError(
616  'SharedTable does not support values of '
617  + 'type '
618  + str(type(v))
619  )
620  sizes[type(v)] += 1
621  # pop tensors from init_dict
622  for k in self.tensors.keys():
623  init_dict.pop(k)
624  # create raw arrays for each type
625  for typ, sz in sizes.items():
626  self.arrays[typ] = RawArray(self.types[typ], sz)
627  # track indices for each key, assign them to their typed rawarray
628  idxs = {typ: 0 for typ in self.types.keys()}
629  for k, v in init_dict.items():
630  val_type = type(v)
631  self.idx[k] = (idxs[val_type], val_type)
632  if val_type == str:
633  v = sys.intern(v)
634  self.arrays[val_type][idxs[val_type]] = v
635  idxs[val_type] += 1
636  # initialize any needed empty arrays
637  for typ, ctyp in self.types.items():
638  if typ not in self.arrays:
639  self.arrays[typ] = RawArray(ctyp, 0)
640  self.lock = Lock()
641 
642  def __len__(self):
643  return len(self.idx) + len(self.tensors)
644 
645  def __iter__(self):
646  return iter([k for k in self.idx] + [k for k in self.tensors])
647 
648  def __contains__(self, key):
649  return key in self.idx or key in self.tensors
650 
651  def __getitem__(self, key):
652  """Returns shared value if key is available."""
653  if key in self.tensors:
654  return self.tensors[key]
655  elif key in self.idx:
656  idx, typ = self.idx[key]
657  return self.arrays[typ][idx]
658  else:
659  raise KeyError('Key "{}" not found in SharedTable'.format(key))
660 
661  def __setitem__(self, key, value):
662  """If key is in table, update it. Otherwise, extend the array to make
663  room. This uses additive resizing not multiplicative, since the number
664  of keys is not likely to change frequently during a run, so do not abuse
665  it.
666  Raises an error if you try to change the type of the value stored for
667  that key--if you need to do this, you must delete the key first.
668  """
669  val_type = type(value)
670  if 'Tensor' in str(val_type):
671  self.tensors[key] = value
672  return
