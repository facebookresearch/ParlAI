349  def act(self, timeout=None, blocking=True):
350  """Sends a message to other agents in the world. If blocking, this
351  will wait for the message to come in so it can be sent. Otherwise
352  it will return None.
353  """
354  if not blocking:
355  # if this is the first act since last sent message start timing
356  if self.message_request_time is None:
357  self.request_message()
358  self.message_request_time = time.time()
359 
360  # If checking timeouts
361  if timeout:
362  # If time is exceeded, timeout
363  if time.time() - self.message_request_time > timeout:
364  raise AgentTimeoutError(timeout, self.worker_id, self.assignment_id)
365 
366  # Get a new message, if it's not None reset the timeout
367  msg = self.get_new_act_message()
368  if msg is not None and self.message_request_time is not None:
369  self.message_request_time = None
370  return msg
371  else:
372  self.request_message()
373  self.message_request_time = time.time()
374 
375  # Timeout in seconds, after which the HIT is expired automatically
376  if timeout:
377  start_time = time.time()
378 
379  # Wait for agent's new message
380  while True:
381  msg = self.get_new_act_message()
382  self.message_request_time = None
383  if msg is not None:
384  return msg
385 
386  # Check if the Turker waited too long to respond
387  if timeout:
388  current_time = time.time()
389  if (current_time - start_time) > timeout:
390  self.message_request_time = None
391  raise AgentTimeoutError(
392  timeout, self.worker_id, self.assignment_id
393  )
394  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
