107  def forward(self, lengths, indices):

109  if lengths.dim() == 1 or lengths.size(1) == 1:
110  lengths_mat = lengths_mat.unsqueeze(0)
111 
112  if lengths_mat.dim() == 1:
113  raise RuntimeError(lengths.shape)
114 
115  input = torch.LongTensor(
116  lengths_mat.size(0), lengths_mat.size(1), torch.max(lengths_mat)
117  )
118  pad = self.padding_idx if self.padding_idx is not None else 0
119  input.fill_(pad)
120  emb_list = []
121  offset = 0
122  for i, row in enumerate(lengths_mat):
123  for j, length in enumerate(row):
124  length = length.item()

127  offset += length

129  for i, row in enumerate(lengths_mat):

131  if self.position_encoding:

147  @staticmethod
148  @lru_cache(maxsize=32)
149  def position_matrix(J, d):
150  m = torch.Tensor(J, d)
151  for k in range(1, d + 1):
152  for j in range(1, J + 1):
153  m[j - 1, k - 1] = (1 - j / J) - (k / d) * (1 - 2 * j / J)
154  return m

167  def __init__(self, embedding_size):
168  super(Hop, self).__init__()

173  self,
174  query_embeddings,
175  in_memory_embeddings,
176  out_memory_embeddings,
177  attention_mask=None,
178  ):
179  attention = torch.bmm(
180  in_memory_embeddings, query_embeddings.unsqueeze(2)
181  ).squeeze(2)
182  if attention_mask is not None:
183  # exclude masked elements from the softmax
184  attention = (
185  attention_mask.float() * attention
186  + (1 - attention_mask.float()) * -1e20
187  )
188  probs = softmax(attention, dim=1).unsqueeze(1)
189  memory_output = torch.bmm(probs, out_memory_embeddings).squeeze(1)
190  query_embeddings = self.linear(query_embeddings)
