72 def _fairseq_opt_wrapper(opt, skip_pretrained_embedding_loading=False):
73  """
74  Marshalls from a dict to a argparse.Namespace object for API compatibility.
75 
76  Also does some necessary post-processing needed for fairseq. Optionally can
77  override pretrained embedding options, which is useful if we're just loading
78  a model from a checkpoint.
79 
80  :param opt: dict. ParlAI options passed around from everywhere.
81  :param skip_pretrained_embedding_loading: bool. Don't preload word embeddings.
82  :return: an argparse.Namespace object for use in fairseq-py.
83  """
84  args = argparse.Namespace()
85 
86  # first set args according to ParlAI options
87  for key in opt:
88  if opt[key] is not None:
89  setattr(args, key, opt[key])
90 
91  # at this point the user *must* have specified an arch
92  if not hasattr(args, "arch"):
93  raise ValueError("--arch/-a must be specified")
94  # fill in default options from the model
95  models.ARCH_CONFIG_REGISTRY[args.arch](args)
96 
97  # post processing of args. See
98  # https://github.com/pytorch/fairseq/blob/v0.5.0/fairseq/options.py#L95
99  if hasattr(args, "lr"):
100  args.lr = options.eval_str_list(args.lr, type=float)
101  if hasattr(args, "update_freq"):
102  args.update_freq = options.eval_str_list(args.update_freq, int)
103  if hasattr(args, "max_sentences_valid"):
104  args.max_sentences_valid = args.max_sentences
105  if args.truncate == -1:
106  # some torch agents use positional embeddings, which must have a max length
107  args.truncate = 1024
108  if not hasattr(args, "max_source_positions"):
109  # fairseq uses a different name for this CLI parameter
110  # Sometimes it's set in model defaults, but not for all models
111  args.max_source_positions = args.truncate
112  # if we don't have source lengths, we don't have target lengths
113  args.max_target_positions = args.truncate
114 
115  # handle modelzoo if possible
116  for k in ("encoder_embed_path", "decoder_embed_path"):
117  if getattr(args, k, None) is None:
118  # not an argument for this model, pretrained embeddings don't matter
119  continue
120  elif skip_pretrained_embedding_loading:
121  # if we want to skip pretrained, then hide the option from fairseq
122  setattr(args, k, None)
123  else:
124  # otherwise we may need to modelzoo adjust the path for fairseq
125  import warnings
126 
127  warnings.warn("We recommend using --embedding-type instead")
128  setattr(args, k, modelzoo_path(opt.get("datapath"), getattr(args, k)))
129 
130  # Here we hardcode a few options that we currently do not support
131  # turn off distributed training
132  args.distributed_world_size = 1
133  args.distributed_rank = 0
134 
135  return args, vars(args)

138 class _FairseqDictionary(DictionaryAgent):
139  """
140  Skeleton dictionary class needed for interaction with fairseq-py.
141 
142  This class mostly just adds some basic API behavior that Fairseq internally
143  expects from dictionaries.
144 
145  It also inserts a fake token at the 0th index of the dictionary, as
146  fairseq-py maintains backwards compatibility with fairseq-lua, which uses
147  1 indexing.
148  """
149 
150  # Name of our fake lua compatibility token
151  _LUA = '__LUACOMPAT__'
152 
153  def __init__(self, *args, **kwargs):
154  super().__init__(*args, **kwargs)
155  # insert the fairseq-lua compatibility token to emulate 1-indexing.
156  # This 1-indexing assumption is baked into a couple of places in fairseq-py,
157  # and is unavoidable at the moment.
158  #
159  # Because of the structure of DictionaryAgent, it's difficult to force
160  # a token in the 0th position without breaking load()ing. I've found
161  # this to be the best way.
162

164  self.add_token(_FairseqDictionary._LUA)
165  # force it to be the "most frequent" token
166  self.freq[_FairseqDictionary._LUA] = self.freq[self.null_token] + 1
167  # sort the list to ensure the lua token is placed first. trim=False to
168  # ensure shuffle is non-destructive.
169  self.sort(trim=False)

492  def _update_metrics(self, metrics, sample):
493  if metrics is None:
494  # probably got an overflow in fp16 mode. don't count this sample
495  return
496 
497  bsz = len(sample['target'])
498  ntok = sample['ntokens']
499  ssize = metrics['sample_size']
500 
501  for k, v in metrics.items():
502  if k in {'ntokens', 'nsentences', 'sample_size'}:
503  # don't need these
504  continue
505  elif k == "nll_loss":
506  # nll loss is always normalized by ntokens
507  self.meters[k].update(v, ntok)
508  elif k == "loss":
509  # loss is explicitly normalized by passed up sample size
510  self.meters[k].update(v, ssize)
511  else:

513  self.meters[k].update(v, bsz)

529  def eval_step(self, batch):
530  """Process batch of inputs.
531 
532  If the batch includes labels, calculate validation metrics as well.
533  If --skip-generation is not set, return a prediction for each input.
534 
535  :param batch: parlai.core.torch_agent.Batch, contains tensorized
536  version of observations.
537  """
538  if batch.text_vec is None:
539  return
540  self.is_training = False
541  samples = self._make_sample(batch)
542  self.model.eval()
543  if batch.label_vec is not None and self.trainer is not None:
544  # Interactive mode won't have a gold label
545  metrics = self.trainer.valid_step(samples)
546  self._update_metrics(metrics, samples)
547 
548  # Output placeholders
549  reranked_cands = None
550  generated_output = None
551 
552  # Grade each of the candidate sequences
553  if batch.candidate_vecs is not None:
554  bsz = len(batch.text_vec)
555  reranked_cands = []
556  # score the candidates for each item in the batch separately, so that
557  # we can support variable number of candidates
558  for i in range(bsz):
559  cands = batch.candidate_vecs[i]
560  if not cands:
561  reranked_cands.append(None)
562  continue
563  ncand = len(cands)
564  # repeat the input many times
565  xs = batch.text_vec[i].unsqueeze(0).expand(ncand, -1)
566  # some models crash if there's leading padding on every example
567  xs = xs[:, : batch.text_lengths[i]]
568  # and appropriately pack the outputs
569  ys, _ = padded_tensor(cands, self.NULL_IDX, self.use_cuda)
570  s = self._make_sample(xs=xs, ys=ys)
571  # perform the actual grading, extract the scores
572  scored = list(self.scorer.score_batched_itr([s], cuda=self.use_cuda))
573  scores = [s[3][0]['score'].item() for s in scored]
574  # intentional hanging comma here; argsort returns a list
575  ranked, = argsort(scores, batch.candidates[i], descending=True)
576  reranked_cands.append(ranked)
577 
578  # Next generate freely to create our response
579  if not self.args.skip_generation:
580  generated_output = self._generate(samples)
581  elif reranked_cands:
582  # we're skiping generation, but we're also grading candidates
583  # so output the highest ranked candidate
584  # In the case of zero candidates, we don't have something to rank,
585  # so we may need to pass on that None
586  generated_output = [
587  ranked and ranked[0] or None for ranked in reranked_cands
588  ]
589  else:
590  # no output at all
591  pass
592 
593  return Output(generated_output, reranked_cands)

595  def _generate(self, samples):
596  no_prev_token = {
597  k: v for k, v in samples['net_input'].items() if k != 'prev_output_tokens'
598  }
599  gens = self.generator.generate(no_prev_token, maxlen=64)
600  bsz = samples['net_input']['src_tokens'].size(0)
601  responses = []
602  for i in range(bsz):
603  beams = gens[i]
604  selected = max(beams, key=lambda x: x["score"])
605  tokens = selected["tokens"]
606  start = 0
607  end = -1
608  for i, t in enumerate(tokens):
609  t = t.item()
610  if t == self.dict.bos_index:
611  # don't include <s> token
612  start = i + 1
613  continue
614  if t == self.dict.eos_index:
615  # stop (and don't include) </s> token
616  end = i
617  break
618  responses.append(self.dict.vec2txt(tokens[start:end]))
619  return responses

621  def report(self):
622  """Return metrics calculated by the model."""
623  # if we haven't initialized yet, just return a dummy object
624  if not hasattr(self, "trainer"):
625  return {}
626 
627  output = {k: v.avg for k, v in self.meters.items()}
628 
629  if "nll_loss" in self.meters:
630  # special case, we used sentence averaging so ppl comes from nll_loss
631  output["ppl"] = np.exp2(self.meters["nll_loss"].avg)
632  else:
633  # normal case, just use loss
634  output["ppl"] = np.exp2(self.meters["loss"].avg)
635 
636  # Fairseq trainer metrics we'll pass up the way
637  trainer_metrics = {"ups", "wps", "gnorm", "clip"}
638  if self.is_training:
639  for k in trainer_metrics:
640  output[k] = self.trainer.meters[k].avg
641 
642  # for display purposes
643  output = {k: round_sigfigs(v, 4) for k, v in output.items()}
644  return output
