141  obs = super()._set_text_vec(*args, **kwargs)

154  def score(self, batch):
155  """Score the batch."""
156  segment_idx = (batch.text_vec * 0).long()
157  if self.sep_last_utt:
158  batch_len = batch.text_vec.size(1)
159  # find where [SEP] token is
160  seps = (batch.text_vec == self.dict.end_idx).nonzero()
161  if len(seps) > 0:
162  for row in seps:
163  # set last utterance to segment 1
164  segment_idx[row[0], list(range(row[1], batch_len))] = 1
165  else:
166  # only one utterance: everything after [CLS] token
167  # should be segment 1
168  segment_idx = (batch.text_vec != self.dict.start_idx).long()
169  mask = (batch.text_vec != self.NULL_IDX).long()
170  token_idx = batch.text_vec * mask
171  return self.model(token_idx, segment_idx, mask)
