213  def parley(self):
214  """RESPONDER is given an image and a comment, and is told to give a
215  response for to the comment"""
216  # Initial Message Value
217  control_msg = {'episode_done': False}
218  control_msg['id'] = 'SYSTEM'
219 
220  '''We only have to worry about 1 agent'''
221  agent = self.agents[0]
222 
223  '''First, we give RESPONDER their personality instructions, and image
224  '''
225  while self.turn_idx < self.num_images:
226  print(self.world_tag + ' is at turn {}...'.format(self.turn_idx))
227  # Send personality + image + comment to turker
228 
229  self.example_num, example = self.agent.example_generator.pop_example()
230  control_msg['text'] = self.get_instruction(
231  tag='start', agent_id=agent.id, turn_num=self.turn_idx + 1
232  )
233 
234  if self.second_resp:
235  control_msg['comment_text'] = (
236  '<b><span style="color:red">'
237  '{}\n</span></b>'.format(example['dialog'][0][1].strip())
238  )
239  control_msg['response_text'] = (
240  '<b><span style="color:blue">'
241  '{}\n</span></b>'.format(example['dialog'][1][1].strip())
242  )
243  else:
244  control_msg['comment_text'] = (
245  '<b><span style="color:red">'
246  '{}\n</span></b>'.format(example['comment'].strip())
247  )
248 
249  img = load_image(
250  os.path.join(self.image_path, '{}.jpg'.format(example['image_hash']))
251  )
252  buffered = BytesIO()
253  img.save(buffered, format='JPEG')
254  encoded = str(base64.b64encode(buffered.getvalue()).decode('ascii'))
255  control_msg['image'] = encoded
256  if self.second_resp:
257  self.pers_idx, personality = (-1, example['dialog'][0][0])
258  else:
259  pers_tup = self.agent.personality_generator.pop_personality()
260  self.pers_idx, personality = pers_tup
261  personality_text = '<b><span style="color:{}">' '{}\n</span></b>'.format(
262  'blue' if not self.second_resp else 'red', personality.strip()
263  )
264  control_msg['personality_text'] = personality_text
265  control_msg['description'] = self.config['task_description']
266  agent.observe(validate(control_msg))
267  time.sleep(1)
268 
269  # Collect comment from turker
270  offensive_counter = 0
271  while offensive_counter < 3:
272  idx = 0
273  acts = self.acts
274  acts[idx] = agent.act(timeout=self.max_resp_time)
275  agent_left = self.check_timeout(acts[idx])
276  if agent_left:
277  break
278  response = acts[idx]['text']
279  offensive = self.offensive_lang_detector.contains_offensive_language(
280  response
281  )
282  if offensive:
283  # Tell Turker to not be offensive!
284  offensive_counter += 1
285  if offensive_counter == 3:
286  break
287  offensive_msg = {'id': 'SYSTEM', 'text': OFFENSIVE_MSG}
288  agent.observe(validate(offensive_msg))
289  else:
290  break
291 
292  if self.chat_done:
293  break
294  ex_to_save = example.copy()
295  key = 'second_response' if self.second_resp else 'first_response'
296  ex_to_save[key] = response
297  ex_to_save['{}_personality'.format(key)] = personality
298  ex_to_save['contains_offensive_language'] = offensive
299  self.data.append(ex_to_save)
300  self.turn_idx += 1
301 
302  if self.turn_idx == self.num_images:
303  control_msg['text'] = CHAT_ENDED_MSG.format(self.num_images)
304  agent.observe(validate(control_msg))
305  self.chat_done = True
306  return
