77  def get_prediction(self):
78  """
79  Core algorithm, which gathers the most common unigrams into a string.
80  """
81  # we always make the same prediction, so cache it for speed
82  if self.unigram_cache is None:
83  most_common = sorted(
84  self.dict.freq.items(), key=lambda x: x[1], reverse=True
85  )
86  most_common = ((u, v) for u, v in most_common if self.is_valid_word(u))
87  most_common = islice(most_common, self.num_words)
88  most_common = (u for u, v in most_common)
89  self.unigram_cache = ' '.join(list(most_common))
90  return self.unigram_cache
