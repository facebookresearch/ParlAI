347  def is_exact_match(self, act, ag, tolerance=0):
348  if act['episode_done']:
349  return False
350 
351  control_msg = {'episode_done': False}
352  control_msg['id'] = 'SYSTEM'
353 
354  text = act['text']
355  if text not in ['', ' ', ' ', ' ']:
356  n_word_match = 0
357  for per in ag.persona_data:
358  per_parse = per.split(' ')
359  regular_words = ['', ' ', 'I', 'I\'m', 'My', 'i']
360  for r_w in regular_words:
361  if r_w in per_parse:
362  per_parse.remove(r_w)
363  per_subseq = [
364  ' '.join(per_parse[i : i + len(per_parse) - tolerance])
365  for i in range(tolerance + 1)
366  ]
367  for pp in per_subseq:
368  if pp in ['', ' ', ' ', ' ']:
369  per_subseq.remove(pp)
370  n_word_match += sum([(paa in text) for paa in per_subseq])
371  if n_word_match > 0:
372  control_msg['text'] = (
373  'We found that you <b><span style="color:red">trivially '
374  'copied character descriptions</span></b>. Please '
375  'rephrase your message again.'
376  )
377  ag.observe(validate(control_msg))
378  return True
379  else:
380  return False
