119  def compute_loss(self, batch, return_output=False):
120  # first compute our regular forced decoding loss
121  token_loss, model_output = super().compute_loss(batch, return_output=True)
122  notnull = batch.label_vec.ne(self.NULL_IDX)
123  num_tokens = notnull.long().sum().item()
124 
125  encoder_states = model_output[2]
126  ctx_know_attn = encoder_states[2]
127 
128  if self.knowledge_alpha == 0.0:
129  loss = token_loss
130  else:
131  _, know_pred = ctx_know_attn.max(1)
132  know_acc = (know_pred == batch.cs_ids).float().sum().item()
133  know_chance = batch.ck_mask.sum(1).float().reciprocal().sum().item()
134  self.metrics['know_chance'] += know_chance
135  self.metrics['bsz'] += batch.text_vec.size(0)
136  self.metrics['know_acc'] += know_acc
137  know_loss = th.nn.functional.cross_entropy(
138  ctx_know_attn, batch.cs_ids, reduction='mean'
139  )
140  self.metrics['know_loss'] += know_loss.item() * batch.text_vec.size(0)
141  # in the original paper the loss was scaled by num_tokens for both
142  # know_loss and token_loss
143  know_loss /= num_tokens
144  loss = (
145  1 - self.knowledge_alpha
146  ) * token_loss + self.knowledge_alpha * know_loss
147  if return_output:
148  return (loss, model_output)
149  else:
150  return loss

180  checked_sentence = '{} {} {}'.format(

184  obs_know = [k.strip() for k in obs.get('knowledge', '').split('\n')]

202  def batchify(self, obs_batch):
203  """
204  Wizard custom batchify, which passes along the knowledge/title.
205 
206  Following the docstring of TorchAgent.batchify, it calls super, then
207  uses an extended version of the torch_agent.Batch namedtuple.
208 
209  The purpose of extending the info is to keep track of some custom
210  metrics.
211  """
212  batch = super().batchify(obs_batch)
213  reordered_observations = [obs_batch[i] for i in batch.valid_indices]
214  is_training = 'labels' in reordered_observations[0]
215 
216  # first parse and compile all the knowledge together
217  all_knowledges = [] # list-of-lists knowledge items for each observation
218  knowledge_counts = [] # how much knowledge each observation gets
219  for obs in reordered_observations:
220  obs_know = self._parse_knowledge(obs)
221  # downsample if desired
222  if (
223  is_training
224  and self.max_knowledge
225  and len(obs_know) > self.max_knowledge
226  ):
227  # offset by one so that we don't choose 0
228  keepers = 1 + np.random.choice(
229  len(obs_know) - 1, self.max_knowledge, False
230  )
231  # correct answer is always the first one
232  keepers[0] = 0
233  obs_know = [obs_know[i] for i in keepers]
234  all_knowledges.append(obs_know)
235  knowledge_counts.append(len(obs_know))
236 
237  # now we want to actually pack this into a tensor, along with the mask
238  N = len(reordered_observations)
239  K = max(knowledge_counts)
240  # round out the array so everything is equally sized
241  for i in range(N):
242  all_knowledges[i] += [''] * (K - knowledge_counts[i])
243  flattened_knowledge = list(chain(*all_knowledges))
244 
245  knowledge_vec = [
246  self._vectorize_text(
247  # the beginning of the sentence is more useful
248  k,
249  truncate=self.knowledge_truncate,
250  add_end=True,
251  truncate_left=False,
252  )
253  for k in flattened_knowledge
254  ]
255  knowledge_vec, _ = padded_tensor(
256  knowledge_vec, self.NULL_IDX, self.use_cuda, left_padded=True
257  )
258  knowledge_vec[:, -1] = self.END_IDX
259  T = knowledge_vec.size(-1)
260  knowledge_vec = knowledge_vec.view(N, K, T)
261 
262  # knowledge mask is a N x K tensor saying which items we're allowed to
263  # attend over
264  bsz = len(reordered_observations)
265  ck_mask = th.zeros(bsz, K, dtype=th.uint8)
266  for i, klen in enumerate(knowledge_counts):
267  ck_mask[i, :klen] = 1
268  ck_mask = ck_mask != 0 # for pytorch 1.0/1.2 uint8/bool compatibility
269  # and the correct labels
270  cs_ids = th.LongTensor(bsz).zero_()
271 
272  if self.use_cuda:
273  knowledge_vec = knowledge_vec.cuda()
274  ck_mask = ck_mask.cuda()
275  cs_ids = cs_ids.cuda()
276 
277  batch['know_vec'] = knowledge_vec
278  batch['ck_mask'] = ck_mask
279  batch['cs_ids'] = cs_ids
280  batch['use_cs_ids'] = is_training
281  batch['knowledge'] = np.array(flattened_knowledge).reshape(N, K)
282  return batch
