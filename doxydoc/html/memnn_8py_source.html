197  def _build_mems(self, mems):
198  """
199  Build memory tensors.
200 
201  During building, will add time features to the memories if enabled.
202 
203  :param mems:
204  list of length batchsize containing inner lists of 1D tensors
205  containing the individual memories for each row in the batch.
206 
207  :returns:
208  3d padded tensor of memories (bsz x num_mems x seqlen)
209  """
210  if mems is None:
211  return None
212  bsz = len(mems)
213  if bsz == 0:
214  return None
215 
216  num_mems = max(len(mem) for mem in mems)
217  if num_mems == 0 or self.memsize <= 0:
218  return None
219  elif num_mems > self.memsize:
220  # truncate to memsize most recent memories
221  num_mems = self.memsize
222  mems = [mem[-self.memsize :] for mem in mems]
223 
224  try:
225  seqlen = max(len(m) for mem in mems for m in mem)
226  if self.use_time_features:
227  seqlen += 1 # add time token to each sequence
228  except ValueError:
229  return None
230 
231  padded = torch.LongTensor(bsz, num_mems, seqlen).fill_(0)
232 
233  for i, mem in enumerate(mems):
234  tf_offset = len(mem) - 1
235  for j, m in enumerate(mem):
236  padded[i, j, : len(m)] = m
237  if self.use_time_features:
238  padded[i, j, -1] = self.dict[self._time_feature(tf_offset - j)]
239 
240  if self.use_cuda:
241  padded = padded.cuda()
242 
243  return padded
