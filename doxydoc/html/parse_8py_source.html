55 def cleanup_text(text):
56  text = text.strip()
57 
58  # Prefer non-unicode special character
59  SWITCH_LIST = [
60  ("\u2019", "'"),
61  ("\u2018", "'"),
62  ("\u201d", '"'),
63  ("\u201c", '"'),
64  ("\u2014", "--"),
65  ("\u2013", "--"),
66  ("\u3002", ". "),
67  ("\u2032", "'"),
68  ("\u3001", ", "),
69  ]
70  for before, after in SWITCH_LIST:
71  text = text.replace(before, after)
72 
73  # fix some broken sentence tokenization
74  text = re.sub(BADSENT, r' . \1', text)
75 
76  ALL_COUNTS.update([t for t in text.split() if len(t) == 1 and ord(t) > 127])
77  return text.strip()
