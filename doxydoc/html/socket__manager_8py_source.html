265  def _safe_send(self, data, force=False):
266  if not self.alive and not force:
267  # Try to wait a half second to send a packet
268  timeout = 0.5
269  while timeout > 0 and not self.alive:
270  time.sleep(0.1)
271  timeout -= 0.1
272  if not self.alive:
273  # don't try to send a packet if we're still dead
274  return False
275  try:
276  with self.send_lock:
277  self.ws.send(data)
278  except websocket.WebSocketConnectionClosedException:
279  # The channel died mid-send, wait for it to come back up
280  return False
281  except BrokenPipeError: # noqa F821 we don't support p2
282  # The channel died mid-send, wait for it to come back up
283  return False
284  except AttributeError:
285  # _ensure_closed was called in parallel, self.ws = None
286  return False
287  except Exception as e:
288  shared_utils.print_and_log(
289  logging.WARN,
290  'Unexpected socket error occured: {}'.format(repr(e)),
291  should_print=True,
292  )
293  return False
294  return True

352  def _send_packet(self, packet, connection_id, send_time):
353  """Sends a packet, blocks if the packet is blocking"""
354  # Send the packet
355  pkt = packet.as_dict()
356  if pkt['data'] is None or packet.status == Packet.STATUS_ACK:
357  return # This packet was _just_ acked.
358  shared_utils.print_and_log(logging.DEBUG, 'Send packet: {}'.format(packet))
359 
360  result = self._safe_send(
361  json.dumps({'type': data_model.SOCKET_ROUTE_PACKET_STRING, 'content': pkt})
362  )
363  if not result:
364  # The channel died mid-send, wait for it to come back up
365  self._safe_put(connection_id, (send_time, packet))
366  return
367 
368  if packet.status != Packet.STATUS_ACK:
369  packet.status = Packet.STATUS_SENT
370 
371  # Handles acks and blocking
372  if packet.requires_ack:
373  if packet.blocking:
374  # Put the packet right back into its place to prevent sending
375  # other packets, then block that connection
376  self._safe_put(connection_id, (send_time, packet))
377  t = time.time() + self.ACK_TIME[packet.type]
378  self.blocking_packets[connection_id] = (t, packet)
379  else:
380  # non-blocking ack: add ack-check to queue
381  t = time.time() + self.ACK_TIME[packet.type]
382  self._safe_put(connection_id, (t, packet))

449  def on_message(*args):
450  """Incoming message handler for ACKs, ALIVEs, HEARTBEATs,
451  PONGs, and MESSAGEs"""
452  packet_dict = json.loads(args[1])
453  if packet_dict['type'] == 'conn_success':
454  self.alive = True
455  return # No action for successful connection
456  packet = Packet.from_dict(packet_dict['content'])
457  if packet is None:
458  return
459  packet_id = packet.id
460  packet_type = packet.type
461  connection_id = packet.get_sender_connection_id()
462  if packet_type == Packet.TYPE_ACK:
463  with self.packet_map_lock:
464  if packet_id not in self.packet_map:
465  # Don't do anything when acking a packet we don't have
466  return
467  # Acknowledgements should mark a packet as acknowledged
468  shared_utils.print_and_log(
469  logging.DEBUG, 'On new ack: {}'.format(args)
470  )
471  self.packet_map[packet_id].status = Packet.STATUS_ACK
472  # If the packet sender wanted to do something on ack
473  if self.packet_map[packet_id].ack_func:
474  self.packet_map[packet_id].ack_func(packet)
475  # clear the stored packet data for memory reasons
476  try:
477  self.packet_map[packet_id].data = None
478  except Exception:
479  pass # state already reduced, perhaps by ack_func
480  elif packet_type == Packet.TYPE_HEARTBEAT:
481  # Heartbeats update the last heartbeat, clears pongs w/o beat
482  self.last_received_heartbeat[connection_id] = packet
483  self.pongs_without_heartbeat[connection_id] = 0
484  elif packet_type == Packet.TYPE_PONG:
485  # Message in response from the router, ensuring we're connected
486  # to it. Redundant but useful for metering from web client.
487  pong_conn_id = packet.get_receiver_connection_id()
488  if self.last_received_heartbeat[pong_conn_id] is not None:
489  self.pongs_without_heartbeat[pong_conn_id] += 1
490  elif packet_id in self.processed_packets:
491  # We don't want to re-process a packet that has already
492  # been recieved, but we do need to tell the sender it is ack'd
493  # re-acknowledge the packet was recieved and already processed
494  self._send_ack(packet)
495  else:
496 
497  # Remaining packet types need to be acknowledged
498  shared_utils.print_and_log(
499  logging.DEBUG, 'On new message: {}'.format(args)
500  )
501  # Call the appropriate callback
502  if packet_type == Packet.TYPE_ALIVE:
503  self.alive_callback(packet)
504  elif packet_type == Packet.TYPE_MESSAGE:
505  self.message_callback(packet)
506 
507  # acknowledge the packet was recieved and processed
508  self._send_ack(packet)
509 
510  # Note to self that this packet has already been processed,
511  # and shouldn't be processed again in the future
512  self.processed_packets.add(packet_id)

577  def channel_thread(self):
578  """Handler thread for monitoring all channels"""
579  # while the thread is still alive
580  while not self.is_shutdown:
581  if self.ws is None:
582  # Waiting for websocket to come back alive
583  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
584  continue
585  for connection_id in self.run.copy():
586  if not self.run[connection_id]:
587  continue
588  try:
589  # Send a heartbeat if needed
590  self._send_needed_heartbeat(connection_id)
591  # Check if client is still alive
592  if self.pongs_without_heartbeat[connection_id] > self.missed_pongs:
593  self.run[connection_id] = False
594  # Setup and run the channel sending thread
595  self.threads[connection_id] = threading.Thread(
596  target=self.socket_dead_callback,
597  name='Socket-dead-{}'.format(connection_id),
598  args=self.worker_assign_ids[connection_id],
599  )
600  self.threads[connection_id].daemon = True
601  self.threads[connection_id].start()
602 
603  # Make sure the queue still exists
604  if connection_id not in self.queues:
605  self.run[connection_id] = False
606  break
607  if self.blocking_packets.get(connection_id) is not None:
608  packet_item = self.blocking_packets[connection_id]
609  if not self.packet_should_block(packet_item):
610  self.blocking_packets[connection_id] = None
611  else:
612  continue
613  try:
614  # Get first item in the queue, check if can send it yet
615  item = self.queues[connection_id].get(block=False)
616  t = item[0]
617  if time.time() < t:
618  # Put the item back into the queue,
619  # it's not time to pop yet
620  self._safe_put(connection_id, item)
621  else:
622  # Try to send the packet
623  packet = item[1]
624  if not packet:
625  # This packet was deleted out from under us
626  continue
627  if packet.status is not Packet.STATUS_ACK:
628  # either need to send initial packet
629  # or resend not-acked packet
630  self._send_packet(packet, connection_id, t)
631  except Empty:
632  pass
633  except Exception as e:
634  shared_utils.print_and_log(
635  logging.WARN,
636  'Unexpected error occurred in socket handling thread: '
637  '{}'.format(repr(e)),
638  should_print=True,
639  )
640  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
