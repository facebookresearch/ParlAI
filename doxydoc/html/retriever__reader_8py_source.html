50  def _split_doc(self, doc):
51  """Given a doc, split it into chunks (by paragraph)."""
52  GROUP_LENGTH = 0
53  docs = []

55  curr_len = 0
56  for split in regex.split(r'\n+', doc):

59  continue
60  # Maybe group paragraphs together until we hit a length limit
61  if len(curr) > 0 and curr_len + len(split) > GROUP_LENGTH:
62  # yield ' '.join(curr)

64  curr = []
