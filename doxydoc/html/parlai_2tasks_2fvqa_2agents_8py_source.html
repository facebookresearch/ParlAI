111  def act(self):
112  if self.asked_question:
113  self.asked_question = False
114  action = {'text': 'Which fact supports this answer?', 'episode_done': True}
115  if self.datatype.startswith('train'):
116  action['labels'] = self.lastY[1]
117  if (
118  self.datatype != 'train'
119  and self.episode_idx + self.step_size >= self.num_episodes()
120  ):
121  self.epochDone = True
122  return action
123 
124  if self.datatype == 'train':
125  self.episode_idx = random.randrange(self.len)
126  else:
127  self.episode_idx = (self.episode_idx + self.step_size) % self.num_episodes()
128 
129  self.asked_question = True
130  qa = self.ques[self.episode_idx]
131  question = qa['question']
132  img_path = self.image_path + qa['img_file']
133 
134  action = {
135  'image': self.image_loader.load(img_path),
136  'text': question,
137  'episode_done': False,
138  }
139 
140  human_readable = qa['fact_surface'].replace('[', '').replace(']', '')
141  self.lastY = [[qa['answer']], [human_readable]]
142 
143  if self.datatype.startswith('train'):
144  action['labels'] = self.lastY[0]
145 
146  return action
