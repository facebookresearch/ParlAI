52 def make_serializable(obj):
53  new_obj = {}
54  for key, val in obj.items():
55  if isinstance(val, (int, str, bytes, dict, list, tuple, bool)):
56  new_obj[key] = val
57  elif isinstance(val, collections.Mapping):
58  new_obj[key] = dict(val)
59  elif isinstance(val, collections.Sequence):
60  new_obj[key] = list(val)
61  elif torch.is_tensor(val):
62  new_obj[key] = {
63  'value': val.tolist(),
64  'deserialized_tensor': True,
65  'type': str(val.dtype),
66  }
67  return new_obj

70 def build_data(opt):
71  if not opt.get('model', False):
72  opt['model'] = 'repeat_label'
73  preprocess = opt.get('pytorch_preprocess', True)
74  opt['dict_file'] = get_pyt_dict_file(opt)
75  dictionary = None
76  if 'dict_maxexs' in opt:
77  # Note: only build dictionary if dict loop args specified
78  dictionary = build_dict(opt, skip_if_built=True)
79  agent = create_agent(opt)
80  # If build teacher not specified, we are simply looking for the file
81  if not opt.get('pytorch_teacher_task', None):
82  df = opt.get('pytorch_datapath')
83  # check if the user set a datafile
84  if not df:
85  raise Exception('Tried to find data but `--pytorch-datapath` is not set')
86  # check if the user provided the already built file
87  if 'pytorch' not in df:
88  df += '.pytorch' + (
89  agent.getID() if opt.get('pytorch_preprocess', True) else ''
90  )
91  if not os.path.isfile(df):
92  raise Exception(
93  'Tried to find data but it is not built, please'
94  'specify `--pytorch-teacher-task`'
95  )
96  else:
97  return df
98 
99  ordered_opt = copy.deepcopy(opt)
100  # we use streaming to build the data
101  dt = opt['datatype'].split(':')[0]
102  ordered_opt['datatype'] = dt + ':ordered:stream'
103  ordered_opt['numthreads'] = 1
104  ordered_opt['batchsize'] = 1
105  ordered_opt['task'] = ordered_opt['pytorch_teacher_task']
106  ordered_opt.pop('pytorch_teacher_dataset')

110  agent = world_data.agents[1]
111  datapath = os.path.join(
112  opt.get('datapath', '.'),
113  '{}_pyt_data'.format(ordered_opt['task'].replace(':', '_')),
114  dt,
115  )
116  if preprocess:
117  datapath += '_{}_preprocess'.format(agent.getID().replace(':', '_'))
118  if os.path.isdir(datapath) and 'data_length' in os.listdir(datapath):
119  # Data already built
120  print("[ pytorch data already built, at {}. ]".format(datapath))
121  return datapath
122  print(
123  '----------\n[ setting up pytorch data, saving to {}/ ]\n----------'.format(
124  datapath
125  )
126  )
127  os.makedirs(datapath, exist_ok=True)
128  num_eps = 0
129  num_exs = 0
130  current = []
131  episode_done = False
132  include_labels = opt.get('pytorch_include_labels', True)
133  context_length = opt.get('pytorch_context_length', -1)
134  context = deque(maxlen=context_length if context_length > 0 else None)
135  total_exs = world_data.num_examples()
136  pbar = tqdm.tqdm(
137  total=total_exs, unit='ex', unit_scale=True, desc='Building pytorch data'
138  )
139  idx_to_char = []
140  cumulative_char_len = 0
141  # pass examples to dictionary
142  with open(os.path.join(datapath, 'data'), 'w') as pytorch_data:
143  while num_exs < total_exs:
144  while not episode_done:
145  # TODO: eventually all teachers should return Messages, so
146  # we should assert this
147  action = Message(teacher.act())
148  current.append(action)
149  episode_done = action.get('episode_done', False)
150 
151  # build separate episodes
152  for ex in current:
153  context.append(ex.get('text', ''))
154  if len(context) > 1:
155  ex.force_set('text', '\n'.join(context))
156  ex.force_set('episode_done', True)
157  labels = ex.get('labels', ex.get('eval_labels', None))
158  if labels is not None and include_labels:
159  context.append(random.choice(labels))
160  # generate observation from new example
161  if preprocess:
162  ex = agent.observe(ex)
163  ex.pop('label_candidates', '')
164  ex['preprocessed'] = True
165  num_eps += 1
166  num_exs += 1
167  pbar.update(1)
168  ex_len = pytorch_data.write(json.dumps(make_serializable(ex)) + "\n")
169  idx_to_char.append(cumulative_char_len)
170  cumulative_char_len += ex_len
171  # reset
172  episode_done = False
173  current.clear()
174  context.clear()
175  pbar.close()
176  with open(os.path.join(datapath, 'char_index'), 'w') as char_index:
177  json.dump(idx_to_char, char_index)
178  with open(os.path.join(datapath, 'data_length'), 'w') as pytorch_data_len:
179  pytorch_data_len.write(json.dumps({'num_eps': num_eps, 'num_exs': num_exs}))
180  if dictionary:
181  dictionary.save(get_pyt_dict_file(opt), sort=True)
182 
183  print('[ pytorch data built. ]')
184  return datapath
