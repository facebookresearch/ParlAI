183  def load_disconnects(self):
184  """Load disconnects from file, populate the disconnects field for any
185  worker_id that has disconnects in the list. Any disconnect that
186  occurred longer ago than the disconnect persist length is ignored
187  """
188  self.disconnects = []
189  # Load disconnects from file
190  file_path = os.path.join(parent_dir, DISCONNECT_FILE_NAME)
191  compare_time = time.time()
192  if os.path.exists(file_path):
193  with open(file_path, 'rb') as f:
194  old_disconnects = pickle.load(f)
195  self.disconnects = [
196  d
197  for d in old_disconnects
198  if (compare_time - d['time']) < DISCONNECT_PERSIST_LENGTH
199  ]
200  # Initialize worker states with proper number of disconnects
201  for disconnect in self.disconnects:
202  worker_id = disconnect['id']
203  if worker_id not in self.mturk_workers:
204  # add this worker to the worker state
205  self.mturk_workers[worker_id] = WorkerState(worker_id)
206  self.mturk_workers[worker_id].disconnects += 1

217  def handle_agent_disconnect(self, worker_id, assignment_id, partner_callback):
218  """Handles a disconnect by the given worker, calls partner_callback
219  on all of the conversation partners of that worker
220  """
221  agent = self._get_agent(worker_id, assignment_id)
222  if agent is not None:
223  # Disconnect in conversation is not workable
224  agent.set_status(AssignState.STATUS_DISCONNECT)
225  # in conversation, inform others about disconnect
226  conversation_id = agent.conversation_id
227  if conversation_id in self.conv_to_agent:
228  conv_participants = self.conv_to_agent[conversation_id]
229  if agent in conv_participants:
230  for other_agent in conv_participants:
231  if agent.assignment_id != other_agent.assignment_id:
232  partner_callback(other_agent)
233  if len(conv_participants) > 1:
234  # The user disconnected from inside a conversation with
235  # another turker, record this as bad behavior
236  self.handle_bad_disconnect(worker_id)

238  def handle_bad_disconnect(self, worker_id):
239  """Update the number of bad disconnects for the given worker, block
240  them if they've exceeded the disconnect limit
241  """
242  if not self.is_sandbox:
243  self.mturk_workers[worker_id].disconnects += 1
244  self.disconnects.append({'time': time.time(), 'id': worker_id})
245  if self.mturk_workers[worker_id].disconnects > MAX_DISCONNECTS:
246  if self.opt['hard_block']:
247  text = (
248  'This worker has repeatedly disconnected from these '
249  'tasks, which require constant connection to complete '
250  'properly as they involve interaction with other '
251  'Turkers. They have been blocked after being warned '
252  'and failing to adhere. This was done in order to '
253  'ensure a better experience for other '
254  'workers who don\'t disconnect.'
255  )
256  self.mturk_manager.block_worker(worker_id, text)
257  shared_utils.print_and_log(
258  logging.INFO,
259  'Worker {} blocked - too many disconnects'.format(worker_id),
260  True,
261  )
262  elif self.opt['disconnect_qualification'] is not None:
263  self.mturk_manager.soft_block_worker(
264  worker_id, 'disconnect_qualification'
265  )
266  shared_utils.print_and_log(
267  logging.INFO,
268  'Worker {} soft blocked - too many disconnects'.format(
269  worker_id
270  ),
271  True,
272  )
