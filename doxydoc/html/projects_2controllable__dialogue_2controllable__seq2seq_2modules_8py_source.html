41 def pad(tensor, length, dim=0, pad=0):

48  :returns: padded tensor if the tensor is shorter than length
49  """

171  def _decode_forced(self, ys, ctrl_inputs, encoder_states):
172  """Decode with teacher forcing."""
173  bsz = ys.size(0)
174  seqlen = ys.size(1)
175 
176  hidden = encoder_states[1]
177  attn_params = (encoder_states[0], encoder_states[2])
178 
179  # input to model is START + each target except the last
180  y_in = ys.narrow(1, 0, seqlen - 1)
181  xs = torch.cat([self._starts(bsz), y_in], 1)
182 
183  scores = []
184  if self.attn_type == 'none':
185  # do the whole thing in one go
186  output, hidden = self.decoder(xs, ctrl_inputs, hidden, attn_params)
187  score = self.output(output)
188  scores.append(score)
189  else:
190  # need to feed in one token at a time so we can do attention
191  for i in range(seqlen):
192  xi = xs.select(1, i).unsqueeze(1)
193  output, hidden = self.decoder(xi, ctrl_inputs, hidden, attn_params)
194  score = self.output(output)
195  scores.append(score)
196 
197  scores = torch.cat(scores, 1)
198  return scores

200  def _decode(self, ctrl_inputs, encoder_states, maxlen):
201  """Decode maxlen tokens."""
202  hidden = encoder_states[1]
203  attn_params = (encoder_states[0], encoder_states[2])
204  bsz = encoder_states[0].size(0)
205 
206  xs = self._starts(bsz) # input start token
207 
208  scores = []
209  for _ in range(maxlen):
210  # generate at most longest_label tokens
211  output, hidden = self.decoder(xs, ctrl_inputs, hidden, attn_params)
212  score = self.output(output)
213  scores.append(score)
214  xs = score.max(2)[1] # next input is current predicted output
215 
216  scores = torch.cat(scores, 1)
217  return scores

219  def _align_inds(self, encoder_states, cand_inds):
220  """Select the encoder states relevant to valid candidates."""
221  enc_out, hidden, attn_mask = encoder_states
222 
223  # LSTM or GRU/RNN hidden state?
224  if isinstance(hidden, torch.Tensor):
225  hid, cell = hidden, None
226  else:
227  hid, cell = hidden
228 
229  if len(cand_inds) != hid.size(1):
230  # if the number of candidates is mismatched from the number of
231  # hidden states, we throw out the hidden states we won't rank with
232  cand_indices = hid.new(cand_inds)
233  hid = hid.index_select(1, cand_indices)
234  if cell is None:
235  hidden = hid
236  else:
237  cell = cell.index_select(1, cand_indices)
238  hidden = (hid, cell)
239 
240  if self.attn_type != 'none':
241  enc_out = enc_out.index_select(0, cand_indices)
242  attn_mask = attn_mask.index_select(0, cand_indices)
243 
244  return enc_out, hidden, attn_mask

246  def _extract_cur(self, encoder_states, index, num_cands):
247  """Extract encoder states at current index and expand them."""
248  enc_out, hidden, attn_mask = encoder_states
249  if isinstance(hidden, torch.Tensor):
250  cur_hid = hidden.select(1, index).unsqueeze(1).expand(-1, num_cands, -1)
251  else:
252  cur_hid = (
253  hidden[0]
254  .select(1, index)
255  .unsqueeze(1)
256  .expand(-1, num_cands, -1)

258  hidden[1]
259  .select(1, index)
260  .unsqueeze(1)
261  .expand(-1, num_cands, -1)
262  .contiguous(),
263  )
264 
265  cur_enc, cur_mask = None, None
266  if self.attn_type != 'none':
267  cur_enc = enc_out[index].unsqueeze(0).expand(num_cands, -1, -1)
268  cur_mask = attn_mask[index].unsqueeze(0).expand(num_cands, -1)

271  def _rank(self, cands, cand_inds, encoder_states):
272  """Rank each cand by the average log-probability of the sequence."""
273  if cands is None:
274  return None
275  encoder_states = self._align_inds(encoder_states, cand_inds)
276 
277  cand_scores = []
278  for batch_idx in range(len(cands)):
279  # we do one set of candidates at a time
280  curr_cs = cands[batch_idx]
281  num_cands = curr_cs.size(0)
282 
283  # select just the one hidden state
284  cur_enc_states = self._extract_cur(encoder_states, batch_idx, num_cands)
285 
286  score = self._decode_forced(curr_cs, None, cur_enc_states)
287  true_score = F.log_softmax(score, dim=2).gather(2, curr_cs.unsqueeze(2))
288  nonzero = curr_cs.ne(0).float()
289  scores = (true_score.squeeze(2) * nonzero).sum(1)
290  seqlens = nonzero.sum(1)
291  scores /= seqlens
292  cand_scores.append(scores)
293 
294  max_len = max(len(c) for c in cand_scores)
295  cand_scores = torch.cat(
296  [pad(c, max_len, pad=self.NULL_IDX).unsqueeze(0) for c in cand_scores], 0
297  )
298  return cand_scores

300  def forward(
301  self,
302  xs,
303  ctrl_inputs=None,
304  ys=None,
305  cands=None,
306  prev_enc=None,
307  maxlen=None,
308  seq_len=None,
309  ):
310  """
311  Get output predictions from the model.
312 
313  :param xs:
314  (bsz x seqlen) LongTensor input to the encoder
315  :param ys:
316  expected output from the decoder. used for teacher forcing
317  to calculate loss.
318  :param cands:
319  set of candidates to rank
320  :param prev_enc:
321  if you know you'll pass in the same xs multiple times, you can pass
322  in the encoder output from the last forward pass to skip
323  recalcuating the same encoder output.
324  :param maxlen:
325  max number of tokens to decode. if not set, will use the length of
326  the longest label this model has seen. ignored when ys is not None.
327  :param seq_len:
328  this is the sequence length of the input (xs), i.e. xs.size(1). we
329  use this to recover the proper output sizes in the case when we
330  distribute over multiple gpus
331  :param ctrl_inputs:
332  (bsz x num_controls) LongTensor containing control vars
333 
334  :returns:
335  scores, candidate scores, and encoder states
336 
337  - scores contains the model's predicted token scores.
338  (bsz x seqlen x num_features)
339  - candidate scores are the score the model assigned to each candidate
340  (bsz x num_cands)
341  - encoder states are the (output, hidden, attn_mask) states from the
342  encoder. feed this back in to skip encoding on the next call.
343  """
344  if ys is not None:
345  # keep track of longest label we've ever seen
346  # we'll never produce longer ones than that during prediction
347  self.longest_label = max(self.longest_label, ys.size(1))
348 
349  encoder_states = self._encode(xs, prev_enc)
350 
351  # rank candidates if they are available
352  cand_scores = None
353  if cands is not None:
354  cand_inds = [i for i in range(cands.size(0))]
355  cand_scores = self._rank(cands, cand_inds, encoder_states)
356 
357  if ys is not None:
358  # use teacher forcing
359  scores = self._decode_forced(ys, ctrl_inputs, encoder_states)
360  else:
361  scores = self._decode(
362  ctrl_inputs, encoder_states, maxlen or self.longest_label
363  )
364 
365  if seq_len is not None:
366  # when using multiple gpus, we need to make sure output of
367  # encoder is correct size for gathering; we recover this with
368  # the parameter seq_len
369  if encoder_states[0].size(1) < seq_len:
370  out_pad_tensor = torch.zeros(
371  encoder_states[0].size(0),
372  seq_len - encoder_states[0].size(1),
373  encoder_states[0].size(2),
374  ).cuda()
375  new_out = torch.cat([encoder_states[0], out_pad_tensor], 1)
376  encoder_states = (new_out, encoder_states[1], encoder_states[2])
377 
378  return scores, cand_scores, encoder_states

381 class ControlEncoder(nn.Module):
382  """Given CT control variable, gives the concatenated control embeddings vector."""
383 
384  def __init__(self, control_settings):
385  super().__init__()
386  self.control_settings = control_settings # see ControllableSeq2seqAgent
387 
388  # int to string mapping giving the canonical ordering of the controls
389  self.idx2ctrl = {
390  d['idx']: control for control, d in self.control_settings.items()
391  }
392 
393  # Initialize control embeddings
394  self.control_embeddings = nn.ModuleDict(
395  {
396  c: nn.Embedding(d['num_buckets'], d['embsize'], sparse=False)
397  for c, d in control_settings.items()
398  }
399  ) # maps from string (ctrl name) to nn.Embedding
400 
401  def forward(self, control_inputs):
402  """
403  Forward pass.
404 
405  :param control_inputs: (bsz x num_control_vars) LongTensor of control
406  variable values (i.e. bucket ids)
407 
408  :returns: control_embs, (bsz x sum of control emb sizes) FloatTensor of
409  control variable embeddings, concatenated
410  """
411  # list length num_control_vars of tensors shape (bsz, 1)
412  control_inputs = torch.split(control_inputs, 1, dim=1)
413 
414  # list length num_control_vars of tensors shape (bsz)
415  control_inputs = [torch.squeeze(t, 1) for t in control_inputs]
416 
417  assert len(control_inputs) == len(self.control_settings)
418 
419  control_embs = []
420  for idx, inputs in enumerate(control_inputs):
421  control = self.idx2ctrl[idx] # string, the ctrl var for this position
422  # Append a tensor shape (bsz, control embedding size)
423  control_embs.append(self.control_embeddings[control](inputs))
424  control_embs = torch.cat(control_embs, dim=1) # shape (bsz, sum of ctrl embszs)
425 
426  return control_embs

507  def forward(self, xs):
508  """Encode sequence.
509 
510  :param xs: (bsz x seqlen) LongTensor of input token indices
511 
512  :returns: encoder outputs, hidden state, attention mask
513  encoder outputs are the output state at each step of the encoding.
514  the hidden state is the final hidden state of the encoder.
515  the attention mask is a mask of which input values are nonzero.
516  """
517  bsz = len(xs)
518 
519  # embed input tokens
520  xs = self.input_dropout(xs)
521  xes = self.dropout(self.lt(xs))
522  attn_mask = xs.ne(0)
523  try:
524  x_lens = torch.sum(attn_mask.int(), dim=1)
525  xes = pack_padded_sequence(xes, x_lens, batch_first=True)
526  packed = True
527  except ValueError:
528  # packing failed, don't pack then
529  packed = False
530 
531  encoder_output, hidden = self.rnn(xes)
532  if packed:
533  encoder_output, _ = pad_packed_sequence(encoder_output, batch_first=True)
534  if self.dirs > 1:
535  # project to decoder dimension by taking sum of forward and back
536  if isinstance(self.rnn, nn.LSTM):
537  hidden = (
538  hidden[0].view(-1, self.dirs, bsz, self.hsz).sum(1),
539  hidden[1].view(-1, self.dirs, bsz, self.hsz).sum(1),
540  )
541  else:
542  hidden = hidden.view(-1, self.dirs, bsz, self.hsz).sum(1)
543 
544  return encoder_output, hidden, attn_mask

607  def forward(self, xs, ctrl_inputs=None, hidden=None, attn_params=None):
608  """Decode from input tokens.
609 
610  :param xs: (bsz x seqlen) LongTensor of input token indices
611  :param ctrl_inputs: (bsz, num_controls) LongTensor
612  :param hidden: hidden state to feed into decoder. default (None)
613  initializes tensors using the RNN's defaults.
614  :param attn_params: (optional) tuple containing attention parameters,
615  default AttentionLayer needs encoder_output states
616  and attention mask (e.g. encoder_input.ne(0))
617 
618  :returns: output state(s), hidden state.
619  output state of the encoder. for an RNN, this is
620  (bsz, seq_len, num_directions * hiddensize).
621  hidden state will be same dimensions as input
622  hidden state. for an RNN, this is a tensor of sizes
623  (bsz, numlayers * num_directions, hiddensize).
624  """
625  # sequence indices => sequence embeddings
626  xes = self.dropout(self.lt(xs)) # shape (bsz, seqlen, embsize)
627 
628  # Concatenate the control embeddings
629  if ctrl_inputs is not None:
630  ctrl_embs = self.dropout(
631  self.control_encoder(ctrl_inputs)
632  ) # shape (bsz, sum of ctrl emb sizes)
633  ctrl_embs_tiled = ctrl_embs.unsqueeze(1).repeat(
634  1, xes.size(1), 1
635  ) # shape (bsz, seqlen, sum of ctrl emb sizes)
636  xes = torch.cat(
637  [xes, ctrl_embs_tiled], 2
638  ) # shape (bsz,seqlen,embsize+sum of ctrl embsz)
639 
640  if self.attn_time == 'pre':
641  # modify input vectors with attention
642  xes, _attw = self.attention(xes, hidden, attn_params)
643 
644  # feed tokens into rnn
645  output, new_hidden = self.rnn(xes, hidden)
646 
647  if self.attn_time == 'post':
648  # modify output vectors with attention
649  output, _attw = self.attention(output, new_hidden, attn_params)
650 
651  return output, new_hidden

654 class OutputLayer(nn.Module):
655  """Takes in final states and returns distribution over candidates."""
656 
657  def __init__(
658  self,
659  num_features,
660  embeddingsize,
661  hiddensize,
662  dropout=0,
663  numsoftmax=1,
664  shared_weight=None,
665  padding_idx=-1,
666  ):
667  """Initialize output layer.
668 
669  :param num_features: number of candidates to rank
670  :param hiddensize: (last) dimension of the input vectors
671  :param embeddingsize: (last) dimension of the candidate vectors
672  :param numsoftmax: (default 1) number of softmaxes to calculate.
673  see arxiv.org/abs/1711.03953 for more info.
674  increasing this slows down computation but can
675  add more expressivity to the embeddings.
676  :param shared_weight: (num_features x esz) vector of weights to use as
677  the final linear layer's weight matrix. default
678  None starts with a new linear layer.
679  :param padding_idx: model should output a large negative number for
680  score at this index. if set to -1 (default),
681  this is disabled. if >= 0, subtracts one from
682  num_features and always outputs -1e20 at this
683  index. only used when shared_weight is not None.
684  setting this param helps protect gradient from
685  entering shared embedding matrices.
686  """
687  super().__init__()
688  self.dropout = nn.Dropout(p=dropout)
689 
690  self.padding_idx = padding_idx if shared_weight is not None else -1
691 
692  # embedding to scores
693  if shared_weight is None:
694  # just a regular linear layer
695  self.e2s = nn.Linear(embeddingsize, num_features, bias=True)
696  else:
697  # use shared weights and a bias layer instead
698  if padding_idx == 0:
699  num_features -= 1 # don't include padding
700  shared_weight = shared_weight.narrow(0, 1, num_features)
701  elif padding_idx > 0:
702  raise RuntimeError('nonzero pad_idx not yet implemented')
703  self.weight = Parameter(shared_weight)
704  self.bias = Parameter(torch.Tensor(num_features))
705  self.reset_parameters()
706  self.e2s = lambda x: F.linear(x, self.weight, self.bias)
707 
708  self.numsoftmax = numsoftmax
709  if numsoftmax > 1:
710  self.esz = embeddingsize
711  self.softmax = nn.Softmax(dim=1)
712  self.prior = nn.Linear(hiddensize, numsoftmax, bias=False)
713  self.latent = nn.Linear(hiddensize, numsoftmax * embeddingsize)
714  self.activation = nn.Tanh()
715  else:
716  # rnn output to embedding
717  if hiddensize != embeddingsize:
718  # learn projection to correct dimensions
719  self.o2e = nn.Linear(hiddensize, embeddingsize, bias=True)
720  else:
721  # no need for any transformation here
722  self.o2e = lambda x: x
723 
724  def reset_parameters(self):
725  """Reset bias param."""
726  if hasattr(self, 'bias'):
727  stdv = 1.0 / math.sqrt(self.bias.size(0))
728  self.bias.data.uniform_(-stdv, stdv)
729 
730  def forward(self, input):
731  """Compute scores from inputs.
732 
733  :param input: (bsz x seq_len x num_directions * hiddensize) tensor of
734  states, e.g. the output states of an RNN
735 
736  :returns: (bsz x seqlen x num_cands) scores for each candidate
737  """
738  # next compute scores over dictionary
739  if self.numsoftmax > 1:
740  bsz = input.size(0)
741  seqlen = input.size(1) if input.dim() > 1 else 1
742 
743  # first compute different softmax scores based on input vec
744  # hsz => numsoftmax * esz
745  latent = self.latent(input)
746  active = self.dropout(self.activation(latent))
747  # esz => num_features
748  logit = self.e2s(active.view(-1, self.esz))
749 
750  # calculate priors: distribution over which softmax scores to use
751  # hsz => numsoftmax
752  prior_logit = self.prior(input).view(-1, self.numsoftmax)
753  # softmax over numsoftmax's
754  prior = self.softmax(prior_logit)
755 
756  # now combine priors with logits
757  prob = self.softmax(logit).view(bsz * seqlen, self.numsoftmax, -1)
758  probs = (prob * prior.unsqueeze(2)).sum(1).view(bsz, seqlen, -1)
759  scores = probs.log()
760  else:
761  # hsz => esz, good time for dropout
762  e = self.dropout(self.o2e(input))
763  # esz => num_features
764  scores = self.e2s(e)
765 
766  if self.padding_idx == 0:
767  pad_score = scores.new(scores.size(0), scores.size(1), 1).fill_(-NEAR_INF)
768  scores = torch.cat([pad_score, scores], dim=-1)
769 
770  return scores

773 class AttentionLayer(nn.Module):
774  """Computes attention between hidden and encoder states.
775 
776  See arxiv.org/abs/1508.04025 for more info on each attention type.
777  """
778 
779  def __init__(
780  self,
781  attn_type,
782  hiddensize,
783  embeddingsize,
784  bidirectional=False,
785  attn_length=-1,
786  attn_time='pre',
787  ):
788  """Initialize attention layer."""
789  super().__init__()
790  self.attention = attn_type
791 
792  if self.attention != 'none':
793  hsz = hiddensize
794  hszXdirs = hsz * (2 if bidirectional else 1)
795  if attn_time == 'pre':
796  # attention happens on the input embeddings
797  input_dim = embeddingsize
798  elif attn_time == 'post':
799  # attention happens on the output of the rnn
800  input_dim = hsz
801  else:
802  raise RuntimeError('unsupported attention time')
803 
804  # linear layer for combining applied attention weights with input
805  self.attn_combine = nn.Linear(hszXdirs + input_dim, input_dim, bias=False)
806 
807  if self.attention == 'local':
808  # local attention over fixed set of output states
809  if attn_length < 0:
810  raise RuntimeError('Set attention length to > 0.')
811  self.max_length = attn_length
812  # combines input and previous hidden output layer
813  self.attn = nn.Linear(hsz + input_dim, attn_length, bias=False)
814  # combines attention weights with encoder outputs
815  elif self.attention == 'concat':
816  self.attn = nn.Linear(hsz + hszXdirs, hsz, bias=False)
817  self.attn_v = nn.Linear(hsz, 1, bias=False)
818  elif self.attention == 'general':
819  # equivalent to dot if attn is identity
820  self.attn = nn.Linear(hsz, hszXdirs, bias=False)
821 
822  def forward(self, xes, hidden, attn_params):
823  """Compute attention over attn_params given input and hidden states.
824 
825  :param xes: input state. will be combined with applied
826  attention.
827  :param hidden: hidden state from model. will be used to select
828  states to attend to in from the attn_params.
829  :param attn_params: tuple of encoder output states and a mask showing
830  which input indices are nonzero.
831 
832  :returns: output, attn_weights
833  output is a new state of same size as input state `xes`.
834  attn_weights are the weights given to each state in the
835  encoder outputs.
836  """
837  if self.attention == 'none':
838  # do nothing, no attention
839  return xes, None
840 
841  if type(hidden) == tuple:
842  # for lstms use the "hidden" state not the cell state
843  hidden = hidden[0]
844  last_hidden = hidden[-1] # select hidden state from last RNN layer
845 
846  enc_out, attn_mask = attn_params
847  bsz, seqlen, hszXnumdir = enc_out.size()
848  numlayersXnumdir = last_hidden.size(1)
849 
850  if self.attention == 'local':
851  # local attention weights aren't based on encoder states
852  h_merged = torch.cat((xes.squeeze(1), last_hidden), 1)
853  attn_weights = F.softmax(self.attn(h_merged), dim=1)
854 
855  # adjust state sizes to the fixed window size
856  if seqlen > self.max_length:
857  offset = seqlen - self.max_length
858  enc_out = enc_out.narrow(1, offset, self.max_length)
859  seqlen = self.max_length
860  if attn_weights.size(1) > seqlen:
861  attn_weights = attn_weights.narrow(1, 0, seqlen)
862  else:
863  hid = last_hidden.unsqueeze(1)
864  if self.attention == 'concat':
865  # concat hidden state and encoder outputs
866  hid = hid.expand(bsz, seqlen, numlayersXnumdir)
867  h_merged = torch.cat((enc_out, hid), 2)
868  # then do linear combination of them with activation
869  active = F.tanh(self.attn(h_merged))
870  attn_w_premask = self.attn_v(active).squeeze(2)
871  elif self.attention == 'dot':
872  # dot product between hidden and encoder outputs
873  if numlayersXnumdir != hszXnumdir:
874  # enc_out has two directions, so double hid
875  hid = torch.cat([hid, hid], 2)
876  enc_t = enc_out.transpose(1, 2)
877  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
878  elif self.attention == 'general':
879  # before doing dot product, transform hidden state with linear
880  # same as dot if linear is identity
881  hid = self.attn(hid)
882  enc_t = enc_out.transpose(1, 2)
883  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
884 
885  # calculate activation scores, apply mask if needed
886  if attn_mask is not None:
887  # remove activation from NULL symbols
888  attn_w_premask.masked_fill_(~attn_mask, -NEAR_INF)
889  attn_weights = F.softmax(attn_w_premask, dim=1)
890 
891  # apply the attention weights to the encoder states
892  attn_applied = torch.bmm(attn_weights.unsqueeze(1), enc_out)
893  # concatenate the input and encoder states
894  merged = torch.cat((xes.squeeze(1), attn_applied.squeeze(1)), 1)
895  # combine them with a linear layer and tanh activation
896  output = torch.tanh(self.attn_combine(merged).unsqueeze(1))
897 
898  return output, attn_weights
