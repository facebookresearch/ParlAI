76 def get_metrics(actions, gt_actions):
77  tp, fp, fn = 0, 0, 0
78  action_set, gt_action_set = set(actions), set(gt_actions)
79  for action in action_set:
80  if action in gt_action_set:
81  tp += 1
82  else:
83  fp += 1
84  for action in gt_action_set:
85  if action not in action_set:
86  fn += 1
87  prec = tp / (tp + fp) if tp + fp > 0 else 0.0
88  recall = tp / (tp + fn) if tp + fn > 0 else 0.0
89  f1 = 2.0 * prec * recall / (prec + recall) if prec + recall > 0 else 0.0
90  return f1
