373  # and copy them over to states (the params loaded from file)

438  def _init_controls(self):
439  """
440  Initialize controls.
441 
442  Sets the following:
443 
444  self.control_vars: list of strings. The CT controls sorted alphabetically.
445 
446  self.control_settings: a dictionary containing info about the CT controls.
447  Each control name maps to a dictionary that contains:
448  'embsize': embedding size for this control
449  'num_buckets': num buckets for this control
450  'set_value': a set value for this control, or None
451  'idx': the index of this control in this list self.control_vars
452 
453  self.wd_features: list of strings, the WD features to use.
454 
455  self.wd_wts: list of floats, the WD weights to use.
456  """
457 
458  # Set self.control_vars, a list of the CT control vars in alphabetical order
459  self.control_vars = (
460  sorted(self.opt['control_vars'].split(','))
461  if self.opt['control_vars'] != ''
462  else []
463  )
464 
465  # Process the control_num_buckets flag (for CT)
466  ctrl_numbucket_override = {}
467  if self.opt['control_num_buckets'] != "":
468  ctrl_numbucket_override = {
469  s.split(':')[0]: int(s.split(':')[1])
470  for s in self.opt['control_num_buckets'].split(',')
471  } # string to int
472 
473  # Process the control_embeddingsize flag (for CT)
474  ctrl_esz_override = {}
475  if self.opt['control_embeddingsize'] != "":
476  ctrl_esz_override = {
477  s.split(':')[0]: int(s.split(':')[1])
478  for s in self.opt['control_embeddingsize'].split(',')
479  } # string to int
480 
481  # Process the set_controls flag, which gives user-supplied settings for CT
482  set_controls = {}
483  if self.opt['set_controls'] != "":
484  set_controls = {} # string to (int or string)
485  for s in self.opt['set_controls'].split(','):
486  control, set_val = s.split(':')[0], s.split(':')[1]
487  if control not in self.control_vars:
488  raise ValueError(
489  "Received --set-controls for control '%s', but "
490  "that is not one of the existing CT controls for "
491  "this model, which are: %s"
492  % (control, ', '.join(self.control_vars))
493  )
494  try:
495  set_val = int(set_val) # set_val should be a string of an int
496  except ValueError:
497  raise ValueError(
498  "Received --set-controls '%s' for CT "
499  "control '%s'. The set value must be an integer."
500  % (set_val, control)
501  )
502  set_controls[control] = int(set_val)
503 
504  # Set self.control_settings for the CT controls
505  self.control_settings = {}
506  for idx, c in enumerate(self.control_vars):
507  d = {}
508  d['embsize'] = (
509  ctrl_esz_override[c]
510  if c in ctrl_esz_override
511  else CONTROL2DEFAULTEMBSIZE[c]
512  )
513  d['num_buckets'] = (
514  ctrl_numbucket_override[c]
515  if c in ctrl_numbucket_override
516  else CONTROL2DEFAULTNUMBUCKETS[c]
517  )
518  if c in set_controls:
519  set_val = set_controls[c]
520  if set_val not in range(d['num_buckets']):
521  raise ValueError(
522  "Received --set-controls '%s' for CT control "
523  "'%s', which has num_buckets=%i. The set value "
524  "must be between 0 and %i."
525  % (set_val, c, d['num_buckets'], d['num_buckets'] - 1)
526  )
527  d['set_value'] = set_controls[c] if c in set_controls else None
528  d['idx'] = idx
529  self.control_settings[c] = d

531  # Get list of WD features and weights, self.wd_features and self.wd_weights

592  """

595  if num_tok > 0:
596  if self.metrics['correct_tokens'] > 0:
597  m['token_acc'] = self.metrics['correct_tokens'] / num_tok
598  m['loss'] = self.metrics['loss'] / num_tok
599  try:
600  m['ppl'] = math.exp(m['loss'])
601  except OverflowError:
602  m['ppl'] = float('inf')

604  m['total_skipped_batches'] = self.metrics['total_skipped_batches']
605  for k, v in m.items():
606  # clean up: rounds to sigfigs and converts tensors to floats
607  m[k] = round_sigfigs(v, 4)

816 
817  This function uses model with the following reqs:
818 
819  - model.encoder takes input returns tuple (enc_out, enc_hidden, attn_mask)
820  - model.decoder takes decoder params and returns decoder outputs after attn
821  - model.output takes decoder outputs and returns distr over dictionary
822 
823  :param model: nn.Module, here defined in modules.py
824  :param batch: Batch structure with input and labels
825  :param beam_size: Size of each beam during the search

830  :param min_n_best: minimum number of completed hypothesis generated
831  from each beam
832  :param max_ts: the maximum length of the decoded sequence
833  :param wd_features: list of strings, the WD features to use
834  :param wd_weights: list of floats, the WD weights to use

837  - beam_preds_scores : list of tuples (prediction, score) for each
838  sample in Batch

842  for any following postprocessing, e.g. dot logging.

846  if wd_wts is None:
847  wd_wts = []

849  enc_out = encoder_states[0]
850  enc_hidden = encoder_states[1]
851  attn_mask = encoder_states[2]

854

857  Beam(
858  beam_size,

864  cuda=current_device,
865  block_ngram=block_ngram,
866  )

870  torch.Tensor([start])
871  .detach()
872  .expand(batch_size, 1)
873  .long()
874  .to(current_device)
875  )

877  decoder_input = decoder_input.repeat(1, beam_size).view(
878  beam_size * batch_size, -1
879  )

931  score = score.view(batch_size, beam_size, -1)
932  score = F.log_softmax(score, dim=-1)

934  if not b.done():

936 
937  # If using WD, update scores_in to reflect the WD features
938  if len(wd_features) > 0:
939

941  # across the whole vocabulary
942  wd_feat_vecs = torch.zeros((beam_size, vocab_size))
943  for hyp_idx in range(beam_size): # For each hypothesis
944

946  partial_hyp = b.partial_hyps[hyp_idx] if ts > 0 else None
947

949  wd_feat_vec = get_wd_features(
950  dictionary,
951  partial_hyp,
952  batch.history[i],
953  wd_features,
954  wd_wts,
955  ) # shape (vocab_size)
956 
957  wd_feat_vecs[hyp_idx, :] = wd_feat_vec
958  wd_feat_vecs = wd_feat_vecs.to(current_device)
959 
960  # Add the WD features to the log probability scores
961  scores_in = scores_in + wd_feat_vecs
962 
963  # Update the beam as usual
964  b.advance(scores_in)
965 
966  decoder_input = torch.cat(
967  [b.get_output_from_current_step() for b in beams]
968  ).unsqueeze(-1)
969  permute_hidden_idx = torch.cat(
970  [
971  beam_size * i + b.get_backtrack_from_current_step()

973  ]
974  )
975  # permute decoder hiddens with respect to chosen hypothesis now

978  hidden[i].data.copy_(
979  hidden[i].data.index_select(dim=1, index=permute_hidden_idx)
980  )
981  else: # GRU
982  hidden.data.copy_(
983  hidden.data.index_select(dim=1, index=permute_hidden_idx)
984  )
985  for b in beams:
986  b.check_finished()
987 
988  beam_preds_scores = [list(b.get_top_hyp()) for b in beams]
989  for pair in beam_preds_scores:
990  pair[0] = Beam.get_pretty_hypothesis(pair[0])
991 
992  n_best_beams = [b.get_rescored_finished(n_best=min_n_best) for b in beams]
993  n_best_beam_preds_scores = []
994  for i, beamhyp in enumerate(n_best_beams):

998  beams[i].get_hyp_from_finished(hyp)
999  )
1000  score = hyp.score
1001  this_beam.append((pred, score))
1002  n_best_beam_preds_scores.append(this_beam)
1003 
1004  return beam_preds_scores, n_best_beam_preds_scores, beams
