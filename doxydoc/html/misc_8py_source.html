132 def load_cands(path, lines_have_ids=False, cands_are_replies=False):
133  """
134  Load global fixed set of candidate labels that the teacher provides.
135 
136  Every example will include these as candidates. The true labels for a
137  specific example are also added to this set, so that it's possible to get
138  the right answer.
139  """
140  if path is None:
141  return None
142  cands = []
143  cnt = 0
144  with open(path) as read:
145  for line in read:
146  line = line.strip().replace('\\n', '\n')
147  if len(line) > 0:
148  cnt = cnt + 1
149  # If lines are numbered we strip them of numbers.
150  if cnt == 1 and line[0:2] == '1 ':
151  lines_have_ids = True
152  # If tabs then the label_candidates are all the replies.
153  if '\t' in line and not cands_are_replies:
154  cands_are_replies = True
155  cands = []
156  if lines_have_ids:
157  space_idx = line.find(' ')
158  line = line[space_idx + 1 :]

161  if len(sp) > 1 and sp[1] != '':
162  cands.append(sp[1])

164  cands.append(line)
165  else:
166  cands.append(line)

183 class Opt(dict):
184  """
185  Class for tracking options.
186 
187  Functions like a dict, but allows us to track the history of arguments
188  as they are set.
189  """
190 
191  def __init__(self, *args, **kwargs):
192  super().__init__(*args, **kwargs)
193  self.history = {}
194  self.deepcopies = []
195 
196  def __setitem__(self, key, val):
197  loc = traceback.format_stack()[-2]
198  self.history.setdefault(key, []).append((loc, val))
199  super().__setitem__(key, val)
200 
201  def __getstate__(self):
202  return (self.history, self.deepcopies, dict(self))
203 
204  def __setstate__(self, state):
205  self.history, self.deepcopies, data = state
206  self.update(data)
207 
208  def __reduce__(self):
209  return (Opt, (), self.__getstate__())
210 
211  def __deepcopy__(self, memo):
212  """Override deepcopy so that history is copied over to new object."""
213  # track location of deepcopy
214  loc = traceback.format_stack()[-3]
215  self.deepcopies.append(loc)
216  # deepcopy the dict
217  memo = deepcopy(dict(self))
218  # make into Opt object
219  memo = Opt(memo)
220  # deepcopy the history
221  memo.history = deepcopy(self.history)
222  # deepcopy the deepcopy history
223  memo.deepcopies = deepcopy(self.deepcopies)
224  return memo

345  def log(self, done, total, report=None):
346  """
347  Log report, time elapsed, and percentage progress towards goal.
348 
349  :param done: number of examples completed so far
350  :param total: total number of elements to be completed. if total > 0,
351  calculates the time remaining and percentage complete.
352  :param report: dict of pairs to log
353 
354  :returns: tuple log string, log dict
355  log string contains time elapsed and string representation of
356  the log dict
357  log dict contains pairs of all items to log, which includes
358  percentage complete and projected time left if total > 0
359  """
360  self.tot_time += self.timer.time()
361  self.timer.reset()
362  log = {}
363  log['exs'] = done
364  if total > 0:
365  log['%done'] = done / total
366  if log["%done"] > 0:
367  time_left = self.tot_time / log['%done'] - self.tot_time
368  log['time_left'] = str(int(time_left)) + 's'
369  z = '%.2f' % (100 * log['%done'])
370  log['%done'] = str(z) + '%'
371  if report:
372  for k, v in report.items():
373  if k not in log:
374  log[k] = v
375  text = str(int(self.tot_time)) + "s elapsed: " + str(log).replace('\\n', '\n')
376  return text, log

398 def round_sigfigs(x, sigfigs=4):
399  """
400  Round value to specified significant figures.
401 
402  :param x: input number
403  :param sigfigs: number of significant figures to return
404 
405  :returns: float number rounded to specified sigfigs
406  """
407  try:
408  if x == 0:
409  return 0
410  return round(x, -math.floor(math.log10(abs(x)) - sigfigs + 1))
411  except (RuntimeError, TypeError):
412  # handle 1D torch tensors
413  # if anything else breaks here please file an issue on Github
414  if hasattr(x, 'item'):
415  return round_sigfigs(x.item(), sigfigs)
416  else:
417  return round_sigfigs(x[0], sigfigs)
418  except (ValueError, OverflowError) as ex:
419  if x in [float('inf'), float('-inf')] or x != x: # inf or nan
420  return x
421  else:
422  raise ex

454  @classmethod

456  cls,
457  observations,
458  dictionary,
459  end_idx=None,
460  null_idx=0,
461  dq=False,
462  eval_labels=True,
463  truncate=None,
464  ):
465  """
466  Pad observations to max width.
467 
468  We check that examples are valid, pad with zeros, and sort by length
469  so that we can use the pack_padded function. The list valid_inds
470  keeps track of which indices are valid and the order in which we sort
471  the examples.
472 
473  dq -- whether we should use deque or list
474  eval_labels -- whether or not we want to consider eval labels
475  truncate -- truncate input and output lengths
476 
477  DEPRECATED. USE PARLAI.CORE.TORCH_AGENT INSTEAD.
478  """
479 
480  def valid(obs):
481  # check if this is an example our model should actually process
482  return 'text' in obs and len(obs['text']) > 0
483 
484  try:
485  # valid examples and their indices
486  valid_inds, exs = zip(
487  *[(i, ex) for i, ex in enumerate(observations) if valid(ex)]
488  )
489  except ValueError:
490  # zero examples to process in this batch, so zip failed to unpack
491  return None, None, None, None, None, None
492 
493  # `x` text is already tokenized and truncated
494  # sort by length so we can use pack_padded
495  if any(['text2vec' in ex for ex in exs]):
496  parsed_x = [ex['text2vec'] for ex in exs]
497  else:
498  parsed_x = [dictionary.txt2vec(ex['text']) for ex in exs]
499 
500  if len(parsed_x) > 0 and not isinstance(parsed_x[0], deque):
501  if dq:
502  parsed_x = [deque(x, maxlen=truncate) for x in parsed_x]
503  elif truncate is not None and truncate > 0:
504  parsed_x = [x[-truncate:] for x in parsed_x]
505 
506  x_lens = [len(x) for x in parsed_x]
507  ind_sorted = sorted(range(len(x_lens)), key=lambda k: -x_lens[k])
508 
509  exs = [exs[k] for k in ind_sorted]
510  valid_inds = [valid_inds[k] for k in ind_sorted]
511  parsed_x = [parsed_x[k] for k in ind_sorted]
512  end_idxs = [x_lens[k] for k in ind_sorted]
513 
514  eval_labels_avail = any(['eval_labels' in ex for ex in exs])
515  labels_avail = any(['labels' in ex for ex in exs])
516  if eval_labels:
517  some_labels_avail = eval_labels_avail or labels_avail
518  else:
519  some_labels_avail = labels_avail
520 
521  max_x_len = max(x_lens)
522 
523  # pad with zeros
524  if dq:
525  parsed_x = [
526  x
527  if len(x) == max_x_len
528  else x + deque((null_idx,)) * (max_x_len - len(x))
529  for x in parsed_x
530  ]
531  else:
532  parsed_x = [
533  x if len(x) == max_x_len else x + [null_idx] * (max_x_len - len(x))
534  for x in parsed_x
535  ]
536  xs = parsed_x
537 
538  # set up the target tensors
539  ys = None
540  labels = None
541  y_lens = None
542  if some_labels_avail:
543  # randomly select one of the labels to update on (if multiple)
544  if labels_avail:
545  labels = [random.choice(ex.get('labels', [''])) for ex in exs]
546  else:
547  labels = [random.choice(ex.get('eval_labels', [''])) for ex in exs]
548  # parse each label and append END
549  if dq:
550  parsed_y = [deque(maxlen=truncate) for _ in labels]
551  for deq, y in zip(parsed_y, labels):
552  deq.extendleft(reversed(dictionary.txt2vec(y)))
553  else:
554  parsed_y = [dictionary.txt2vec(label) for label in labels]
555  if end_idx is not None:
556  for y in parsed_y:
557  y.append(end_idx)
558 
559  y_lens = [len(y) for y in parsed_y]
560  max_y_len = max(y_lens)
561 
562  if dq:
563  parsed_y = [
564  y
565  if len(y) == max_y_len
566  else y + deque((null_idx,)) * (max_y_len - len(y))
567  for y in parsed_y
568  ]
569  else:
570  parsed_y = [
571  y if len(y) == max_y_len else y + [null_idx] * (max_y_len - len(y))
572  for y in parsed_y
573  ]
574  ys = parsed_y
575 
576  return xs, ys, labels, valid_inds, end_idxs, y_lens

649 def _ellipse(lst, max_display=5, sep='|'):
650  """
651  Like join, but possibly inserts an ellipsis.
652 
653  :param lst: The list to join on
654  :param int max_display: the number of items to display for ellipsing.
655  If -1, shows all items
656  :param string sep: the delimiter to join on
657  """
658  # copy the list (or force it to a list if it's a set)
659  choices = list(lst)
660  # insert the ellipsis if necessary
661  if max_display > 0 and len(choices) > max_display:
662  ellipsis = '...and {} more'.format(len(choices) - max_display)
663  choices = choices[:max_display] + [ellipsis]
664  return sep.join(str(c) for c in choices)

667 def display_messages(msgs, prettify=False, ignore_fields='', max_len=1000):
668  """
669  Return a string describing the set of messages provided.
670 
671  If prettify is true, candidates are displayed using prettytable.
672  ignore_fields provides a list of fields in the msgs which should not be
673  displayed.
674  """
675  lines = []
676  episode_done = False
677  ignore_fields = ignore_fields.split(',')
678  for index, msg in enumerate(msgs):
679  if msg is None or (index == 1 and 'agent_reply' in ignore_fields):
680  # We only display the first agent (typically the teacher) if we
681  # are ignoring the agent reply.
682  continue
683  if msg.get('episode_done'):
684  episode_done = True
685  # Possibly indent the text (for the second speaker, if two).
686  space = ''
687  if len(msgs) == 2 and index == 1:
688  space = ' '
689  # Only display rewards !=0 as they are confusing in non-RL tasks.
690  if msg.get('reward', 0) != 0:
691  lines.append(space + '[reward: {r}]'.format(r=msg['reward']))
692  for key in msg:
693  if key not in DISPLAY_MESSAGE_DEFAULT_FIELDS and key not in ignore_fields:
694  if type(msg[key]) is list:
695  line = '[' + key + ']:\n ' + _ellipse(msg[key], sep='\n ')
696  else:
697  line = '[' + key + ']: ' + clip_text(str(msg.get(key)), max_len)
698  lines.append(space + line)
699  if type(msg.get('image')) == str:
700  lines.append(msg['image'])
701  if msg.get('text', ''):
702  text = clip_text(msg['text'], max_len)
703  ID = '[' + msg['id'] + ']: ' if 'id' in msg else ''
704  lines.append(space + ID + text)
705  for field in {'labels', 'eval_labels', 'label_candidates', 'text_candidates'}:
706  if msg.get(field) and field not in ignore_fields:
707  lines.append('{}[{}: {}]'.format(space, field, _ellipse(msg[field])))
708 
709  if episode_done:
710  lines.append('- - - - - - - - - - - - - - - - - - - - -')
711 
712  return '\n'.join(lines)

844 def padded_tensor(
845  items,
846  pad_idx=0,
847  use_cuda=False,
848  left_padded=False,
849  max_len=None,
850  fp16friendly=False,
851 ):
852  """
853  Create a right-padded matrix from an uneven list of lists.
854 
855  Returns (padded, lengths), where padded is the padded matrix, and lengths
856  is a list containing the lengths of each row.
857 
858  Matrix is right-padded (filled to the right) by default, but can be
859  left padded if the flag is set to True.
860 
861  Matrix can also be placed on cuda automatically.
862 
863  :param list[iter[int]] items: List of items
864  :param bool sort: If True, orders by the length
865  :param int pad_idx: the value to use for padding
866  :param bool use_cuda: if true, places `padded` on GPU
867  :param bool left_padded:
868  :param int max_len: if None, the max length is the maximum item length
869  :param bool fp16friendly: if True, pads the time dimension to be a multiple of 8.
870 
871  :returns: (padded, lengths) tuple
872  :rtype: (Tensor[int64], list[int])
873  """
874  # hard fail if we don't have torch
875  if not __TORCH_AVAILABLE:
876  raise ImportError(
877  "Cannot use padded_tensor without torch; go to http://pytorch.org"
878  )
879 
880  # number of items
881  n = len(items)
882  # length of each item
883  lens = [len(item) for item in items]
884  # max in time dimension
885  t = max(lens) if max_len is None else max_len
886 
887  # if input tensors are empty, we should expand to nulls
888  t = max(t, 1)
889 
890  if fp16friendly and (t % 8 != 0):
891  # pad to be a multiple of 8 to ensure we use the tensor cores
892  t += 8 - (t % 8)
893 
894  if isinstance(items[0], torch.Tensor):
895  # keep type of input tensors, they may already be cuda ones
896  output = items[0].new(n, t)
897  else:
898  output = torch.LongTensor(n, t)
899  output.fill_(pad_idx)
900 
901  for i, (item, length) in enumerate(zip(items, lens)):
902  if length == 0:
903  # skip empty items
904  continue
905  if not isinstance(item, torch.Tensor):
906  # put non-tensors into a tensor
907  item = torch.LongTensor(item)
908  if left_padded:
909  # place at end
910  output[i, t - length :] = item
911  else:
912  # place at beginning
913  output[i, :length] = item
914 
915  if use_cuda:
916  output = output.cuda()
917  return output, lens

920 def padded_3d(tensors, pad_idx=0, use_cuda=0, dtype=TORCH_LONG, fp16friendly=False):
921  """
922  Make 3D padded tensor for list of lists of 1D tensors or lists.
923 
924  :param tensors:
925  list of lists of 1D tensors (or lists)
926  :param pad_idx:
927  padding to fill tensor with
928  :param use_cuda:
929  whether to call cuda() before returning
930  :param bool fp16friendly:
931  if True, pads the final dimension to be a multiple of 8.
932 
933  :returns:
934  3D tensor with the maximum dimensions of the inputs
935  """
936  a = len(tensors)
937  b = max(len(row) for row in tensors)
938  c = max(len(item) for row in tensors for item in row)
939 
940  # pad empty tensors
941  if fp16friendly and c % 8 != 0:
942  c += 8 - (c % 8)
943  c = max(c, 1)
944 
945  output = torch.full((a, b, c), pad_idx, dtype=dtype)
946 
947  for i, row in enumerate(tensors):
948  for j, item in enumerate(row):
949  if len(item) == 0:
950  continue
951  if not isinstance(item, torch.Tensor):
952  item = torch.Tensor(item, dtype=dtype)
953  output[i, j, : len(item)] = item
954 
955  if use_cuda:
956  output = output.cuda()
957 
958  return output
