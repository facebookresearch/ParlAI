20 def _path(opt, filename, add_suffix=False):
21  build(opt)
22  if add_suffix:

54  # Use 'in' to also capture 'train:ordered:stream'

69  path = _path(opt, opt[eval_file_flag], add_suffix=False)

161  def _setup_data(self, path): # Make private method for ParlAIDialogTeacher
162  """Reads data in the fbdialog format.
163 
164  Returns ``((x,y,r,c), new_episode?)`` tuples.
165  """
166  print("[ Loading Self-Feeding text data:" + path + "]")
167  self.episodes = []
168  self.num_exs = 0
169  self.max_train = self.opt.get('max_train', 0)
170  with open(path, 'r') as f:
171  for line in f.readlines():
172  if self.max_train and self.num_exs >= self.max_train:
173  break
174  parley = json.loads(line)
175 
176  # NOTE: History is trimmed here, not by TorchAgent (except in
177  # interactive mode)
178  if self.opt['history_size'] == 0:
179  parley['context'] = '__null__'
180  elif self.opt['history_size'] > 0:
181  utterances = re.split(r'__p\d__', parley['context'])[1:]
182  trimmed = utterances[-self.opt['history_size'] :]
183  parley['context'] = add_person_tokens(trimmed, last_speaker=1)
184 
185  # WARNING: STRIPPING AWAY MEMORIES
186  parley['memories'] = []
187 
188  episode = {
189  'text': parley['context'],
190  'labels': [parley['response']],
191  'label_candidates': parley.get('candidates', []),
192  'reward': parley.get('reward', 0),
193  'episode_done': True,
194  }
195 
196  # Convert integer labels (e.g., polarization dataset) to strings
197  episode['labels'] = [str(l) for l in episode['labels']]
198 
199  self.num_exs += 1
200  self.episodes.append([episode])

203 class SelfFeedingMTLTeacher(core_agents.MultiTaskTeacher):
204  """Creates a teacher that is actually a set of teachers each based on
205  a task string--each of these teachers will get called in turn,
206  either randomly or in order.
207  They are all in the same world (they are the same agent switching tasks).
208 
209  More specifically, this child class of MultiTaskTeacher supports multitask learning
210  with batches (ensuring that all batches only have data from a single task at a time)
211  """
212 
213  def __init__(self, opt, shared=None):
214  if opt['task'] == 'self_feeding:SelfFeedingMTLTeacher':
215  # This does not happen except in tests such as test_new_tasks.py
216  opt = copy.deepcopy(opt)
217  opt['task'] = 'self_feeding:SelfFeedingTeacher'
218  super().__init__(opt, shared)
219 
220  # TODO: allow user to specify other strategies for creating sampling_prob
221  # uniform, batch_proportion, etc.
222  num_batches = np.array([t.num_examples() / t.bsz for t in self.tasks])
223  self.sampling_prob = num_batches / np.sum(num_batches)
224 
225  self.task_idx_assignment = -1 # This will be updated by BatchWorld
226  self.new_task = True
227  self.random = opt.get('datatype') == 'train'

233  def observe(self, observation):
234  return self.tasks[self.task_idx].observe(observation)

236  def batch_act(self, batch_observation):
237  # We use only one task for each batch
238  task_idx = self.get_task_index()
239  actions = []
240  for _ in range(self.tasks[task_idx].bsz):
241  action = self.tasks[task_idx].act()
242  # Pass the name of the task currently being worked on
243  action['subtask'] = self.tasks[task_idx].opt['subtask']
244  actions.append(action)
245  return actions

247  def act(self):
248  self.task_idx = self.get_task_index()
249  if self.task_idx < 0:
250  return {'episode_done': True}
251  action = self.tasks[self.task_idx].act()
252  # Pass the name of the task currently being worked on
253  action['subtask'] = self.tasks[self.task_idx].opt['subtask']
254  return action
255 
256  def get_task_index(self):
257  if self.opt['datatype'] == 'train':
258  return np.random.choice(range(len(self.tasks)), p=self.sampling_prob)
259  else:
260  for i, subtask in enumerate(self.tasks):
261  if not subtask.epoch_done():
262  return i
263  # If this is reached, all tasks are done, so return sentinel
264  return -1
265 
266  # We get most metrics from the agents, not the teachers
267  def report(self):
268  m = {'exs': sum(t.report()['exs'] for t in self.tasks)}
269  return m
