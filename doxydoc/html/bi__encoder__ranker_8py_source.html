169  def score_candidates(self, batch, cand_vecs, cand_encs=None):
170  # Encode contexts first
171  token_idx_ctxt, segment_idx_ctxt, mask_ctxt = to_bert_input(
172  batch.text_vec, self.NULL_IDX
173  )
174  embedding_ctxt, _ = self.model(
175  token_idx_ctxt, segment_idx_ctxt, mask_ctxt, None, None, None
176  )
177 
178  if cand_encs is not None:
179  return embedding_ctxt.mm(cand_encs.t())
180 
181  if len(cand_vecs.size()) == 2 and cand_vecs.dtype == torch.long:
182  # train time. We compare with all elements of the batch
183  token_idx_cands, segment_idx_cands, mask_cands = to_bert_input(
184  cand_vecs, self.NULL_IDX
185  )
186  _, embedding_cands = self.model(
187  None, None, None, token_idx_cands, segment_idx_cands, mask_cands
188  )
189  return embedding_ctxt.mm(embedding_cands.t())
190 
191  # predict time with multiple candidates
192  if len(cand_vecs.size()) == 3:
193  csize = cand_vecs.size() # batchsize x ncands x sentlength
194  cands_idx_reshaped = cand_vecs.view(csize[0] * csize[1], csize[2])
195  token_idx_cands, segment_idx_cands, mask_cands = to_bert_input(
196  cands_idx_reshaped, self.NULL_IDX
197  )
198  _, embedding_cands = self.model(
199  None, None, None, token_idx_cands, segment_idx_cands, mask_cands
200  )
201  embedding_cands = embedding_cands.view(
202  csize[0], csize[1], -1
203  ) # batchsize x ncands x embed_size
204  embedding_cands = embedding_cands.transpose(
205  1, 2
206  ) # batchsize x embed_size x ncands
207  embedding_ctxt = embedding_ctxt.unsqueeze(1) # batchsize x 1 x embed_size
208  scores = torch.bmm(
209  embedding_ctxt, embedding_cands
210  ) # batchsize x 1 x ncands
211  scores = scores.squeeze(1) # batchsize x ncands
212  return scores
213 
214  # otherwise: cand_vecs should be 2D float vector ncands x embed_size
215  return embedding_ctxt.mm(cand_vecs.t())

267 def to_bert_input(token_idx, null_idx):
268  """ token_idx is a 2D tensor int.
269  return token_idx, segment_idx and mask
270  """
271  segment_idx = token_idx * 0
272  mask = token_idx != null_idx
273  # nullify elements in case self.NULL_IDX was not 0
274  token_idx = token_idx * mask.long()
275  return token_idx, segment_idx, mask
