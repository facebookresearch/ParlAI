334  def format_model_reply(self, text):
335  switch_list = [(' .', '.'), (' ,', ','), (' ?', '?'), (' !', '!'), (" ' ", "'")]
336  # add the spaces so that
337  new_text = text.lower()
338 
339  # normalize in case of human:
340  for new, old in switch_list:
341  new_text = new_text.replace(old, new).replace(' ', ' ')
342 
343  # split on punctuation to find sentence boundaries
344  # capitalize stuff
345  tokens = new_text.split(' ')
346  for i in range(len(tokens)):
347  if i == 0:
348  tokens[i] = uppercase(tokens[i])
349  elif tokens[i] in ('i', "i'm", "i've", "i'll", "i'd"):
350  tokens[i] = uppercase(tokens[i])
351  elif tokens[i] in '?.!' and i < len(tokens) - 1:
352  tokens[i + 1] = uppercase(tokens[i + 1])
353  new_text = ' '.join(tokens)
354  new_text = ' ' + new_text + ' '
355 
356  for tup in switch_list:
357  new_text = new_text.replace(tup[0], tup[1])
358 
359  # get rid of surrounding whitespace
360  new_text = new_text.strip()
361  new_text = new_text.replace(' ', ' ')
362 
363  return new_text
