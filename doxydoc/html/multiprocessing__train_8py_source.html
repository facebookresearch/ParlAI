36 def multiprocess_train(
37  rank, opt, port=61337, rank_offset=0, gpu=None, hostname='localhost'
38 ):
39  """
40  Subprocess which initializes distributed training, and begins training.
41 
42  This should be launched n times for n GPUs; this is handled either in main
43  or via srun.
44 
45  :param int rank: This process's rank - 1. (Starts at -1 ... n - 2). See comments.
46  :param opt: command line options
47  :param int port: A TCP port to use. This will need to be changed to run
48  multiple distributed training setups on the same machine.
49  :param int gpu: Which GPU to use. Defaults to using rank and local devices,
50  but must be manually specified when using many-hosts.
51  :param str hostname: Hostname of the main server.
52  """
53  # Set per-host options
54  opt = copy.deepcopy(opt)
55  # we need to manually adjust the rank differently in multiprocessing
56  # and distributed train
57  rank = rank + rank_offset
58  opt['rank'] = rank
59  if gpu is None:
60  # default assumption is local GPUs
61  gpu = rank % torch.cuda.device_count()
62  opt['gpu'] = gpu
63  # make sure we don't just use whatever GPU was saved in the model file
64  if 'override' not in opt:
65  opt['override'] = {}
66  opt['override']['gpu'] = gpu
67 
68  # Suppress output of workers except the main host.
69  if opt.get('verbose') or rank != 0:
70  print_prefix = '[rank:{:3d}]'.format(rank)
71  else:
72  print_prefix = None
73  suppress_output = not opt.get('verbose') and rank != 0
74 
75  with distributed_utils.override_print(suppress_output, print_prefix):
76  # perform distributed setup, ensuring all hosts are ready
77  torch.cuda.set_device(opt['gpu'])
78  dist.init_process_group(
79  backend="nccl",
80  init_method="tcp://{}:{}".format(hostname, port),
81  world_size=opt['distributed_world_size'],
82  rank=rank,
83  )
84  print("Distributed group initialized")
85 
86  # manual_seed can be a noop without this
87  torch.cuda.init()
88  # make sure all parameters will be in sync
89  torch.manual_seed(42)
90  # force a sync so that no one gets ahead, and all are seeded together
91  distributed_utils.sync_object(None)
92 
93  # Run the actual training
94  return single_train.TrainLoop(opt).train()

97 def launch_and_train(opt, port):
98  """Perform a fork() to many processes."""
99  # Launch multiple subprocesses
100  spawncontext = torch.multiprocessing.spawn(
101  multiprocess_train,
102  # need to give rank offset as 1 to cover the fact that the main
103  # process is rank 0, but that spawn() doesn't let you control rank
104  (opt, port, 1),
105  nprocs=opt['distributed_world_size'] - 1, # main proc will also run loop
106  join=False,
107  )
108 
109  try:
110  retval = multiprocess_train(0, opt, port)
111  spawncontext.join()
112  return retval
113  except KeyboardInterrupt:
114  # tell the subprocesses to stop too
115  for p in spawncontext.processes:
116  if p.is_alive():
117  os.kill(p.pid, signal.SIGINT)
118  raise
