32  else:
33  try:

35  except KeyError:

38  "['LSTM', 'GRU', 'RNN_TANH' or 'RNN_RELU']"
39  )
40  self.rnn = nn.RNN(
41  ninp, nhid, nlayers, nonlinearity=nonlinearity, dropout=dropout
42  )
43  self.decoder = nn.Linear(nhid, ntoken)
44 
45  # Optionally tie weights as in:
46  # "Using the Output Embedding to Improve Language Models" (Press & Wolf 2016)
47  # https://arxiv.org/abs/1608.05859
48  # and
49  # "Tying Word Vectors and Word Classifiers: A Loss Framework for

62  self.decoder.bias.data.fill_(0)
63  self.decoder.weight.data.uniform_(-initrange, initrange)

69  def forward(self, input, hidden, no_pack=False):
70  emb = self.drop(self.encoder(input))
71  # if eval, pack padded sequence (we don't pack during training because
72  # we have no padding in our input samples)
73  if not self.training and not no_pack:
74  emb_lens = [x for x in torch.sum((input > 0).int(), dim=0).data]
75  emb_packed = pack_padded_sequence(emb, emb_lens, batch_first=False)
76  packed_output, hidden = self.rnn(emb_packed, hidden)
77  output, _ = pad_packed_sequence(packed_output, batch_first=False)
78  else:
79  output, hidden = self.rnn(emb, hidden)
80  output = self.drop(output)
81  decoded = self.decoder(
82  output.view(output.size(0) * output.size(1), output.size(2))
83  )
84  return decoded.view(output.size(0), output.size(1), decoded.size(1)), hidden
