59  def score_candidates(self, batch, cand_vecs, cand_encs=None):

61  raise Exception(

63  )
64  num_cands_per_sample = cand_vecs.size(1)
65  bsz = cand_vecs.size(0)
66  text_idx = (
67  batch.text_vec.unsqueeze(1)

75  )
76  scores = self.model(tokens, segments)
