29 def mod_labels(ys, task):
30  if ys is not None:
31  # replace comma-labeled babi tasks with spaces
32  # this is more friendly to our tokenizer which makes commas full tokens
33  # this way models won't be penalized for not generating a comma
34  if task == '8':
35  # holding: labels like 'milk,cookies,football'
36  # replace with spaces 'milk football cookies'
37  ys = [y.replace(',', ' ') for y in ys]
38  elif task == '19':
39  # pathfinding: labels like 'n,e' or 's,w'
40  # replace with spaces, 'n e'
41  ys = [y.replace(',', ' ') for y in ys]
42 
43  return ys
