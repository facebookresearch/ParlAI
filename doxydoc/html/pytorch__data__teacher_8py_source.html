44 class BatchSortCache(object):
45  """
46  Object that encapsulates the functionality of the batch sort cache.
47 
48  Maps episode length to dictionary with following keys:
49 
50  - ``current_idx``: which episode in the list are we at (if simply indexing
51  into list)
52  - ``ep_list``: list of episodes of the length of the key
53  - ``bucket_complete``: if there are no more episodes left to consider in the
54  bucket
55  """
56 
57  @classmethod
58  def create(cls):
59  """Singleton factory."""
60  if not hasattr(cls, 'length_to_eps'):
61  # Maps episode length to list of episodes
62  cls.length_to_eps = {}
63  # Set of episode indices already in the cache
64  cls.ep_indices = set()
65  # List of batches if popping batches
66  cls.batches = []
67  # If all episodes have been loaded into memory
68  cls.load_complete = Value(ctypes.c_bool, False)
69  # Lock to access batches
70  cls.batches_lock = Lock()
71  # Lock to access length_to_eps
72  cls.cache_lock = Lock()
73  # Lock for condition variables
74  cls.fill_cache_lock = RLock()
75  # Condition notifying Loader to add to cache
76  cls.add_to_cache_cv = Condition(lock=cls.fill_cache_lock)
77  # Condition notifying teacher that cache has episodes
78  cls.cache_filled_cv = Condition(lock=cls.fill_cache_lock)
79 
80  @classmethod
81  def destroy(cls):
82  """Singleton destroyer."""
83  if hasattr(cls, 'length_to_eps'):
84  del cls.length_to_eps
85  del cls.ep_indices
86  del cls.batches
87  del cls.load_complete
88  del cls.batches_lock
89  del cls.cache_lock
90  del cls.fill_cache_lock
91  del cls.add_to_cache_cv
92  del cls.cache_filled_cv

94  @classmethod
95  def batch_cache(cls, function):
96  """Create the cache of batches."""
97  max_cache_size = 10000 # Max unseen eps
98  min_cache_size = 1000 # Min unseen eps
99 
100  def get_cache_size():
101  """Return number of available episodes."""
102  return sum(
103  len(v['ep_list']) - v['current_idx']
104  for k, v in cls.length_to_eps.items()
105  )
106 
107  def get_available_buckets(bsz):
108  """Return buckets where there are enough episodes for a batch."""
109  if cls.load_complete.value:
110  return {
111  k: v
112  for k, v in cls.length_to_eps.items()
113  if not v['bucket_complete']
114  or len(v['ep_list']) - v['current_idx'] > 0
115  }
116  else:
117  return {
118  k: v
119  for k, v in cls.length_to_eps.items()
120  if len(v['ep_list']) - v['current_idx'] >= bsz
121  }
122 
123  def reset():
124  """Reset the indices into the buckets."""
125  with cls.cache_lock:
126  for idx in cls.length_to_eps:
127  cls.length_to_eps[idx]['current_idx'] = 0
128  cls.length_to_eps[idx]['bucket_complete'] = False
129 
130  def consolidate(caller):
131  """Consolidate remaining episodes into batches."""
132  cls.load_complete.value = True
133  bsz = caller.bsz
134  batch = []
135  sorted_lengths = sorted(cls.length_to_eps.keys())
136  with cls.cache_lock:
137  if caller.batch_cache_type == 'index':
138  for length in sorted_lengths:
139  current_idx = cls.length_to_eps[length]['current_idx']
140  ep_list = cls.length_to_eps[length]['ep_list']
141  unseen_eps = ep_list[current_idx:]
142  cls.length_to_eps[length]['ep_list'] = ep_list[:current_idx]
143  batch = unseen_eps + batch
144  while len(batch) >= bsz:
145  cls.length_to_eps[length]['ep_list'] += batch[:bsz]
146  batch = batch[bsz:]
147  if len(batch) > 0:
148  cls.length_to_eps[-1] = {
149  'current_idx': 0,
150  'ep_list': batch,
151  'bucket_complete': False,
152  }
153  elif caller.batch_cache_type == 'pop':
154  for length in sorted_lengths:
155  batch += cls.length_to_eps[length]['ep_list']
156  with cls.batches_lock:
157  while len(batch) >= bsz:
158  cls.batches.append(batch[:bsz])
159  batch = batch[bsz:]
160  if len(batch) > 0:
161  with cls.batches_lock:
162  cls.batches.append(batch)
163 
164  def flatten(l):
165  """Flatten a list."""
166  return [item for sublist in l for item in sublist]
167 
168  def put_in_cache(ep_idx, episode, caller):
169  """Put episode `ep_idx` into cache."""
170  length = ep_length(episode[caller.batch_sort_field])
171  lengths = [length] + flatten(
172  [
173  [length + i, length + (i * -1)]
174  for i in range(1, caller.batch_length_range)
175  ]
176  )
177  lengths = [max(i, 1) for i in lengths]
178  in_cache = ep_idx in cls.ep_indices
179  # first check if episode can go in existing bucket
180  if not in_cache:
181  for l in lengths:
182  if l in cls.length_to_eps:
183  with cls.cache_lock:
184  cls.length_to_eps[l]['ep_list'] += [(ep_idx, episode)]
185  cls.ep_indices.add(ep_idx)
186  in_cache = True
187  break
188  # otherwise, make a new bucket
189  if not in_cache:
190  with cls.cache_lock:
191  cls.length_to_eps[length] = {
192  'current_idx': 0,
193  'ep_list': [(ep_idx, episode)],
194  'bucket_complete': False,
195  }
196  cls.ep_indices.add(ep_idx)
197  if ep_idx == caller.dataset.num_episodes() - 1:
198  consolidate(caller)
199  with cls.add_to_cache_cv:
200  cls.cache_filled_cv.notify_all()
201 
202  @wraps(function)
203  def wrapper(*args):
204  """Wrap a function."""
205  # TODO: refactor
206  caller = args[0]
207  batch_sort = caller.batch_sort
208  batch_cache_type = caller.batch_cache_type
209  bsz = caller.bsz
210  if not batch_sort or not caller.datatype.startswith('train'):
211  return function(*args)
212  # If Loader, put episodes in cache
213  if isinstance(caller, LoaderProcess):
214  with cls.add_to_cache_cv:
215  while (
216  get_cache_size() >= max_cache_size
217  and len(get_available_buckets(bsz)) > 0
218  ):
219  cls.cache_filled_cv.notify_all()
220  cls.add_to_cache_cv.wait()
221  idx_and_batch = function(*args)
222  if idx_and_batch is None:
223  return None
224  for ep_index, ep in idx_and_batch[1]:
225  put_in_cache(ep_index, ep, caller)
226  return idx_and_batch
227  # If teacher, return batch of episodes
228  else:
229  teacher = caller
230  num_batches = teacher.num_batches
231  while True:
232  with cls.cache_filled_cv:
233  while not cls.load_complete.value and (
234  get_cache_size() <= min_cache_size
235  or len(get_available_buckets(bsz)) == 0
236  ):
237  cls.add_to_cache_cv.notify()
238  cls.cache_filled_cv.wait()
239  available_buckets = get_available_buckets(bsz)
240  if cls.load_complete.value and batch_cache_type == 'pop':
241  return teacher.batch_idx + 1, random.choice(cls.batches)
242  batch = None
243  available_buckets = get_available_buckets(bsz)
244  if len(available_buckets) != 0:
245  # Pick length index at random
246  length = random.choice(list(available_buckets.keys()))
247  with cls.cache_lock:
248  current_idx = cls.length_to_eps[length]['current_idx']
249  ep_list = cls.length_to_eps[length]['ep_list']
250  num_eps = len(ep_list)
251  if num_eps - current_idx >= bsz:
252  if batch_cache_type == 'pop':
253  batch = ep_list[:bsz]
254  cls.length_to_eps[length]['ep_list'] = ep_list[bsz:]
255  else:
256  batch = ep_list[current_idx : current_idx + bsz]
257  cls.length_to_eps[length]['current_idx'] = (
258  current_idx + bsz
259  )
260  elif cls.load_complete.value and num_eps > 0:
261  if batch_cache_type == 'pop':
262  batch = ep_list
263  elif num_eps - current_idx > 0:
264  batch = ep_list[current_idx:]
265  cls.length_to_eps[length]['current_idx'] = (
266  num_eps - 1
267  )
268  cls.length_to_eps[length]['bucket_complete'] = True
269 
270  if batch is not None:
271  if batch_cache_type == 'pop':
272  with cls.batches_lock:
273  cls.batches.append(batch)
274  elif teacher.batch_idx + 1 >= num_batches:
275  reset()
276  return teacher.batch_idx + 1, batch
277 
278  return wrapper

281 def ep_length(val):
282  """Determine the length of an episode, given the specified value."""
283  if isinstance(val, (int, bytes, bool)):
284  return 1
285  if isinstance(val, str):
286  return len(val.replace('\n', ' ').split(' '))
287  if isinstance(val, (collections.Mapping, collections.Sequence, torch.Tensor)):
288  if isinstance(val, collections.Mapping) and val.get(
289  'deserialized_tensor', False
290  ):
291  return len(val['value'])
292  return len(val)

295 def get_dataset_classes(opt):
296  """
297  Get datasets from the options.
298 
299  To use a custom dataset (as opposed to the StreamDataset or ParlAIDataset),
300  you can subclass the pytorch Dataset class and specify its location on the
301  command line.
302 
303  For example, the VQA v1 task provides a custom dataset, which can
304  be specified on the command line as follows: ``-pytd vqa_v1:VQADataset``
305 
306  Note that if the dataset is named ``DefaultDataset``, then you do
307  not need to specify its name following the colon; e.g., it
308  would just be: ``-pytd vqa_v1``
309  """
310  if 'stream' in opt.get('datatype'):
311  default_dataset = StreamDataset
312  else:
313  default_dataset = ParlAIDataset
314  dataset_name = opt.get('pytorch_teacher_dataset')
315  task_name = opt.get('pytorch_teacher_task')
316  datasets = []
317  if task_name is not None:
318  datasets += [
319  (default_dataset, default_collate, task) for task in task_name.split(',')
320  ]
321  if not dataset_name:
322  return datasets
323  sps = [d.strip() for d in dataset_name.split(',')]
324  for sp in sps:
325  full_task_name = sp
326  repo = 'parlai'
327  if sp.startswith('internal:'):
328  # To switch to local repo, useful for non-public projects
329  # (make a directory called 'parlai_internal' with your private agents)
330  repo = 'parlai_internal'
331  sp = sp[9:]
332  sp = sp.split(':')
333  if '.' in sp[0]:
334  module_name = sp[0]
335  else:
336  dataset = sp[0].lower()
337  module_name = '{}.tasks.{}.agents'.format(repo, dataset)
338  if len(sp) > 1:
339  sp[1] = sp[1][0].upper() + sp[1][1:]
340  dataset = sp[1]
341  if '.' not in sp[0] and 'Dataset' not in dataset:
342  # Reformat from underscore to CamelCase and append "Dataset" to
343  # class name by default if a complete path is not given.
344  words = dataset.split('_')
345  teacher_name = ''
346  for w in words:
347  teacher_name += w[0].upper() + w[1:]
348  dataset = teacher_name + 'Dataset'
349  else:
350  dataset = 'DefaultDataset'
351  my_module = importlib.import_module(module_name)
352  dataset_class = getattr(my_module, dataset)
353 
354  collate = default_collate
355  if hasattr(dataset_class, 'collate'):
356  collate = dataset_class.collate
357  elif opt.get('model', False):
358  agent_class = get_agent_module(opt.get('model'))
359  if hasattr(agent_class, 'collate'):
360  collate = agent_class.collate
361  datasets.append((dataset_class, collate, full_task_name))
362  return datasets

448 def deserialize(obj):
449  """Deserializes lists into Tensors."""
450  keys = list(obj.keys())
451  for key in keys:
452  if type(obj[key]) is dict and obj[key].get('deserialized_tensor', False):
453  dtype = STR_TO_TORCH_DTYPE[obj[key]['type']]
454  val = obj[key]['value']
455  del obj[key]
456  obj[key] = torch.as_tensor(val, dtype=dtype)
457  return obj

600 class PytorchDataTeacher(FixedDialogTeacher):
601  """
602  A teacher that loads data using Pytorch Datasets.
603 
604  For details on how to use, please follow the tutorial here:
605  http://parl.ai/static/docs/tutorial_worlds.html#multiprocessed-pytorch-dataloader
606  """
607 
608  def __init__(self, opt, shared=None):
609  opt['batch_sort'] = False
610  super().__init__(opt, shared)
611  self.use_batch_act = self.bsz > 1
612  self.num_workers = opt['numworkers']
613  self.batch_sort = (
614  opt.get('pytorch_teacher_batch_sort') and 'train' in self.datatype
615  )
616  self.batch_cache_type = opt.get('batch_sort_cache_type')
617  self.batch_sort_field = opt.get('batch_sort_field')
618  # One can specify a collate function to use for preparing a batch
619  self.opt = opt.copy()
620  self.is_shared = shared is not None
621  dataset_classes = self._get_dataset_class(opt)
622  self.ordered = 'ordered' in self.datatype or (
623  'stream' in self.datatype and not opt.get('shuffle')
624  )
625  if self.ordered:
626  # force index for ordered, so that we see every example
627  warn_once(
628  '\nNote: You are using PytorchDataTeacher with ordered '
629  'examples. Please specify `--shuffle` if you would like '
630  'to have examples loaded in randomized order.\n'
631  )
632  self.batch_cache_type = 'index'
633 
634  if not shared:
635  BatchSortCache.create()
636  if len(dataset_classes) > 1:
637  datasets = []

639  dataset_opt = opt.copy()
640  dataset_opt['pytorch_teacher_task'] = task_name
641  dataset_opt['task'] = task_name
642  datasets.append(class_name(dataset_opt))

644  self.id = ','.join([d[2] for d in dataset_classes])
645  self.dataset = ParlAIConcatDataset(datasets)
646  else:
647  class_name, self.collate_fn, task_name = dataset_classes[0]
648  self.id = task_name

651  data_sampler = sampler.SequentialSampler(self.dataset)
