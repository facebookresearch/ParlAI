20 def _transpose_hidden_state(hidden_state):
21  """
22  Transpose the hidden state so that batch is the first dimension.
23 
24  RNN modules produce (num_layers x batchsize x dim) hidden state, but
25  DataParallel expects batch size to be first. This helper is used to
26  ensure that we're always outputting batch-first, in case DataParallel
27  tries to stitch things back together.
28  """
29  if isinstance(hidden_state, tuple):
30  return tuple(map(_transpose_hidden_state, hidden_state))
31  elif torch.is_tensor(hidden_state):
32  return hidden_state.transpose(0, 1)
33  else:
34  raise ValueError("Don't know how to transpose {}".format(hidden_state))

150  def reorder_encoder_states(self, encoder_states, indices):
151  """Reorder encoder states according to a new set of indices."""
152  enc_out, hidden, attn_mask = encoder_states
153 
154  # make sure we swap the hidden state around, apropos multigpu settings
155  hidden = _transpose_hidden_state(hidden)
156 
157  # LSTM or GRU/RNN hidden state?
158  if isinstance(hidden, torch.Tensor):
159  hid, cell = hidden, None
160  else:
161  hid, cell = hidden
162 
163  if not torch.is_tensor(indices):
164  # cast indices to a tensor if needed
165  indices = torch.LongTensor(indices).to(hid.device)
166 
167  hid = hid.index_select(1, indices)
168  if cell is None:
169  hidden = hid
170  else:
171  cell = cell.index_select(1, indices)
172  hidden = (hid, cell)
173 
174  if self.attn_type != 'none':
175  enc_out = enc_out.index_select(0, indices)
176  attn_mask = attn_mask.index_select(0, indices)
177 
178  # and bring it back to multigpu friendliness
179  hidden = _transpose_hidden_state(hidden)
180 
181  return enc_out, hidden, attn_mask

272  def forward(self, xs):
273  """Encode sequence.
274 
275  :param xs: (bsz x seqlen) LongTensor of input token indices
276 
277  :returns: encoder outputs, hidden state, attention mask
278  encoder outputs are the output state at each step of the encoding.
279  the hidden state is the final hidden state of the encoder.
280  the attention mask is a mask of which input values are nonzero.
281  """
282  bsz = len(xs)
283 
284  # embed input tokens
285  xs = self.input_dropout(xs)
286  xes = self.dropout(self.lt(xs))
287  attn_mask = xs.ne(0)
288  try:
289  x_lens = torch.sum(attn_mask.int(), dim=1)
290  xes = pack_padded_sequence(xes, x_lens, batch_first=True)
291  packed = True
292  except ValueError:
293  # packing failed, don't pack then
294  packed = False
295 
296  encoder_output, hidden = self.rnn(xes)
297  if packed:
298  # total_length to make sure we give the proper length in the case
299  # of multigpu settings.
300  # https://pytorch.org/docs/stable/notes/faq.html#pack-rnn-unpack-with-data-parallelism
301  encoder_output, _ = pad_packed_sequence(
302  encoder_output, batch_first=True, total_length=xs.size(1)
303  )
304 
305  if self.dirs > 1:
306  # project to decoder dimension by taking sum of forward and back
307  if isinstance(self.rnn, nn.LSTM):
308  hidden = (
309  hidden[0].view(-1, self.dirs, bsz, self.hsz).sum(1),
310  hidden[1].view(-1, self.dirs, bsz, self.hsz).sum(1),
311  )
312  else:
313  hidden = hidden.view(-1, self.dirs, bsz, self.hsz).sum(1)
314 
315  return encoder_output, _transpose_hidden_state(hidden), attn_mask

368  def forward(self, xs, encoder_output, incremental_state=None):
369  """Decode from input tokens.
370 
371  :param xs: (bsz x seqlen) LongTensor of input token indices
372  :param encoder_output: output from RNNEncoder. Tuple containing
373  (enc_out, enc_hidden, attn_mask) tuple.
374  :param incremental_state: most recent hidden state to the decoder.
375  If None, the hidden state of the encoder is used as initial state,
376  and the full sequence is computed. If not None, computes only the
377  next forward in the sequence.
378 
379  :returns: (output, hidden_state) pair from the RNN.
380 
381  - output is a bsz x time x latentdim matrix. If incremental_state is
382  given, the time dimension will be 1. This value must be passed to
383  the model's OutputLayer for a final softmax.
384  - hidden_state depends on the choice of RNN
385  """
386  enc_state, enc_hidden, attn_mask = encoder_output
387  # in case of multi gpu, we need to transpose back out the hidden state
388  attn_params = (enc_state, attn_mask)
389 
390  if incremental_state is not None:
391  # we're doing it piece by piece, so we have a more important hidden
392  # seed, and we only need to compute for the final timestep
393  hidden = _transpose_hidden_state(incremental_state)
394  # only need the last timestep then
395  xs = xs[:, -1:]
396  else:
397  # starting fresh, or generating from scratch. Use the encoder hidden
398  # state as our start state
399  hidden = _transpose_hidden_state(enc_hidden)
400 
401  if isinstance(hidden, tuple):
402  hidden = tuple(x.contiguous() for x in hidden)
403  else:
404  hidden = hidden.contiguous()
405 
406  # sequence indices => sequence embeddings
407  seqlen = xs.size(1)
408  xes = self.dropout(self.lt(xs))
409 
410  if self.attn_time == 'pre':
411  # modify input vectors with attention
412  # attention module requires we do this one step at a time
413  new_xes = []
414  for i in range(seqlen):
415  nx, _ = self.attention(xes[:, i : i + 1], hidden, attn_params)
416  new_xes.append(nx)
417  xes = torch.cat(new_xes, 1).to(xes.device)
418 
419  if self.attn_time != 'post':
420  # no attn, we can just trust the rnn to run through
421  output, new_hidden = self.rnn(xes, hidden)
422  else:
423  # uh oh, post attn, we need run through one at a time, and do the
424  # attention modifications
425  new_hidden = hidden
426  output = []
427  for i in range(seqlen):
428  o, new_hidden = self.rnn(xes[:, i, :].unsqueeze(1), new_hidden)
429  o, _ = self.attention(o, new_hidden, attn_params)
430  output.append(o)
431  output = torch.cat(output, dim=1).to(xes.device)
432 
433  return output, _transpose_hidden_state(new_hidden)

441 class OutputLayer(nn.Module):
442  """Takes in final states and returns distribution over candidates."""
443 
444  def __init__(
445  self,
446  num_features,
447  embeddingsize,
448  hiddensize,
449  dropout=0,
450  numsoftmax=1,
451  shared_weight=None,
452  padding_idx=-1,
453  ):
454  """Initialize output layer.
455 
456  :param num_features: number of candidates to rank
457  :param hiddensize: (last) dimension of the input vectors
458  :param embeddingsize: (last) dimension of the candidate vectors
459  :param numsoftmax: (default 1) number of softmaxes to calculate.
460  see arxiv.org/abs/1711.03953 for more info.
461  increasing this slows down computation but can
462  add more expressivity to the embeddings.
463  :param shared_weight: (num_features x esz) vector of weights to use as
464  the final linear layer's weight matrix. default
465  None starts with a new linear layer.
466  :param padding_idx: model should output a large negative number for
467  score at this index. if set to -1 (default),
468  this is disabled. if >= 0, subtracts one from
469  num_features and always outputs -1e20 at this
470  index. only used when shared_weight is not None.
471  setting this param helps protect gradient from
472  entering shared embedding matrices.
473  """
474  super().__init__()
475  self.dropout = nn.Dropout(p=dropout)
476 
477  self.padding_idx = padding_idx
478  rng = 1.0 / math.sqrt(num_features)
479  self.bias = Parameter(torch.Tensor(num_features).uniform_(-rng, rng))
480 
481  # embedding to scores
482  if shared_weight is None:
483  # just a regular linear layer
484  self.shared = False
485  self.weight = Parameter(
486  torch.Tensor(num_features, embeddingsize).normal_(0, 1)
487  )
488  else:
489  # use shared weights and a bias layer instead
490  self.shared = True
491  self.weight = shared_weight.weight
492 
493  self.numsoftmax = numsoftmax
494  if numsoftmax > 1:
495  self.esz = embeddingsize
496  self.softmax = nn.Softmax(dim=1)
497  self.prior = nn.Linear(hiddensize, numsoftmax, bias=False)
498  self.latent = nn.Linear(hiddensize, numsoftmax * embeddingsize)
499  self.activation = nn.Tanh()
500  else:
501  # rnn output to embedding
502  if hiddensize != embeddingsize:
503  # learn projection to correct dimensions
504  self.o2e = nn.Linear(hiddensize, embeddingsize, bias=True)
505  else:
506  # no need for any transformation here
507  self.o2e = Identity()
508 
509  def forward(self, input):
510  """Compute scores from inputs.
511 
512  :param input: (bsz x seq_len x num_directions * hiddensize) tensor of
513  states, e.g. the output states of an RNN
514 
515  :returns: (bsz x seqlen x num_cands) scores for each candidate
516  """
517  # next compute scores over dictionary
518  if self.numsoftmax > 1:
519  bsz = input.size(0)
520  seqlen = input.size(1) if input.dim() > 1 else 1
521 
522  # first compute different softmax scores based on input vec
523  # hsz => numsoftmax * esz
524  latent = self.latent(input)
525  active = self.dropout(self.activation(latent))
526  # esz => num_features
527  logit = F.linear(active.view(-1, self.esz), self.weight, self.bias)
528 
529  # calculate priors: distribution over which softmax scores to use
530  # hsz => numsoftmax
531  prior_logit = self.prior(input).view(-1, self.numsoftmax)
532  # softmax over numsoftmax's
533  prior = self.softmax(prior_logit)
534 
535  # now combine priors with logits
536  prob = self.softmax(logit).view(bsz * seqlen, self.numsoftmax, -1)
537  probs = (prob * prior.unsqueeze(2)).sum(1).view(bsz, seqlen, -1)
538  scores = probs.log()
539  else:
540  # hsz => esz, good time for dropout
541  e = self.dropout(self.o2e(input))
542  # esz => num_features
543  scores = F.linear(e, self.weight, self.bias)
544 
545  if self.padding_idx >= 0:
546  scores[:, :, self.padding_idx] = -NEAR_INF
547 
548  return scores

551 class AttentionLayer(nn.Module):
552  """Computes attention between hidden and encoder states.
553 
554  See arxiv.org/abs/1508.04025 for more info on each attention type.
555  """
556 
557  def __init__(
558  self,
559  attn_type,
560  hiddensize,
561  embeddingsize,
562  bidirectional=False,
563  attn_length=-1,
564  attn_time='pre',
565  ):
566  """Initialize attention layer."""
567  super().__init__()
568  self.attention = attn_type
569 
570  if self.attention != 'none':
571  hsz = hiddensize
572  hszXdirs = hsz * (2 if bidirectional else 1)
573  if attn_time == 'pre':
574  # attention happens on the input embeddings
575  input_dim = embeddingsize
576  elif attn_time == 'post':
577  # attention happens on the output of the rnn
578  input_dim = hsz
579  else:
580  raise RuntimeError('unsupported attention time')
581 
582  # linear layer for combining applied attention weights with input
583  self.attn_combine = nn.Linear(hszXdirs + input_dim, input_dim, bias=False)
584 
585  if self.attention == 'local':
586  # local attention over fixed set of output states
587  if attn_length < 0:
588  raise RuntimeError('Set attention length to > 0.')
589  self.max_length = attn_length
590  # combines input and previous hidden output layer
591  self.attn = nn.Linear(hsz + input_dim, attn_length, bias=False)
592  # combines attention weights with encoder outputs
593  elif self.attention == 'concat':
594  self.attn = nn.Linear(hsz + hszXdirs, hsz, bias=False)
595  self.attn_v = nn.Linear(hsz, 1, bias=False)
596  elif self.attention == 'general':
597  # equivalent to dot if attn is identity
598  self.attn = nn.Linear(hsz, hszXdirs, bias=False)
599 
600  def forward(self, xes, hidden, attn_params):
601  """Compute attention over attn_params given input and hidden states.
602 
603  :param xes: input state. will be combined with applied
604  attention.
605  :param hidden: hidden state from model. will be used to select
606  states to attend to in from the attn_params.
607  :param attn_params: tuple of encoder output states and a mask showing
608  which input indices are nonzero.
609 
610  :returns: output, attn_weights
611  output is a new state of same size as input state `xes`.
612  attn_weights are the weights given to each state in the
613  encoder outputs.
614  """
615  if self.attention == 'none':
616  # do nothing, no attention
617  return xes, None
618 
619  if type(hidden) == tuple:
620  # for lstms use the "hidden" state not the cell state
621  hidden = hidden[0]
622  last_hidden = hidden[-1] # select hidden state from last RNN layer
623 
624  enc_out, attn_mask = attn_params
625  bsz, seqlen, hszXnumdir = enc_out.size()
626  numlayersXnumdir = last_hidden.size(1)
627 
628  if self.attention == 'local':
629  # local attention weights aren't based on encoder states
630  h_merged = torch.cat((xes.squeeze(1), last_hidden), 1)
631  attn_weights = F.softmax(self.attn(h_merged), dim=1)
632 
633  # adjust state sizes to the fixed window size
634  if seqlen > self.max_length:
635  offset = seqlen - self.max_length
636  enc_out = enc_out.narrow(1, offset, self.max_length)
637  seqlen = self.max_length
638  if attn_weights.size(1) > seqlen:
639  attn_weights = attn_weights.narrow(1, 0, seqlen)
640  else:
641  hid = last_hidden.unsqueeze(1)
642  if self.attention == 'concat':
643  # concat hidden state and encoder outputs
644  hid = hid.expand(bsz, seqlen, numlayersXnumdir)
645  h_merged = torch.cat((enc_out, hid), 2)
646  # then do linear combination of them with activation
647  active = F.tanh(self.attn(h_merged))
648  attn_w_premask = self.attn_v(active).squeeze(2)
649  elif self.attention == 'dot':
650  # dot product between hidden and encoder outputs
651  if numlayersXnumdir != hszXnumdir:
652  # enc_out has two directions, so double hid
653  hid = torch.cat([hid, hid], 2)
654  enc_t = enc_out.transpose(1, 2)
655  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
656  elif self.attention == 'general':
657  # before doing dot product, transform hidden state with linear
658  # same as dot if linear is identity
659  hid = self.attn(hid)
660  enc_t = enc_out.transpose(1, 2)
661  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
662 
663  # calculate activation scores, apply mask if needed
664  if attn_mask is not None:
665  # remove activation from NULL symbols
666  attn_w_premask.masked_fill_((~attn_mask), -NEAR_INF)
667  attn_weights = F.softmax(attn_w_premask, dim=1)
668 
669  # apply the attention weights to the encoder states
670  attn_applied = torch.bmm(attn_weights.unsqueeze(1), enc_out)
671  # concatenate the input and encoder states
672  merged = torch.cat((xes.squeeze(1), attn_applied.squeeze(1)), 1)
673  # combine them with a linear layer and tanh activation
674  output = torch.tanh(self.attn_combine(merged).unsqueeze(1))
