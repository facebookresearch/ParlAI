154 def pre_process(fname, num_ex, alt_speaker):
155  """
156  Pre-process the given file to bring the conversation in a certain format.
157  :param fname: File name to be processed
158  :param num_ex: Number of conversations to be used
159  :param alt_speaker: Name of other speaker to be used
160 
161  :return: List of tuples of the form: (speaker, speech)
162  """
163  conversation = []
164  with open(fname) as f:
165  lines = f.readlines()
166  random.shuffle(lines)
167  lines = lines[:num_ex]
168  for line in lines:
169  data = json.loads(line)
170  dialogue = data["dialog"]
171  for item in dialogue:
172  if item["speaker"] == "human_evaluator":
173  speaker = "human"
174  else:
175  speaker = alt_speaker
176  conversation += [(speaker, item["text"])]
177  conversation += [(END_OF_CONVO, END_OF_CONVO)]
178 
179  return conversation
