25 class MTurkMultiAgentDialogWorld(MTurkTaskWorld):
26  """Basic world where each agent gets a turn in a round-robin fashion,
27  receiving as input the actions of all other agents since that agent last
28  acted.
29  """
30 
31  def __init__(self, opt, agents=None, shared=None):
32  # Add passed in agents directly.
33  self.agents = agents
34  self.acts = [None] * len(agents)
35  self.episodeDone = False
36 
37  def parley(self):
38  """For each agent, get an observation of the last action each of the
39  other agents took. Then take an action yourself.
40  """
41  acts = self.acts
42  for index, agent in enumerate(self.agents):
43  try:
44  acts[index] = agent.act(timeout=None)
45  except TypeError:
46  acts[index] = agent.act() # not MTurkAgent
47  if acts[index]['episode_done']:
48  self.episodeDone = True
49  for other_agent in self.agents:
50  if other_agent != agent:
51  other_agent.observe(validate(acts[index]))
52 
53  def episode_done(self):
54  return self.episodeDone
55 
56  def shutdown(self):
57  """Shutdown all mturk agents in parallel, otherwise if one mturk agent
58  is disconnected then it could prevent other mturk agents from
59  completing.
60  """
61  global shutdown_agent
62 
63  def shutdown_agent(agent):
64  try:
65  agent.shutdown(timeout=None)
66  except Exception:
67  agent.shutdown() # not MTurkAgent
68 
69  Parallel(n_jobs=len(self.agents), backend='threading')(
70  delayed(shutdown_agent)(agent) for agent in self.agents
