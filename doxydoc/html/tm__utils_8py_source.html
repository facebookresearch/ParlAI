39 def gen_ep_cheatsheet(convo):
40  """
41  Generates a cheatsheet for a particular conversation (episode).
42  The index and it's significance in the cheatsheet is shown below:
43  0: First index of a USER that has an ASSISTANT reply to it
44  1: Last index of a USER that has an ASSISTANT reply to it
45  2: First index of an ASSISTANT that has a USER reply to it
46  3: Last index of an ASSISTANT that has a USER reply to it
47  4: Number of examples for USER speech as text and ASSISTANT speech as label
48  5: Number of examples for ASSISTANT speech as text and USER speech as label
49  :param convo:
50  The dialogue between USER and ASSISTANT [after smoothening]
51  """
52  cheatsheet = [-1, -1, -1, -1, 0, 0]
53  # Assumed that length of convo is greater than two due to filtering cond
54  for idx in range(1, len(convo)):
55  # find first USER with reply
56  if convo[idx - 1]['speaker'] == "USER" and convo[idx]['speaker'] == "ASSISTANT":
57  if cheatsheet[0] == -1:
58  cheatsheet[0] = idx - 1
59  # find last USER with reply
60  cheatsheet[1] = idx - 1
61  # find first ASSISTANT with reply
62  if convo[idx - 1]['speaker'] == "ASSISTANT" and convo[idx]['speaker'] == "USER":
63  if cheatsheet[2] == -1:
64  cheatsheet[2] = idx - 1
65  # find last ASSISTANT with reply
66  cheatsheet[3] = idx - 1
67 
68  # Calculate num_examples only if atleast one user was found
69  if cheatsheet[1] != -1:
70  # Calculate number of user examples
71  cheatsheet[4] = (cheatsheet[1] - cheatsheet[0]) // 2 + 1
72  # Calculate num_examples only if atleast one user was found
73  if cheatsheet[3] != -1:
74  # Calculate number of assistant examples
75  cheatsheet[5] = (cheatsheet[3] - cheatsheet[2]) // 2 + 1
76 
77  return cheatsheet

94 def join_speech(utt1, utt2):
95  """
96  Joins two conversations using a '\n'
97  Join texts and doesn't care about segments
98  Assumption: utt1 is the one being popped off from the stack and the utt2
99  is the one trying to be added. This is so that I check for 'ctr'
100  field only in one of the parameters.
101  :param utt1:
102  An utterance (json object) from either USER or ASSISTANT
103  :param utt2:
104  Next utterance(json object) from a speaker same as utt1
105  """
106  new_utt = {}
107  new_utt["index"] = utt1["index"]
108  new_utt["text"] = utt1["text"] + "\n" + utt2["text"]
109  new_utt["speaker"] = utt1["speaker"]
110  if 'ctr' in utt1:
111  new_utt['ctr'] = utt1['ctr'] + 1
112  else:
113  new_utt['ctr'] = 2
114  return new_utt

117 def smoothen_convo(conversation, opt):
118  """
119  Aggregates contiguous responses by the same speaker in the data so that data
120  eventually contains alternations between USER and ASSISTANT
121  :param conversation:
122  The dialogue between USER and ASSISTANT with possible multiple contiguous
123  speeches by the same speaker
124  :param opt:
125  options dict, mainly useful for accessing value of exclude_invalid_data
126  """
127  dialogue = conversation['utterances']
128  conversation_stack = []
129  for speech in dialogue:
130  if (
131  conversation_stack
132  and speech["speaker"] == conversation_stack[-1]["speaker"]
133  ):
134  conversation_stack.append(join_speech(conversation_stack.pop(), speech))
135  else:
136  conversation_stack.append(speech)
137  processed_conversation = []
138  corrupt = False

140  if (

144  ):
145  corrupt = True
