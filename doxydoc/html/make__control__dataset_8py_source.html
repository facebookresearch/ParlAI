41 
42  print('[ starting to convert.. ]')
43  print('[ saving output to {} ]'.format(outfile))
44  fw = open(outfile, 'w')
45  log_timer = TimeLogger()
46 
47  for _ in range(num_examples):
48  world.parley()
49  world.acts[0]['labels'] = world.acts[0].get(
50  'labels', world.acts[0].pop('eval_labels', None)
51  )
52 
53  # Need to get history in order to compute control values
54  hist = ConvAI2History(world.acts[0]['text'], assume_persontokens=False)
55  response = world.acts[0]['labels'][0]
56 
57  # Compute control values
58  for ctrl in controls:
59  ctrl_val = eval_attr(response, hist, ctrl)
60  if ctrl == 'avg_nidf':
61  assert ctrl_val >= 0
62  assert ctrl_val <= 1
63  elif ctrl == 'question':
64  assert ctrl_val in [0, 1]
65  elif ctrl == 'lastuttsim':
66  if ctrl_val is not None:
67  assert ctrl_val >= -1
68  assert ctrl_val <= 1
69  else:
70  raise Exception('unexpected ctrl name: %s' % ctrl)
71  world.acts[0][ctrl] = ctrl_val # add control value to act
72 
73  # Write to file
74  txt = msg_to_str(world.acts[0], ignore_fields=ignorefields)
75  fw.write(txt + '\n')
76  if world.acts[0].get('episode_done', False):
77  fw.write('\n')
78 
79  if log_timer.time() > opt['log_every_n_secs']:
80  text, _log = log_timer.log(world.total_parleys, world.num_examples())
81  print(text)
82 
83  if world.epoch_done():
84  print('EPOCH DONE')
