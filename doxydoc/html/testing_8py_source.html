82 class retry(object):
83  """
84  Decorator for flaky tests. Test is run up to ntries times, retrying on failure.
85 
86  On the last time, the test will simply fail.
87 
88  >>> @retry(ntries=10)
89  ... def test_flaky(self):
90  ... import random
91  ... self.assertLess(0.5, random.random())
92  """
93 
94  def __init__(self, ntries=3):
95  self.ntries = ntries
96 
97  def __call__(self, testfn):
98  """Call testfn(), possibly multiple times on failureException."""
99  from functools import wraps
100 
101  @wraps(testfn)
102  def _wrapper(testself, *args, **kwargs):
103  for _ in range(self.ntries - 1):
104  try:
105  return testfn(testself, *args, **kwargs)
106  except testself.failureException:
107  pass
108  # last time, actually throw any errors there may be
109  return testfn(testself, *args, **kwargs)
110 
111  return _wrapper

218 def train_model(opt):
219  """
220  Run through a TrainLoop.
221 
222  If model_file is not in opt, then this helper will create a temporary
223  directory to store the model, dict, etc.
224 
225  :return: (stdout, valid_results, test_results)
226  :rtype: (str, dict, dict)
227  """
228  import parlai.scripts.train_model as tms
229 
230  with capture_output() as output:
231  with tempdir() as tmpdir:
232  if 'model_file' not in opt:
233  opt['model_file'] = os.path.join(tmpdir, 'model')
234  if 'dict_file' not in opt:
235  opt['dict_file'] = os.path.join(tmpdir, 'model.dict')
236  parser = tms.setup_args()
237  # needed at the very least to set the overrides.
238  parser.set_params(**opt)
239  parser.set_params(log_every_n_secs=10)
240  popt = parser.parse_args(print_args=False)
241  # in some rare cases, like for instance if the model class also
242  # overrides its default params, the params override will not
243  # be taken into account.
244  for k, v in opt.items():
245  popt[k] = v
246  tl = tms.TrainLoop(popt)
247  valid, test = tl.train()
248 
249  return (output.getvalue(), valid, test)

253  """
254  Run through an evaluation loop.
255 
256  :param opt:
257  Any non-default options you wish to set.
258  :param bool skip_valid:

268  by disabling autocleanup
269  """
270  import parlai.scripts.eval_model as ems
271 
272  parser = ems.setup_args()
273  parser.set_params(**opt)
274  parser.set_params(log_every_n_secs=10)
275  popt = parser.parse_args(print_args=False)
276 
277  if popt.get('model_file') and not popt.get('dict_file'):
278  popt['dict_file'] = popt['model_file'] + '.dict'
279 
280  with capture_output() as output:
281  popt['datatype'] = 'valid'
282  valid = None if skip_valid else ems.eval_model(popt)
283  popt['datatype'] = 'test'
284  test = None if skip_test else ems.eval_model(popt)
285 
286  return (output.getvalue(), valid, test)
