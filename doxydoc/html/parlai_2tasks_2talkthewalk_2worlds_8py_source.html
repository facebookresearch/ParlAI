248  def step_aware(self, action, loc, boundaries):
249  orientations = ['N', 'E', 'S', 'W']
250  steps = dict()
251  steps['N'] = [0, 1]
252  steps['E'] = [1, 0]
253  steps['S'] = [0, -1]
254  steps['W'] = [-1, 0]
255 
256  new_loc = copy.deepcopy(loc)
257  if action == 'ACTION:TURNLEFT':
258  # turn left
259  new_loc[2] = (new_loc[2] - 1) % 4
260 
261  if action == 'ACTION:TURNRIGHT':
262  # turn right
263  new_loc[2] = (new_loc[2] + 1) % 4
264 
265  if action == 'ACTION:FORWARD':
266  # move forward
267  orientation = orientations[loc[2]]
268  new_loc[0] = new_loc[0] + steps[orientation][0]
269  new_loc[1] = new_loc[1] + steps[orientation][1]
270 
271  new_loc[0] = min(max(new_loc[0], boundaries[0]), boundaries[2])
272  new_loc[1] = min(max(new_loc[1], boundaries[1]), boundaries[3])
273  return new_loc
