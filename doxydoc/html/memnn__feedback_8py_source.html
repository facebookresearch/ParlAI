255  def backward(self, loss, retain_graph=False):
256  # zero out optimizer and take one optimization step
257  for o in self.optimizers.values():
258  o.zero_grad()
259  loss.backward(retain_graph=retain_graph)

263  o.step()

265  def parse_cands(self, cand_answers):
266  """Returns:
267  cand_answers = tensor (vector) of token indices for answer candidates
268  cand_answers_lengths = tensor (vector) with lengths of each answer candidate
269  """
270  parsed_cands = [to_tensors(c, self.dict) for c in cand_answers]
271  cand_answers_tensor = torch.cat([x[1] for x in parsed_cands])
272  max_cands_len = max([len(c) for c in cand_answers])
273  cand_answers_lengths = torch.LongTensor(
274  len(cand_answers), max_cands_len
275  ).zero_()
276  for i in range(len(cand_answers)):
277  if len(parsed_cands[i][0]) > 0:
278  cand_answers_lengths[i, -len(parsed_cands[i][0]) :] = parsed_cands[i][0]
279  cand_answers_tensor = Variable(cand_answers_tensor)
280  cand_answers_lengths = Variable(cand_answers_lengths)
281  return cand_answers_tensor, cand_answers_lengths

283  def get_cand_embeddings_with_added_beta(self, cands, selected_answer_inds):
284  # add beta_word to the candidate selected by the learner to indicate learner's answer
285  cand_answers_with_beta = copy.deepcopy(cands)
286 
287  for i in range(len(cand_answers_with_beta)):
288  cand_answers_with_beta[i][selected_answer_inds[i]] += ' ' + self.beta_word
289 
290  # get candidate embeddings after adding beta_word to the selected candidate
291  cand_answers_tensor_with_beta, cand_answers_lengths_with_beta = self.parse_cands(
292  cand_answers_with_beta
293  )
294  cands_embeddings_with_beta = self.model.answer_embedder(
295  cand_answers_lengths_with_beta, cand_answers_tensor_with_beta
296  )
297  if self.opt['cuda']:
298  cands_embeddings_with_beta = cands_embeddings_with_beta.cuda()
299  return cands_embeddings_with_beta

301  def predict(self, xs, answer_cands, ys=None, feedback_cands=None):
302  is_training = ys is not None
303  if is_training and 'FP' not in self.model_setting:
304  # Subsample to reduce training time
305  answer_cands = [
306  list(set(random.sample(c, min(32, len(c))) + self.labels))
307  for c in answer_cands
308  ]
309  else:
310  # rank all cands to increase accuracy
311  answer_cands = [list(set(c)) for c in answer_cands]
312 
313  self.model.train(mode=is_training)
314 
315  # Organize inputs for network (see contents of xs and ys in batchify method)
316  inputs = [Variable(x, volatile=is_training) for x in xs]
317 
318  if self.decoder:
319  output_embeddings = self.model(*inputs)
320  self.decoder.train(mode=is_training)
321  output_lines, loss = self.decode(output_embeddings, ys)
322  predictions = self.generated_predictions(output_lines)
323  self.backward(loss)
324  return predictions
325 
326  scores = None
327  if is_training:
328  label_inds = [
329  cand_list.index(self.labels[i])
330  for i, cand_list in enumerate(answer_cands)
331  ]
332 
333  if 'FP' in self.model_setting:
334  if len(feedback_cands) == 0:
335  print(
336  'FP is not training... waiting for negative feedback examples'
337  )
338  else:
339  cand_answers_embs_with_beta = self.get_cand_embeddings_with_added_beta(
340  answer_cands, label_inds
341  )
342  scores, forward_prediction_output = self.model(
343  *inputs, answer_cands, cand_answers_embs_with_beta
344  )
345  fp_scores = self.model.get_score(
346  feedback_cands, forward_prediction_output, forward_predict=True
347  )
348  feedback_label_inds = [
349  cand_list.index(self.feedback_labels[i])
350  for i, cand_list in enumerate(feedback_cands)
351  ]
352  if self.opt['cuda']:
353  feedback_label_inds = Variable(
354  torch.cuda.LongTensor(feedback_label_inds)
355  )
356  else:
357  feedback_label_inds = Variable(
358  torch.LongTensor(feedback_label_inds)
359  )
360  loss_fp = self.loss_fn(fp_scores, feedback_label_inds)
361  if loss_fp.data[0] > 100000:
362  raise Exception(
363  "Loss might be diverging. Loss:", loss_fp.data[0]
364  )
365  self.backward(loss_fp, retain_graph=True)
366 
367  if self.opt['cuda']:
368  label_inds = Variable(torch.cuda.LongTensor(label_inds))
369  else:
370  label_inds = Variable(torch.LongTensor(label_inds))
371 
372  if scores is None:
373  output_embeddings = self.model(*inputs)
374  scores = self.model.get_score(answer_cands, output_embeddings)
375 
376  predictions = self.ranked_predictions(answer_cands, scores)
377 
378  if is_training:
379  update_params = True
380  # don't perform regular training if in FP mode
381  if self.model_setting == 'FP':
382  update_params = False
383  elif 'RBI' in self.model_setting:
384  if len(self.rewarded_examples_inds) == 0:
385  update_params = False
386  else:
387  self.rewarded_examples_inds = torch.LongTensor(
388  self.rewarded_examples_inds
389  )
390  if self.opt['cuda']:
391  self.rewarded_examples_inds = self.rewarded_examples_inds.cuda()
392  # use only rewarded examples for training
393  loss = self.loss_fn(
394  scores[self.rewarded_examples_inds, :],
395  label_inds[self.rewarded_examples_inds],
396  )
397  else:
398  # regular IM training
399  loss = self.loss_fn(scores, label_inds)
400 
401  if update_params:
402  self.backward(loss)
403  return predictions

411  def decode(self, output_embeddings, ys=None):
412  batchsize = output_embeddings.size(0)
413  hn = output_embeddings.unsqueeze(0).expand(
414  self.opt['rnn_layers'], batchsize, output_embeddings.size(1)
415  )
416  x = self.model.answer_embedder(
417  Variable(torch.LongTensor([1])), Variable(self.START_TENSOR)
418  )
419  xes = x.unsqueeze(1).expand(x.size(0), batchsize, x.size(1))
420 
421  loss = 0
422  output_lines = [[] for _ in range(batchsize)]
423  done = [False for _ in range(batchsize)]
424  total_done = 0
425  idx = 0
426  while (total_done < batchsize) and idx < self.longest_label:
427  # keep producing tokens until we hit END or max length for each ex
428  if self.opt['cuda']:
429  xes = xes.cuda()
430  hn = hn.contiguous()
431  preds, scores = self.decoder(xes, hn)
432  if ys is not None:
433  y = Variable(ys[0][:, idx])
434  temp_y = y.cuda() if self.opt['cuda'] else y
435  loss += self.loss_fn(scores, temp_y)
436  else:
437  y = preds
438  # use the true token as the next input for better training
439  xes = self.model.answer_embedder(
440  Variable(torch.LongTensor(preds.numel()).fill_(1)), y
441  ).unsqueeze(0)
442 
443  for b in range(batchsize):
444  if not done[b]:
445  token = self.dict.vec2txt(preds.data[b])
446  if token == self.END:
447  done[b] = True
448  total_done += 1
449  else:
450  output_lines[b].append(token)
451  idx += 1
452  return output_lines, loss

460  def parse(self, text):
461  """Returns:
462  query = tensor (vector) of token indices for query
463  query_length = length of query
464  memory = tensor (matrix) where each row contains token indices for a memory
465  memory_lengths = tensor (vector) with lengths of each memory
466  """
467  sp = text.split('\n')
468  query_sentence = sp[-1]
469  query = self.dict.txt2vec(query_sentence)
470  query = torch.LongTensor(query)
471  query_length = torch.LongTensor([len(query)])
472 
473  sp = sp[:-1]
474  sentences = []
475  for s in sp:
476  sentences.extend(s.split('\t'))
477  if len(sentences) == 0:
478  sentences.append(self.dict.null_token)
479 
480  num_mems = min(self.mem_size, len(sentences))
481  memory_sentences = sentences[-num_mems:]
482  memory = [self.dict.txt2vec(s) for s in memory_sentences]
483  memory = [torch.LongTensor(m) for m in memory]
484  memory_lengths = torch.LongTensor([len(m) for m in memory])
485  memory = torch.cat(memory)
486  return (query, memory, query_length, memory_lengths)

488  def batchify(self, obs):
489  """Returns:
490  xs = [memories, queries, memory_lengths, query_lengths]
491  ys = [labels, label_lengths] (if available, else None)
492  cands = list of candidates for each example in batch
493  valid_inds = list of indices for examples with valid observations
494  """
495  exs = [ex for ex in obs if 'text' in ex]
496  valid_inds = [i for i, ex in enumerate(obs) if 'text' in ex]
497  if not exs:
498  return [None] * 5
499 
500  if 'RBI' in self.model_setting:
501  self.rewarded_examples_inds = [
502  i
503  for i, ex in enumerate(obs)
504  if 'text' in ex and ex.get('reward', 0) > 0
505  ]
506 
507  parsed = [self.parse(ex['text']) for ex in exs]
508  queries = torch.cat([x[0] for x in parsed])
509  memories = torch.cat([x[1] for x in parsed])
510  query_lengths = torch.cat([x[2] for x in parsed])
511  memory_lengths = torch.LongTensor(len(exs), self.mem_size).zero_()
512  for i in range(len(exs)):
513  if len(parsed[i][3]) > 0:
514  memory_lengths[i, -len(parsed[i][3]) :] = parsed[i][3]
515  xs = [memories, queries, memory_lengths, query_lengths]
516 
517  ys = None
518  self.labels = [random.choice(ex['labels']) for ex in exs if 'labels' in ex]
519 
520  if len(self.labels) == len(exs):
521  parsed = [self.dict.txt2vec(l) for l in self.labels]
522  parsed = [torch.LongTensor(p) for p in parsed]
523  label_lengths = torch.LongTensor([len(p) for p in parsed]).unsqueeze(1)
524  self.longest_label = max(self.longest_label, label_lengths.max())
525  padded = [
526  torch.cat(
527  (
528  p,
529  torch.LongTensor(self.longest_label - len(p)).fill_(
530  self.END_TENSOR[0]
531  ),
532  )
533  )
534  for p in parsed
535  ]
536  labels = torch.stack(padded)
537  ys = [labels, label_lengths]
538 
539  feedback_cands = []
540  if 'FP' in self.model_setting:
541  self.feedback_labels = [
542  ex['feedback']
543  for ex in exs
544  if 'feedback' in ex and ex['feedback'] is not None
545  ]
546  self.feedback_cands = self.feedback_cands | set(self.feedback_labels)
547 
548  if (
549  len(self.feedback_labels) == len(exs)
550  and len(self.feedback_cands) > self.num_feedback_cands
551  ):
552  feedback_cands = [
553  list(
554  set(
555  random.sample(self.feedback_cands, self.num_feedback_cands)
556  + [feedback]
557  )
558  )
559  for feedback in self.feedback_labels
560  ]
561 
562  cands = [ex['label_candidates'] for ex in exs if 'label_candidates' in ex]
563  # Use words in dict as candidates if no candidates are provided
564  if len(cands) < len(exs):
565  cands = build_cands(exs, self.dict)
566  # Avoid rebuilding candidate list every batch if its the same
567  if self.last_cands != cands:
568  self.last_cands = cands
569  self.last_cands_list = [list(c) for c in cands]
570  cands = self.last_cands_list
571  return xs, ys, cands, valid_inds, feedback_cands

573  def batch_act(self, observations):
574  batchsize = len(observations)
575  batch_reply = [{'id': self.getID()} for _ in range(batchsize)]
576 
577  xs, ys, cands, valid_inds, feedback_cands = self.batchify(observations)
578 
579  if xs is None or len(xs[1]) == 0:
580  return batch_reply
581 
582  # Either train or predict
583  predictions = self.predict(xs, cands, ys, feedback_cands)
584 
585  for i in range(len(valid_inds)):
586  batch_reply[valid_inds[i]]['text'] = predictions[i][0]
587  batch_reply[valid_inds[i]]['text_candidates'] = predictions[i]
588  return batch_reply

620  cands = []

622  if 'label_candidates' in ex:
