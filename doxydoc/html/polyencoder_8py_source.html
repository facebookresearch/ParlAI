148  def score_candidates(self, batch, cand_vecs, cand_encs=None):
149  """Score candidates.
150 
151  The Poly-encoder encodes the candidate and context independently. Then,
152  the model applies additional attention before ultimately scoring a
153  candidate.
154  """
155  bsz = batch.text_vec.size(0)
156  ctxt_rep, ctxt_rep_mask, ctxt_pos, _ = self.model(ctxt_tokens=batch.text_vec)
157 
158  if cand_encs is not None:
159  if bsz == 1:
160  cand_rep = cand_encs
161  else:
162  cand_rep = cand_encs.expand(bsz, cand_encs.size(1), -1)
163  # bsz x num cands x seq len
164  elif len(cand_vecs.shape) == 3:
165  _, _, _, cand_rep = self.model(cand_tokens=cand_vecs)
166  # bsz x seq len (if batch cands) or num_cands x seq len (if fixed cands)
167  elif len(cand_vecs.shape) == 2:
168  _, _, _, cand_rep = self.model(cand_tokens=cand_vecs.unsqueeze(1))
169  num_cands = cand_rep.size(0) # will be bsz if using batch cands
170  cand_rep = cand_rep.expand(num_cands, bsz, -1).transpose(0, 1).contiguous()
171 
172  scores = self.model(
173  ctxt_rep=ctxt_rep,
174  ctxt_rep_mask=ctxt_rep_mask,
175  cand_rep=cand_rep,
176  ctxt_pos=ctxt_pos,
177  )
178  return scores

187 class PolyEncoderModule(torch.nn.Module):
188  """Poly-encoder model.
189 
190  See https://arxiv.org/abs/1905.01969 for more details

193  def __init__(self, opt, dict, null_idx):
194  super(PolyEncoderModule, self).__init__()
195  self.null_idx = null_idx
196  self.encoder_ctxt = self.get_encoder(opt, dict, null_idx, 'none_with_pos_embs')
197  self.encoder_cand = self.get_encoder(opt, dict, null_idx, opt['reduction_type'])
198 
199  self.type = opt['polyencoder_type']
200  self.n_codes = opt['poly_n_codes']
201  self.attention_type = opt['poly_attention_type']
202  self.attention_keys = opt.get('polyencoder_attention_keys', 'context')
203  self.attention_num_heads = opt['poly_attention_num_heads']
204  self.codes_attention_type = opt['codes_attention_type']
205  self.codes_attention_num_heads = opt['codes_attention_num_heads']
206  embed_dim = opt['embedding_size']
207 
208  # In case it's a polyencoder with code.
209  if self.type == 'codes':
210  # experimentally it seems that random with size = 1 was good.
211  codes = torch.empty(self.n_codes, embed_dim)
212  codes = torch.nn.init.uniform_(codes)
213  self.codes = torch.nn.Parameter(codes)
214 
215  # The attention for the codes.
216  if self.codes_attention_type == 'multihead':
217  self.code_attention = MultiHeadAttention(
218  self.codes_attention_num_heads, embed_dim, opt['dropout']
219  )
220  elif self.codes_attention_type == 'sqrt':
221  self.code_attention = PolyBasicAttention(
222  self.type, self.n_codes, dim=2, attn='sqrt', get_weights=False
223  )
224  elif self.codes_attention_type == 'basic':
225  self.code_attention = PolyBasicAttention(
226  self.type, self.n_codes, dim=2, attn='basic', get_weights=False
227  )
228 
229  # The final attention (the one that takes the candidate as key)
230  if self.attention_type == 'multihead':
231  self.attention = MultiHeadAttention(
232  self.attention_num_heads, opt['embedding_size'], opt['dropout']
233  )
234  else:
235  self.attention = PolyBasicAttention(
236  self.type,
237  self.n_codes,
238  dim=2,
239  attn=self.attention_type,
240  get_weights=False,
241  )

284  def attend(self, attention_layer, queries, keys, values, mask):
285  """Apply attention.
286 
287  :param attention_layer:
288  nn.Module attention layer to use for the attention
289  :param queries:
290  the queries for attention
291  :param keys:
292  the keys for attention
293  :param values:
294  the values for attention
295  :param mask:
296  mask for the attention keys
297 
298  :return:
299  the result of applying attention to the values, with weights computed
300  wrt to the queries and keys.
301  """
302  if keys is None:
303  keys = values
304  if isinstance(attention_layer, PolyBasicAttention):
305  return attention_layer(queries, keys, mask_ys=mask, values=values)
306  elif isinstance(attention_layer, MultiHeadAttention):
307  return attention_layer(queries, keys, values, mask)
308  else:
309  raise Exception('Unrecognized type of attention')

311  def encode(self, ctxt_tokens, cand_tokens):
312  """Encode a text sequence.
313 
314  :param ctxt_tokens:
315  2D long tensor, batchsize x sent_len
316  :param cand_tokens:
317  3D long tensor, batchsize x num_cands x sent_len
318  Note this will actually view it as a 2D tensor
319  :return:
320  (ctxt_rep, ctxt_mask, ctxt_pos, cand_rep)
321  - ctxt_rep 3D float tensor, batchsize x n_codes x dim
322  - ctxt_mask byte: batchsize x n_codes (all 1 in case
323  of polyencoder with code. Which are the vectors to use
324  in the ctxt_rep)
325  - ctxt_pos 3D float tensor, batchsize x sent_len x dim
326  - cand_rep (3D float tensor) batchsize x num_cands x dim
327  """
328  cand_embed = None
329  ctxt_rep = None
330  ctxt_rep_mask = None
331  ctxt_pos = None
332  if cand_tokens is not None:
333  assert len(cand_tokens.shape) == 3
334  bsz = cand_tokens.size(0)
335  num_cands = cand_tokens.size(1)
336  cand_embed = self.encoder_cand(cand_tokens.view(bsz * num_cands, -1))
337  cand_embed = cand_embed.view(bsz, num_cands, -1)
338 
339  if ctxt_tokens is not None:
340  assert len(ctxt_tokens.shape) == 2
341  bsz = ctxt_tokens.size(0)
342  # get context_representation. Now that depends on the cases.
343  ctxt_out, ctxt_mask, ctxt_pos = self.encoder_ctxt(ctxt_tokens)
344  att_keys = ctxt_out if self.attention_keys == 'context' else ctxt_pos
345  dim = ctxt_out.size(2)
346 
347  if self.type == 'codes':
348  ctxt_rep = self.attend(
349  self.code_attention,
350  queries=self.codes.repeat(bsz, 1, 1),
351  keys=att_keys,
352  values=ctxt_out,
353  mask=ctxt_mask,
354  )
355  ctxt_pos = None # we don't need this anymore
356  ctxt_rep_mask = ctxt_rep.new_ones(bsz, self.n_codes).byte()
357 
358  elif self.type == 'n_first':
359  # Expand the output if it is not long enough
360  if ctxt_out.size(1) < self.n_codes:
361  difference = self.n_codes - ctxt_out.size(1)
362  extra_rep = ctxt_out.new_zeros(bsz, difference, dim)
363  ctxt_rep = torch.cat([ctxt_out, extra_rep], dim=1)
364  ctxt_pos = torch.cat([ctxt_pos, extra_rep], dim=1)
365  extra_mask = ctxt_mask.new_zeros(bsz, difference)
366  ctxt_rep_mask = torch.cat([ctxt_mask, extra_mask], dim=1)
367  else:
368  ctxt_rep = ctxt_out[:, 0 : self.n_codes, :]
369  ctxt_pos = ctxt_pos[:, 0 : self.n_codes, :]
370  ctxt_rep_mask = ctxt_mask[:, 0 : self.n_codes]
371 
372  return ctxt_rep, ctxt_rep_mask, ctxt_pos, cand_embed

374  def score(self, ctxt_rep, ctxt_rep_mask, ctxt_pos, cand_embed):
375  """Score the candidates.
376 
377  :param ctxt_rep:
378  3D float tensor, bsz x ctxt_len x dim
379  :param ctxt_rep_mask:
380  2D byte tensor, bsz x ctxt_len, in case there are some elements
381  of the ctxt that we should not take into account.
382  :param ctx_pos: 3D float tensor, bsz x sent_len x dim
383  :param cand_embed: 3D float tensor, bsz x num_cands x dim
384 
385  :return: scores, 2D float tensor: bsz x num_cands
386  """
387  # Attention keys determined by self.attention_keys
388  # 'context' == use context final rep; otherwise use context position embs
389  keys = ctxt_rep if self.attention_keys == 'context' else ctxt_pos
390  # reduces the context representation to a 3D tensor bsz x num_cands x dim
391  ctxt_final_rep = self.attend(
392  self.attention, cand_embed, keys, ctxt_rep, ctxt_rep_mask
393  )
394  scores = torch.sum(ctxt_final_rep * cand_embed, 2)
395  return scores

397  def forward(
398  self,
399  ctxt_tokens=None,
400  cand_tokens=None,
401  ctxt_rep=None,
402  ctxt_rep_mask=None,
403  ctxt_pos=None,
404  cand_rep=None,
405  ):
406  """Forward pass of the model.
407 
408  Due to a limitation of parlai, we have to have one single model
409  in the agent. And because we want to be able to use data-parallel,
410  we need to have one single forward() method.
411  Therefore the operation_type can be either 'encode' or 'score'.
412 
413  :param ctxt_tokens:
414  tokenized contexts
415  :param cand_tokens:
416  tokenized candidates
417  :param ctxt_rep:
418  (bsz x num_codes x hsz)
419  encoded representation of the context. If self.type == 'codes', these
420  are the context codes. Otherwise, they are the outputs from the
421  encoder
422  :param ctxt_rep_mask:
423  mask for ctxt rep
424  :param ctxt_pos:
425  position embeddings for the ctxt_rep. If self.type == 'codes', these
426  are None, as their use is earlier in the pipeline.
427  :param cand_rep:
428  encoded representation of the candidates
429  """
430  if ctxt_tokens is not None or cand_tokens is not None:
431  return self.encode(ctxt_tokens, cand_tokens)
432  elif (
433  ctxt_rep is not None and ctxt_rep_mask is not None and cand_rep is not None
434  ):
435  # ctxt_pos can be none, if we are using codes (not first M)
436  return self.score(ctxt_rep, ctxt_rep_mask, ctxt_pos, cand_rep)
437  raise Exception('Unsupported operation')
