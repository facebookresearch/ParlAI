25 def get_word_counts(opt, count_inputs):
26  """Goes through the dataset specified in opt and gets word counts.
27 
28  Inputs:
29  count_inputs: If True, include both input and reply when counting words
30  and utterances. Otherwise, only include reply text.
31 
32  Returns:
33  word_counter_per_sent: a Counter mapping each word to the number of
34  utterances in which it appears.
35  num_sents: int. number of utterances counted
36  """
37  # Create repeat label agent and assign it to the specified task
38  agent = RepeatLabelAgent(opt)
39  world = create_task(opt, agent)
40 
41  # Count word frequency for all words in dataset
42  word_counter_per_sent = Counter()
43  num_sents = 0
44  count = 0
45  log_timer = TimeLogger()
46  while True:
47  count += 1
48 
49  world.parley()
50  reply = world.acts[0].get('labels', world.acts[0].get('eval_labels'))[0]
51 
52  words = reply.split()
53  words_no_dups = list(set(words)) # remove duplicates
54  word_counter_per_sent.update(words_no_dups)
55  num_sents += 1
56 
57  # Optionally count words in input text
58  if count_inputs:
59  input = world.acts[0]['text']
60  input = input.split('\n')[-1] # e.g. in ConvAI2, this removes persona
61  words = input.split()
62  words_no_dups = list(set(words)) # remove duplicates
63  word_counter_per_sent.update(words_no_dups)
64  num_sents += 1
65 
66  if log_timer.time() > opt['log_every_n_secs']:
67  text, _log = log_timer.log(world.total_parleys, world.num_examples())
68  print(text)
69 
70  if world.epoch_done():
71  print('EPOCH DONE')
72  break
73 
74  return word_counter_per_sent, num_sents

128 def load_word2nidf(opt):
129  """
130  Loads word count stats from word2count.pkl file in data/controllable_dialogue,
131  computes NIDF for all words, and returns the word2nidf dictionary.
132 
133  Returns:
134  word2nidf: dict mapping words to their NIDF score (float between 0 and 1)
135  """
136  word2count_fp = os.path.join(opt['datapath'], CONTROLLABLE_DIR, 'word2count.pkl')
137  print("Loading word count stats from %s..." % word2count_fp)
138  with open(word2count_fp, "rb") as f:
139  data = pickle.load(f)
140  num_sents = data['num_sents']
141  print('num_sents: ', num_sents)
142  word2count = data['word2count']
143  min_c = min(word2count.values()) # max count
144  max_c = max(word2count.values()) # min count
145  word2nidf = {
146  w: (math.log(max_c) - math.log(c)) / (math.log(max_c) - math.log(min_c))
147  for w, c in word2count.items()
148  }
149  print("Done loading word2nidf dictionary.")
150  return word2nidf
