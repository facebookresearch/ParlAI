6 """Base script for model-agnostic perplexity evaluation.
7 
8 While resistent to choices of model-added tokens like START and END, this
9 requires fixing a specific vocabulary. Be sure to use the same build_dict
10 parameters for all comparisons.
11 
12 Tokens which are present in the data being evaluated but not in the vocabulary
13 do not contribute to the perplexity score, but they are still sent to the model
14 so the model can update its state. If the token is in the vocabulary but
15 receives a probability score of zero by the model, the model will get a
16 perplexity score of `inf`.
17 
18 This requires agents to implement the following function:
19 
20 def next_word_probability(self, partial_out):
21  Return probability distribution over next words given a partial true output.
22  This is used to calculate the per-word perplexity.
23 
24  Arguments:
25  partial_out -- list of previous "true" words
26 
27  Returns a dict, where each key is a word and each value is a probability
28  score for that word. Unset keys assume a probability of zero.
29 
30  e.g.
31  (previous observation: {'text': 'Run test program.'})
32  [] => {'hello': 1.0}
33  ['hello'] => {'world': 1.0}
34 """

54 class PerplexityWorld(World):
55  """Instead of just calling act/observe on each agent, this world just calls
56  act on the teacher and then calls `next_word_probability` on the agent.
57 
58  The label for each example is parsed by the provided tokenizer, and then
59  for each word in the parsed label the model is given the input and all of
60  the tokens up to the current word and asked to predict the current word.
61 
62  The model must return a probability of any words it thinks are likely in
63  the form of a dict mapping words to scores. If the scores do not sum to 1,
64  they are normalized to do so. If the correct word is not present or has a
65  probablity of zero, it will be assigned a probability of 1e-8.
66 
67  The API of the next_word_probability function which agents must implement
68  is mentioned in the documentation for this file.
69  """

106  def parley(self):
107  action = self.task.act()
108  self.acts[0] = action.copy()
109 
110  # hide labels from model
111  labels = action.get('eval_labels', action.get('labels', None))
112  if 'label_candidates' in action:
113  action.pop('label_candidates')
114  if labels is None:
115  # empty example, move on
116  return
117 
118  parsed = self.dict.tokenize(labels[0])
119  loss = 0
120  num_tokens = 0
121  num_unk = 0
122  self.agent.observe(action)
123  for i in range(len(parsed)):
124  if parsed[i] in self.dict:
125  # only score words which are in the dictionary
126  probs = self.agent.next_word_probability(parsed[:i])
127  # get probability of correct answer, divide by total prob mass
128  prob_true = probs.get(parsed[i], 0)
129  if prob_true > 0:
130  prob_true /= sum((probs.get(k, 0) for k in self.dict.keys()))
131  loss -= math.log(prob_true)
132  else:
133  loss = float('inf')
134  num_tokens += 1
135  else:
136  num_unk += 1
137  with self._lock():
138  self.metrics['exs'] += 1
139  self.metrics['loss'] += loss
140  self.metrics['num_tokens'] += num_tokens
141  self.metrics['num_unk'] += num_unk

180 def eval_ppl(opt, build_dict=None, dict_file=None):

182

186  - keys(): returns an iterator over all tokens in the dictionary

190  the functions above.

192  "dictionary_class" argument (defaults to
193  parlai.core.dict:DictionaryAgent).
194

196  determine the dictionary used for the evaluation.
