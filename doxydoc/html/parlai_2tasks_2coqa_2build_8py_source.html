18 def make_parlai_format(outpath, dtype, data):
19  print('building parlai:' + dtype)
20  with open(os.path.join(outpath, dtype + '.txt'), 'w') as fout:
21  for each in data:
22  output = []
23  story = each['story'].replace('\n', '\\n')
24  for question, ans in zip(each['questions'], each['answers']):
25  question_txt = ''
26  if question['turn_id'] == 1:
27  # include the context in the very first turn
28  question_txt = story + '\\n' + question['input_text']
29  else:
30  question_txt = question['input_text']
31  output.append(
32  'text:{question}\tlabels:{labels}'.format(
33  question=question_txt,
34  labels=ans['input_text'].replace('|', ' __PIPE__ '),
35  )
36  )
37  if question['turn_id'] < len(each['questions']):
38  output.append('\n')
39  output.append('\t\tepisode_done:True\n')
40  fout.write(''.join(output))
