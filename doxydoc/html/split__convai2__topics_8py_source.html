115 def main(opt):
116  """Converts a Fbdialog file of episodes into two self-feeding files (split by topic)
117 
118  All conversations including a word in the provided topic's bag of words will be
119  separated from conversations without those words.
120  """
121  on_topic_exs = []
122  off_topic_exs = []
123  num_episodes = 0
124  for episode in extract_fb_episodes(opt['infile']):
125  num_episodes += 1
126  if opt['min_unit'] == 'episode':
127  if includes_topic(episode, TOPIC):
128  on_topic_exs.extend(episode_to_examples(episode, opt['history_size']))
129  else:
130  off_topic_exs.extend(episode_to_examples(episode, opt['history_size']))
131  elif opt['min_unit'] == 'example':
132  for example in episode_to_examples(episode, opt['history_size']):
133  if includes_topic([example], TOPIC):
134  on_topic_exs.append(example)
135  else:
136  off_topic_exs.append(example)
137 
138  if opt['shuffle']:
139  random.shuffle(on_topic_exs)
140  random.shuffle(off_topic_exs)
141 
142  total = len(on_topic_exs) + len(off_topic_exs)
143  on_pct = len(on_topic_exs) / total
144  print(
145  f"Separated {total} examples (from {num_episodes} episodes) into "
146  f"{len(off_topic_exs)} off-topic and {len(on_topic_exs)} "
147  f"({on_pct * 100:.1f}%) on-topic"
148  )
149 
150  outfile_base, outfile_ext = os.path.splitext(opt['outfile'])
151  unit_prefix = opt['min_unit'][:3]
152  topic_prefix = TOPIC_NAME[:3]
153  on_topic_filename = f"{outfile_base}_{unit_prefix}_{topic_prefix}{outfile_ext}"
154  with open(on_topic_filename, 'w') as outfile:
155  for ex in on_topic_exs:
156  outfile.write(json.dumps(ex.to_dict()) + '\n')
157  off_topic_filename = f"{outfile_base}_{unit_prefix}_no{topic_prefix}{outfile_ext}"
158  with open(off_topic_filename, 'w') as outfile:
159  for ex in off_topic_exs:
160  outfile.write(json.dumps(ex.to_dict()) + '\n')
