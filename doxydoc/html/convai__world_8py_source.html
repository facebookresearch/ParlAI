98  def _get_updates(self):
99  """Make HTTP request to Router Bot for new messages
100  Expecting server response to be like {'ok': True, "result": [...]}
101  :return: list of new messages received since last request
102  """
103  res = requests.get(self.bot_url + '/getUpdates')
104  if res.status_code != 200:
105  print(res.text)
106  self._increase_delay()
107  return {'ok': False, "result": []}
108  elif self.router_bot_pull_delay > self.minimum_pull_delay:
109  self.router_bot_pull_delay = self.minimum_pull_delay
110  return res.json()

119  def _send_message(self, observation, chatID):
120  """Make HTTP request to Router Bot to post new message
121  :param observation: message that will be sent to server
122  :param chatID: id of chat
123  :return: None
124  """
125  if self._is_end_of_conversation(observation['text']):
126  data = {
127  'text': '/end',

130  else:
131  data = {'text': observation['text'], 'evaluation': 0}
132  message = {'chat_id': chatID, 'text': json.dumps(data)}

134  headers = {'Content-Type': 'application/json'}
135 
136  res = requests.post(

138  )

140  print(res.text)

204  def pull_new_messages(self):
205  """Requests the server for new messages and processes every message.
206  If a message starts with '/start' string then a new chat will be created and
207  the message will be added to stack.
208  If a message has the same chat id as already existing chat then it will be
209  added to message stack for this chat.
210  Any other messages will be ignored.
211  If after processing all messages message stack is still empty then new request
212  to server will be performed.
213  :return: None
214  """
215  print('Waiting for new messages from server...', flush=True)
216  while True:
217  time.sleep(self.router_bot_pull_delay)
218  msgs = self._get_updates()
219  if len(msgs["result"]) > 0:
220  for msg in msgs["result"]:
221  print('\nProceed message: %s' % msg)
222  text = self._get_message_text(msg)
223  chatID = self._get_chat_id(msg)
224 
225  if self.chats.get(chatID, None) is not None:
226  print('Message was recognized as part of chat #%s' % chatID)
227  self.messages.append((chatID, text))
228  elif self._is_begin_of_conversation(text):
229  print(
230  'Message was recognised as start of new chat #%s' % chatID
231  )
232  if self.bot_capacity == -1 or 0 <= self.bot_capacity > (
233  len(self.chats) - len(self.finished_chats)
234  ):
235  self._init_chat(chatID)
236  text = self._strip_start_message(text)
237  self.messages.append((chatID, text))
238  print(
239  'New world and agents for chat #%s are created.'
240  % chatID
241  )
242  else:
243  print(
244  'Cannot start new chat #%s due to bot capacity'
245  'limit reached.' % chatID
246  )
247  else:
248  print(
249  'Message was not recognized as part of any chat.'
250  'Message skipped.'
251  )
252  if len(self.messages) > 0:
253  break
254  else:
255  print('Waiting for new messages from server...', flush=True)

257  def parley(self):
258  """Pops next message from stack, gets corresponding chat, agents, world
259  and performs communication between agents.
260  Result of communication will be send to server.
261  If message stack is empty then server will be requested for new messages.
262  :return: None
263  """
264  print('Try to cleanup finished chat before new parley.')
265  self.cleanup_finished_chats()
266 
267  if len(self.messages) == 0:
268  print('Message stack is empty. Try to request new messages from server.')
269  self.pull_new_messages()
270 
271  print('Pop next message from stack')
272 
273  (chatID, text) = self.messages.pop(0)
274  episode_done = self._is_end_of_conversation(text)
275  local_agent = self.chats.get(chatID, None)
276 
277  if local_agent is not None:
278  self.curr_chatID = chatID
279  msg = {
280  'id': 'MasterBot#%s' % chatID,
281  'text': text,
282  'episode_done': episode_done,
283  }
284  local_agent.observe(msg)
285  reply = local_agent.act()
286  self.last_exchanges[chatID] = [msg, reply]
287  if self._is_end_of_conversation(reply['text']) or reply['episode_done']:
288  episode_done = True
289 
290  if self._is_skip_response(reply['text']):
291  print('Skip response from agent for chat #%s' % chatID)
292  else:
293  print('Send response from agent to chat #%s: %s' % (chatID, reply))
294  self._send_message(reply, chatID)
295  else:
296  print('Message was not recognized as part of any chat. Message skipped.')
297 
298  if episode_done:
299  self.finished_chats.add(chatID)
