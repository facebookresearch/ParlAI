87 def get_word_stats(text, agent_dict, bins=(0, 100, 1000, 100000)):
88  """
89  Function which takes text sequence and dict, returns word freq and length statistics
90 
91  :param sequence: text sequence
92  :param agent_dict: can be external dict or dict from the model
93  :param bins: list with range boundaries
94  :return: freqs dictionary, num words, avg word length, avg char length
95  """
96  pred_list = agent_dict.tokenize(text)
97  pred_freq = [agent_dict.freq[word] for word in pred_list]
98  freqs = {i: 0 for i in bins}
99  for f in pred_freq:
100  for b in bins:
101  if f <= b:
102  freqs[b] += 1
103  break
104 
105  wlength = len(pred_list)
106  clength = len(text) # including spaces
107  return freqs, len(pred_freq), wlength, clength
