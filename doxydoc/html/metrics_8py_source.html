62 def aggregate_task_reports(reports, tasks, micro=False):
63  """
64  Aggregate separate task reports into a single report.
65 
66  :param reports: list of report dicts from separate tasks
67  :param tasks: list of tasks
68  :param micro: average per example if True, else average over t
69 
70  :return: aggregated report dicts
71  """
72  if len(reports) == 1:
73  # singular task
74  return reports[0]
75  # multiple tasks, aggregate metrics
76  metrics = {}
77  exs = {}
78  total_report = {'tasks': {}}
79  # collect metrics from all reports
80  for i, report in enumerate(reports):
81  total_report['tasks'][tasks[i]] = report
82  for metric, val in report.items():
83  if metric == 'exs':
84  exs[tasks[i]] = val
85  else:
86  metrics.setdefault(metric, {})[tasks[i]] = val
87  # now aggregate
88  total_exs = sum(exs.values())
89  total_report['exs'] = total_exs
90  for metric, task_vals in metrics.items():
91  if all([isinstance(v, Number) for v in task_vals.values()]):
92  if micro:
93  # average over the number of examples
94  vals = [task_vals[task] * exs[task] for task in tasks]
95  total_report[metric] = round_sigfigs(sum(vals) / total_exs, 4)
96  else: # macro
97  # average over tasks
98  vals = task_vals.values()
99  total_report[metric] = round_sigfigs(sum(vals) / len(vals), 4)
100  # add a warning describing how metrics were averaged across tasks.
101  total_report['warning'] = 'metrics are averaged across tasks'
102  if micro:
103  total_report['warning'] += ' and weighted by the number of examples ' 'per task'
104  return total_report

118 def _prec_recall_f1_score(pred_items, gold_items):
119  """
120  Compute precision, recall and f1 given a set of gold and prediction items.
121 
122  :param pred_items: iterable of predicted values
123  :param gold_items: iterable of gold values
124 
125  :return: tuple (p, r, f1) for precision, recall, f1
126  """
127  common = Counter(gold_items) & Counter(pred_items)
128  num_same = sum(common.values())
129  if num_same == 0:
130  return 0, 0, 0
131  precision = 1.0 * num_same / len(pred_items)
132  recall = 1.0 * num_same / len(gold_items)
133  f1 = (2 * precision * recall) / (precision + recall)
134  return precision, recall, f1

137 def _f1_score(guess, answers):
138  """Return the max F1 score between the guess and *any* answer."""
139  if guess is None or answers is None:
140  return 0
141  g_tokens = normalize_answer(guess).split()
142  scores = [
143  _prec_recall_f1_score(g_tokens, normalize_answer(a).split()) for a in answers
144  ]
145  return max(f1 for p, r, f1 in scores)

148 def _bleu(guess, answers):
149  """Compute approximate BLEU score between guess and a set of answers."""
150  if nltkbleu is None:
151  # bleu library not installed, just return a default value
152  return None
153  # Warning: BLEU calculation *should* include proper tokenization and
154  # punctuation etc. We're using the normalize_answer for everything though,
155  # so we're over-estimating our BLEU scores. Also note that NLTK's bleu is
156  # going to be slower than fairseq's (which is written in C), but fairseq's
157  # requires that everything be in arrays of ints (i.e. as tensors). NLTK's
158  # works with strings, which is better suited for this module.
159  return nltkbleu.sentence_bleu(
160  [normalize_answer(a).split(" ") for a in answers],
161  normalize_answer(guess).split(" "),
162  smoothing_function=nltkbleu.SmoothingFunction(epsilon=1e-12).method1,
163  )

166 def _rouge(guess, answers):
167  global rouge
168  """Compute ROUGE score between guess and *any* answers. Return the best."""
169  if rouge is None:
170  return None, None, None
171  evaluator = rouge.Rouge(metrics=['rouge-n', 'rouge-l'], max_n=2)
172  try:
173  scores = [
174  evaluator.get_scores(normalize_answer(guess), normalize_answer(a))
175  for a in answers
176  ]
177  except LookupError:
178  warn_once(
179  'ROUGE requires nltk punkt tokenizer. Please run '
180  '`python -c "import nltk; nltk.download(\'punkt\')`'
181  )
182  rouge = None
183  return None, None, None
184 
185  scores_rouge1 = [score['rouge-1']['r'] for score in scores]
186  scores_rouge2 = [score['rouge-2']['r'] for score in scores]
187  scores_rougeL = [score['rouge-l']['r'] for score in scores]
188  return max(scores_rouge1), max(scores_rouge2), max(scores_rougeL)

191 def aggregate_metrics(reporters):
192  """Aggregate metrics from multiple reports."""
193  # reporters is a list of teachers or worlds
194  m = {}
195  m['tasks'] = {}
196  sums = {}
197  num_tasks = 0
198  total = 0
199  for i in range(len(reporters)):
200  task_id = reporters[i].getID()
201  task_report = reporters[i].report()
202  for each_metric, value in task_report.items():
203  if isinstance(value, float):
204  sums[each_metric] = 0.0
205  m[each_metric] = 0.0
206  elif isinstance(value, Number):
207  sums[each_metric] = 0
208  m[each_metric] = 0
209 
210  for i in range(len(reporters)):
211  task_id = reporters[i].getID()
212  task_report = reporters[i].report()
213  while task_id in m['tasks']:
214  # prevent name clobbering if using multiple tasks with same ID
215  task_id += '_'
216  m['tasks'][task_id] = task_report
217  total += task_report['exs']
218  found_any = False
219  for k in sums.keys():
220  if k in task_report:
221  sums[k] += task_report[k]
222  found_any = True
223  if found_any:
224  num_tasks += 1
225  m['exs'] = total
226  m['accuracy'] = 0
227  if num_tasks > 0:
228  for k in sums.keys():
229  m[k] = round_sigfigs(sums[k] / num_tasks, 4)
230  return m

291  def _update_ranking_metrics(self, observation, labels):

293  if text_cands is None:
294  return
295  else:
