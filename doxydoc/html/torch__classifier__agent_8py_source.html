121  # set up threshold, only used in binary classification

216  def train_step(self, batch):
217  """Train on a single batch of examples."""
218  if batch.text_vec is None:
219  return Output()
220  self.model.train()
221  self.optimizer.zero_grad()
222 
223  # calculate loss
224  labels = self._get_labels(batch)
225  scores = self.score(batch)
226  loss = self.criterion(scores, labels)
227  loss.backward()
228  self.update_params()
229 
230  # update metrics
231  self.metrics['loss'] += loss.item()
232  self.metrics['examples'] += len(batch.text_vec)
233 
234  # get predictions
235  _, prediction_id = torch.max(scores.cpu(), 1)
236  preds = [self.class_list[idx] for idx in prediction_id]
237  self._update_confusion_matrix(batch, preds)
238 
239  return Output(preds)

241  def eval_step(self, batch):
242  """Train on a single batch of examples."""
243  if batch.text_vec is None:
244  return
245 
246  self.model.eval()
247  scores = self.score(batch)
248  probs = F.softmax(scores, dim=1)
249  if self.threshold is None:
250  _, prediction_id = torch.max(probs.cpu(), 1)
251  else:
252  ref_prob = probs.cpu()[:, 0]
253  # choose ref class if Prob(ref class) > threshold
254  prediction_id = ref_prob <= self.threshold
255  preds = [self.class_list[idx] for idx in prediction_id]
256 
257  if batch.labels is None:
258  # interactive mode
259  if self.opt.get('print_scores', False):
260  preds = self._format_interactive_output(probs, prediction_id)
261  else:
262  labels = self._get_labels(batch)
263  loss = self.criterion(scores, labels)
264  self.metrics['loss'] += loss.item()
265  self.metrics['examples'] += len(batch.text_vec)
266  self._update_confusion_matrix(batch, preds)
267 
268  return Output(preds)

277  def _report_prec_recall_metrics(self, confmat, class_name, metrics):
278  """
279  Use the confusion matrix to compute precision and recall.
280 
281  :param confmat:
282  the confusion matrics
283  :param str class_name:
284  the class name to compute P/R for
285  :param metrics:
286  metrics dictionary to modify
287  :return:
288  the number of examples of each class.
289  """
290  # TODO: document these parameter types.
291  eps = 0.00001 # prevent divide by zero errors
292  true_positives = confmat[(class_name, class_name)]
293  num_actual_positives = (
294  sum([confmat[(class_name, c)] for c in self.class_list]) + eps
295  )
296  num_predicted_positives = (
297  sum([confmat[(c, class_name)] for c in self.class_list]) + eps
298  )
299 
300  recall_str = 'class_{}_recall'.format(class_name)
301  prec_str = 'class_{}_prec'.format(class_name)
302  f1_str = 'class_{}_f1'.format(class_name)
303 
304  # update metrics dict
305  metrics[recall_str] = true_positives / num_actual_positives
306  metrics[prec_str] = true_positives / num_predicted_positives
307  metrics[f1_str] = 2 * (
308  (metrics[recall_str] * metrics[prec_str])
309  / (metrics[recall_str] + metrics[prec_str] + eps)
310  )
311 
312  return num_actual_positives

314  def report(self):
315  """Report loss as well as precision, recall, and F1 metrics."""
316  m = super().report()
317  examples = self.metrics['examples']
318  if examples > 0:
319  m['examples'] = examples
320  m['mean_loss'] = self.metrics['loss'] / examples
321 
322  # get prec/recall metrics
323  confmat = self.metrics['confusion_matrix']
324  if self.opt.get('get_all_metrics'):
325  metrics_list = self.class_list
326  else:
327  # only give prec/recall metrics for ref class
328  metrics_list = [self.ref_class]
329 
330  examples_per_class = []
331  for class_i in metrics_list:
332  class_total = self._report_prec_recall_metrics(confmat, class_i, m)
333  examples_per_class.append(class_total)
334 
335  if len(examples_per_class) > 1:
336  # get weighted f1
337  f1 = 0
338  total_exs = sum(examples_per_class)
339  for i in range(len(self.class_list)):
340  f1 += (examples_per_class[i] / total_exs) * m[
341  'class_{}_f1'.format(self.class_list[i])
342  ]
343  m['weighted_f1'] = f1
344 
345  for k, v in m.items():
346  m[k] = round_sigfigs(v, 4)
347 
348  return m
