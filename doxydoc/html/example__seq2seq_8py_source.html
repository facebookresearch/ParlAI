47 class DecoderRNN(nn.Module):
48  """Generates a sequence of tokens in response to context."""
49 
50  def __init__(self, output_size, hidden_size, numlayers):
51  """Initialize decoder.
52 
53  :param input_size: size of embedding
54  :param hidden_size: size of GRU hidden layers
55  :param numlayers: number of GRU layers
56  """
57  super().__init__()
58  self.hidden_size = hidden_size
59 
60  self.embedding = nn.Embedding(output_size, hidden_size)
61  self.gru = nn.GRU(
62  hidden_size, hidden_size, num_layers=numlayers, batch_first=True
63  )
64  self.out = nn.Linear(hidden_size, output_size)
65  self.softmax = nn.LogSoftmax(dim=2)
66 
67  def forward(self, input, hidden):
68  """Return encoded state.
69 
70  :param input: batch_size x 1 tensor of token indices.
71  :param hidden: past (e.g. encoder) hidden state
72  """
73  emb = self.embedding(input)
74  rel = F.relu(emb)
75  output, hidden = self.gru(rel, hidden)
76  scores = self.softmax(self.out(output))
77  return scores, hidden

196  def v2t(self, vector):
197  """Convert vector to text.
198 
199  :param vector: tensor of token indices.
200  1-d tensors will return a string, 2-d will return a list of strings
201  """
202  if vector.dim() == 1:
203  output_tokens = []
204  # Remove the final END_TOKEN that is appended to predictions
205  for token in vector:
206  if token == self.END_IDX:
207  break
208  else:
209  output_tokens.append(token)
210  return self.dict.vec2txt(output_tokens)
211  elif vector.dim() == 2:
212  return [self.v2t(vector[i]) for i in range(vector.size(0))]
213  raise RuntimeError(
214  'Improper input to v2t with dimensions {}'.format(vector.size())
215  )

222  def train_step(self, batch):
223  """Train model to produce ys given xs.
224 
225  :param batch: parlai.core.torch_agent.Batch, contains tensorized
226  version of observations.
227 
228  Return estimated responses, with teacher forcing on the input sequence
229  (list of strings of length batchsize).
230  """
231  xs, ys = batch.text_vec, batch.label_vec
232  if xs is None:
233  return
234  bsz = xs.size(0)
235  starts = self.START.expand(bsz, 1) # expand to batch size
236  loss = 0
237  self.zero_grad()
238  self.encoder.train()
239  self.decoder.train()
240  target_length = ys.size(1)
241  # save largest seen label for later
242  self.longest_label = max(target_length, self.longest_label)
243 
244  _encoder_output, encoder_hidden = self.encoder(xs)
245 
246  # Teacher forcing: Feed the target as the next input
247  y_in = ys.narrow(1, 0, ys.size(1) - 1)
248  decoder_input = torch.cat([starts, y_in], 1)
249  decoder_output, decoder_hidden = self.decoder(decoder_input, encoder_hidden)
250 
251  scores = decoder_output.view(-1, decoder_output.size(-1))
252  loss = self.criterion(scores, ys.view(-1))
253  loss.backward()
254  self.update_params()
255 
256  _max_score, predictions = decoder_output.max(2)
257  return Output(self.v2t(predictions))

259  def eval_step(self, batch):
260  """Generate a response to the input tokens.
261 
262  :param batch: parlai.core.torch_agent.Batch, contains tensorized
263  version of observations.
264 
265  Return predicted responses (list of strings of length batchsize).
266  """
267  xs = batch.text_vec
268  if xs is None:
269  return
270  bsz = xs.size(0)
271  starts = self.START.expand(bsz, 1) # expand to batch size
272  # just predict
273  self.encoder.eval()
274  self.decoder.eval()
275  _encoder_output, encoder_hidden = self.encoder(xs)
276 
277  predictions = []
278  done = [False for _ in range(bsz)]
279  total_done = 0
280  decoder_input = starts
281  decoder_hidden = encoder_hidden
282 
283  for _ in range(self.longest_label):
284  # generate at most longest_label tokens
285  decoder_output, decoder_hidden = self.decoder(decoder_input, decoder_hidden)
286  _max_score, preds = decoder_output.max(2)
287  predictions.append(preds)
288  decoder_input = preds # set input to next step
289 
290  # check if we've produced the end token
291  for b in range(bsz):
292  if not done[b]:
293  # only add more tokens for examples that aren't done
294  if preds[b].item() == self.END_IDX:
295  # if we produced END, we're done
296  done[b] = True
297  total_done += 1
298  if total_done == bsz:
299  # no need to generate any more
300  break
301  predictions = torch.cat(predictions, 1)
302  return Output(self.v2t(predictions))
