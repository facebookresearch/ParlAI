167  def get(self, episode_idx, entry_idx=0):
168  ep = self.data[episode_idx]
169  action = {
170  'image_id': int(ep['filename'].replace('.jpg', '')),
171  'episode_done': True,
172  'labels': [s['raw'] for s in ep['sentences']],
173  }
174  if self.use_intro:
175  action['text'] = QUESTION
176  if 'train' not in self.datatype:
177  if self.num_cands > 0:
178  labels = action['labels']
179  cands_to_sample = [c for c in self.cands if c not in labels]
180  cands = (
181  random.Random(episode_idx).sample(cands_to_sample, self.num_cands)
182  + labels
183  )
184  random.shuffle(cands)
185  action['label_candidates'] = cands
186  else:
187  action['label_candidates'] = self.cands

190  def next_example(self):
191  """Returns the next example from this dataset after starting to queue

194  ready = None
195  # pull up the currently queued example
196  if self.example is not None:

198  # move the image we loaded in the background into the example
199  image = self.data_queue.get()
200  self.example['image'] = image
201  ready = (self.example, self.imageEpochDone)
202  # get the next base example: super().next_example() calls self.get()
203  self.example, self.imageEpochDone = super().next_example()
204  if self.image_mode != 'none' and 'image_id' in self.example:
205  # load the next image in the background
206  image_id = self.example['image_id']
207  self.submit_load_request(image_id)
208  # Try to return the previously cached example
209  if ready is None:
210  return self.next_example()
