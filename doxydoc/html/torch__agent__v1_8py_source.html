439  def _project_vec(self, vec, target_dim, method='random'):
440  """If needed, project vector to target dimensionality.
441 
442  Projection methods implemented are the following:
443 
444  random - random gaussian matrix multiplication of input vector
445 
446  :param vec: one-dimensional vector
447  :param target_dim: dimension of returned vector
448  :param method: projection method. will be used even if the dim is
449  not changing if method ends in "-force".
450  """
451  pre_dim = vec.size(0)
452  if pre_dim != target_dim or method.endswith('force'):
453  if method.startswith('random'):
454  # random projection
455  if not hasattr(self, 'proj_rp'):
456  self.proj_rp = torch.Tensor(pre_dim, target_dim).normal_()
457  # rescale so we're not destroying norms too much
458  # http://scikit-learn.org/stable/modules/random_projection.html#gaussian-random-projection
459  self.proj_rp /= target_dim
460  return torch.mm(vec.unsqueeze(0), self.proj_rp)
461  else:
462  # TODO: PCA
463  # TODO: PCA + RP
464  # TODO: copy
465  raise RuntimeError(
466  'Projection method not implemented: {}' ''.format(method)
467  )
468  else:
469  return vec

513  def _vectorize_text(
514  self, text, add_start=False, add_end=False, truncate=None, truncate_left=True
515  ):
516  """Return vector from text.
517 
518  :param text: String to vectorize.
519  :param add_start: Add the start token to the front of the tensor.
520  :param add_end: Add the end token to the end of the tensor.
521  :param truncate: Truncate to this many tokens >= 0, or None.
522  :param truncate_left: Truncate from the left side (keep the rightmost
523  tokens). You probably want this True for inputs,
524  False for targets.
525  """
526  vec = self.dict.txt2vec(text)
527  if truncate is None or len(vec) + add_start + add_end < truncate:
528  # simple: no truncation
529  if add_start:
530  vec.insert(0, self.START_IDX)
531  if add_end:
532  vec.append(self.END_IDX)
533  elif truncate_left:
534  # don't check add_start, we know are truncating it
535  if add_end:
536  # add the end token first
537  vec.append(self.END_IDX)
538  vec = vec[len(vec) - truncate :]
539  else:
540  # truncate from the right side
541  # don't check add_end, we know we are truncating it
542  vec = vec[: truncate - add_start]
543  if add_start:
544  # always keep the start token if it's there
545  vec.insert(0, self.START_IDX)
546  tensor = torch.LongTensor(vec)
547  return tensor

558  def _set_text_vec(self, obs, truncate, split_lines):
559  """Sets the 'text_vec' field in the observation.
560 
561  Useful to override to change vectorization behavior"""
562 
563  if 'text_vec' in obs:
564  # check truncation of pre-computed vectors
565  obs['text_vec'] = self._check_truncate(obs['text_vec'], truncate)
566  if split_lines and 'memory_vecs' in obs:
567  obs['memory_vecs'] = [
568  self._check_truncate(m, truncate) for m in obs['memory_vecs']
569  ]
570  elif 'text' in obs:
571  # convert 'text' into tensor of dictionary indices
572  # we don't add start and end to the input
573  if split_lines:
574  # if split_lines set, we put most lines into memory field
575  obs['memory_vecs'] = []
576  for line in obs['text'].split('\n'):
577  obs['memory_vecs'].append(
578  self._vectorize_text(line, truncate=truncate)
579  )
580  # the last line is treated as the normal input
581  obs['text_vec'] = obs['memory_vecs'].pop()
582  else:
583  obs['text_vec'] = self._vectorize_text(obs['text'], truncate=truncate)

636  def vectorize(
637  self, obs, add_start=True, add_end=True, truncate=None, split_lines=False
638  ):
639  """Make vectors out of observation fields and store in the observation.
640 
641  In particular, the 'text' and 'labels'/'eval_labels' fields are
642  processed and a new field is added to the observation with the suffix
643  '_vec'.
644 
645  If you want to use additional fields on your subclass, you can override
646  this function, call super().vectorize(...) to process the text and
647  labels, and then process the other fields in your subclass.
648 
649  :param obs: Single observation from observe function.
650  :param add_start: default True, adds the start token to each label.
651  :param add_end: default True, adds the end token to each label.
652  :param truncate: default None, if set truncates all vectors to the
653  specified length. Note that this truncates to the
654  rightmost for inputs and the leftmost for labels
655  and, when applicable, candidates.
656  :param split_lines: If set, returns list of vectors instead of a single
657  vector for input text, one for each substring after
658  splitting on newlines.
659  """

665  def batchify(
666  self,
667  obs_batch,
668  sort=False,
669  is_valid=lambda obs: 'text_vec' in obs or 'image' in obs,
670  ):
671  """Create a batch of valid observations from an unchecked batch.
672 
673  A valid observation is one that passes the lambda provided to the
674  function, which defaults to checking if the preprocessed 'text_vec'
675  field is present which would have been set by this agent's 'vectorize'
676  function.

679  explanation of each field.

682  this function and return your own "Batch" namedtuple: copy the Batch

684  fields that you want to be able to access. You can then call

687

689  :param sort: Default False, orders the observations by length of
690  vectors. Set to true when using
691  torch.nn.utils.rnn.pack_padded_sequence.

693  the label vectors if available.

695  observation, determines if an observation is valid
696  """
697  if len(obs_batch) == 0:
698  return Batch()

701

703  return Batch()

706

715  x_lens, xs, x_lens, valid_inds, exs, descending=True
716  )

723  if some_labels_avail:
724  field = 'labels' if labels_avail else 'eval_labels'
725 
726  label_vecs = [ex.get(field + '_vec', self.EMPTY) for ex in exs]
727  labels = [ex.get(field + '_choice') for ex in exs]

734  )

737  cands, cand_vecs = None, None
738  if any('label_candidates_vecs' in ex for ex in exs):
739  cands = [ex.get('label_candidates', None) for ex in exs]
740  cand_vecs = [ex.get('label_candidates_vecs', None) for ex in exs]
741 
742  # IMAGE
743  imgs = None
744  if any('image' in ex for ex in exs):
745  imgs = [ex.get('image', None) for ex in exs]
746 
747  # MEMORIES
748  mems = None
749  if any('memory_vecs' in ex for ex in exs):
750  mems = [ex.get('memory_vecs', None) for ex in exs]

755  label_vec=ys,

1098  def advance(self, softmax_probs):
1099  """Advance the beam one step."""
1100  voc_size = softmax_probs.size(-1)
1101  current_length = len(self.all_scores) - 1
1102  if current_length < self.min_length:
1103  # penalize all eos probs to make it decode longer
1104  for hyp_id in range(softmax_probs.size(0)):
1105  softmax_probs[hyp_id][self.eos] = -NEAR_INF
1106  if len(self.bookkeep) == 0:
1107  # the first step we take only the first hypo into account since all
1108  # hypos are the same initially
1109  beam_scores = softmax_probs[0]
1110  else:
1111  # we need to sum up hypo scores and curr softmax scores before topk
1112  # [beam_size, voc_size]
1113  beam_scores = softmax_probs + self.scores.unsqueeze(1).expand_as(
1114  softmax_probs
1115  )
1116  for i in range(self.outputs[-1].size(0)):
1117  if self.block_ngram > 0:
1118  current_hypo = self.partial_hyps[i][1:]
1119  current_ngrams = []
1120  for ng in range(self.block_ngram):
1121  ngrams = Beam.find_ngrams(current_hypo, ng)
1122  if len(ngrams) > 0:
1123  current_ngrams.extend(ngrams)
1124  counted_ngrams = Counter(current_ngrams)
1125  if any(v > 1 for k, v in counted_ngrams.items()):
1126  # block this hypothesis hard
1127  beam_scores[i] = -NEAR_INF
1128 
1129  # if previous output hypo token had eos
1130  # we penalize those word probs to never be chosen
1131  if self.outputs[-1][i] == self.eos:
1132  # beam_scores[i] is voc_size array for i-th hypo
1133  beam_scores[i] = -NEAR_INF
1134 
1135  flatten_beam_scores = beam_scores.view(-1) # [beam_size * voc_size]
1136  with torch.no_grad():
1137  best_scores, best_idxs = torch.topk(
1138  flatten_beam_scores, self.beam_size, dim=-1
1139  )
1140 
1141  self.scores = best_scores
1142  self.all_scores.append(self.scores)
1143  # get the backtracking hypothesis id as a multiple of full voc_sizes
1144  hyp_ids = best_idxs / voc_size
1145  # get the actual word id from residual of the same division
1146  tok_ids = best_idxs % voc_size
1147 
1148  self.outputs.append(tok_ids)
1149  self.bookkeep.append(hyp_ids)
1150  self.partial_hyps = [
1151  self.partial_hyps[hyp_ids[i]] + [tok_ids[i].item()]
1152  for i in range(self.beam_size)
1153  ]
1154 
1155  # check new hypos for eos label, if we have some, add to finished
1156  for hypid in range(self.beam_size):
1157  if self.outputs[-1][hypid] == self.eos:
1158  # this is finished hypo, adding to finished
1159  eostail = self.HypothesisTail(
1160  timestep=len(self.outputs) - 1,
1161  hypid=hypid,
1162  score=self.scores[hypid],
1163  tokenid=self.eos,
1164  )
1165  self.finished.append(eostail)
1166  self.n_best_counter += 1
1167 
1168  if self.outputs[-1][0] == self.eos:
1169  self.eos_top = True
1170  if self.eos_top_ts is None:
1171  self.eos_top_ts = len(self.outputs) - 1
1172 
1173  def done(self):
1174  """Return whether beam search is complete."""
1175  return self.eos_top and self.n_best_counter >= self.min_n_best
