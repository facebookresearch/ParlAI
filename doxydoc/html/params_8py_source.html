100 def get_model_name(opt):
101  """Get the model name from either `--model` or `--model-file`."""
102  model = opt.get('model', None)
103  if model is None:
104  # try to get model name from model opt file
105  model_file = opt.get('model_file', None)
106  if model_file is not None:
107  model_file = modelzoo_path(opt.get('datapath'), model_file)
108  optfile = model_file + '.opt'
109  if os.path.isfile(optfile):
110  new_opt = load_opt_file(optfile)
111  model = new_opt.get('model', None)
112  return model

115 def str2bool(value):
116  """Convert 'yes', 'false', '1', etc. into a boolean."""
117  v = value.lower()
118  if v in ('yes', 'true', 't', '1', 'y'):
119  return True
120  elif v in ('no', 'false', 'f', 'n', '0'):
121  return False
122  else:
123  raise argparse.ArgumentTypeError('Boolean value expected.')

131 def str2class(value):
132  """
133  From import path string, returns the class specified.
134 
135  For example, the string 'parlai.agents.drqa.drqa:SimpleDictionaryAgent'
136  returns <class 'parlai.agents.drqa.drqa.SimpleDictionaryAgent'>.
137  """
138  if ':' not in value:
139  raise RuntimeError('Use a colon before the name of the class.')
140  name = value.split(':')
141  module = importlib.import_module(name[0])
142  return getattr(module, name[1])

145 def class2str(value):
146  """Inverse of params.str2class()."""
147  s = str(value)
148  s = s[s.find('\'') + 1 : s.rfind('\'')] # pull out import path
149  s = ':'.join(s.rsplit('.', 1)) # replace last period with ':'
150  return s

153 def fix_underscores(args):
154  """
155  Convert underscores to hyphens in args.
156 
157  For example, converts '--gradient_clip' to '--gradient-clip'.
158 
159  :param args: iterable, possibly containing args strings with underscores.
160  """
161  if args:
162  new_args = []
163  for a in args:
164  if type(a) is str and a.startswith('-'):
165  a = a.replace('_', '-')
166  new_args.append(a)
167  args = new_args
168  return args

171 class CustomHelpFormatter(argparse.ArgumentDefaultsHelpFormatter):
172  """
173  Produce a custom-formatted `--help` option.
174 
175  See https://goo.gl/DKtHb5 for details.
176  """
177 
178  def __init__(self, *args, **kwargs):
179  kwargs['max_help_position'] = 8
180  kwargs['width'] = 130
181  super().__init__(*args, **kwargs)

183  def _format_action_invocation(self, action):
184  if not action.option_strings or action.nargs == 0:
185  return super()._format_action_invocation(action)
186  default = self._get_default_metavar_for_optional(action)
187  args_string = self._format_args(action, default)
188  return ', '.join(action.option_strings) + ' ' + args_string

191 class ParlaiParser(argparse.ArgumentParser):
192  """
193  Provide an opt-producer and CLI argument parser.
194 
195  Pseudo-extension of ``argparse`` which sets a number of parameters
196  for the ParlAI framework. More options can be added specific to other
197  modules by passing this object and calling ``add_arg()`` or
198  ``add_argument()`` on it.
199 
200  For example, see ``parlai.core.dict.DictionaryAgent.add_cmdline_args``.
201 
202  :param add_parlai_args:
203  (default True) initializes the default arguments for ParlAI
204  package, including the data download paths and task arguments.
205  :param add_model_args:
206  (default False) initializes the default arguments for loading
207  models, including initializing arguments from that model.
208  """
209 
210  def __init__(
211  self, add_parlai_args=True, add_model_args=False, description='ParlAI parser'
212  ):
213  """Initialize the ParlAI argparser."""
214  super().__init__(
215  description=description,
216  allow_abbrev=False,
217  conflict_handler='resolve',
218  formatter_class=CustomHelpFormatter,
219  )
220  self.register('type', 'bool', str2bool)
221  self.register('type', 'floats', str2floats)
222  self.register('type', 'class', str2class)
223  self.parlai_home = os.path.dirname(
224  os.path.dirname(os.path.dirname(os.path.realpath(__file__)))
225  )
226  os.environ['PARLAI_HOME'] = self.parlai_home
227 
228  self.add_arg = self.add_argument
229 
230  # remember which args were specified on the command line
231  self.overridable = {}
232 
233  if add_parlai_args:
234  self.add_parlai_args()

236  self.add_model_args()
