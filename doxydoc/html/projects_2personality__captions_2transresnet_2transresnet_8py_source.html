142  self.fixed_cands_enc = None
143  if self.fcp is not None:
144  with open(self.fcp) as f:

147  print('loading saved cand encodings')

150  cands_enc_file, map_location=lambda cpu, _: cpu

153  print('Extracting cand encodings')
154  self.model.eval()
155  pbar = tqdm.tqdm(
156  total=len(self.fixed_cands),
157  unit='cand',
158  unit_scale=True,
159  desc='Extracting candidate encodings',
160  )
161  fixed_cands_enc = []
162  for _, batch in enumerate(
163  [
164  self.fixed_cands[i : i + 50]
165  for i in range(0, len(self.fixed_cands) - 50, 50)
166  ]
167  ):
168  embedding = self.model(None, None, batch)[1].detach()
169  fixed_cands_enc.append(embedding)
170  pbar.update(50)
171  self.fixed_cands_enc = torch.cat(fixed_cands_enc, 0)
172  torch.save(self.fixed_cands_enc, cands_enc_file)

222  def eval_step(self, valid_obs, image_feats, personalities):
223  """
224  Model eval step.
225 
226  :param valid_obs:
227  list of valid observations
228 
229  :param image_feats:
230  list of image features, one per example
231 
232  :param personalities:
233  list of personalities, one per example
234 
235  :return:
236  the total loss, number of correct examples, the total number of
237  examples evaluated, the ranked position of each correct caption,
238  and the ranked lists of candidates (one per example)
239  """
240  med_rank = None
241  chosen_captions = None
242  if 'label_candidates' in valid_obs[0] or self.fixed_cands is not None:
243  # User provides candidates, used as negatives for evaluation
244  candidates_encoded = None
245  if self.fixed_cands is not None:
246  candidates_encoded = self.fixed_cands_enc
247  candidates = self.fixed_cands
248  else:
249  candidates = [v['label_candidates'] for v in valid_obs]
250  if self.one_cand_set:
251  candidates_encoded = self.model(None, None, candidates[0])[
252  1
253  ].detach()
254  chosen_captions = self.model.choose_best_caption(
255  image_feats,
256  personalities,
257  candidates,
258  candidates_encoded=candidates_encoded,
259  k=-1 if self.fixed_cands is None else 100,
260  )
261  # calculate median ranks
262  num_examples = len(chosen_captions)
263  loss = -1
264  if self.fixed_cands is not None:
265  num_correct = 0
266  else:
267  comments = [v['eval_labels'] for v in valid_obs]
268  med_rank = []
269  for i, c_list in enumerate(chosen_captions):
270  lowest_rank = len(c_list) + 1
271  for _, c in enumerate(comments[i]):
272  lowest_rank = min(lowest_rank, c_list.index(c) + 1)
273  med_rank.append(lowest_rank)
274  num_correct = sum(
275  [
276  1 if chosen_captions[i][0] in chosen_captions[i] else 0
277  for i in range(len(chosen_captions))
278  ]
279  )
280  else:
281  comments = [random.choice(v['eval_labels']) for v in valid_obs]
282  loss, num_correct, num_examples = self.model.eval_batch(
283  image_feats, personalities, comments
284  )
285 
286  return loss, num_correct, num_examples, med_rank, chosen_captions

288  def batch_act(self, observations):
289  """
290  Act on a batch of observations.
291 
292  :param observations:
293  list of observations
294 
295  :return:
296  A list of acts, one for each observation
297  """

300  image_feats = self.extract_image_feats(valid_obs)
301  personalities = [v.get('text', '') for v in valid_obs]
302 
303  chosen_captions = None
304  med_rank = None
305  if is_training:
306  loss, num_correct, num_examples = self.train_step(
307  valid_obs, image_feats, personalities
308  )
309  else:
310  loss, num_correct, num_examples, med_rank, chosen_captions = self.eval_step(
311  valid_obs, image_feats, personalities
312  )
313 
314  self.update_metrics(loss, num_correct, num_examples, med_rank)
315  result = [
316  {'text': 'No Response During Training'} for _ in range(len(observations))
317  ]
318  if chosen_captions is not None:
319  for i, index_obs in enumerate(valid_indexes):
320  result[index_obs]['text'] = chosen_captions[i][0]
321  result[index_obs]['text_candidates'] = chosen_captions[i]
322  return result

324  def extract_image_feats(self, obs):
325  """
326  Extract image features from the observations.
327 
328  :param obs:
329  list of observations
330 
331  :return:
332  list of image features
333  """
334  tmp_image_feats = [v.get('image') for v in obs]
335  for i, im in enumerate(tmp_image_feats):
336  try:
337  # Check if given img features of form [1, <dim>, 1, 1]
338  if len(im.size()) == 4:
339  tmp_image_feats[i] = im[0, :, 0, 0]
340  except TypeError: # No Image Feats Given
341  tmp_image_feats[i] = self.blank_image_features
342  image_feats = []
343  for img in tmp_image_feats:
344  image_feats.append(img.detach())
345  return image_feats

347  def filter_valid_obs(self, observations, is_training):
348  """Filter out invalid observations."""
349  label_key = 'labels' if is_training else 'eval_labels'
350  valid_obs = []

355  if self.fixed_cands is not None:

405  def receive_metrics(self, metrics_dict):
406  """

408

417  m = metrics_dict['hits@1/100']

419  self.freeze_impatience = 0
420  self.freeze_best_metric = m
421  print('performance not good enough to unfreeze the model.')
422  else:
423  self.freeze_impatience += 1
424  print('Growing impatience for unfreezing')
425  if self.freeze_impatience >= self.freeze_patience:
426  self.is_frozen = False
427  print(

429  'Reloading the best model so far.'
430  )
431  self._build_model(self.model_file)
432  if self.use_cuda:
433  self.model = self.model.cuda()
434  print('Unfreezing.')
