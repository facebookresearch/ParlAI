55 def find_ngrams(token_dict, text, n):
56  """
57  Break text into ngrams that appear in ``token_dict``.
58 
59  :param token_dict:
60  ``dict`` to check for ngrams
61  :param text:
62  ``str`` to look for ngrams in
63  :param n:
64  ``int`` max size of ngrams
65  """
66  # base case
67  if n <= 1:
68  return text
69  # tokens committed to output
70  saved_tokens = []
71  # tokens remaining to be searched in sentence
72  search_tokens = text[:]
73  # tokens stored until next ngram found
74  next_search = []
75  while len(search_tokens) >= n:
76  ngram = ' '.join(search_tokens[:n])
77  if ngram in token_dict:
78  # first, search previous unmatched words for smaller ngrams
79  sub_n = min(len(next_search), n - 1)
80  saved_tokens.extend(find_ngrams(token_dict, next_search, sub_n))
81  next_search.clear()
82  # then add this ngram
83  saved_tokens.append(ngram)
84  # then pop this ngram from the remaining words to search
85  search_tokens = search_tokens[n:]
86  else:
87  next_search.append(search_tokens.pop(0))
88  remainder = next_search + search_tokens
89  sub_n = min(len(remainder), n - 1)
90  saved_tokens.extend(find_ngrams(token_dict, remainder, sub_n))
91  return saved_tokens

551  for token, freq in self.freq.items():
552  if freq < min_freq:

561  def _remove_non_bpe(self):
562  """Set the dictionary vocab to the bpe vocab, merging counts."""
563  to_remove = []
564  to_add = []
565  for token, freq in self.freq.items():
566  tokens = self.bpe_tokenize(token)
567  if len(tokens) != 1:
568  for t in tokens:
569  to_add.append((t, freq))

588  def load(self, filename):
589  """
590  Load pre-existing dictionary in 'token[<TAB>count]' format.
591 
592  Initialize counts from other dictionary, or 0 if they aren't included.
593  """
594  print('Dictionary: loading dictionary from {}'.format(filename))
595 
596  lower_special = self.null_token == self.null_token.lower()
597  SPECIAL_TOKENS = {'__UNK__', '__NULL__', '__END__', '__START__'}
598  with codecs.open(filename, 'r', encoding='utf-8', errors='ignore') as read:
599  for line in read:
600  split = line.strip().split('\t')
601  token = unescape(split[0])
602  if lower_special and token in SPECIAL_TOKENS:
603  token = token.lower()
604  cnt = int(split[1]) if len(split) > 1 else 0
605  self.freq[token] = cnt
606  self.add_token(token)
607  print('[ num words = %d ]' % len(self))

609  def save(self, filename=None, append=False, sort=True):
610  """
611  Save dictionary to file.
612 
613  Format is 'token<TAB>count' for every token in the dictionary, sorted
614  by count with the most frequent words first.
615 
616  If ``append`` (default ``False``) is set to ``True``, appends instead of
617  overwriting.
618 
619  If ``sort`` (default ``True``), then first sort the dictionary before saving.
620  """
621  filename = self.opt['dict_file'] if filename is None else filename
622 
623  if self.tokenizer == 'bpe':
624  needs_removal = self.bpehelper.finalize(
625  self.freq, num_symbols=self.maxtokens, minfreq=self.minfreq
626  )
627  if needs_removal:
628  self._remove_non_bpe()
629  elif filename != self.opt['dict_file']:
630  # need to copy over the old codecs file
631  self.bpehelper.copy_codecs_file(filename + '.codecs')
632  if sort:
633  self.sort(trim=False)
634  elif self.tokenizer == 'gpt2':
635  # never remove or sort tokens from gpt2
636  pass
637  elif sort:
638  self.sort(trim=True)
639 
640  print('Dictionary: saving dictionary to {}'.format(filename))
641 
642  make_dir(os.path.dirname(filename))
643  mode = 'a' if append else 'w'
644  with open(filename, mode, encoding='utf-8') as write:
645  for i in self.ind2tok.keys():
646  tok = self.ind2tok[i]
647  cnt = self.freq[tok]
648  write.write('{tok}\t{cnt}\n'.format(tok=escape(tok), cnt=cnt))
649

652  json.dump(self.opt, handle)

654  def sort(self, trim=True):
655  """
656  Sort the dictionary.
657 
658  Inline operation. Rearranges the dictionary so that the elements with
659  the lowest index have the highest counts. This reindexes the dictionary
660  according to the sorted frequencies, breaking ties alphabetically by
661  token.
662 
663  :param bool trim:
664  If True, truncate the dictionary based on minfreq and maxtokens.
665  """
666  if trim and self.tokenizer == 'gpt2':
667  raise RuntimeError("You should not trim the dictionary when using gpt-2.")
668  # sort first by count, then alphabetically
669  if trim:
670  self.remove_tail(self.minfreq)
671  sorted_pairs = sorted(self.freq.items(), key=lambda x: (-x[1], x[0]))
672  new_tok2ind = {}
673  new_ind2tok = {}
674  for i, (tok, _) in enumerate(sorted_pairs):
675  new_tok2ind[tok] = i
676  new_ind2tok[i] = tok
677  self.tok2ind = new_tok2ind
678  self.ind2tok = new_ind2tok
679  if trim:
680  self.resize_to_max(self.maxtokens)
681  assert len(self.freq) == len(self.ind2tok) == len(self.tok2ind)
682  return sorted_pairs

702
