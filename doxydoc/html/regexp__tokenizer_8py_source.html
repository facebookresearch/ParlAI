48  '(?<ellipses>%s)|(?P<punct>%s)|(?P<nonws>%s)'
49  % (
50  self.DIGIT,
51  self.TITLE,
52  self.ABBRV,
53  self.NEGATION,
54  self.HYPHEN,
55  self.CONTRACTION1,
56  self.ALPHA_NUM,
57  self.CONTRACTION2,

64  self.PUNCT,
65  self.NON_WS,
66  ),
67  flags=regex.IGNORECASE + regex.UNICODE + regex.MULTILINE,
68  )
69  if len(kwargs.get('annotators', {})) > 0:
70  logger.warning(
71  '%s only tokenizes! Skipping annotators: %s'
72  % (type(self).__name__, kwargs.get('annotators'))
73  )
74  self.annotators = set()

77  def tokenize(self, text):
78  data = []
79  matches = [m for m in self._regexp.finditer(text)]
80  for i in range(len(matches)):
81  # Get text
82  token = matches[i].group()
83 
84  # Make normalizations for special token types
85  if self.substitutions:
86  groups = matches[i].groupdict()
87  if groups['sdquote']:
88  token = "``"
89  elif groups['edquote']:
90  token = "''"
91  elif groups['ssquote']:
92  token = "`"
93  elif groups['esquote']:
94  token = "'"
95  elif groups['dash']:
96  token = '--'
97  elif groups['ellipses']:
98  token = '...'
99 
100  # Get whitespace
101  span = matches[i].span()
102  start_ws = span[0]
103  if i + 1 < len(matches):
104  end_ws = matches[i + 1].span()[0]
105  else:
106  end_ws = span[1]
107 
108  # Format data
109  data.append((token, text[start_ws:end_ws], span))
110  return Tokens(data, self.annotators)
