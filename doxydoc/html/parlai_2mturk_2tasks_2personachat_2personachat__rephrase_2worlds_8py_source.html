232  def is_exact_match(self, act, ag, persona_data, tolerance=0):
233  if act['episode_done']:
234  return False
235 
236  control_msg = {'episode_done': False}
237  control_msg['id'] = 'SYSTEM'
238 
239  text = act['text']
240  if text not in ['', ' ', ' ', ' ']:
241  n_word_match = 0
242  for per in [persona_data]:
243  per_parse = per.split(' ')
244  regular_words = ['', ' ', 'I', 'I\'m', 'My', 'i']
245  for r_w in regular_words:
246  if r_w in per_parse:
247  per_parse.remove(r_w)
248  per_subseq = [
249  ' '.join(per_parse[i : i + len(per_parse) - tolerance])
250  for i in range(tolerance + 1)
251  ]
252  for pp in per_subseq:
253  if pp in ['', ' ', ' ', ' ']:
254  per_subseq.remove(pp)
255  n_word_match += sum([(paa in text) for paa in per_subseq])
256  if n_word_match > 0:
257  control_msg['text'] = (
258  'We found that you <b><span style="color:red">trivially '
259  'copied character descriptions</span></b>. Please '
260  'rephrase your message again.'
261  )
262  ag.observe(validate(control_msg))
263  return True
264  else:
265  return False
