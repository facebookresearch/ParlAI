99  if self.fcp is not None:
100  with open(self.fcp) as f:

102  cands_enc_file = "{}.cands_enc".format(self.fcp)
103  print("loading saved cand encodings")
104  if os.path.isfile(cands_enc_file):
105  self.fixed_cands_enc = torch.load(
106  cands_enc_file, map_location=lambda cpu, _: cpu
107  )
108  else:
109  print("Extracting cand encodings")
110  self.model.eval()
111  pbar = tqdm.tqdm(
112  total=len(self.fixed_cands),
113  unit="cand",
114  unit_scale=True,
115  desc="Extracting candidate encodings",
116  )
117  fixed_cands_enc = []
118  for _, batch in enumerate(
119  [
120  self.fixed_cands[i : i + 50]
121  for i in range(0, len(self.fixed_cands) - 50, 50)
122  ]
123  ):
124  embedding = self.model.forward_text_encoder(batch).detach()
125  fixed_cands_enc.append(embedding)
126  pbar.update(50)
127  self.fixed_cands_enc = torch.cat(fixed_cands_enc, 0)
128  torch.save(self.fixed_cands_enc, cands_enc_file)

183  def eval_step(self, valid_obs, image_feats, personalities, dialogue_histories):
184  """
185  Model eval step.
186 
187  :param valid_obs:
188  list of valid observations
189 
190  :param image_feats:
191  list of image features, one per example
192 
193  :param personalities:
194  list of personalities, one per example
195 
196  :param dialogue_histories:
197  list of dialogue histories, one per example
198 
199  :return:
200  the total loss, number of correct examples,
201  the ranked position of each correct caption,
202  and the ranked lists of candidates (one per example)
203  """
204  self.model.eval()
205  med_rank = None
206  chosen_responses = None
207  candidates_encoded = None
208 
209  if self.fixed_cands is not None:
210  candidates_encoded = self.fixed_cands_enc
211  candidates = self.fixed_cands
212  else:
213  candidates = [v["label_candidates"] for v in valid_obs]
214  chosen_responses = self.model.choose_best_response(
215  image_feats,
216  personalities,
217  dialogue_histories,
218  candidates,
219  candidates_encoded=candidates_encoded,
220  k=-1 if self.fixed_cands is None else 100,
221  batchsize=len(valid_obs),
222  )
223  loss = -1
224  if self.fixed_cands is not None:
225  num_correct = 0
226  else:
227  labels = [v.get("eval_labels") for v in valid_obs]
228  equality_list = [
229  1 if chosen_responses[i][0] in labels[i] else 0
230  for i in range(len(labels))
231  ]
232  # calculate med ranks
233  med_rank = []
234  for i, e_list in enumerate(chosen_responses):
235  lowest_rank = len(e_list) + 1
236  for c in labels[i]:
237  lowest_rank = min(lowest_rank, e_list.index(c) + 1)
238  med_rank.append(lowest_rank)
239  num_correct = sum(equality_list)
240 
241  return loss, num_correct, med_rank, chosen_responses

283  def extract_texts(self, obs):
284  """
285  Extract the personalities and dialogue histories from observations.
286 
287  Additionally determine which dialogue round we are in.
288 
289  Note that this function assumes that the personality is the
290  last line of the `text` field in the observation.
291 
292  :param obs:
293  list of observations
294 
295  :return:
296  a list of personalities, a list of dialogue histories, and the
297  current dialogue round (either first, second, or third+)
298  """
299  splits = [v.get("text").split("\n") for v in obs]
300  if self.personality_override:
301  splits = [s + [self.personality_override] for s in splits]
302  personalities = [t[-1] for t in splits]
303  dialogue_histories = None
304  dialogue_round = "first_round"
305  if len(splits[0]) >= 2:
306  dialogue_round = "second_round" if len(splits[0]) == 2 else "third_round+"
307  dialogue_histories = ["\n".join(t[:-1]) for t in splits]
308 
309  return personalities, dialogue_histories, dialogue_round

356  def receive_metrics(self, metrics_dict):
357  """
358  Receive the metrics from validation.
359 
360  Unfreeze text encoder weights after a certain number of rounds without improvement.
361 
362  Override to account for different dialogue rounds.
363 
364  :param metrics_dict:
365  the metrics dictionary
366  """
367  if "tasks" in metrics_dict:
368  metrics_dict = metrics_dict["tasks"]["internal:comment_battle:imageDialog"]
369  if self.freeze_patience != -1 and self.is_frozen:
370  m_key = "hits@1/100"
371  ms = [
372  metrics_dict[r].get(m_key, -1)
373  for r in ["first_round", "second_round", "third_round+"]
374  ]
375  m = sum(ms) / len([m for m in ms if m >= 0])
376  if m > self.freeze_best_metric:
377  self.freeze_impatience = 0
378  self.freeze_best_metric = m
379  print("performance not good enough to unfreeze the model.")
380  else:
381  self.freeze_impatience += 1
382  print("Growing impatience for unfreezing")
383  if self.freeze_impatience >= self.freeze_patience:
384  self.is_frozen = False
