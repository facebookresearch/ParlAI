179  def load_disconnects(self):
180  """Load disconnects from file, populate the disconnects field for any
181  worker_id that has disconnects in the list. Any disconnect that
182  occurred longer ago than the disconnect persist length is ignored
183  """
184  self.disconnects = []
185  # Load disconnects from file
186  file_path = os.path.join(parent_dir, DISCONNECT_FILE_NAME)
187  compare_time = time.time()
188  if os.path.exists(file_path):
189  with open(file_path, 'rb') as f:
190  old_disconnects = pickle.load(f)
191  self.disconnects = [
192  d
193  for d in old_disconnects
194  if (compare_time - d['time']) < DISCONNECT_PERSIST_LENGTH
195  ]
196  # Initialize worker states with proper number of disconnects
197  for disconnect in self.disconnects:
198  worker_id = disconnect['id']
199  if worker_id not in self.mturk_workers:
200  # add this worker to the worker state
201  self.mturk_workers[worker_id] = WorkerState(worker_id)
202  self.mturk_workers[worker_id].disconnects += 1
203 
204  def save_disconnects(self):
205  """Saves the local list of disconnects to file"""
206  file_path = os.path.join(parent_dir, DISCONNECT_FILE_NAME)
207  if os.path.exists(file_path):
208  os.remove(file_path)
209  with open(file_path, 'wb') as f:
210  pickle.dump(self.disconnects, f, pickle.HIGHEST_PROTOCOL)

212  def handle_agent_disconnect(self, worker_id, assignment_id, partner_callback):
213  """Handles a disconnect by the given worker, calls partner_callback
214  on all of the conversation partners of that worker
215  """
216  agent = self._get_agent(worker_id, assignment_id)
217  if agent is not None:
218  # Disconnect in conversation is not workable
219  agent.set_status(AssignState.STATUS_DISCONNECT)
220  # in conversation, inform others about disconnect
221  conversation_id = agent.conversation_id
222  if conversation_id in self.conv_to_agent:
223  conv_participants = self.conv_to_agent[conversation_id]
224  if agent in conv_participants:
225  for other_agent in conv_participants:
226  if agent.assignment_id != other_agent.assignment_id:
227  partner_callback(other_agent)
228  if len(conv_participants) > 1:
229  # The user disconnected from inside a conversation with
230  # another turker, record this as bad behavior
231  self.handle_bad_disconnect(worker_id)

233  def handle_bad_disconnect(self, worker_id):
234  """Update the number of bad disconnects for the given worker, block
235  them if they've exceeded the disconnect limit
236  """
237  if not self.is_sandbox:
238  self.mturk_workers[worker_id].disconnects += 1
239  self.disconnects.append({'time': time.time(), 'id': worker_id})
240  if self.mturk_workers[worker_id].disconnects > MAX_DISCONNECTS:
241  if self.opt['hard_block']:
242  text = (
243  'This worker has repeatedly disconnected from these '
244  'tasks, which require constant connection to complete '
245  'properly as they involve interaction with other '
246  'Turkers. They have been blocked after being warned '
247  'and failing to adhere. This was done in order to '
248  'ensure a better experience for other '
249  'workers who don\'t disconnect.'
250  )
251  self.mturk_manager.block_worker(worker_id, text)
252  shared_utils.print_and_log(
253  logging.INFO,
254  'Worker {} blocked - too many disconnects'.format(worker_id),
255  True,
256  )
257  elif self.opt['disconnect_qualification'] is not None:
258  self.mturk_manager.soft_block_worker(
259  worker_id, 'disconnect_qualification'
260  )
261  shared_utils.print_and_log(
262  logging.INFO,
263  'Worker {} soft blocked - too many disconnects'.format(
264  worker_id
265  ),
266  True,
267  )

290  def get_agent_work_status(self, assignment_id):
291  """Get the current status of an assignment's work"""
292  client = mturk_utils.get_mturk_client(self.is_sandbox)
293  try:
294  response = client.get_assignment(AssignmentId=assignment_id)
295  status = response['Assignment']['AssignmentStatus']
296  worker_id = self.assignment_to_worker_id[assignment_id]
297  agent = self._get_agent(worker_id, assignment_id)
298  if agent is not None and status == MTurkAgent.ASSIGNMENT_DONE:
299  agent.hit_is_complete = True
300  return status
301  except ClientError as e:
302  # If the assignment isn't done, asking for the assignment will fail
303  not_done_message = (
304  'This operation can be called with a status '
305  'of: Reviewable,Approved,Rejected'
306  )
307  if not_done_message in e.response['Error']['Message']:
308  return MTurkAgent.ASSIGNMENT_NOT_DONE
309  else:
310  shared_utils.print_and_log(
311  logging.WARN,
312  'Unanticipated error in `get_agent_work_status`: '
313  + e.response['Error']['Message'],
314  should_print=True,
315  )
316  # Assume not done if status check seems to be faulty.
317  return MTurkAgent.ASSIGNMENT_NOT_DONE
