172  def __init__(
173  self,
174  opt,
175  field='text',
176  vec_type='deque',
177  maxlen=None,
178  size=-1,
179  p1_token='__p1__',
180  p2_token='__p2__',

183  self.field = field
184  self.dict = dict_agent
185  self.delimiter = opt.get('delimiter', '\n')
186  self.delimiter_tok = self.parse(self.delimiter)
187  self.size = size
188  self.split_on_newln = opt.get('split_lines', False)
189 
190  # set up history objects
191  if vec_type != 'deque' and vec_type != 'list':
192  raise RuntimeError('Type {} is not supported for history'.format(vec_type))
193  self.vec_type = vec_type
194  self.max_len = maxlen
195 
196  self.history_strings = []
197  self.history_raw_strings = []
198  self.history_vecs = []
199 
200  # person token args
201  self.add_person_tokens = opt.get('person_tokens', False)
202  self.add_p1_after_newln = opt.get('add_p1_after_newln', False)
203  self.p1_token = p1_token
204  self.p2_token = p2_token
205 
206  # tracking when to clear history
207  self.reset_on_next_update = False

209  def parse(self, text):
210  """Tokenize text with the given dictionary."""
211  return self.dict.txt2vec(text)

213  def reset(self):
214  """Clear the history."""
215  self.history_raw_strings = []
216  self.history_strings = []
217  self.history_vecs = []

219  def _update_strings(self, text):
220  if self.size > 0:
221  while len(self.history_strings) >= self.size:
222  self.history_strings.pop(0)
223  self.history_strings.append(text)
224 
225  def _update_raw_strings(self, text):
226  if self.size > 0:
227  while len(self.history_raw_strings) >= self.size:
228  self.history_raw_strings.pop(0)
229  self.history_raw_strings.append(text)
230 
231  def _update_vecs(self, text):
232  if self.size > 0:
233  while len(self.history_vecs) >= self.size:
234  self.history_vecs.pop(0)
235  self.history_vecs.append(self.parse(text))

237  def update_history(self, obs, add_next=None):
238  """
239  Update the history with the given observation.
240 
241  :param add_next:
242  string to append to history prior to updating it with the
243  observation
244  """
245  if self.reset_on_next_update:
246  # this is the first example in a new episode, clear the previous
247  # history
248  self.reset()
249  self.reset_on_next_update = False
250 
251  if add_next is not None:
252  self._update_raw_strings(add_next)
253  if self.add_person_tokens:
254  add_next = self._add_person_tokens(add_next, self.p2_token)
255  # update history string
256  self._update_strings(add_next)
257  # update history vecs
258  self._update_vecs(add_next)
259 
260  if self.field in obs and obs[self.field] is not None:
261  if self.split_on_newln:
262  next_texts = obs[self.field].split('\n')
263  else:
264  next_texts = [obs[self.field]]
265  for text in next_texts:
266  self._update_raw_strings(text)
267  if self.add_person_tokens:
268  text = self._add_person_tokens(
269  obs[self.field], self.p1_token, self.add_p1_after_newln
270  )
271  # update history string
272  self._update_strings(text)
273  # update history vecs
274  self._update_vecs(text)
275 
276  if obs.get('episode_done'):
277  # end of this episode, clear the history when we see a new example
278  self.reset_on_next_update = True

280  def get_history_str(self):
281  """Return the string version of the history."""
282  if len(self.history_strings) > 0:
283  return self.delimiter.join(self.history_strings)
284  return None
285 
286  def get_history_vec(self):
287  """Return a vectorized version of the history."""
288  if len(self.history_vecs) == 0:
289  return None
290 
291  if self.vec_type == 'deque':
292  history = deque(maxlen=self.max_len)
293  for vec in self.history_vecs[:-1]:
294  history.extend(vec)
295  history.extend(self.delimiter_tok)
296  history.extend(self.history_vecs[-1])
297  else:
298  # vec type is a list
299  history = []
300  for vec in self.history_vecs[:-1]:
301  history += vec
302  history += self.delimiter_tok
303  history += self.history_vecs[-1]
304 
305  return history

307  def get_history_vec_list(self):
308  """Return a list of history vecs."""
309  return self.history_vecs

651  opt = self.opt
652

657  print('[ Using CUDA ]')
658  if not shared and opt['gpu'] != -1:
659  torch.cuda.set_device(opt['gpu'])
660  # indicate whether using fp16
661  self.fp16 = self.use_cuda and self.opt.get('fp16', False)
662 
663  if shared is None:
664  # intitialize any important structures from scratch
665  self.replies = {} # past replies
666  self._replies_are_shared = False
667  self.dict = self.build_dictionary()
668 
669  if opt.get('fp16'):
670  # Volta cores revert to FP32 hardware if tensors are not multiples
671  # of 8 in all dimensions. This INCLUDES the embeddings layer! As
672  # such, we need some extra magic to ensure the dictionary is padded
673  # with extra tokens to make it a multiple of 8.
674  if len(self.dict) % 8 != 0:
675  for i in range(8 - len(self.dict) % 8):
676  self.dict['__FP16_PAD_{}__'.format(i)] = 1

678  self.metrics = {}
679  # gradient norms
680  self.metrics['gnorm'] = 0.0
681  # gradient clipping rate
682  self.metrics['clip'] = 0.0
683  # number of calls to optimizer.step()
684  self.metrics['updates'] = 0
685  else:
686  # copy initialized data from shared table
687  self.opt = shared['opt']
688  self.dict = shared['dict']
689  self.model = shared['model']
690  self.criterion = shared['criterion']
691  self.metrics = shared['metrics']

693  # if we're not using batching (e.g. mturk), then replies really need
694  # to stay separated
695  self.replies = {}
696  self._replies_are_shared = False

698  self.replies = shared['replies']
699  self._replies_are_shared = True

805  def init_optim(self, params, optim_states=None, saved_optim_type=None):
806  """
807  Initialize optimizer with model parameters.
808 
809  :param params:
810  parameters from the model
811 
812  :param optim_states:
813  optional argument providing states of optimizer to load
814 
815  :param saved_optim_type:
816  type of optimizer being loaded, if changed will skip loading
817  optimizer states
818  """
819  opt = self.opt
820 
821  # set up optimizer args
822  lr = opt['learningrate']
823  kwargs = {'lr': lr}
824  if opt.get('weight_decay'):
825  kwargs['weight_decay'] = opt['weight_decay']
826  if opt.get('momentum') > 0 and opt['optimizer'] in ['sgd', 'rmsprop', 'qhm']:
827  # turn on momentum for optimizers that use it
828  kwargs['momentum'] = opt['momentum']
829  if opt['optimizer'] == 'sgd' and opt.get('nesterov', True):
830  # for sgd, maybe nesterov
831  kwargs['nesterov'] = opt.get('nesterov', True)
832  elif opt['optimizer'] == 'qhm':
833  # qhm needs a nu
834  kwargs['nu'] = opt.get('nus', (0.7,))[0]
835  elif opt['optimizer'] == 'adam':
836  # turn on amsgrad for adam
837  # amsgrad paper: https://openreview.net/forum?id=ryQu7f-RZ
838  kwargs['amsgrad'] = True
839  elif opt['optimizer'] == 'qhadam':
840  # set nus for qhadam
841  kwargs['nus'] = opt.get('nus', (0.7, 1.0))
842  if opt['optimizer'] in ['adam', 'sparseadam', 'fused_adam', 'adamax', 'qhadam']:
843  # set betas for optims that use it
844  kwargs['betas'] = opt.get('betas', (0.9, 0.999))

846  if opt.get('adam_eps'):
847  kwargs['eps'] = opt['adam_eps']
848

850  self.optimizer = optim_class(params, **kwargs)
851  if self.fp16:
852  self.optimizer = fp16_optimizer_wrapper(self.optimizer)
853 
854  # TODO: we might want to hard reset optimizers here in the
855  # case of fine tuning. Some rudimentary experiments seemed to
856  # indicate that keeping adam weights around was desirable, so this
857  # will remain the behavior for the time being.
858  if optim_states and saved_optim_type != opt['optimizer']:
859  # we changed from adam to adamax, or sgd to adam, or similar
860  print('WARNING: not loading optim state since optim class changed.')

863  optimstate_fp16 = 'loss_scaler' in optim_states
864  if self.fp16 and optimstate_fp16:
865  # previously trained in fp16, now we're training in fp16.
866  # ideally no action needed, but APEX broke backwards
867  # compatibility and this is the hack around it.
868  optim_states['loss_scaler'] = self.optimizer.state_dict()['loss_scaler']
869  elif optimstate_fp16 and not self.fp16:
870  # old optimizer was fp16 but now we're doing fp32,
871  # drop the fp16 wrapper from the state_dict and just load
872  # the fp16 weights into the fp32 tensors
873  optim_states = optim_states['optimizer_state_dict']
874  elif not optimstate_fp16 and self.fp16:
875  # old optimizer was fp32, but now we're doing fp16.
876  # this is a bit clunky, but alternatives are worse
877  self.optimizer.optimizer.load_state_dict(optim_states)
878  return
879  else:
880  # previously trained in fp32, loading in fp32.
881  # no special treatment needed.
882  pass
883 
884  # finally, try to actually load the optimizer state
885  try:
886  self.optimizer.load_state_dict(optim_states)
887  except ValueError:
888  print('WARNING: not loading optim state since model params changed.')

890  def build_lr_scheduler(self, states=None, hard_reset=False):
891  """
892  Create the learning rate scheduler, and assign it to self.scheduler.
893 
894  This scheduler will be updated upon a call to receive_metrics.
895  May also create self.warmup_scheduler, if appropriate.
896 
897  :param state_dict states: Possible state_dict provided by model
898  checkpoint, for restoring LR state
899 
900  :param bool hard_reset: If true, the LR scheduler should ignore the
901  state dictionary.
902  """
903  # first make sure there are no null pointers
904  if states is None:
905  states = {}
906  optimizer = self.optimizer
907  if self.fp16:
908  # lr schedulers don't work with apex, they expect the "real" optimizer
909  optimizer = optimizer.optimizer
910 
911  warmup_updates = self.opt.get('warmup_updates', -1)
912  updates_so_far = states.get('number_training_updates', 0)
913  if warmup_updates > 0 and (updates_so_far < warmup_updates or hard_reset):
914 
915  def _warmup_lr(step):
916  start = self.opt['warmup_rate']
917  end = 1.0
918  progress = min(1.0, step / self.opt['warmup_updates'])
919  lr_mult = start + (end - start) * progress
920  return lr_mult
921 
922  self.warmup_scheduler = optim.lr_scheduler.LambdaLR(optimizer, _warmup_lr)
923  else:
924  self.warmup_scheduler = None
925 
926  patience = self.opt.get('lr_scheduler_patience', 3)
927  decay = self.opt.get('lr_scheduler_decay', 0.5)
928 
929  if self.opt.get('lr_scheduler') == 'none':
930  self.scheduler = None
931  elif decay == 1.0:
932  warn_once(
933  "Your LR decay is set to 1.0. Assuming you meant you wanted "
934  "to disable learning rate scheduling. Adjust --lr-scheduler-decay "
935  "if this is not correct."
936  )
937  self.scheduler = None
938  elif self.opt.get('lr_scheduler') == 'reduceonplateau':
939  self.scheduler = optim.lr_scheduler.ReduceLROnPlateau(
940  optimizer, 'min', factor=decay, patience=patience, verbose=True
941  )
942  elif self.opt.get('lr_scheduler') == 'fixed':
943  self.scheduler = optim.lr_scheduler.StepLR(optimizer, patience, gamma=decay)
944  elif self.opt.get('lr_scheduler') == 'invsqrt':
945  if self.opt.get('warmup_updates', -1) <= 0:
946  raise ValueError(
947  '--lr-scheduler invsqrt requires setting --warmup-updates'
948  )
949  warmup_updates = self.opt['warmup_updates']
950  decay_factor = np.sqrt(max(1, warmup_updates))
951 
952  def _invsqrt_lr(step):
953  return decay_factor / np.sqrt(max(1, step))
954 
955  self.scheduler = optim.lr_scheduler.LambdaLR(optimizer, _invsqrt_lr)
956  else:
957  raise ValueError(
958  "Don't know what to do with lr_scheduler '{}'".format(
959  self.opt.get('lr_scheduler')
960  )
961  )
962 
963  # time to load LR state from the checkpoint, if possible.
964  if (
965  # there is already an old LR scheduler saved on disk
966  states
967  and
968  # and the old LR scheduler is different
969  states.get('lr_scheduler_type') != self.opt['lr_scheduler']
970  and
971  # and we're not already using a fresh scheduler
972  not hard_reset
973  ):
974  # the LR scheduler changed, start things fresh
975  warn_once("LR scheduler is different from saved. Starting fresh!")
976  hard_reset = True
977 
978  if hard_reset:
979  # We're not going to use the LR schedule, let's just exit
980  return
981 
982  # do the actual loading (if possible)
983  if 'number_training_updates' in states:
984  self._number_training_updates = states['number_training_updates']
985  if self.scheduler and 'lr_scheduler' in states:
986  self.scheduler.load_state_dict(states['lr_scheduler'])
987  if states.get('warmup_scheduler') and getattr(self, 'warmup_scheduler', None):
988  self.warmup_scheduler.load_state_dict(states['warmup_scheduler'])
