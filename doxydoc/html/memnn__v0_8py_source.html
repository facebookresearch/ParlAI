228  def predict(self, xs, cands, ys=None):
229  is_training = ys is not None
230  if is_training:
231  # Subsample to reduce training time
232  cands = [
233  list(set(random.sample(c, min(32, len(c))) + self.labels))
234  for c in cands
235  ]
236  else:
237  # rank all cands to increase accuracy
238  cands = [list(set(c)) for c in cands]
239 
240  self.model.train(mode=is_training)
241  # Organize inputs for network (see contents of xs and ys in batchify method)
242  output_embeddings = self.model(*xs)
243 
244  if self.decoder is None:
245  scores = self.score(cands, output_embeddings)
246  if is_training:
247  label_inds = [
248  cand_list.index(self.labels[i]) for i, cand_list in enumerate(cands)
249  ]
250  if self.use_cuda:
251  label_inds = torch.cuda.LongTensor(label_inds)
252  else:
253  label_inds = torch.LongTensor(label_inds)
254  loss = self.rank_loss(scores, label_inds)
255  predictions = self.ranked_predictions(cands, scores)
256  else:
257  self.decoder.train(mode=is_training)
258 
259  output_lines, loss = self.decode(output_embeddings, ys)
260  predictions = self.generated_predictions(output_lines)
261 
262  if is_training:
263  for o in self.optimizers.values():
264  o.zero_grad()
265  loss.backward()
266  for o in self.optimizers.values():
267  o.step()
268  return predictions

270  def score(self, cands, output_embeddings):
271  last_cand = None
272  max_len = max([len(c) for c in cands])
273  scores = output_embeddings.data.new(len(cands), max_len)
274  for i, cand_list in enumerate(cands):
275  if last_cand != cand_list:
276  candidate_lengths, candidate_indices = to_tensors(cand_list, self.dict)
277  candidate_embeddings = self.model.answer_embedder(
278  candidate_lengths, candidate_indices
279  )
280  if self.use_cuda:
281  candidate_embeddings = candidate_embeddings.cuda()
282  last_cand = cand_list
283  scores[i, : len(cand_list)] = self.model.score.one_to_many(
284  output_embeddings[i].unsqueeze(0), candidate_embeddings
285  ).squeeze(0)
286  return scores

295  def decode(self, output_embeddings, ys=None):
296  batchsize = output_embeddings.size(0)
297  hn = output_embeddings.unsqueeze(0).expand(
298  self.opt['rnn_layers'], batchsize, output_embeddings.size(1)
299  )
300  x = self.model.answer_embedder(torch.LongTensor([1]), self.START_TENSOR)
301  xes = x.unsqueeze(1).expand(x.size(0), batchsize, x.size(1))
302 
303  loss = 0
304  output_lines = [[] for _ in range(batchsize)]
305  done = [False for _ in range(batchsize)]
306  total_done = 0
307  idx = 0
308  while (total_done < batchsize) and idx < self.longest_label:
309  # keep producing tokens until we hit END or max length for each ex
310  if self.use_cuda:
311  xes = xes.cuda()
312  hn = hn.contiguous()
313  preds, scores = self.decoder(xes, hn)
314  if ys is not None:
315  y = ys[0][:, idx]
316  temp_y = y.cuda() if self.use_cuda else y
317  loss += self.gen_loss(scores, temp_y)
318  else:
319  y = preds
320  # use the true token as the next input for better training
321  xes = self.model.answer_embedder(
322  torch.LongTensor(preds.numel()).fill_(1), y
323  ).unsqueeze(0)
324 
325  for b in range(batchsize):
326  if not done[b]:
327  token = self.dict.vec2txt(preds[b])
328  if token == self.END:
329  done[b] = True
330  total_done += 1
331  else:
332  output_lines[b].append(token)
333  idx += 1
334  return output_lines, loss

342  def parse(self, memory):
343  """Returns:
344  query = tensor (vector) of token indices for query
345  query_length = length of query
346  memory = tensor (matrix) where each row contains token indices for a memory
347  memory_lengths = tensor (vector) with lengths of each memory
348  """
349  query = memory.pop()
350  query = torch.LongTensor(query)
351  query_length = torch.LongTensor([len(query)])
352 
353  if len(memory) == 0:
354  memory.append([self.NULL_IDX])
355 
356  memory = [torch.LongTensor(m) for m in memory]
357  memory_lengths = torch.LongTensor([len(m) for m in memory])
358  memory = torch.cat(memory)
359 
360  return (query, memory, query_length, memory_lengths)

362  def batchify(self, obs):
363  """Returns:
364  xs = [memories, queries, memory_lengths, query_lengths]
365  ys = [labels, label_lengths] (if available, else None)
366  cands = list of candidates for each example in batch
367  valid_inds = list of indices for examples with valid observations
368  """
369  exs = [ex for ex in obs if 'text' in ex and len(ex['text']) > 0]
370  valid_inds = [
371  i for i, ex in enumerate(obs) if 'text' in ex and len(ex['text']) > 0
372  ]
373  if not exs:
374  return [None] * 4
375 
376  parsed = [self.parse(ex['text']) for ex in exs]
377  queries = torch.cat([x[0] for x in parsed])
378  memories = torch.cat([x[1] for x in parsed])
379  query_lengths = torch.cat([x[2] for x in parsed])
380  memory_lengths = torch.LongTensor(len(exs), self.mem_size).zero_()
381  for i in range(len(exs)):
382  if len(parsed[i][3]) > 0:
383  memory_lengths[i, -len(parsed[i][3]) :] = parsed[i][3]
384  xs = [memories, queries, memory_lengths, query_lengths]
385 
386  ys = None
387  self.labels = [random.choice(ex['labels']) for ex in exs if 'labels' in ex]
388  if len(self.labels) == len(exs):
389  parsed = [self.dict.txt2vec(l) for l in self.labels]
390  parsed = [torch.LongTensor(p) for p in parsed]
391  label_lengths = torch.LongTensor([len(p) for p in parsed]).unsqueeze(1)
392  self.longest_label = max(self.longest_label, label_lengths.max())
393  padded = [
394  torch.cat(
395  (
396  p,
397  torch.LongTensor(self.longest_label - len(p)).fill_(
398  self.END_TENSOR[0]
399  ),
400  )
401  )
402  for p in parsed
403  ]
404  labels = torch.stack(padded)
405  ys = [labels, label_lengths]
406 
407  cands = [ex['label_candidates'] for ex in exs if 'label_candidates' in ex]
408  # Use words in dict as candidates if no candidates are provided
409  if len(cands) < len(exs):
410  cands = build_cands(exs, self.dict)
411  # Avoid rebuilding candidate list every batch if its the same
412  if self.last_cands != cands:
413  self.last_cands = cands
414  self.last_cands_list = [list(c) for c in cands]
415  cands = self.last_cands_list
416  return xs, ys, cands, valid_inds

418  def batch_act(self, observations):
419  batchsize = len(observations)
420  batch_reply = [{'id': self.getID()} for _ in range(batchsize)]
421 
422  xs, ys, cands, valid_inds = self.batchify(observations)
423 
424  if xs is None or len(xs[1]) == 0:
425  return batch_reply
426 
427  # Either train or predict
428  predictions = self.predict(xs, cands, ys)
429 
430  for i in range(len(valid_inds)):
431  self.answers[valid_inds[i]] = predictions[i][0]
432  batch_reply[valid_inds[i]]['text'] = predictions[i][0]
433  batch_reply[valid_inds[i]]['text_candidates'] = predictions[i]
434  return batch_reply
