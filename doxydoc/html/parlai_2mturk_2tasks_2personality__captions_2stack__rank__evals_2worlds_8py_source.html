143  def parley(self):
144  """CHOOSER is given an image and 2 comments/captions, and is asked
145  to choose more engaging comment.
146  """
147  # Initial Message Value
148  control_msg = {'episode_done': False}

150

153 
154  """First, we give CHOOSER the image

160  self.example_num, example = self.agent.example_generator.pop_example()
161  img = load_image(
162  os.path.join(self.image_path, '{}.jpg'.format(example['image_hash']))
163  )
164  buffered = BytesIO()
165  img.save(buffered, format="JPEG")
166  encoded = str(base64.b64encode(buffered.getvalue()).decode('ascii'))
167  control_msg['image'] = encoded
168  """
169  Setup comments for ranking
170  """
171  comments = [(ck, example[ck]) for ck in [self.ck1, self.ck2]]

174  if self.show_personality:
175  control_msg['personality'] = (
176  '<b><span style="color:blue">'
177  '{}\n</span></b>'.format(example['personality'].strip())
178  )
179 
180  best_pick = None
181  control_msg['text'] = PICK_BEST_MSG.format(self.turn_idx + 1)
182  control_msg['new_eval'] = True
183  agent.observe(validate(control_msg))
184  time.sleep(1)
