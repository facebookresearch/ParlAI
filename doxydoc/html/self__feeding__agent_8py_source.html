306  def act(self):
307  if self.opt['request_feedback'] or self.opt['request_rating']:
308  # rating request ->
309  # 0/+1 -> normal
310  # -1 -> feedback request
311  # feedback request -> topic request -> normal
312  positivity = -1 # actual positivity values are in [0, 1]
313 
314  # Preprocessing (for special states that transition to NORMAL)
315  if self.status in [NEWTOPIC_REQUESTED, RATING_ACCEPTED]:
316  self.status = NORMAL
317 
318  if self.status == NORMAL:
319  positivity = self.predict_satisfaction(self.observation)
320  if self.opt["display_sat_estimate"]:
321  print(f"[ satisfaction estimate ]: {positivity}")
322  if self.do_request_feedback(positivity):
323  action = self.make_action(self.make_feedback_request())
324  self.status = FEEDBACK_REQUESTED
325  elif self.do_request_rating(positivity):
326  action = self.make_action(self.make_rating_request())
327  self.requested_rating = True
328  self.status = RATING_REQUESTED
329 
330  elif self.status == RATING_REQUESTED:
331  rating = self.extract_rating()
332  if rating == -1:
333  action = self.make_action(
334  self.make_feedback_request(), reward=rating
335  )
336  self.status = FEEDBACK_REQUESTED
337 
338  else:
339  action, reply = self.make_rating_response(rating)
340  self.status = RATING_ACCEPTED
341  # Store just the non-template part of the response in the history
342  self.history.update_history({"text": reply})
343 
344  elif self.status == FEEDBACK_REQUESTED:
345  action = self.make_action(THANKS + ' ' + NEWTOPIC)
346  self.status = NEWTOPIC_REQUESTED
347  self.history.reset()
348 
349  else:
350  raise Exception(f"Unrecognized status: {self.status}")
351 
352  if self.status == NORMAL:
353  action = super().act()
354  self.history.update_history(action)
355 
356  return action

417  self.optimizer.zero_grad()

470  def dialog_step(self, batch):
471  batchsize = batch.text_vec.size(0)
472  if self.model.training:
473  cands, cand_vecs, label_inds = self._build_candidates(
474  batch, source=self.opt['candidates'], mode='train'
475  )
476  else:
477  cands, cand_vecs, label_inds = self._build_candidates(
478  batch, source=self.opt['eval_candidates'], mode='eval'
479  )
480  if self.encode_candidate_vecs and self.eval_candidates in [
481  'fixed',
482  'vocab',
483  ]:
484  # if we cached candidate encodings for a fixed list of candidates,
485  # pass those into the score_candidates function
486  if self.eval_candidates == 'fixed':
487  cand_vecs = self.fixed_candidate_encs
488  elif self.eval_candidates == 'vocab':
489  cand_vecs = self.vocab_candidate_encs
490 
491  scores = self.model.score_dialog(batch.text_vec, cand_vecs)
492  _, ranks = scores.sort(1, descending=True)
493 
494  if self.model.training:
495  # Get predictions but not full rankings for the sake of speed
496  cand_ranked = None
497  if cand_vecs.dim() == 2:
498  preds = [cands[ordering[0]] for ordering in ranks]
499  elif cand_vecs.dim() == 3:
500  preds = [cands[i][ordering[0]] for i, ordering in enumerate(ranks)]
501  else:
502  # Return full rankings to calculate hits@ metrics
503  cand_ranked = []
504  for i, ordering in enumerate(ranks):
505  if cand_vecs.dim() == 2:
506  cand_list = cands
507  elif cand_vecs.dim() == 3:
508  cand_list = cands[i]
509  cand_ranked.append([cand_list[rank] for rank in ordering])
510  preds = [cand_ranked[i][0] for i in range(batchsize)]
511 
512  if label_inds is None:
513  loss = None
514  else:
515  loss = self.criterion(scores, label_inds)
516  self.update_dia_metrics(loss, ranks, label_inds, batchsize)
517  return loss, preds, cand_ranked

553  def satisfaction_step(self, batch):
554  batchsize = batch.text_vec.size(0)
555  # probs is a [batchsize] torch.FloatTensor with values in [0, 1]
556  if self.opt['regex']:
557  contexts = [self.dict.vec2txt(vec) for vec in batch.text_vec]
558  # HACK: --use-cuda flag
559  probs = self.rating_classifier.predict_proba(contexts).cuda()
560  else:
561  probs = self.model.score_satisfaction(batch.text_vec)
562  # preds is a [batchsize] torch.LongTensor with values in {0, 1}
563  preds = (probs > self.opt['rating_threshold']).long()
564 
565  if batch.labels is None: # TODO: not sure if it's None, list of Nones, etc.
566  loss = None
567  else:
568  # labels will be a [batchsize] torch.LongTensor with values in {0, 1}
569  labels = torch.LongTensor([int(l) == 1 for l in batch.labels]).cuda()
570 
571  loss = self.satisfaction_criterion(probs, labels.float())
572  self.update_sat_metrics(loss, preds, labels, batchsize)
573 
574  return loss, preds

576  def update_dia_metrics(self, loss, ranks, label_inds, batchsize):
577  self.metrics['dia_exs'] += batchsize
578  self.metrics['dia_loss'] += loss.item()
579  if label_inds is not None:
580  for b in range(batchsize):
581  rank = (ranks[b] == label_inds[b]).nonzero().item()
582  self.metrics['dia_rank'] += 1 + rank
583  self.metrics['dia_correct'] += rank == 0
584 
585  def update_fee_metrics(self, loss, ranks, label_inds, batchsize):
586  self.metrics['fee_exs'] += batchsize
587  self.metrics['fee_loss'] += loss.item()
588  if label_inds is not None:
589  for b in range(batchsize):
590  rank = (ranks[b] == label_inds[b]).nonzero().item()
591  self.metrics['fee_rank'] += 1 + rank
592  self.metrics['fee_correct'] += rank == 0
593 
594  def update_sat_metrics(self, loss, preds, labels, batchsize):
595  # tp/fp/tn/fn are w/r/t the rarer class (negative responses)
596  self.metrics['sat_exs'] += batchsize
597  self.metrics['sat_loss'] += loss.item()
598  a = self.opt['target_class']
599  b = not self.opt['target_class']
600  assert a in [0, 1]
601  assert b in [0, 1]
602  self.metrics['sat_tp'] += ((preds == labels) * (labels == a)).sum().item()
603  self.metrics['sat_fp'] += ((preds != labels) * (labels == b)).sum().item()
604  self.metrics['sat_tn'] += ((preds == labels) * (labels == b)).sum().item()
605  self.metrics['sat_fn'] += ((preds != labels) * (labels == a)).sum().item()

724  def predict_satisfaction_by_uncertainty(self, batch):
725  # Use the dialog model's confidence to predict the rating on previous response
726  # HACK: this test is run using a model that was trained on dialog but is now
727  # being evaluated on satisfaction. We use a satisfaction dataset so that we have
728  # access to the satisfaction labels. Therefore, we do a sloppy hack here to pull
729  # out what would have been the candidates for the dialog task. We use histsz=4
730  # and ignore the last response (the user's feedback) and use the penultimate
731  # utterances as the candidates. The (up to) two utterances before that are
732  # context.
733 
734  # pull out dialog candidates from text_vecs since this is a satisfaction task
735  assert self.opt['history_size'] > 2
736  text_vecs = []
737  cand_vecs = []
738  for vec in batch.text_vec:
739  last_p1 = (vec == self.dict.txt2vec('__p1__')[0]).nonzero()[-1].item()
740  last_p2 = (vec == self.dict.txt2vec('__p2__')[0]).nonzero()[-1].item()
741  text_vecs.append(vec[:last_p2])
742  cand_vecs.append(vec[last_p2 + 1 : last_p1])
743  text_padded, _ = padded_tensor(text_vecs)
744  cand_padded, _ = padded_tensor(cand_vecs)
745  scores = self.model.score_dialog(text_padded, cand_padded)
746  confidences = F.softmax(scores, dim=1).cpu().detach().numpy()
747 
748  preds = []
749  for example in confidences:
750  ranked_confidences = sorted(list(example), reverse=True)
751  if self.opt['uncertainty_style'] == 'mag':
752  # If the most confident choice isn't confident enough, predict that
753  # the response the bot gives will be bad (pred=0)
754  mag = ranked_confidences[0]
755  preds.append(mag > self.opt['uncertainty_threshold'])
756  elif self.opt['uncertainty_style'] == 'gap':
757  # If the gap between the first and second most confident choices isn't
758  # large enough, predict that the response the bot gives will be bad
759  # (pred=0)
760  gap = ranked_confidences[0] - ranked_confidences[1]
761  preds.append(gap > self.opt['uncertainty_threshold'])
762 
763  loss = torch.tensor(0)
764  preds = torch.LongTensor(preds)
765  labels = torch.LongTensor([int(l) == 1 for l in batch.labels])
766  batchsize = len(labels)
767  self.update_sat_metrics(loss, preds, labels, batchsize)
768  return preds
