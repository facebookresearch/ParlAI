15 def _circleci_parallelism(suite):
16  """Allow for parallelism in CircleCI for speedier tests.."""
17  if int(os.environ.get('CIRCLE_NODE_TOTAL', 0)) <= 1:
18  # either not running on circleci, or we're not using parallelism.
19  return suite
20  # tests are automatically sorted by discover, so we will get the same ordering
21  # on all hosts.
22  total = int(os.environ['CIRCLE_NODE_TOTAL'])
23  index = int(os.environ['CIRCLE_NODE_INDEX'])
24 
25  # right now each test is corresponds to a /file/. Certain files are slower than
26  # others, so we want to flatten it
27  tests = [testfile._tests for testfile in suite._tests]
28  tests = list(chain.from_iterable(tests))
29  random.Random(42).shuffle(tests)
30  tests = [t for i, t in enumerate(tests) if i % total == index]
31  return unittest.TestSuite(tests)
