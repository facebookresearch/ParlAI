173  else:
174  control_msg['description'] = config_responses['task_description']
175  self.example_id, igc_example = self.agent.example_generator.pop_example()
176  img = self.image_loader.load(self.image_id_to_path(self.example_id))
177  buffered = BytesIO()
178  img.save(buffered, format="JPEG")
179  encoded = str(base64.b64encode(buffered.getvalue()).decode('ascii'))
180  control_msg['image'] = encoded
181  control_msg['context'] = igc_example['context']
182  """
183  Setup Options for rating
184  """
185  if self.d_rnd == 'questions':
186  options = [(k, v) for k, v in igc_example['questions'].items()]

188  control_msg['question'] = igc_example['question']
189  options = [(k, v) for k, v in igc_example['responses'].items()]
190  random.shuffle(options)
191  options, dup_dict = self.filter_option_duplicates(options)
192  control_msg['options'] = [c[1] for c in options]
193  # Collect rating from turker
194  rate_msg = RATE_MSG if self.d_rnd == 'questions' else RATE_RESPONSE_MSG
195  control_msg['text'] = rate_msg.format(self.turn_idx + 1)
196  control_msg['new_eval'] = True

198  time.sleep(1)

200  # First timeout check
201  self.check_timeout(act)
202  if self.chat_done:
203  break
204  try:
