12 def xavier_init(module):
13  """Xavier initializer for module parameters."""
14  for parameter in module.parameters():
15  if len(parameter.data.shape) == 1:
16  # 1D vector means bias
17  parameter.data.fill_(0)
18  else:
19  fan_in = parameter.data.size(0)
20  fan_out = parameter.data.size(1)
21  parameter.data.normal_(0, math.sqrt(2 / (fan_in + fan_out)))
