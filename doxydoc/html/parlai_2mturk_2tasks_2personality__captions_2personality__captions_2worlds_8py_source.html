271  def parley(self):
272  """
273  COMMENTER is given an image, and is told to give a comment for
274  the image
275  """
276  # Initial Message Value
277  control_msg = {'episode_done': False}
278  control_msg['id'] = 'SYSTEM'
279 
280  # First, we give COMMENTER their personality instructions, and image
281  while self.turn_idx < self.num_images:
282  print(self.world_tag + ' is at turn {}...'.format(self.turn_idx))
283  # Send personality + image to turker
284  if not self.multiple_personality:
285  pers_tup = self.agent.personality_generator.pop_personality()
286  self.pers_idx, personality = pers_tup
287  img_tup = self.agent.image_generator.pop_image()
288  self.image_hash, image_path = img_tup
289  img = load_image(image_path)
290  else:
291  pair_tup = self.agent.personality_and_image_generator.pop_pair()
292  self.pair_idx, personality, self.image_hash, image_path = pair_tup
293  img = load_image(image_path)
294  buffered = BytesIO()
295  img.save(buffered, format='JPEG')
296  encoded = str(base64.b64encode(buffered.getvalue()).decode('ascii'))
297  control_msg['image'] = encoded
298  if self.data_type == 'personality':
299  personality_text = (
300  '<b><span style="color:blue">'
301  '{}\n</span></b>'.format(personality.strip())
302  )
303  control_msg['personality_text'] = personality_text
304  control_msg['text'] = self.get_instruction(
305  tag='start', agent_id=self.agent.id, turn_num=self.turn_idx + 1
306  )
307  control_msg['description'] = self.config['task_description']
308  control_msg['task_type_title'] = self.task_type_title
309  self.agent.observe(validate(control_msg))
310  time.sleep(1)
311 
312  # Collect comment from turker
313  offensive_counter = 0
314  while offensive_counter < 3:
315  idx = 0
316  acts = self.acts
317  acts[idx] = self.agent.act(timeout=self.max_resp_time)
318  agent_left = self.check_timeout(acts[idx])
319  if agent_left:
320  break
321  comment = acts[idx]['text']
322  offensive = self.offensive_lang_detector.contains_offensive_language(
323  comment
324  )
325  if offensive:
326  # Tell Turker to not be offensive!
327  offensive_msg = {'id': 'SYSTEM', 'text': OFFENSIVE_MSG}
328  self.agent.observe(validate(offensive_msg))
329  offensive_counter += 1
330  else:
331  break
332 
333  if self.chat_done:
334  break
335  self.data.append(
336  {
337  'comment': comment,
338  'personality': personality,
339  'image_hash': self.image_hash,
340  'image_path': image_path,
341  'contains_offensive_language': offensive,
342  }
343  )
344  self.turn_idx += 1
345 
346  if self.turn_idx == self.num_images:
347  control_msg['text'] = CHAT_ENDED_MSG.format(self.num_images)
348  control_msg['eval'] = True
349  self.agent.observe(validate(control_msg))
350  act = self.agent.act(timeout=self.max_resp_time)
351  self.check_timeout(act)
352  try:
353  ev_val = int(act['eval'])
354  ev_text = act['text']
355  except BaseException: # If there is a disconnect here
356  ev_val, ev_text = (-1, 'Eval not received')
357  self.eval = {ev_val: ev_text}
358  self.chat_done = True
359  return
