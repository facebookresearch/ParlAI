53  def __init__(self, opt, shared=None):
54  """Runs subprocess command to set up remote partner.
55  Only run the subprocess command once: if using multiple threads, tell
56  the partner how many paired agents to set up so that they can manage
57  the multithreading effectively in their environment. (We don't run
58  subprocess.Popen for each thread.)
59  """
60  self.opt = copy.deepcopy(opt)
61  self.address = opt['remote_address']
62  if opt.get('remote_host') and self.address == 'localhost':
63  self.address = '*'
64  self.socket_type = zmq.REP if opt['remote_host'] else zmq.REQ
65  if shared and 'port' in shared:
66  # for multithreading, use specified port
67  self.port = shared['port']
68  else:
69  if 'port' in opt:
70  self.port = opt['port']
71  else:
72  raise RuntimeError(
73  'You need to run RemoteAgent.'
74  + 'add_cmdline_args(argparser) before '
75  + 'calling this class to set up options.'
76  )
77  if opt.get('remote_cmd'):
78  # if available, command to launch partner instance, passing on
79  # some shared parameters from ParlAI
80  # useful especially if "remote" agent is running locally, e.g.
81  # in a different language than python
82  self.process = subprocess.Popen(
83  '{cmd} {port} {numthreads} {args}'.format(
84  cmd=opt['remote_cmd'],
85  port=opt['port'],
86  numthreads=opt['numthreads'],
87  args=opt.get('remote_args', ''),
88  ).split()
89  )
90  self.connect()
91  super().__init__(opt, shared)

93  def connect(self):
94  """Bind or connect to ZMQ socket. Requires package zmq."""
95  context = zmq.Context()
96  self.socket = context.socket(self.socket_type)
97  self.socket.setsockopt(zmq.LINGER, 1)
98  host = 'tcp://{}:{}'.format(self.address, self.port)
99  if self.socket_type == zmq.REP:
100  self.socket.bind(host)
101  else:
102  self.socket.connect(host)
103  print('python thread connected to ' + host)

105  def act(self):
106  """Send message to paired agent listening over zmq."""
107  if self.observation is not None:
108  text = json.dumps(sanitize(self.observation))
109  self.socket.send_unicode(text)
110  reply = self.socket.recv_unicode()

124  def shutdown(self):
125  """Shut down paired listener with <END> signal."""
126  if hasattr(self, 'socket'):
127  try:
128  self.socket.send_unicode('<END>', zmq.NOBLOCK)
129  except zmq.error.ZMQError:
130  # may need to listen first
131  try:
132  self.socket.recv_unicode(zmq.NOBLOCK)
133  self.socket.send_unicode('<END>', zmq.NOBLOCK)
134  except zmq.error.ZMQError:
135  # paired process is probably dead already
136  pass
137  if hasattr(self, 'process'):
138  # try to let the subprocess clean up, but don't wait too long
139  try:
140  self.process.communicate(timeout=1)
141  except subprocess.TimeoutExpired:
142  self.process.kill()

145 class ParsedRemoteAgent(RemoteAgentAgent):

168  def act(self):
169  parsed = {}
170  for k, v in self.observation.items():
171  if type(v) == str:
172  # We split on newlines because we don't treat them as charactes
173  # in the default dictionary but our receiving agent might want
174  # to know where the newlines are in the text block.
175  parsed[k] = self.parse(v, split_lines=True)
176  else:
177  # not a string, see if it's an iterable of strings
178  try:
179  # if there are newlines in the label, it's part of the label
180  parsed[k] = [self.parse(s) for s in v]
181  except TypeError:
182  # oops, it's not. just pass it on.
183  parsed[k] = v
184 
185  super().observe(parsed)
186  reply = super().act()
187  unparsed = {}
188  for k, v in reply.items():
189  # TODO(ahm): this fails if remote agent sends anything other than
190  # vectors, which means that pretty much only .text will work
191  unparsed[k] = self.parse(v)
192  return unparsed

194  def parse(self, s, split_lines=False):
195  """Returns a parsed (list of indices) version of a string s.
196  Optionally return list of vectors for each line in the string in case
197  you need to know where those are.
198  """
199  if split_lines:
200  return [self.dict.parse(line, vec_type=list) for line in s.split('\n')]
201  else:
202  return self.dict.parse(s, vec_type=list)
