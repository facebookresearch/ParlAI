22 def load_embeddings(opt, word_dict):
23  """Initialize embeddings from file of pretrained vectors."""
24  embeddings = torch.Tensor(len(word_dict), opt['embedding_dim'])
25  embeddings.normal_(0, 1)
26  opt['embedding_file'] = modelzoo_path(opt.get('datapath'), opt['embedding_file'])
27  # Fill in embeddings
28  if not opt.get('embedding_file'):
29  raise RuntimeError('Tried to load embeddings with no embedding file.')
30  with open(opt['embedding_file']) as f:
31  for line in f:
32  parsed = line.rstrip().split(' ')
33  if len(parsed) > 2:
34  assert len(parsed) == opt['embedding_dim'] + 1
35  w = normalize_text(parsed[0])
36  if w in word_dict:
37  vec = torch.Tensor([float(i) for i in parsed[1:]])
38  embeddings[word_dict[w]].copy_(vec)
39 
40  # Zero NULL token
41  embeddings[word_dict['__NULL__']].fill_(0)
42 
43  return embeddings

66 def vectorize(opt, ex, word_dict, feature_dict):
67  """Turn tokenized text inputs into feature vectors."""
68  # Index words
69  document = torch.LongTensor([word_dict[w] for w in ex['document']])
70  question = torch.LongTensor([word_dict[w] for w in ex['question']])
71 
72  # Create extra features vector
73  features = torch.zeros(len(ex['document']), len(feature_dict))
74 
75  # f_{exact_match}
76  if opt['use_in_question']:
77  q_words_cased = set([w for w in ex['question']])
78  q_words_uncased = set([w.lower() for w in ex['question']])
79  for i in range(len(ex['document'])):
80  if ex['document'][i] in q_words_cased:
81  features[i][feature_dict['in_question']] = 1.0
82  if ex['document'][i].lower() in q_words_uncased:
83  features[i][feature_dict['in_question_uncased']] = 1.0
84 
85  # f_{tf}
86  if opt['use_tf']:
87  counter = Counter([w.lower() for w in ex['document']])
88  l = len(ex['document'])
89  for i, w in enumerate(ex['document']):
90  features[i][feature_dict['tf']] = counter[w.lower()] * 1.0 / l
91 
92  if opt['use_time'] > 0:
93  # Counting from the end, each (full-stop terminated) sentence gets
94  # its own time identitfier.
95  sent_idx = 0
96 
97  def _full_stop(w):
98  return w in {'.', '?', '!'}
99 
100  for i, w in reversed(list(enumerate(ex['document']))):
101  sent_idx = sent_idx + 1 if _full_stop(w) else max(sent_idx, 1)
102  if sent_idx < opt['use_time']:
103  features[i][feature_dict['time=T%d' % sent_idx]] = 1.0
104  else:
105  features[i][feature_dict['time>=T%d' % opt['use_time']]] = 1.0
106 
107  # Maybe return without target
108  if ex['target'] is None:
109  return document, features, question
110 
111  # ...or with target
112  start = torch.LongTensor([ex['target'][0]])
113  end = torch.LongTensor([ex['target'][1]])
114 
115  return document, features, question, start, end

118 def batchify(batch, null=0, cuda=False):
119  """Collate inputs into batches."""
120  NUM_INPUTS = 3
121  NUM_TARGETS = 2
122  NUM_EXTRA = 2
123 
124  # Get elements
125  docs = [ex[0] for ex in batch]
126  features = [ex[1] for ex in batch]
127  questions = [ex[2] for ex in batch]
128  text = [ex[-2] for ex in batch]
129  spans = [ex[-1] for ex in batch]
130 
131  # Batch documents and features
132  max_length = max([d.size(0) for d in docs])
133  x1 = torch.LongTensor(len(docs), max_length).fill_(null)
134  x1_mask = torch.ByteTensor(len(docs), max_length).fill_(1)
135  x1_f = torch.zeros(len(docs), max_length, features[0].size(1))
136  for i, d in enumerate(docs):
137  x1[i, : d.size(0)].copy_(d)
138  x1_mask[i, : d.size(0)].fill_(0)
139  x1_f[i, : d.size(0)].copy_(features[i])
140 
141  # Batch questions
142  max_length = max([q.size(0) for q in questions])
143  x2 = torch.LongTensor(len(questions), max_length).fill_(null)
144  x2_mask = torch.ByteTensor(len(questions), max_length).fill_(1)
145  for i, q in enumerate(questions):
146  x2[i, : q.size(0)].copy_(q)
147  x2_mask[i, : q.size(0)].fill_(0)
148 
149  # Pin memory if cuda
150  if cuda:
151  x1 = x1.pin_memory()
152  x1_f = x1_f.pin_memory()
153  x1_mask = x1_mask.pin_memory()
154  x2 = x2.pin_memory()
155  x2_mask = x2_mask.pin_memory()
156 
157  # Maybe return without targets
158  if len(batch[0]) == NUM_INPUTS + NUM_EXTRA:
159  return x1, x1_f, x1_mask, x2, x2_mask, text, spans
160 
161  # ...Otherwise add targets
162  elif len(batch[0]) == NUM_INPUTS + NUM_EXTRA + NUM_TARGETS:
163  y_s = torch.cat([ex[3] for ex in batch])
164  y_e = torch.cat([ex[4] for ex in batch])
165  return x1, x1_f, x1_mask, x2, x2_mask, y_s, y_e, text, spans
166 
167  # ...Otherwise wrong number of inputs
168  raise RuntimeError('Wrong number of inputs per batch')
