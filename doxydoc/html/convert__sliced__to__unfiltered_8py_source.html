25 def main(config):
26  """Creates .unfiltered files from .sliced files
27 
28  input: a .sliced file of logs (in ParlaiDialog format) from Mturk task 1, each of
29  which starts with an initial prompt or topic request, and ends with a y_exp
30  output: a .unfiltered file (in self-feeding format) with every utterance output by
31  bot used as a label (i.e., act as though the bot was a human and we want to
32  train in a normal supervised way).
33  """
34  examples = []
35  episodes = [e for e in extract_parlai_episodes(config['infile'])]
36  for episode in episodes:
37  history = []
38  num_parleys = len(episode)
39  for i, parley in enumerate(episode):
40  if i == 0: # Don't include the topic request
41  history.append(parley.response)
42  continue
43  elif i == num_parleys - 1:
44  # ultimate turn was correction request and explanation
45  continue
46  else:
47  example = Parley(
48  context=add_person_tokens(history, last_speaker=1),
49  response=parley.context, # What the bot said
50  )
51  examples.append(example)
52  history.append(parley.context)
53  history.append(parley.response)
54 
55  # Write new episodes to self-feeding format
56  with open(config['outfile'], 'w') as outfile:
57  for ex in examples:
58  outfile.write(json.dumps(ex.to_dict()) + '\n')
59 
60  print(
61  f"Extracted {len(examples)} self-feeding episodes out of "
62  f"{len(episodes)} parlai episodes and wrote them to {config['outfile']}."
63  )
64 
65 
66 if __name__ == '__main__':
67  config = setup_args()
68  assert config['infile'].endswith('.sliced')
69  assert config['outfile'].endswith('.unfiltered')
70  main(config)
