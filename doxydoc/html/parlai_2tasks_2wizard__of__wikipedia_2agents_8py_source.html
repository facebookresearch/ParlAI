47 def _get_chosen_title_and_sent(wizard_entry, k_dict):
48  """
49  Return a nicely extracted title and chosen sentence.
50  :return: pair (title, sentence)
51  """
52  title_dict = wizard_entry.get('checked_passage', 'none')
53  sentence_dict = wizard_entry.get('checked_sentence', {})
54  title = None
55  sentence = None
56  if sentence_dict == {}:
57  title = sentence = TOKEN_NOCHOSEN
58  else:
59  sentence = _first_val(sentence_dict)
60  if sentence == TOKEN_NOCHOSEN:
61  title = TOKEN_NOCHOSEN
62  else:
63  title = ''
64  # cand_title1 is the title from the `checked_passage`
65  cand_title1 = _first_val(title_dict) if title_dict else ''
66  # cand_title2 is the extracted title of the passage from the
67  # sentence dict, which is e.g. `self_Vermont_Syrup_0`
68  cand_title2 = ' '.join(_first_key(sentence_dict).split('_')[1:-1])
69  if (
70  cand_title1
71  and cand_title1 in k_dict
72  and sentence in k_dict[cand_title1]
73  ):
74  title = cand_title1
75  elif cand_title2 in k_dict and sentence in k_dict[cand_title2]:
76  title = cand_title2
77  else: # neither candidate title is the right one
78  for t, passage in k_dict.items():
79  if sentence in passage:
80  title = t
81  break
82 
83  return title, sentence

686  def get_span(self, one, two):

688  return None

690  two_space = self.space_punctuation(two)

706  t_2.append(temp_2)
