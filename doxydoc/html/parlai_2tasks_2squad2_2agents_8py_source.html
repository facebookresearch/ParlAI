154 class TitleTeacher(DefaultTeacher):
155  """This version of SquAD inherits from the Default Teacher. The only
156  difference is that the 'text' field of an observation will contain
157  the title of the article separated by a newline from the paragraph and the
158  query.
159  Note: The title will contain underscores, as it is the part of the link for
160  the Wikipedia page; i.e., the article is at the site:
161  https://en.wikipedia.org/wiki/{TITLE}
162  Depending on your task, you may wish to remove underscores.
163  """
164

166  self.id = 'squad_title'

174  for article in self.squad:
175  title = article['title']
176  # each paragraph is a context for the attached questions
177  for paragraph in article['paragraphs']:
178  # each question is an example
179  for qa in paragraph['qas']:
180  question = qa['question']

186  yield ('\n'.join([title, context, question]), answers), True

189 class SentenceIndexTeacher(IndexTeacher):

211  article = self.squad[article_idx]
212  paragraph = article['paragraphs'][paragraph_idx]
213  qa = paragraph['qas'][qa_idx]
214  context = paragraph['context']
215  question = qa['question']
216 
217  answers = []
218  if not qa['is_impossible']:
219  answers = [a['text'] for a in qa['answers']]
220 
221  # temporarily remove '.', '?', '!' from answers for proper sentence
222  # tokenization
223  edited_answers = []
224  for answer in answers:
225  new_answer = answer.replace('.', '').replace('?', '').replace('!', '')
226  context = context.replace(answer, new_answer)
227  edited_answers.append(new_answer)
228 
229  edited_sentences = self.sent_tok.tokenize(context)
230  sentences = []
231 
232  for sentence in edited_sentences:
233  for i in range(len(edited_answers)):
234  sentence = sentence.replace(edited_answers[i], answers[i])
235  sentences.append(sentence)
236 
237  for i in range(len(edited_answers)):
238  context = context.replace(edited_answers[i], answers[i])
239 
240  labels = []
241  label_starts = []
242  for sentence in sentences:
243  for answer in answers:
244  if answer in sentence and sentence not in labels:
245  labels.append(sentence)
246  label_starts.append(context.index(sentence))
247  break
248  if len(labels) == 0:
249  labels.append('')

252  if qa['is_impossible']:
253  plausible = qa['plausible_answers']
