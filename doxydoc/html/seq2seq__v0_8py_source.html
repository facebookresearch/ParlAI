85 class Seq2seqAgent(Agent):
86  """Agent which takes an input sequence and produces an output sequence.
87 
88  This model supports encoding the input and decoding the output via one of
89  several flavors of RNN. It then uses a linear layer (whose weights can
90  be shared with the embedding layer) to convert RNN output states into
91  output tokens. This model currently uses greedy decoding, selecting the
92  highest probability token at each time step.
93 
94  For more information, see the following papers:
95  - Neural Machine Translation by Jointly Learning to Align and Translate
96  `(Bahdanau et al. 2014) <arxiv.org/abs/1409.0473>`_
97  - Sequence to Sequence Learning with Neural Networks
98  `(Sutskever et al. 2014) <arxiv.org/abs/1409.3215>`_
99  - Effective Approaches to Attention-based Neural Machine Translation
100  `(Luong et al. 2015) <arxiv.org/abs/1508.04025>`_
101  """

436  try:
437  import torchtext.vocab as vocab
438  except ImportError as ex:
439  print('Please install torch text with `pip install torchtext`')
440  raise ex
441  pretrained_dim = 300
442  if opt['embedding_type'].startswith('glove'):
443  if 'twitter' in opt['embedding_type']:
444  init = 'glove-twitter'
445  name = 'twitter.27B'
446  pretrained_dim = 200
447  else:
448  init = 'glove'
449  name = '840B'
450  embs = vocab.GloVe(
451  name=name,

456  )
457  elif opt['embedding_type'].startswith('fasttext'):

464  )

501 
502  if self.use_cuda:
503  self.criterion.cuda()

527  self.model.decoder.e2s.weight.requires_grad = False

543  )
544  if self.use_cuda:
545  for state in self.optimizer.state.values():
546  for k, v in state.items():
547  if isinstance(v, torch.Tensor):
548  state[k] = v.cuda()

557 
558  Print out each added key and each overriden key.
559  Only override args specific to the model.

575  # skip non-model args

578  print('[ Adding new option: | {k}: {v} | ]'.format(k=k, v=v))
579  elif self.opt[k] != v:

582  k=k, old=self.opt[k], v=v

584  )

587  print(
588  '[ No dictionary path detected, trying to load previous '

632  def report(self):
633  """Report loss and perplexity from model's perspective.
634 
635  Note that this includes predicting __END__ and __UNK__ tokens and may
636  differ from a truly independent measurement.
637  """
638  m = {}
639  num_tok = self.metrics['num_tokens']
640  if num_tok > 0:
641  if self.metrics['correct_tokens'] > 0:
642  m['token_acc'] = self.metrics['correct_tokens'] / num_tok
643  m['loss'] = self.metrics['loss'] / num_tok
644  try:
645  m['ppl'] = math.exp(m['loss'])
646  except OverflowError:
647  m['ppl'] = float('inf')
648  if self.metrics['total_skipped_batches'] > 0:
649  m['total_skipped_batches'] = self.metrics['total_skipped_batches']
650  for k, v in m.items():
651  # clean up: rounds to sigfigs and converts tensors to floats
652  m[k] = round_sigfigs(v, 4)
653  return m

703  Update the model using the targets if available, otherwise rank
704  candidates as well if they are available and param is set.
705  """
706  predictions, cand_preds = None, None
707  if is_training:
708  self.model.train()
709  self.zero_grad()
710  out = None
711  try:
712  out = self.model(xs, ys, rank_during_training=cands is not None)
713  # generated response
714  _preds, scores, cand_preds = out[0], out[1], out[2]
715 
716  score_view = scores.view(-1, scores.size(-1))
717  loss = self.criterion(score_view, ys.view(-1))
718  # save loss to metrics
719  y_ne = ys.ne(self.NULL_IDX)
720  target_tokens = y_ne.long().sum().item()
721  correct = ((ys == _preds) * y_ne).sum().item()
722  self.metrics['correct_tokens'] += correct
723  self.metrics['loss'] += loss.item()
724  self.metrics['num_tokens'] += target_tokens
725  loss /= target_tokens # average loss per token
726  loss.backward()
727  except RuntimeError as e:
728  # catch out of memory exceptions during fwd/bck (skip batch)
729  if 'out of memory' in str(e):
730  print(
731  '| WARNING: ran out of memory, skipping batch. '
732  'if this happens frequently, decrease batchsize or '
733  'truncate the inputs to the model.'
734  )
735  self.metrics['total_skipped_batches'] += 1
736  return predictions, cand_preds
737  else:
738  raise e
739  self.update_params()
740  else:
741  self.model.eval()
742  out = self.model(
743  xs,
744  ys=None,
745  cands=cands,
746  valid_cands=valid_cands,
747  beam_size=self.beam_size,
748  topk=self.topk,
749  )
750  predictions, cand_preds = out[0], out[2]
751 
752  if ys is not None:
753  # calculate loss on targets
754  out = self.model(xs, ys)
755  scores = out[1]
756  score_view = scores.view(-1, scores.size(-1))
757  loss = self.criterion(score_view, ys.view(-1))
758  # save loss to metrics
759  target_tokens = ys.ne(self.NULL_IDX).long().sum().item()
760  self.metrics['loss'] += loss.item()
761  self.metrics['num_tokens'] += target_tokens
762

786  if ys is not None:

938 class PerplexityEvaluatorAgent(Seq2seqAgent):
939  """Subclass for doing standardized perplexity evaluation.
940 
941  This is designed to be used in conjunction with the PerplexityWorld at
942  parlai/scripts/eval_ppl.py. It uses the `next_word_probability` function
943  to calculate the probability of tokens one token at a time.
944  """
945 
946  def __init__(self, opt, shared=None):
947  super().__init__(opt, shared)
948  self.prev_enc = None
949  self.last_xs = None

951  def next_word_probability(self, partial_out):
952  """Return probability distribution over next words given an input and
953  partial true output. This is used to calculate the per-word perplexity.
954 
955  Arguments:
956  observation -- input observation dict
957  partial_out -- list of previous "true" words
958 
959  Returns a dict, where each key is a word and each value is a probability
960  score for that word. Unset keys assume a probability of zero.
961 
962  e.g.
963  {'text': 'Run test program.'}, ['hello'] => {'world': 1.0}
964  """
965  obs = self.observation
966  obs['eval_labels'] = [' '.join(partial_out)]
967  batch = self.vectorize([obs])
968 
969  xs, ys = batch[0], batch[1]
970  if (
971  self.prev_enc is not None
972  and self.last_xs is not None
973  and (
974  xs.shape[1] != self.last_xs.shape[1]
975  or (xs == self.last_xs).sum().item() != xs.shape[1]
976  )
977  ):
978  # reset prev_enc, this is a new input
979  self.prev_enc = None
980  self.last_xs = xs
981 
982  self.model.eval()
983  # no need to predict farther ahead
984  # if you pass in any ys, this will be ignored
985  self.model.longest_label = 1
986  out = self.model(
987  xs, ys=(ys if len(partial_out) > 0 else None), prev_enc=self.prev_enc
988  )
989  scores, self.prev_enc = out[1], out[4]
990  # scores is bsz x seqlen x num_words, so select probs of current index
991  probs = F.softmax(scores.select(1, -1), dim=1).squeeze()
992  dist = mydefaultdict(lambda: 1e-7) # default probability for any token
993  for i in range(len(probs)):
994  dist[self.dict[i]] = probs[i].item()
995  return dist
