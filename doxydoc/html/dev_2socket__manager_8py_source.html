218  def _safe_send(self, data, force=False):
219  if not self.alive and not force:
220  # Try to wait a half second to send a packet
221  timeout = 0.5
222  while timeout > 0 and not self.alive:
223  time.sleep(0.1)
224  timeout -= 0.1
225  if not self.alive:
226  # don't try to send a packet if we're still dead
227  return False
228  try:
229  with self.send_lock:
230  self.ws.send(data)
231  except websocket.WebSocketConnectionClosedException:
232  # The channel died mid-send, wait for it to come back up
233  return False
234  except BrokenPipeError: # noqa F821 we don't support p2
235  # The channel died mid-send, wait for it to come back up
236  return False
237  except AttributeError:
238  # _ensure_closed was called in parallel, self.ws = None
239  return False
240  except Exception as e:
241  shared_utils.print_and_log(
242  logging.WARN,
243  'Unexpected socket error occured: {}'.format(repr(e)),
244  should_print=True,
245  )
246  return False
247  return True

290  def _send_packet(self, packet, send_time):
291  """Sends a packet, blocks if the packet is blocking"""
292  # Send the packet
293  pkt = packet.as_dict()
294  if pkt['data'] is None or packet.status == Packet.STATUS_SENT:
295  return # This packet was _just_ sent.
296  shared_utils.print_and_log(logging.DEBUG, 'Send packet: {}'.format(packet))
297 
298  result = self._safe_send(json.dumps({'type': pkt['type'], 'content': pkt}))
299  if not result:
300  # The channel died mid-send, wait for it to come back up
301  self.sending_queue.put((send_time, packet))
302  return
303 
304  packet.status = Packet.STATUS_SENT
305  if packet.ack_func is not None:
306  packet.ack_func(packet)

361  def on_disconnect(*args):
362  """Disconnect event is a no-op for us, as the server reconnects

373  def on_message(*args):
374  """Incoming message handler for SERVER_PONG, MESSAGE_BATCH,
375  AGENT_DISCONNECT, SNS_MESSAGE, SUBMIT_MESSAGE, AGENT_ALIVE
376  """
377  packet_dict = json.loads(args[1])
378  if packet_dict['type'] == 'conn_success': # TODO make socket func
379  self.alive = True
380  return
381  # The packet inherits the socket function type
382  packet_dict['content']['type'] = packet_dict['type']
383  packet = Packet.from_dict(packet_dict['content'])
384  if packet is None:
385  return
386  packet_id = packet.id
387  packet_type = packet.type
388  if packet_id in self.processed_packets:
389  return # no need to handle already-processed packets
390 
391  # Note to self that this packet has already been processed,
392  # and shouldn't be processed again in the future
393  self.processed_packets.add(packet_id)
394  if packet_type == data_model.SERVER_PONG:
395  # Incoming pong means our ping was returned
396  self.pings_without_pong = 0
397  elif packet_type == data_model.AGENT_ALIVE:
398  # agent is connecting for the first time
399  self.alive_callback(packet)
400  self.processed_packets.add(packet_id)
401  elif packet_type == data_model.MESSAGE_BATCH:
402  # Any number of agents are included in this message batch,
403  # so process each individually
404  batched_packets = packet.data['messages']
405  for batched_packet_dict in batched_packets:
406  batched_packet_dict['type'] = data_model.AGENT_MESSAGE
407  batched_packet = Packet.from_dict(batched_packet_dict)
408  self.message_callback(batched_packet)
409  elif packet_type == data_model.AGENT_DISCONNECT:
410  # Server detected an agent disconnect, extract and remove
411  disconnected_id = packet.data['connection_id']
412  worker_id, assign_id = self.worker_assign_ids[disconnected_id]
413  self.socket_dead_callback(worker_id, assign_id)
414  elif packet_type == data_model.SNS_MESSAGE:
415  # Treated as a regular message
416  self.message_callback(packet)
417  elif packet_type == data_model.SUBMIT_MESSAGE:
418  # Treated as a regular message
419  self.message_callback(packet)

421  def run_socket(*args):
422  url_base_name = self.server_url.split('https://')[1]
423  protocol = "wss"
424  if url_base_name in ['localhost', '127.0.0.1']:
425  protocol = "ws"
426  while self.keep_running:
427  try:
428  sock_addr = "{}://{}:{}/".format(protocol, url_base_name, self.port)

438  except Exception as e:

442  )

461  )
462  try:

464  except Exception:
465  pass

475  def channel_thread(self):
476  """Handler thread for monitoring all channels"""
477  # while the thread is still alive
478  while not self.is_shutdown:
479  if self.ws is None:
480  # Waiting for websocket to come back alive
481  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
482  continue
483  self._try_send_world_ping()
484  try:
485  # Get first item in the queue, check if can send it yet
486  item = self.sending_queue.get(block=False)
487  t = item[0]
488  if time.time() < t:
489  # Put the item back into the queue,
490  # it's not time to pop yet
491  self.sending_queue.put(item)
492  else:
493  # Try to send the packet
494  packet = item[1]
495  if not packet:
496  # This packet was deleted out from under us
497  continue
498  if packet.status is not Packet.STATUS_SENT:
499  # either need to send initial packet
500  # or resend after a failed send
501  self._send_packet(packet, t)
502  except Empty:
503  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
504  except Exception as e:
505  shared_utils.print_and_log(
506  logging.WARN,
507  'Unexpected error occurred in socket handling thread: '
508  '{}'.format(repr(e)),
509  should_print=True,
510  )
